28   def get(key:String) : Future[Option[ChannelBuffer]] = {
29     val p = Promise[Option[ChannelBuffer]]
30     val ck = getCacheKey(key)
31     ck onSuccess { cKey =>
32       val f = client.get(cKey)
33       f onSuccess { data =>
34         p.setValue(data)
35       } onFailure { ex =>
36         p.setException(ex)
37       }
38     }
39     p
40   }

96   def getCacheKey(key:String) : Future[String] = {
97     val p = Promise[String]
98     val currentSet = _getCurrentSet
99     currentSet onSuccess { cKey =>
100       p.setValue(_getKey(key, cKey))
101     } onFailure { ex =>
102       p.setValue(_getKey(key, -1))
103     }
104     p
105   }
106 
107   private def _getKey(key:String, set:Int) : String = {
108     val setKeyPrefix = if (set > -1) set.toString else ""
109     setKeyPrefix + "." + config.namespace + "." + key
110   }
111 
112   private def _getCurrentSet : Future[Int] = {
113     val p = Promise[Int]
114     if (config.setKey.isEmpty) {
115       p.setValue(-1)
116     } else {
117       val f = client.get(config.setKey)
118       f onSuccess { reply =>
119         reply match {
120           case Some(buffer) => p.setValue(buffer.readInt())
121           case None => p.setValue(-1)
122         }
123       } onFailure { reply =>
124         p.setValue(-1)
125       }
126     }
127     p
128   }
