109   override def preStart(): Unit = {
110     super.preStart()
111 
112     // Register as a handler
113     MessageService.registerMediator(self)
114     // Register for cluster information
115     if (cluster.isDefined) {
116       require(!cluster.get.isTerminated, "Cluster node must not be terminated")
117       cluster.get.subscribe(self, initialStateMode = InitialStateAsSnapshot,
118         classOf[MemberEvent], classOf[ReachabilityEvent])
119 
120       // Set a timeout so that we will not be stuck if there are no other nodes to gossip with
121       context.system.scheduler.scheduleOnce(15 seconds, self, InitialShareTimeout)
122     }
123     else {
124       log.info("Clustering is disabled to message handling will only operate locally")
125       context.parent ! MessagingStarted
126     }

128     log.info("Message processor started: {}", context.self.path)
129   }

131   override def postStop(): Unit = {
132     // Cancel our schedule sharing task
133     if (shareTask.isDefined) {
134       shareTask.get.cancel()
135     }

137     // Un-register as a handler
138     MessageService.unregisterMediator(self)
139     if (cluster.isDefined && !cluster.get.isTerminated) {
140       // Un-register for cluster information
141       cluster.get unsubscribe self
142     }

259   /**
260    * Add the subscription
261    * @param message the Subscribe message
262    */
263   private def subscribe(message: Subscribe): Unit = {
264     val lr = registry(selfAddress)
265     val v = lr.clock.counter + 1
266     val currentTime = System.currentTimeMillis
267     val newClock = VectorClock(v, currentTime)
268     val sub = Subscription(message.ref, message.localOnly)
269 
270     message.topics foreach {
271       topic =>
272         val newLr = registry(selfAddress).copy(
273           address = selfAddress,
274           clock = lr.clock.copy(counter = v, time = currentTime),
275           availableRemote = true,
276           content = lr.content.get(topic).map {
277             topicEntry =>
278               val newEntry = TopicEntry(
279                 topic,
280                 newClock,
281                 topicEntry.subscriptions ++ List(sub))
282               lr.content.updated(topic, newEntry)
283           }.getOrElse {
284             lr.content + (topic -> TopicEntry(topic, newClock, Set(sub)))
285           }
286         )
287 
288         registry += (selfAddress -> newLr)
289 
290         log.info("The actor [{}] is subscribing to the topic {}", message.ref.path, topic)
291         // Make sure we have a topic actor
292         val encode = URLEncoder.encode(topic, "utf-8")
293         context.child(encode) match {
294           case Some(childRef) => childRef forward message
295           case None =>
296           // Do nothing since the topic actor does not currently exist and it will be created
297           // whenever an actual message is send to it
298         }
299 
300         publishSubscriptionEvent(true, topic, message.ref)
301         sender() ! SubscribeAck(message)
302     }
303 
304     context watch message.ref
305   }

307   /**
308    * Remove the subscription
309    * @param message the Unsubscribe message
310    */
311   private def unsubscribe(message: Unsubscribe): Unit = {
312     val v = registry(selfAddress).clock.counter + 1
313     val currentTime = System.currentTimeMillis
314 
315     message.topics foreach {
316       topic =>
317         val lr = registry(selfAddress)
318         // Update the registry
319         lr.content.get(topic).map {
320           topicEntry =>
321             val newEntry = (selfAddress -> lr.copy(
322               clock = lr.clock.copy(counter = v, time = currentTime),
323               content = lr.content.updated(topic, topicEntry.copy(
324                 topic = topic,
325                 clock = lr.clock.copy(counter = v, time = currentTime),
326                 subscriptions = topicEntry.subscriptions.filterNot(_.subscriber == message.ref)))
327             ))
328 
329             // Now check to see if we have any remaining subscriptions for this topic. If not, then we
330             // can remove the topic all together
331             if (newEntry._2.content(topic).subscriptions.isEmpty) {
332               registry += (selfAddress -> newEntry._2.copy(
333                 content = newEntry._2.content - topic))
334             }
335             else {
336               registry += newEntry
337             }
338 
339         }.getOrElse(log.warning("Received unsubscribe from topic {} which was not subscribed to", topic))
340 
341         log.info("The actor [{}] is unsubscribing to the topic {}", message.ref.path, topic)
342 
343         // Send a message to the topic actor to remove the subscription
344         context.child(URLEncoder.encode(topic, "utf-8")) match {
345           case Some(g) => g forward message
346           case None =>
347           // Do nothing since the topic actor does not currently exist and it will be created
348           // whenever an actual message is send to it
349         }
350 
351         publishSubscriptionEvent(false, topic, message.ref)
352         sender() ! UnsubscribeAck(message)
353     }
354 
355     // If we have no more subscriptions for this actor then we should unwatch it
356     val refs = (for {
357       entries <- registry.values
358       subs <- entries.content.collect {
359         case m => m._2.subscriptions
360       }
361       sub <- subs
362       if (sub.subscriber.equals(message.ref))
363     } yield sub).toSet
364 
365     if (refs.isEmpty) {
366       context unwatch message.ref
367     }
368   }

370   /**
371    * An actor we have been watching has been terminated. We will now
372    * update our internal registry for subscriptions that are related
373    * to the given actor
374    * @param actorRef the actor that has terminated
375    */
376   private def terminated(actorRef: ActorRef): Unit = {
377 
378     // Get all of the topics for this actor
379     val topics = (for {
380       content <- registry(selfAddress).content
381       sub <- content._2.subscriptions
382       if (sub.subscriber.equals(actorRef))
383     } yield content._1).toSeq.distinct
384 
385     log.info("The actor [{}] has been terminated and it's subscriptions will be removed", actorRef.path)
386     unsubscribe(Unsubscribe(topics, actorRef))
387   }

389   /**
390    * Forward the given message to the actor for the
391    * given topic.
392    * @param message the message to forward
393    */
394   private def forwardToTopic(message: MessageCommand): Unit = {
395     val encode = URLEncoder.encode(message.topic, "utf-8")
396 
397     context.child(encode) match {
398       case Some(g) => g forward message
399       case None =>
400         // If there is no topic actor then we need to "seed" it and then forward the message
401         val refs = (for {
402           entries <- registry.values
403           subs <- entries.content.collect {
404             case m if m._1.equals(message.topic) => m._2.subscriptions
405           }
406           if (!subs.isEmpty)
407           sub <- subs
408         } yield sub).toSet
409 
410         // TODO - What if no subscribers
411         if (!refs.isEmpty) {
412           val childRef = context.actorOf(MessagingTopicActor.props(selfAddress, trashInterval, refs), name = encode)
413           childRef forward message
414         }
415         else {
416           log.warning("The message to the topic {} could not be pushed because there were no subscribers", message.topic)
417         }
418     }
419   }

456   /**
457    * Determine if the local state is newer then the remote state and then
458    * act accordingly. If there are deltas then send those back to the caller.
459    * @param remoteVersions the state from a remote peer
460    * @param sourcePath the Path for the remote actor that is sharing this information
461    */
462   private def verifyVersions(remoteVersions: Map[Address, Long], sourcePath: ActorPath): Unit = {
463 
464     if (!nodes(sender().path.address)) {
465       log.info("Ignoring received subscription information status from unknown node [{}] ", sender().path)
466     }
467     else {
468       log.debug(s"Verifying versions from {}", sourcePath)
469 
470       // See if our data is "newer" then the remote service
471       val delta = collectDelta(remoteVersions)
472       val newer = remoteHasNewerVersion(remoteVersions)
473       if (delta.isEmpty && !newer) {
474         if (shareTask.isEmpty) {
475           self ! InitialShareComplete
476         }
477       }
478       else {
479         if (delta.nonEmpty) {
480           log.debug("The subscription information for {} is newer then {} so we are sending the delta back", self.path, sourcePath)
481           sender() ! Delta(delta)
482         }
483         // Now check and see if any of the remote data is newer then ours
484         if (newer) {
485           log.debug("The subscription information for {} is newer then {} so we are asking for the delta back", sourcePath, self.path)
486           sender() ! Status(versions = localVersions) // it will reply with Delta
487         }
488       }
489     }
490   }

492   /**
493    * Determine the deltas between the local state and one from a remote peer
494    * @param remoteVersions the state from a remote peer
495    * @return the deltas
496    */
497   private def collectDelta(remoteVersions: Map[Address, Long]): immutable.Iterable[RegistryEntry] = {
498     // Combines the local and remote versions. If the local version does not exist then it
499     // will be represented by a version of 0.
500     val local = localVersions
501 
502     // Get the newer local nodes
503     local.collect {
504       // If the remote entry is missing or if our local version is newer then pull it out here
505       case (add, v) if !remoteVersions.contains(add) || v > remoteVersions(add) =>
506         val entry = registry(add)
507         val remote = remoteVersions.get(add)
508         if (remote.isDefined) {
509           log.debug(remoteVersions.mkString(","))
510           log.debug(s"Version {} for {} is older then the local version of {}", remote.get, add, v)
511         }
512         else {
513           log.debug(s"The registry for {} is not present in the gossiped subscriptions. Adding it to the deltas.", add)
514         }
515 
516         // Grab the non-local subscriptions for this node
517         val nonLocal = entry.content.filter {
518           case (_, value) => remote.isEmpty || value.clock.counter > remote.get
519         } map {
520           d =>
521             d._1 -> d._2.copy(subscriptions = d._2.subscriptions.filter(_.localOnly == false))
522         }
523         registry(add).copy(content = nonLocal)
524 
525     }
526   }

528   /**
529    * Update our internal state of subscriptions with the remote deltas
530    * @param deltas The deltas
531    */
532   private def updateDeltas(deltas: immutable.Iterable[RegistryEntry]): Unit = {
533     // Reply from Status message in the gossip chat
534     // the Delta contains potential updates (newer versions) from the other node
535 
536     // Only accept deltas/buckets from known nodes, otherwise there is a risk of
537     // adding back entries when nodes are removed
538     if (nodes(sender().path.address)) {
539 
540       // Don't update local registrations
541       deltas.filterNot(_.address == selfAddress) foreach {
542         b =>
543           if (nodes(b.address)) {
544             val entry = registry(b.address)
545             if (b.clock.counter > entry.clock.counter) {
546 
547               // Diff the existing subscriptions with the new version.
548               val removals = (for {
549                 kp <- entry.content
550                 sub <- kp._2.subscriptions
551                 // If the topic is not in the new set or if the specific actor path is not then we know that is has been removed
552                 if (!b.content.contains(kp._1) || b.content(kp._1).subscriptions.filter(_.subscriber.path == sub.subscriber.path).isEmpty)
553               } yield (kp._2.topic, sub.subscriber)).toSet
554 
555               val adds = (for {
556                 kp <- b.content
557                 sub <- kp._2.subscriptions
558                 // If the topic is not in the old set or if the specific actor path is not then we know that is has been added
559                 if (!entry.content.contains(kp._1) || entry.content(kp._1).subscriptions.filter(_.subscriber.path == sub.subscriber.path).isEmpty)
560               } yield (kp._2.topic, sub.subscriber)).toSet
561 
562               // Merge the existing subscriptions with the deltas
563               val newEntry = entry.copy(
564                 clock = entry.clock.copy(counter = b.clock.counter, time = b.clock.time ),
565                 content = entry.content ++ b.content
566               )
567               // Now update the registry
568               registry += (b.address -> newEntry)
569 
570               // These were removed
571               removals foreach {
572                 r => publishSubscriptionEvent(false, r._1, r._2)
573               }
574               // These were added
575               adds foreach {
576                 r => publishSubscriptionEvent(true, r._1, r._2)
577               }
578 
579               log.debug("There is a difference in subscription data so the system will update using the deltas sent to it from {}: {} [remote], {} [local]",
580                 b.address, b.clock, entry.clock)
581 
582               // Now we need to update the subscriptions for any existing topic
583               updateTopicSubscriptions
584             }
585           }
586       }
587 
588       // The sharing process has completed
589       if (shareTask.isEmpty) self ! InitialShareComplete
590     }
591   }

593   /**
594    * Update the child topic actors with the local copy of subscriptions
595    */
596   private def updateTopicSubscriptions: Unit = {
597     context.children foreach {
598       child =>
599         val topic = URLDecoder.decode(child.path.name, "utf-8")
600 
601         val refs = (for {
602           entry <- registry.values
603           if (entry.availableRemote) // Only allow subscriptions for nodes that are currently available
604           subs <- entry.content.collect {
605             case m if m._1.equals(topic) => m._2.subscriptions
606           }
607           if (!subs.isEmpty)
608           sub <- subs
609         } yield sub).toSet
610 
611         log.debug("Sending the updated the subscriptions for the topic {}: {}", topic, refs.map(_.subscriber.path).mkString(","))
612         child ! UpdateSubscriptions(refs)
613     }
614   }

616   /**
617    * Update the availability of subscriptions for a given node
618    * @param members a sequence of tuples with Address and MemberStatus.
619    */
620   private def updateNodeStatus(members: Seq[(Address, MemberStatus)]): Unit = {
621     members foreach {
622       member =>
623 
624         def updateNodeRemoteStatus(entry: RegistryEntry, available: Boolean) = {
625           if (entry != null) {
626             // Update the status of the given node
627             registry += (member._1 -> entry.copy(availableRemote = available))
628           }
629         }
630 
631         member._2 match {
632           case MemberStatus.Joining => // Do nothing
633 
634           case MemberStatus.Up =>
635             // Add the member to the list of nodes
636             nodes += member._1
637             updateNodeRemoteStatus(registry(member._1), true)
638 
639           case Unreachable =>
640             // We will just update the node status so that no messages are sent to those locations, but the subscriptions are retained until the
641             // node is actually removed
642             updateNodeRemoteStatus(registry(member._1), false)
643 
644           case Reachable =>
645             // The node has become reachable again so reinstate it.
646             updateNodeRemoteStatus(registry(member._1), true)
647 
648           case _ => // All other status are either from the node being removed or on it's way out the door
649             // Remove the member from the list of nodes and registry
650             removeNode(member)
651         }
652     }
653 
654     // Now we need to update the subscriptions for any existing topic
655     updateTopicSubscriptions
656   }

658   /**
659    * Remove the member from the list of nodes and registry
660    * @param member
661    */
662   private def removeNode(member: (Address, MemberStatus)): Unit = {
663     if (member._1 == selfAddress) {
664       // If we are asked to be removed then just shut ourself down
665       context stop self
666     }
667     else {
668       nodes -= member._1
669       // Notify listeners of the subscription removals
670       registry(member._1).content.values foreach {
671         c => c.subscriptions foreach {
672           s => publishSubscriptionEvent(false, c.topic, s.subscriber)
673         }
674       }
675       registry -= member._1
676       log.info("The node [{}] is no longer available [{}] and it's subscriptions will be removed", member._1, member._2)
677     }
678   }

680   /**
681    * Publish the subscription command for all listening on the pre-defined topic
682    * {@link com.webtrends.portfolio.communication.MessageService.MessageSubscriptionEventTopic}.
683    * @param topic the topic that changed
684    * @param subscriber the actor ref that is involved
685    */
