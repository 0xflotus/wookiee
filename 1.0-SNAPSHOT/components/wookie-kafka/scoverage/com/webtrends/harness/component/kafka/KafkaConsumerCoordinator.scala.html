82   /**
83    *
84    * @return

86   def initial:Receive = healthReceive orElse configReceive orElse {
87     case NodeData(nodeId, data) =>
88       val assignments = Serialization.read(data)(formats, manifest[List[PartitionAssignment]])
89       processAssignmentEvent(assignments)
90 
91     // ZK data has changed
92     case ZookeeperChildEvent(event) =>
93       // Node registration change
94       event.getType match {
95         // Node data has changed. If the node is local, notify the node actor
96         case CHILD_UPDATED | CHILD_ADDED | CHILD_REMOVED =>
97           val nodeId = event.getData.getPath.split("/").last
98           if (hostname == nodeId) {
99             log.debug(s"Node data for [$nodeId] has been updated. Notifying it")
100             self ! NodeData(nodeId, new String(event.getData.getData, utf8))
101           }
102 
103         case _ =>
104         // To remove a compiler warning
105       }
106 
107     case PrepareForShutdown =>
108       //Stop all workers
109       stopAllWorkers
110       context.become(awaitingShutdown)
111 
112     case BroadcastToWorkers(msg) =>
113       workers withFilter(_._2.started) foreach(_._2.actor.forward(msg))
114   }
115 
116   /**
117    * We got the prepare for shutdown event
118    * @return
119    */
120   def awaitingShutdown:Receive = healthReceive orElse configReceive orElse {
121     case msg:Any => //Just ignore all other messages
122       log.debug(s"Waiting for shutdown ignoring $msg")
123   }
124 
125 
126   def stopAllWorkers = {
127     log.info(s"Stopping ${workers.size} workers")
128     workers foreach(_._2.stopWorker())
129   }
130 
131   override def renewConfiguration() = {
132     super.renewConfiguration()
133     kafkaProxy ! ConfigChange()
134     renewTopicAgeThresholds()
135   }
136 
137   // Will stop all actors that are no longer needed and start any that are now needed
138   def stopUnneededWorkers(assigns: List[PartitionAssignment]) {
139     val specs = assigns.map(_.assignmentName)
140     workers foreach { worker =>
141       if (specs.contains(worker._1)) {
142         worker._2.startWorker()
143       } else {
144         worker._2.stopWorker()
145         workerKafkaHealth.remove(worker._1)
146       }
147     }
148   }
149 
150   def createWorkerIfNeeded(assign: PartitionAssignment) {
151     val spec = assign.assignmentName
152     if (!workers.contains(spec)) {
153       log.info(s"Creating worker ${assign.host}, ${assign.topic}_${assign.partition}")
154       val ref = context.actorOf(Props(topicWorker, kafkaProxy, assign, offsetManager).withDispatcher("worker-dispatcher"), spec)
155       workers.put(spec, new WorkerRef(true, ref))
156     }
157   }
158 
159   def processAssignmentEvent(assigns: List[PartitionAssignment]) {
160     log.debug(s"Processing ${assigns.length} assignments.")
161     stopUnneededWorkers(assigns)
162     assigns foreach { assign => createWorkerIfNeeded(assign) }
163   }
164 
165   // Used to keep track of workers and their states, will start on construction if start=true
166   class WorkerRef(start: Boolean, ref: ActorRef) {
167     var started = start
168     val actor = ref
169     if (started) {
170       startWorker()
171     }
172 
173     override def toString = s"WorkerRef(started=$started, actor=$actor)"
174 
175     def stopWorker() {
176       ref ! PartitionConsumerWorker.Stop
177       started = false
178     }
179 
180     def startWorker() {
181       ref ! Start
182       started = true
183     }
184   }
