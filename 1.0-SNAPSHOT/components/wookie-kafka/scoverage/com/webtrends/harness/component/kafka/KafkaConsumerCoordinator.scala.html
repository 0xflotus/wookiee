60   val offsetManager = context.actorOf(OffsetManager.props(appRootPath), "offset-manager")

62   val workers = mutable.Map.empty[String,WorkerRef]

64   override def preStart() {
65     register(self, ZookeeperChildEventRegistration(self, distributorPaths.assignmentPath))
66     renewTopicAgeThresholds()
67   }

69   override def postStop() {
70     unregister(self, ZookeeperChildEventRegistration(self, distributorPaths.assignmentPath))

79   def receive = initial

81   /**
82    *
83    * @return
84    */
85   def initial:Receive = healthReceive orElse configReceive orElse {
86     case NodeData(nodeId, data) =>
87       val assignments = Serialization.read(data)(formats, manifest[List[PartitionAssignment]])
88       processAssignmentEvent(assignments)
89 
90     // ZK data has changed
91     case ZookeeperChildEvent(event) =>
92       // Node registration change
93       event.getType match {
94         // Node data has changed. If the node is local, notify the node actor
95         case CHILD_UPDATED | CHILD_ADDED | CHILD_REMOVED =>
96           val nodeId = event.getData.getPath.split("/").last
97           if (hostname == nodeId) {
98             log.debug(s"Node data for [$nodeId] has been updated. Notifying it")
99             self ! NodeData(nodeId, new String(event.getData.getData, utf8))
100           }
101 
102         case _ =>
103         // To remove a compiler warning
104       }
105 
106     case PrepareForShutdown =>
107       //Stop all workers
108       stopAllWorkers
109       context.become(awaitingShutdown)
110 
111     case BroadcastToWorkers(msg) =>
112       workers withFilter(_._2.started) foreach(_._2.actor.forward(msg))
113   }
114 
115   /**
116    * We got the prepare for shutdown event
117    * @return
118    */
119   def awaitingShutdown:Receive = healthReceive orElse configReceive orElse {
120     case msg:Any => //Just ignore all other messages
121       log.debug(s"Waiting for shutdown ignoring $msg")
122   }
123 
124 
125   def stopAllWorkers = {
126     log.info(s"Stopping ${workers.size} workers")
127     workers foreach(_._2.stopWorker())
128   }
129 
130   override def renewConfiguration() = {
131     super.renewConfiguration()
132     kafkaProxy ! ConfigChange
133     renewTopicAgeThresholds()
134   }
135 
136   // Will stop all actors that are no longer needed and start any that are now needed
137   def stopUnneededWorkers(assigns: List[PartitionAssignment]) {
138     val specs = assigns.map(_.assignmentName)
139     workers foreach { worker =>
140       if (specs.contains(worker._1)) worker._2.startWorker() else worker._2.stopWorker()
141     }
142   }
143 
144   def createWorkerIfNeeded(assign: PartitionAssignment) {
145     val spec = assign.assignmentName
146     if (!workers.contains(spec)) {
147       log.info(s"Creating worker ${assign.host}, ${assign.topic}_${assign.partition}")
148       val ref = context.actorOf(Props(topicWorker, kafkaProxy, assign, offsetManager), spec)
149       workers.put(spec, new WorkerRef(true, ref))
150     }
151   }
152 
153   def processAssignmentEvent(assigns: List[PartitionAssignment]) {
154     log.debug(s"Processing ${assigns.length} assignments.")
155     stopUnneededWorkers(assigns)
156     assigns foreach { assign => createWorkerIfNeeded(assign) }
157   }
158 
159   // Used to keep track of workers and their states, will start on construction if start=true
160   class WorkerRef(start: Boolean, ref: ActorRef) {
161     var started = start
162     val actor = ref
163     if (started) {
164       startWorker()
165     }
166 
167     override def toString = s"WorkerRef(started=$started, actor=$actor)"
168 
169     def stopWorker() {
170       ref ! PartitionConsumerWorker.Stop
171       started = false
172     }
173 
174     def startWorker() {
175       ref ! Start
176       started = true
177     }
178   }
