61 class KafkaConsumerDistributor(sourceProxy: ActorRef) extends Actor
62   with ActorLoggingAdapter
63   with ZookeeperAdapter
64   with ZookeeperEventAdapter
65   with KafkaSettings
66   with Stash {
67 
68   import KafkaConsumerDistributor._
69 
70   implicit val timeout = Timeout(5 seconds)
71 
72   //Registration zk path
73   val zkNodePath = s"${distributorPaths.nodePath}/$hostname"
74   val hcName = "consumer-distributor"
75   var lastOutOfSyncUpdate = 0L
76   var assignmentLeader: Option[ActorRef] = None
77 
78 
79   override def preStart() {
80     // Register for ZooKeeper State events. When registering we will be sent a message
81     // for the current state
82     register(self, ZookeeperStateEventRegistration(self))
83     register(self, ZookeeperChildEventRegistration(self, distributorPaths.nodePath))
84     register(self, ZookeeperLeaderEventRegistration(self, distributorPaths.leaderPath))
85 
86     self ! RegisterSelf
87   }
88 
89   override def postStop() {
90     unregisterNode()
91     unregister(self, ZookeeperStateEventRegistration(self))
92     unregister(self, ZookeeperChildEventRegistration(self, distributorPaths.nodePath))
93     unregister(self, ZookeeperLeaderEventRegistration(self, distributorPaths.leaderPath))
94   }
95 
96   def receive = initializing
97 
98   /**
99    * Initializing the Distributor, if we can't then
100    * we will schedule a message to keep trying
101    * @return
102    */
103   def initializing: Receive = {
104     case RegisterSelf => registerNode()
105 
106     case NodeRegistrationResult(result) =>
107       //If we can't register, we keep trying
108       result match {
109         case true =>
110           log.info(s"$hostname registered!, moving to registered state")
111           context.become(registered)
112           unstashAll()
113         case false =>
114           context.system.scheduler.scheduleOnce(3 seconds,
115             self,
116             RegisterSelf)
117       }
118 
119     case CheckHealth => sender() ! HealthComponent(hcName,
120                                                     ComponentState.DEGRADED,
121                                                     "Distributor has not registered with ZK yet")
122 
123     case msg => stash()
124   }
125 
126   /**
127    * Once we are registered, we can accept other messages
128    * @return
129    */
130   def registered: Receive = {
131       // Our ZK state has changed
132       case ZookeeperStateEvent(state) =>
133         state match {
134           // If our ZK state is flaky, we may need to reestablish the local node registrations
135           case conn @ (CONNECTED | RECONNECTED) =>
136             log.info(s"Moving back to initializing due to Zk connection State =${conn} ")
137             context.become(initializing)
138             self ! RegisterSelf
139           case _ =>
140             log.debug(s"Zk state event=${state}")
141             relinquishLeadership()
142         }
143 
144       // ZK data has changed
145       case ZookeeperChildEvent(event) =>
146         handleChildEvent(event)
147 
148       // A change in leadership has occurred
149       case ZookeeperLeadershipEvent(leader) =>
150         if (leader) takeLeadership() else relinquishLeadership()
151 
152       case GetRegisteredNodes =>
153         getRegisteredNodes(sender())
154 
155       case GetNodeAssignments =>
156         getNodeAssignments(sender())
157 
158       case GetAssignmentLeader => sender ! AssignmentLeaderResult(assignmentLeader)
159 
160       case CheckHealth =>
161         val msg = assignmentLeader.map(_ => "LEADER - ")
162                                   .getOrElse("") + "Distributor Registered with Zk"
163 
164         sender() ! HealthComponent(hcName, ComponentState.NORMAL, msg)
165 
166       case msg => log.debug(s"Unknown message $msg")
167   }
168 
169   def registerNode() = {
170     log.debug(s"Registering node: $hostname")
171     // When the service is restarted, the ephemeral node from the previous process may not have been deleted yet.
172     // Make sure it has been deleted before attempting to add the new one.
173     deleteNode(zkNodePath).onComplete {
174      case _ =>
175        log.debug(s"Node deleted, $zkNodePath!")
176        setData(zkNodePath, hostname.getBytes(utf8), create = true, ephemeral = true).onComplete {
177          case Success(data) =>
178             log.info(s"Successfully initialized node registration for node [$hostname]")
179 
180             self ! NodeRegistrationResult(true)
181           case Failure(fail) =>
182             log.error(s"Unable to initialize node registration for node [$hostname]. ${fail.getMessage()}")
183             self  ! NodeRegistrationResult(false)
184         }
185     }
186   }
187 
188   def unregisterNode() = {
189     deleteNode(zkNodePath).onComplete {
190       case Success(path) =>
191         log.info(s"Successfully removed node registration for node [$hostname]")
192 
193       case Failure(fail) =>
194         log.error(s"Unable to remove node registration for node [$hostname]. ${fail.getMessage()}")
195     }
196   }
197 
198   def getRegisteredNodes(requestingActor: ActorRef) = {
199     getChildren(s"${distributorPaths.nodePath}", includeData = false).onComplete {
200       case Success(nodes) =>
201         requestingActor ! RegisteredNodes(Some(nodes.map{node => node._1}.toList))
202       case Failure(fail) =>
203         requestingActor ! RegisteredNodes(None)
204         log.error(s"Not requesting node data update because " +
205           s"I am unable to get list of nodes at ${distributorPaths.nodePath}. ${fail.getMessage()}")
206     }
207   }
208 
209   def getNodeAssignments(requestingActor: ActorRef) = {
210     log.debug(s"Fetching assigments for ${distributorPaths.assignmentPath}")
211     getChildren(s"${distributorPaths.assignmentPath}", includeData = true).onComplete {
212       case Success(nodes) =>
213         requestingActor ! NodeAssignments(Some(nodes.map { node => node._1 -> new String(node._2.get, utf8) }.toMap))
214       case Failure(fail) =>
215         log.error(s"Unable to get list of nodes at $appRootPath. ${fail.getMessage()}")
216         requestingActor ! NodeAssignments(None)
217     }
218   }
219 
220   def takeLeadership() = {
221     assignmentLeader.fold({
222       log.info(s"Taking leadership of ${distributorPaths.leaderPath}")
223       val leaderActor = context.actorOf(AssignmentDistributorLeader.props(sourceProxy), "assignment-leader")
224       assignmentLeader = Some(leaderActor)
225       leaderActor ! RefreshNodeAssignments
226     })( _ => log.info(s"Asked to take leadership when I am already the leader"))
227   }
228 
229   def relinquishLeadership() = {
230     assignmentLeader.foreach { actor =>
231       log.info(s"Relinquishing leadership of ${distributorPaths.leaderPath}")
232       context.stop(_)
233     }
234     assignmentLeader = None
235   }
236 
237   def handleChildEvent(event: PathChildrenCacheEvent) = {
238     val eventPath = event.getData.getPath
239 
240     // Node registration change
241     if (eventPath.startsWith(distributorPaths.nodePath)) {
242       event.getType match {
243         case CHILD_ADDED | CHILD_REMOVED =>
244           assignmentLeader.foreach { leader =>
245             log.info("Node child registration changed")
246             val now = System.currentTimeMillis()
247             // Limit churn caused by other nodes restarting or being in a wonky state.
248             if (now > lastOutOfSyncUpdate + 30000) {
249               lastOutOfSyncUpdate = now
250               leader ! RefreshNodeAssignments
251             }
252           }
253 
254         case _ =>
255         // To remove a compiler warning
256       }
257     }
258   }
