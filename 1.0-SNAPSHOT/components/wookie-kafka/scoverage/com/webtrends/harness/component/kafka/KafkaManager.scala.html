28 /**
29  * This class manages the creation of both the KafkaConsumerCoordinator (if 'consumer' is configured)
30  * and the KafkaWriter (if 'producer' is configured). One can retrieve either of these actors by
31  * making a call to GetCoordinator or GetWriters
32  */
33 object KafkaManager {
34   // User this to retrieve the coordinator in charge of managing Workers
35   case object GetCoordinator
36   //Will return the distributor
37   case object GetDistributor
38   // Kafka producer started up if 'producer' is configured
39   var producer: Option[ActorRef] = None
40 }
41 
42 class KafkaManager(name: String) extends Component(name) with KafkaSettings {
43   import KafkaManager._
44 
45   // Consumer coordinator started up if 'consumer' is configured
46   var coordinator: Option[ActorRef] = None
47 
48   // Consumer proxy, started up if 'consumer' is configured. Used to retrieve topic
49   // information and the data to consume
50   var proxy: Option[ActorRef] = None
51 
52   //Distributor actor will start up if a 'consumer' is configured
53   var distributor: Option[ActorRef] = None
54 
55 
56   override def receive = super.receive orElse configReceive orElse {
57     // Use this call to get the Kafka Consumer Coordinator, if configured
58     case GetCoordinator =>
59       sender ! coordinator
60 
61     case GetDistributor =>
62       sender ! distributor
63 
64     case SystemReady =>
65       if (kafkaConfig.hasPath("producer")) {
66         startProducer()
67       }
68       if (kafkaConfig.hasPath("consumer")) {
69         startCoordinator()
70       }
71     case PrepareForShutdown =>
72       coordinator.foreach(_ ! PrepareForShutdown)
73       distributor.foreach(_ ! PrepareForShutdown)
74   }
75 
76   def startProducer() {
77     log.info("Starting producer as wookie-kafka config contained 'producer' config")
78     producer = Some(context.actorOf(KafkaProducer.props(), "producer"))
79   }
80 
81   /**
82    * Start up all necessary actors
83    */
84   def startCoordinator() {
85     if(coordinator.isEmpty) {
86       log.info(s"Starting coordinator class")
87       proxy = Some(context.actorOf(KafkaConsumerProxy.props(), "consumer-proxy"))
88       coordinator = Some(context.actorOf(Props(leader, proxy.get), "consumer-coordinator"))
89       distributor = Some(context.actorOf(KafkaConsumerDistributor.props(proxy.get), "consumer-distributor"))
90     }
91   }
92 
93   override def renewConfiguration() {
94     log.info("Received config change message, checking hosts for changes...")
95     super.renewConfiguration()
96     if (coordinator.isDefined) {
97       coordinator.get ! ConfigChange()
98     }
99   }
100 
101   override def postStop() {
102     stop
103   }
104 
105   // Stops the coordinator, unregistering this node from zookeeper
106   override def stop {
107     context.children foreach { child â‡’
108       log.info(s"Stopping child [${child.path}]")
109       context.stop(child)
110     }
111     coordinator = None
112     proxy = None
113     distributor = None
114     producer = None
115   }
116 }
