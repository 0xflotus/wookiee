80     if(scheduler.isEmpty) scheduler =

91   def distributeAssignments(assignmentInfo: DistributeAssignments) = {
92     val nodes = assignmentInfo.nodes
93     val assignsWithIndex = assignmentInfo.topicResp.partitionsByTopic.zipWithIndex
94     val assignsByNode = (for (
95       node <- nodes.zipWithIndex
96     ) yield {
97       node._1 -> assignsWithIndex.filter { assign =>
98         (assign._2 >= node._2) && (assign._2 - node._2) % nodes.length == 0 }.map { x => x._1 }
99     }).toMap
100 
101     log.debug(s"Setting new node assignments: ${assignsByNode.toString()}")
102     setAssignmentsForNodes(assignsByNode.map { case (nodeId, partitionsByTopic) =>
103       nodeId -> Serialization.write(partitionsByTopic)(formats)
104     })
105   }
106 
107   def setAssignmentsForNodes(data: Map[String, String]) = {
108     // Delete data for nodes which aren't specified in the updated data
109     getChildren(s"${paths.assignmentPath}", includeData = true).onComplete {
110       case Success(existingData) =>
111         // Delete any entries that don't have new data
112         val nodesToDelete = existingData.filter{ case (nodeId, existingData) => !data.contains(nodeId)}
113 
114         Future.traverse(nodesToDelete) { case (nodeId, _) =>
115           deleteNode(s"${paths.assignmentPath}/$nodeId")
116         }.onFailure {
117           case ex => log.error(s"Unable to delete assignment nodes - ${nodesToDelete.map(_._1)}", ex)
118         }
119       case Failure(fail) =>
120         log.error(s"Unable to get list of nodes at ${paths.assignmentPath} " +
121                   s"while attempting to delete stale node data. ${fail.getMessage()}")
122     }
123 
124     Future.traverse(data) { case (nodeId, nodeData) =>
125       setData(s"${paths.assignmentPath}/$nodeId", nodeData.getBytes(utf8), create = true, ephemeral = false)
126     }.onComplete {
127       case Success(path) =>
128         log.debug(s"Successfully set node data")
129       case Failure(fail) =>
130         log.error(s"Unable to set node data. ${fail.getMessage()}")
131     }
132   }

206   }
