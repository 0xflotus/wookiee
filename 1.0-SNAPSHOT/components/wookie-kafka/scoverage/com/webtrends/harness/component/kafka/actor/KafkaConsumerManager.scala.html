52 class KafkaConsumerManager extends Actor with KafkaSettings
53 with ActorLoggingAdapter with ZookeeperAdapter with ZookeeperEventAdapter {
54 
55   def configuredTopics = topicMap.keys.toList
56   import KafkaConsumerManager._
57 
58   val actorName = "Kafka Consumer Manager"
59   val bufferSize = 1024*1024
60 
61   // List of kafka brokers and information about them
62   var brokers = new mutable.HashMap[String, BrokerSpec]()
63 
64   // Holder of consumers connected to each kafka broker
65   val consumersByHost = new mutable.HashMap[String, SimpleConsumer]()
66 
67   context.parent ! HealthComponent(actorName, ComponentState.NORMAL, "Proxy has not acquired brokers yet")
68 
69   override def preStart() = {
70     requestBrokerInfo()
71   }
72 
73   def receive: Receive = configReceive orElse {
74     case TopicPartitionReq =>
75       sender ! TopicPartitionResp(getPartitionLeaders)
76 
77     case FetchConsumer(host) =>
78       sender ! consumersByHost.get(host)
79   }
80 
81   override def renewConfiguration() = {
82     super.renewConfiguration()
83     requestBrokerInfo()
84     log.info("Config renewed, new broker list: {}", brokers)
85   }
86 
87   def requestBrokerInfo() = {
88     log.debug("Using kafka-hosts to set consumed hosts, {}", kafkaSources)
89     brokers.clear()
90     kafkaSources foreach { s =>
91       s._2 foreach { broker =>
92         val hostPort: Array[String] = broker.split(":")
93         val host: String = hostPort(0)
94         val port: Int = if (hostPort.length == 2) hostPort(1).toInt else 9092
95         brokers.put(host, BrokerSpec(host, port, s._1))
96       }
97     }
98 
99     log.debug("Processing brokers {}", brokers.toString())
100     brokers.values.foreach { b =>
101       if (!consumersByHost.contains(b.host)) {
102         consumersByHost.put(b.host, new SimpleConsumer(b.host, b.port, 15000, bufferSize, clientId))
103       }
104     }
105     context.parent ! HealthComponent(actorName, ComponentState.NORMAL, s"Proxy has been configured.")
106   }
107 
108   def getPartitionLeaders: Set[PartitionAssignment] = {
109     val partitionsByTopic = new mutable.HashSet[PartitionAssignment]()
110     val topicMetaRequest = new TopicMetadataRequest(versionId = 1, correlationId = 0, clientId = clientId, topics = Seq())
111 
112     // Get our partition meta data for the configured topics
113     val processedClusters = new mutable.HashSet[String]()
114     val unprocessed = new mutable.HashSet[String]
115     for (bro <- brokers.values
116          if !processedClusters.contains(bro.cluster)) {
117       try {
118         val consumer = consumersByHost(bro.host)
119         val topicsMetaResp = consumer.send(topicMetaRequest)
120         for ( topicMeta <- topicsMetaResp.topicsMetadata.filter { meta => configuredTopics.contains(meta.topic) };
121               partMeta <- topicMeta.partitionsMetadata )
122           yield {
123             partMeta.leader match {
124               case Some(broker) =>
125                 log.debug(s"Leader found for topic [${topicMeta.topic}:${partMeta.partitionId}]: ${broker.host}")
126                 partitionsByTopic.add(PartitionAssignment(topicMeta.topic, partMeta.partitionId, brokers(broker.host).cluster, broker.host))
127               case None =>
128                 log.error(s"No leader found for topic [${topicMeta.topic}:${partMeta.partitionId}]")
129             }
130             processedClusters.add(bro.cluster)
131           }
132       } catch {
133         case e: Throwable =>
134           log.error(s"Unable to get topic meta data from ${bro.host}, will retry soon", e)
135           unprocessed.add(bro.host)
136       }
137     }
138 
139     if (unprocessed.nonEmpty) {
140       log.warn(s"Some brokers despondent: ${unprocessed.mkString(",")}. Remaining brokers will start their workers.")
141       context.parent ! HealthComponent(actorName, ComponentState.DEGRADED, s"Brokers despondent: ${unprocessed.mkString(",")}.")
142     } else {
143       log.debug("Successfully processed brokers {}", brokers.toString())
144       context.parent ! HealthComponent(actorName, ComponentState.NORMAL, "Successfully fetched broker data")
145     }
146     partitionsByTopic.toSet[PartitionAssignment]
147   }
