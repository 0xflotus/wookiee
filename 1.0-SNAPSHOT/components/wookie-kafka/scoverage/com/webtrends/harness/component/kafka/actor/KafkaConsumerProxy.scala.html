67   val bufferSize = 1024*1024
68   val refreshLock = new ReentrantLock()
69   var brokers = Seq[BrokerSpec]()
70   val consumersByHost = new mutable.HashMap[String,KafkaConsumer]()
71   val hostsByTopicParts = new mutable.HashMap[(String, Int), Option[List[String]]]()

76   override def preStart() = {

78   }

80   def receive:Receive = configReceive orElse {
81     // We will first pull the broker data from ZK and use that
82     // To make topic meta data requests to get the partition information
83     case KafkaRefreshReq(light) =>
84       if (!refreshLock.isLocked) {
85         Future {
86           refreshLock.lock()
87           try {
88             log.debug("Refreshing kafka broker/partition data")
89             if (!light) clearKafkaMappings()
90             requestBrokerInfo()
91           } catch {
92             case e: Exception =>
93               log.error("Failed to fetch broker info from ZK. Retrying in 10 seconds", e)
94               health = HealthComponent("kafka-proxy", ComponentState.DEGRADED, "Failed to fetch broker info from ZK. Retrying in 10 seconds")
95               context.system.scheduler.scheduleOnce(10 seconds) {
96                 self ! KafkaRefreshReq(light)
97               }
98           } finally {
99             refreshLock.unlock()
100           }
101         }
102       }
103 
104     case TopicPartitionReq =>
105       sender ! TopicPartitionResp(partitionsByTopic.toSet[PartitionAssignment])
106 
107     case FetchConsumer(host) =>
108       sender ! consumersByHost.get(host)
109 
110     case CheckHealth => sender ! health
111   }
112 
113   override def renewConfiguration() = {
114     super.renewConfiguration()
115     self ! KafkaRefreshReq(light = false)
116   }
117 
118   def requestBrokerInfo() = {
119     log.debug("Using kafka-hosts to set consumed hosts, {}", kafkaSources)
120     var brokerData = Seq[BrokerSpec]()
121     clusters.clear()
122     kafkaSources foreach { s =>
123       s._2 foreach { broker =>
124         val hostPort: Array[String] = broker.split(":")
125         val host: String = hostPort(0)
126         val port: Int = if (hostPort.length == 2) hostPort(1).toInt else 9092
127         brokerData = brokerData :+ BrokerSpec(host, port, s._1)
128         clusters.put(host, s._1)
129       }
130     }
131 
132     processBrokerData(brokerData)
133   }
134 
135   def processBrokerData(newBrokers: Seq[BrokerSpec]) = {
136     log.debug("Processing brokers {}", newBrokers.toString())
137     brokers = newBrokers
138 
139     brokers.filter { it => partitionsByTopic.forall { part => part.host != it.host } }.foreach { b =>
140       try {
141         val oldConsumer = consumersByHost.get(b.host)
142         consumersByHost.put(b.host, new KafkaConsumer(b.host, b.port, 15000, bufferSize, clientId))
143         oldConsumer.foreach { it =>
144           it.closed = true
145           it.close()
146         }
147       } catch {
148         case e: Throwable =>
149           log.error(s"Exception creating simple consumer for ${b.host}: ${e.getMessage}")
150       }
151     }
152 
153     refreshPartitionLeaders()
154   }
155 
156   def refreshPartitionLeaders() = {
157     val topicMetaRequest = new TopicMetadataRequest(
158       versionId = 1,
159       correlationId = 0,
160       clientId = clientId,
161       topics = Seq())
162 
163     // Get our partition meta data for the configured topics and map the topic/partition tuples to consumers
164     var failed = false
165     val processedClusters = new mutable.HashSet[String]()
166     val clusterSet = clusters.values.toSet
167     for (bro <- brokers
168          if !clusterSet.forall(x => processedClusters.contains(x))
169          if !processedClusters.contains(clusters(bro.host))) {
170       try {
171         val consumer = consumersByHost(bro.host)
172         val topicsMetaResp = consumer.send(topicMetaRequest)
173 
174         for ( topicMeta <- topicsMetaResp.topicsMetadata.filter { meta => configuredTopics.contains(meta.topic) };
175               partMeta <- topicMeta.partitionsMetadata )
176           yield {
177             val key = (topicMeta.topic, partMeta.partitionId)
178             partMeta.leader match {
179               case Some(broker) =>
180                 log.debug(s"Leader found for topic [${topicMeta.topic}:${partMeta.partitionId}]: ${broker.host}")
181                 partitionsByTopic.add(PartitionAssignment(topicMeta.topic, partMeta.partitionId, clusters(broker.host), broker.host))
182                 if (hostsByTopicParts.contains(key) && hostsByTopicParts.get(key).isDefined) {
183                   hostsByTopicParts.put(key, Some(hostsByTopicParts.get(key).get.get :+ broker.host))
184                 } else {
185                   hostsByTopicParts.put(key, Some(List(broker.host)))
186                 }
187               case None =>
188                 log.error(s"No leader found for topic [${topicMeta.topic}:${partMeta.partitionId}]")
189                 hostsByTopicParts.put(key, None)
190             }
191             processedClusters.add(clusters(bro.host))
192           }
193       } catch {
194         case e: Throwable =>
195           log.error(s"Unable to get topic meta data from ${bro.host}, retrying in 10 seconds", e)
196           failed = true
197       }
198     }
199 
200     val unprocessed = clusterSet.filter(x => !processedClusters.contains(x))
201     if (hostsByTopicParts.isEmpty || failed || unprocessed.nonEmpty) {
202       log.warn(s"Some brokers despondent: ${unprocessed.mkString(",")}. Retrying in 10 seconds. Remaining brokers will start their workers")
203       health = HealthComponent("kafka-proxy", ComponentState.DEGRADED, s"Brokers despondent: ${unprocessed.mkString(",")}. Retrying in 10 seconds")
204       context.system.scheduler.scheduleOnce(10 seconds) { self ! KafkaRefreshReq(hostsByTopicParts.nonEmpty) }
205     } else {
206       log.debug("Successfully processed brokers {}", brokers.toString())
207       health = HealthComponent("kafka-proxy", ComponentState.NORMAL, "Successfully fetched broker data")
208     }
209   }
210 
211   // We only want to initiate a broker refresh if one has not already been initiated. When one is needed,
212   // clear out kafka mappings and kick it off. Clearing out the mappings means that any FetchReq messages received
213   // before the refresh has finished will fail. Workers are responsible for handling the failure and retrying
214   def refreshBrokerInfoIfNeeded() = {
215     partitionsByTopic.synchronized {
216       partitionsByTopic.isEmpty match {
217         case false =>
218           clearKafkaMappings()
219           self ! KafkaRefreshReq(light = true)
220         case true =>
221         // Do nothing. A refresh is already in flight.
222       }
223     }
224   }
225 
226   def clearKafkaMappings() = {
227     hostsByTopicParts.clear()
228     partitionsByTopic.clear()
229   }
