57 class KafkaConsumerProxy extends Actor with KafkaSettings
58   with ActorLoggingAdapter with ZookeeperAdapter with ZookeeperEventAdapter {
59 
60   def configuredTopics = topicMap.keys.toList
61   val formats = Serialization.formats(NoTypeHints)
62   import KafkaConsumerProxy._
63   import context.dispatcher
64 
65   val bufferSize = 1024*1024
66 
67   var brokers = Seq[BrokerSpec]()
68   val consumersByHost = new mutable.HashMap[String,KafkaConsumer]()
69   val hostsByTopicParts = new mutable.HashMap[(String, Int), Option[List[String]]]()
70   val partitionsByTopic = new mutable.HashSet[PartitionAssignment]()
71   val clusters = new mutable.HashMap[String, String]()
72   var health = HealthComponent("kafka-proxy", ComponentState.NORMAL, "Proxy has not acquired brokers yet")
73 
74   override def preStart() = {
75     self ! KafkaRefreshReq(light = false)
76   }
77 
78   def receive:Receive = configReceive orElse {
79     // We will first pull the broker data from ZK and use that
80     // To make topic meta data requests to get the partition information
81     case KafkaRefreshReq(light) =>
82       try {
83         log.debug("Refreshing kafka broker/partition data")
84         if (!light) clearKafkaMappings()
85         requestBrokerInfo()
86       } catch {
87         case e: Exception =>
88           log.error("Failed to fetch broker info from ZK. Retrying in 10 seconds", e)
89           health = HealthComponent("kafka-proxy", ComponentState.DEGRADED, "Failed to fetch broker info from ZK. Retrying in 10 seconds")
90           context.system.scheduler.scheduleOnce(10 seconds) { self ! KafkaRefreshReq(light) }
91       }
92 
93     case TopicPartitionReq =>
94       sender ! TopicPartitionResp(partitionsByTopic.toSet[PartitionAssignment])
95 
96     case FetchConsumer(host) =>
97       sender ! consumersByHost.get(host)
98 
99     case CheckHealth => sender ! health
100   }
101 
102   override def renewConfiguration() = {
103     super.renewConfiguration()
104     self ! KafkaRefreshReq(light = false)
105   }
106 
107   def requestBrokerInfo() = {
108     log.info("Using kafka-hosts to set consumed hosts, {}", kafkaSources)
109     var brokerData = Seq[BrokerSpec]()
110     clusters.clear()
111     kafkaSources foreach { s =>
112       s._2 foreach { broker =>
113         val hostPort: Array[String] = broker.split(":")
114         val host: String = hostPort(0)
115         val port: Int = if (hostPort.length == 2) hostPort(1).toInt else 9092
116         brokerData = brokerData :+ BrokerSpec(host, port, s._1)
117         clusters.put(host, s._1)
118       }
119     }
120 
121     processBrokerData(brokerData)
122   }
123 
124   def processBrokerData(newBrokers: Seq[BrokerSpec]) = {
125     log.debug("Processing brokers {}", newBrokers.toString())
126     brokers = newBrokers
127 
128     brokers.filter { it => partitionsByTopic.forall { part => part.host != it.host } }.foreach { b =>
129       try {
130         val oldConsumer = consumersByHost.get(b.host)
131         consumersByHost.put(b.host, new KafkaConsumer(b.host, b.port, 15000, bufferSize, clientId))
132         oldConsumer.foreach { it =>
133           it.closed = true
134           it.close()
135         }
136       } catch {
137         case e: Throwable =>
138           log.error(s"Exception creating simple consumer for ${b.host}: ${e.getMessage}")
139       }
140     }
141 
142     refreshPartitionLeaders()
143   }
144 
145   def refreshPartitionLeaders() = {
146     val topicMetaRequest = new TopicMetadataRequest(
147       versionId = 1,
148       correlationId = 0,
149       clientId = clientId,
150       topics = Seq())
151 
152     // Get our partition meta data for the configured topics and map the topic/partition tuples to consumers
153     var failed = false
154     val processedClusters = new mutable.HashSet[String]()
155     val clusterSet = clusters.values.toSet
156     for (bro <- brokers
157         if !clusterSet.forall(x => processedClusters.contains(x))
158         if !processedClusters.contains(clusters(bro.host))) {
159       try {
160         val consumer = consumersByHost(bro.host)
161         val topicsMetaResp = consumer.send(topicMetaRequest)
162 
163         for ( topicMeta <- topicsMetaResp.topicsMetadata.filter { meta => configuredTopics.contains(meta.topic) };
164               partMeta <- topicMeta.partitionsMetadata )
165         yield {
166           val key = (topicMeta.topic, partMeta.partitionId)
167           partMeta.leader match {
168             case Some(broker) =>
169               log.debug(s"Leader found for topic [${topicMeta.topic}:${partMeta.partitionId}]: ${broker.host}")
170               partitionsByTopic.add(PartitionAssignment(topicMeta.topic, partMeta.partitionId, clusters(broker.host), broker.host))
171               if (hostsByTopicParts.contains(key) && hostsByTopicParts.get(key).isDefined) {
172                 hostsByTopicParts.put(key, Some(hostsByTopicParts.get(key).get.get :+ broker.host))
173               } else {
174                 hostsByTopicParts.put(key, Some(List(broker.host)))
175               }
176             case None =>
177               log.error(s"No leader found for topic [${topicMeta.topic}:${partMeta.partitionId}]")
178               hostsByTopicParts.put(key, None)
179           }
180           processedClusters.add(clusters(bro.host))
181         }
182       } catch {
183         case e: Throwable =>
184           log.error(s"Unable to get topic meta data from ${bro.host}, retrying in 10 seconds", e)
185           failed = true
186       }
187     }
188 
189     val unprocessed = clusterSet.filter(x => !processedClusters.contains(x))
190     if (hostsByTopicParts.isEmpty || failed || unprocessed.nonEmpty) {
191       health = HealthComponent("kafka-proxy", ComponentState.DEGRADED, s"Brokers despondent: ${unprocessed.mkString(",")}. Retrying in 10 seconds")
192       context.system.scheduler.scheduleOnce(10 seconds) { self ! KafkaRefreshReq(hostsByTopicParts.nonEmpty) }
193     } else {
194       health = HealthComponent("kafka-proxy", ComponentState.NORMAL, "Successfully fetched broker data")
195     }
196   }
197 
198   // We only want to initiate a broker refresh if one has not already been initiated. When one is needed,
199   // clear out kafka mappings and kick it off. Clearing out the mappings means that any FetchReq messages received
200   // before the refresh has finished will fail. Workers are responsible for handling the failure and retrying
201   def refreshBrokerInfoIfNeeded() = {
202     partitionsByTopic.synchronized {
203       partitionsByTopic.isEmpty match {
204         case false =>
205           clearKafkaMappings()
206           self ! KafkaRefreshReq(light = true)
207         case true =>
208         // Do nothing. A refresh is already in flight.
209       }
210     }
211   }
212 
213   def clearKafkaMappings() = {
214     hostsByTopicParts.clear()
215     partitionsByTopic.clear()
216   }
