82   def receive = configReceive orElse {
83     // We will first pull the broker data from ZK and use that
84     // To make topic meta data requests to get the partition information
85     case KafkaRefreshReq(light) =>
86       try {
87         log.info("Refreshing kafka broker/partition data")
88         if (!light) clearKafkaMappings()
89         requestBrokerInfo()
90         health = HealthComponent("kafka-proxy", ComponentState.NORMAL, "Successfully fetched broker data")
91       } catch {
92         case e: Exception =>
93           log.error("Failed to fetch broker info from ZK. Retrying in 10 seconds", e)
94           health = HealthComponent("kafka-proxy", ComponentState.DEGRADED, "Failed to fetch broker info from ZK. Retrying in 10 seconds")
95           context.system.scheduler.scheduleOnce(10 seconds) { self ! KafkaRefreshReq(light) }
96       }
97 
98     case TopicPartitionReq =>
99       sender ! TopicPartitionResp(partitionsByTopic.toSet[PartitionAssignment])
100 
101     case FetchReq(topic,part,offset,host) =>
102       try {
103         if (consumersByHost(host).isDefined) {
104           sender ! fetchTopicPart(consumersByHost(host).get, topic, part, offset, clientId)
105         } else {
106           throw new LeaderNotAvailableException()
107         }
108       } catch {
109         case ex: OffsetOutOfRangeException =>
110           sender ! FetchErrorBadOffset(KafkaUtil.getDesiredAvailableOffset(consumersByHost(host).get, topic, part, offset, clientId))
111 
112         case ex: InvalidMessageSizeException =>
113           log.error(s"Failed to fetch message for $topic:$part from $host because message was too big for buffer, you need to increase the buffer size!")
114           sender ! FetchErrorPartitionNotAvailable(offset)
115 
116         // Catch all. Respond with a generic error message that will result in the workers waiting before retrying
117         // and kick off a refresh of our kafka info
118         case ex: Exception =>
119           log.error(s"Failed to fetch from [$host] [$topic] [$part] [$offset] [${ex.getClass.toString}] [${ex.getMessage}]")
120           sender ! FetchErrorPartitionNotAvailable(offset)
121           refreshBrokerInfoIfNeeded()
122       }
123 
124     case CheckHealth => sender() ! health
125   }

132   def fetchTopicPart(consumer: SimpleConsumer, topic: String, part: Int, offset: Long, clientId: String): FetchRes = {
133     val fetch = new FetchRequestBuilder()
134       .clientId(clientId)
135       .addFetch(topic, part, offset, 150000)
136       .build()
137 
138     val fetchResponse = consumer.fetch(fetch)
139     if (fetchResponse.hasError) {
140       ErrorMapping.maybeThrowException(fetchResponse.errorCode(topic, part))
141     }
142 
143     val msgSet = fetchResponse.messageSet(topic, part)
144 
145     val msgSeq = KafkaMessageSet(msgSet, offset) // This is needed since if Kafka is compressing the messages, the fetch request will return an entire compressed block even if the requested offset isn't the beginning of the compressed block. Thus a message we saw previously may be returned again.
146 
147     FetchRes(msgSeq, if (msgSet.size > 0) msgSet.last.nextOffset else offset)
148   }

150   def requestBrokerInfo() = {
151     log.info("Using kafka-hosts to set consumed hosts, {}", kafkaSources)
152     var brokerData = Seq[BrokerSpec]()
153     kafkaSources foreach { s =>
154       s._2 foreach { broker =>
155         val hostPort: Array[String] = broker.split(":")
156         val host: String = hostPort(0)
157         val port: Int = if (hostPort.length == 2) hostPort(1).toInt else 9092
158         brokerData = brokerData :+ BrokerSpec(host, port, s._1)
159         clusters.put(host, s._1)
160       }
161     }
162 
163     processBrokerData(brokerData)
164   }
165 
166   def processBrokerData(newBrokers: Seq[BrokerSpec]) = {
167     log.debug("Processing brokers {}", newBrokers.toString())
168     brokers = Some(newBrokers)
169 
170     brokers.get.filter { it => partitionsByTopic.forall { part => part.host != it.host } }.map { b =>
171       try {
172         consumersByHost.put(b.host, Some(new SimpleConsumer(b.host, b.port, 15000, bufferSize, clientId)))
173       } catch {
174         case e: Throwable =>
175           log.error(s"Exception creating simple consumer for ${b.host}: ${e.getMessage}")
176           consumersByHost.put(b.host, None)
177       }
178     }
179 
180     refreshPartitionLeaders()
181   }
182 
183   def refreshPartitionLeaders() = {
184     val topicMetaRequest = new TopicMetadataRequest(
185       versionId = 1,
186       correlationId = 0,
187       clientId = clientId,
188       topics = Seq())
189 
190     // Get our partition meta data for the configured topics and map the topic/partition tuples to consumers
191     var failed = false
192     val processedClusters = new mutable.HashSet[String]()
193     val clusterSet = clusters.values.toSet
194     for (consumer <- consumersByHost.filter(_._2.isDefined)
195         if !clusterSet.forall(x => processedClusters.contains(x))
196         if !processedClusters.contains(clusters(consumer._1))) {
197       try {
198         val topicsMetaResp = consumer._2.get.send(topicMetaRequest)
199 
200         for ( topicMeta <- topicsMetaResp.topicsMetadata.filter { meta => configuredTopics.contains(meta.topic) };
201               partMeta <- topicMeta.partitionsMetadata )
202         yield {
203           val key = (topicMeta.topic, partMeta.partitionId)
204           partMeta.leader match {
205             case Some(broker) =>
206               log.info(s"Leader found for topic [${topicMeta.topic}:${partMeta.partitionId}]: ${broker.host}")
207               partitionsByTopic.add(PartitionAssignment(topicMeta.topic, partMeta.partitionId, clusters(broker.host), broker.host))
208               if (hostsByTopicParts.contains(key) && hostsByTopicParts.get(key).isDefined) {
209                 hostsByTopicParts.put(key, Some(hostsByTopicParts.get(key).get.get :+ broker.host))
210               } else {
211                 hostsByTopicParts.put(key, Some(List(broker.host)))
212               }
213             case None =>
214               log.error(s"No leader found for topic [${topicMeta.topic}:${partMeta.partitionId}]")
215               hostsByTopicParts.put(key, None)
216           }
217           processedClusters.add(clusters(consumer._1))
218         }
219       } catch {
220         case e: Throwable =>
221           log.error(s"Unable to get topic meta data from ${consumer._2.get.host}, retrying in 10 seconds", e)
222           failed = true
223       }
224     }
225 
226     if (hostsByTopicParts.isEmpty || failed || !clusterSet.forall(x => processedClusters.contains(x))) {
227       health = HealthComponent("kafka-proxy", ComponentState.DEGRADED, "Failed to fetch broker info from ZK. Retrying in 10 seconds")
228       context.system.scheduler.scheduleOnce(10 seconds) { self ! KafkaRefreshReq(hostsByTopicParts.nonEmpty) }
229     }
230   }
231 
232   // We only want to initiate a broker refresh if one has not already been initiated. When one is needed,
233   // clear out kafka mappings and kick it off. Clearing out the mappings means that any FetchReq messages received
234   // before the refresh has finished will fail. Workers are responsible for handling the failure and retrying
235   def refreshBrokerInfoIfNeeded() = {
236     brokers match {
237       case Some(brokers) =>
238         clearKafkaMappings()
239         self ! KafkaRefreshReq(false)
240       case None =>
241         // Do nothing. A refresh is already in flight.
242     }
243   }
244 
245   def clearKafkaMappings() = {
246     consumersByHost.foreach { case (host, consumer) =>
247       if (consumer.isDefined)
248         consumer.get.close()
249     }
250     consumersByHost.clear()
251     hostsByTopicParts.clear()
252     partitionsByTopic.clear()
253     clusters.clear()
254     brokers = None
255   }
