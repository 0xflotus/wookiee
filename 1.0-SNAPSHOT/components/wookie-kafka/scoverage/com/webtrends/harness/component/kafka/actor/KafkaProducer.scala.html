79   def receive = healthReceive orElse {
80 
81     case EventToWrite(topic, event) => sendData(topic, Seq(event))
82 
83     case EventsToWrite(topic, events) => sendData(topic, events)
84 
85     case msg: EventsToWriteWithAck =>

91    * 
92    * @param topic
93    * @param ackId
94    * @param eventMessages
95    * @return
96    */
97   def sendData(topic: String, eventMessages: Seq[Array[Byte]], ackId: String = ""): Either[String, Throwable] ={
98     Try {
99       ensureProducer()
100       
101       val keyedMessages = eventMessages.withFilter(_.length > 0)
102                                        .map(new KeyedMessage[String, Array[Byte]](topic, _))
103 
104       //Only Send if we have 1 valid messages
105       if(keyedMessages.nonEmpty) {
106         dataProducer.foreach(_.send(keyedMessages: _*))
107       }
108 
109       self ! SetHealth(ComponentState.NORMAL, "Messages sending successfully")
110       Left(ackId)
111     } recover {
112       case ex:Exception =>
113         log.error("Unable To write Event", ex)
114         self ! SetHealth(ComponentState.CRITICAL, "Last message failed to send")
115         resetProducer()
116         Right(ex)
117     } get
118   }

131   def resetProducer() = {
132     if (dataProducer.isDefined) {
133       dataProducer.get.close()
134       dataProducer = None
135     }
136     dataProducer = Some(new Producer(new ProducerConfig(configToProps(context.system.settings.config.getConfig("wookie-kafka.producer")))))
137     log.info("Kafka Producer Started")
138   }

140   def getKafkaNodes(config: Config): String = {
141     if (config.hasPath("metadata.broker.list")) {
142       return config.getString("metadata.broker.list")
143     }
144     if (!config.hasPath("broker.root")) {
145       throw new ExceptionInInitializerError("Must either set metadata.broker.list or broker.root in producer config")
146     }
147 
148     val brokerRoot = config.getString("broker.root")
149     val brokerPath = brokerRoot + "/brokers/ids"
150     val rawBrokerFuture = getChildren(brokerPath)
151     var brokers = ""
152     val rawBrokerData = Await.result(rawBrokerFuture, 10 seconds)
153     rawBrokerData foreach { data =>
154       val dataPath = brokerRoot + "/brokers/ids/" + data._1
155       val dataFut = Await.result(getData(dataPath), 10 seconds)
156 
157       val brokerData = Serialization.read(new String(dataFut, utf8))(formats, manifest[BrokerSpec])
158       if (!brokers.isEmpty) {
159         brokers = brokers + ","
160       }
161       brokers = brokers + brokerData.host + ":" + brokerData.port
162     }
163 
164     if (brokers.isEmpty) {
165       throw new ExceptionInInitializerError("No brokers found in ZK! Can't write to anywhere. Broker root: " + brokerRoot)
166     }
167     brokers
168   }

170   def configToProps(config: Config): Properties = {
171     val props: Properties = new Properties
172     import scala.collection.JavaConversions._
173     config.entrySet.foreach { entry =>
174       entry.getValue.valueType match {
175         case ConfigValueType.STRING =>
176           props.put(entry.getKey, config.getString(entry.getKey))
177         case ConfigValueType.NUMBER =>
178           props.put(entry.getKey, config.getNumber(entry.getKey).toString)
179         case _ =>
180         // Ignore other types
181       }
182     }
183     val brokerList = getKafkaNodes(context.system.settings.config.getConfig("wookie-kafka.producer"))
184     props.put("metadata.broker.list", brokerList)
185     props
186   }
