82   def receive:Receive = healthReceive orElse {
83     case EventToWrite(topic, event) => sendData(topic, Seq(event))
84 
85     case EventsToWrite(topic, events) => sendData(topic, events)
86 
87     case msg: EventsToWriteWithAck =>
88       sender() ! EventWriteAck(sendData(msg.topic, msg.events, msg.ackId))

92   def sendData(topic: String, eventMessages: Seq[Array[Byte]], ackId: String = ""): Either[String, Throwable] ={
93     Try {
94       ensureProducer()
95 
96       val keyedMessages = eventMessages.withFilter(_.length > 0)
97         .map(keyedMessage(topic, _))
98 
99       //Only Send if we have 1 valid messages
100       if(keyedMessages.nonEmpty) {
101         dataProducer.foreach(_.send(keyedMessages: _*))
102       }
103 
104       setHealth(ComponentState.NORMAL, "Messages sending successfully")
105       Left(ackId)
106     } recover {
107       case ex:Exception =>
108         log.error("Unable To write Event", ex)
109         setHealth(ComponentState.CRITICAL, "Last message failed to send")
110         resetProducer()
111         Right(ex)
112     } get
113   }

115   private def keyedMessage(topic: String, value: Array[Byte]): KeyedMessage[String, Array[Byte]] = {
116     val keyedMessage =
117       if(evenPartitionDistribution)
118         new KeyedMessage[String, Array[Byte]](topic, partitionKey.toString, value)
119       else
120         new KeyedMessage[String, Array[Byte]](topic, value)
121 
122     partitionKey += 1
123 
124     keyedMessage
125   }

144   def getKafkaNodes(config: Config): String = {
145     if (config.hasPath("metadata.broker.list")) {
146       return config.getString("metadata.broker.list")
147     }
148     if (!config.hasPath("broker.root")) {
149       throw new ExceptionInInitializerError("Must either set metadata.broker.list or broker.root in producer config")
150     }
151 
152     val brokerRoot = config.getString("broker.root")
153     val brokerPath = brokerRoot + "/brokers/ids"
154     val rawBrokerFuture = getChildren(brokerPath)
155     var brokers = ""
156     val rawBrokerData = Await.result(rawBrokerFuture, 10 seconds)
157     rawBrokerData foreach { data =>
158       val dataPath = brokerRoot + "/brokers/ids/" + data._1
159       val dataFut = Await.result(getData(dataPath), 10 seconds)
160 
161       val brokerData = Serialization.read(new String(dataFut, utf8))(formats, manifest[BrokerSpec])
162       if (!brokers.isEmpty) {
163         brokers = brokers + ","
164       }
165       brokers = brokers + brokerData.host + ":" + brokerData.port
166     }
167 
168     if (brokers.isEmpty) {
169       throw new ExceptionInInitializerError("No brokers found in ZK! Can't write to anywhere. Broker root: " + brokerRoot)
170     }
171     brokers
172   }

174   def configToProps(config: Config): Properties = {
175     val props: Properties = new Properties
176     import scala.collection.JavaConversions._
177     config.entrySet.foreach { entry =>
178       entry.getValue.valueType match {
179         case ConfigValueType.STRING =>
180           props.put(entry.getKey, config.getString(entry.getKey))
181         case ConfigValueType.NUMBER =>
182           props.put(entry.getKey, config.getNumber(entry.getKey).toString)
183         case _ =>
184         // Ignore other types
185       }
186     }
187     val brokerList = getKafkaNodes(context.system.settings.config.getConfig("harness-kafka.producer"))
188     props.put("key.serializer.class","kafka.serializer.StringEncoder")
189     props.put("metadata.broker.list", brokerList)
190     props
191   }
