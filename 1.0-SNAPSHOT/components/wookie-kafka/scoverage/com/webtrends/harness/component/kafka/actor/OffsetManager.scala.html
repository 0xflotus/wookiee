29 class OffsetManager(appRoot: String, timeout:Timeout) extends Actor

36   def receive: Receive = {

42   /**
43    * Get the state from zk and send back offset data response
44    * @param req
45    */
46   def getOffsetState(req: GetOffsetData) = {
47     val originalSender = sender()
48     val parent = context.parent
49     //Path to write to zookeeper
50     val path = s"$offsetPath/${req.path}"
51     getData(path).onComplete {
52       case Success(data) =>
53          Try {
54            originalSender ! OffsetDataResponse(Left(OffsetData(data)))
55            parent ! healthy(path)
56          } recover {
57            case ex =>
58              log.error("Error retrieving state from zk", ex)
59              originalSender ! OffsetDataResponse(Right(ex))
60              parent ! unhealthy(path)
61          }
62       case Failure(err) =>
63         err match {
64           case ex: NoNodeException =>
65             originalSender ! OffsetDataResponse(Left(OffsetData(Array.empty[Byte])))
66             parent ! healthy(path)
67           case ex =>
68             log.error("Unable to get state from Zk", ex)
69             originalSender ! OffsetDataResponse(Right(ex))
70             parent ! unhealthy(path)
71         }
72     }
73   }
74 
75   def storeOffsetState(req: StoreOffsetData) = {
76     val path = s"$offsetPath/${req.path}"
77     val originalSender = sender()
78     val parent = context.parent
79     //ZNode should be create and not be ephemeral
80     setData(path, req.offsetData.data, true, false).onComplete {
81       case Success(bytesWritten) =>
82         originalSender ! OffsetDataResponse(Left(OffsetData(req.offsetData.data)))
83         parent ! healthy(path)
84       case Failure(ex) =>
85         log.error("Uanble to write zk state", ex)
86         originalSender ! OffsetDataResponse(Right(ex))
87         parent ! unhealthy(path)
88     }
89   }
90 
91   private def unhealthy(path: String) =
92     ZKHealthState(path, healthy = false, s"Failed to fetch offset state $path from ZK")
93 
94   private def healthy(path: String) =
95     ZKHealthState(path, healthy = true, s"Successfully fetched state $path from ZK")
