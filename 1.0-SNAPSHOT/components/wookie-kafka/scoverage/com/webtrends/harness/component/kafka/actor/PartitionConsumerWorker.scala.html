71 class PartitionConsumerWorker(kafkaProxy: ActorRef, assign: PartitionAssignment, offsetManager: ActorRef)
72   extends LoggingFSM[State, Lock] with Actor with ZookeeperAdapter with KafkaSettings {
73 
74   import context._
75   import ConsumptionFetchMode._
76 
77   val decoder = utf8.newDecoder
78 
79   val topic = assign.topic
80   val partition = assign.partition
81   val host = assign.host
82   val cluster = assign.cluster
83   val name = assign.assignmentName
84 
85   val lockPath = s"$appRootPath/locks/${cluster}_${topic}_$partition"
86 
87   // Harness 3.0 does not provide distributed lock support. Need to pull the curator instance and create our own
88   val curator = Curator(zkConf.get).client
89 
90   val zkCommitRate = kafkaConfig.getLong("zk-offset-commit-rate-millis")
91 
92   // Add scheduled jobs to this list, they will be cancelled on Stop
93   var scheduler: List[Cancellable] = List.empty
94 
95   // The acknowledged offset
96   protected var ackedOffset = 0L
97 
98   // OVERRIDE to set custom offset writing
99   def formatAckedOffset() = ackedOffset.toString
100 
101   // Last offset that was sent over to kafka proxy
102   protected var lastSentToProxyOffset = 0L
103 
104   // Last offset stored
105   protected var lastSentToStorage = 0L
106 
107   // Each partition worker needs to lock the partition it is working on to prevent data duplication.
108   // We need to ensure that the initial node completes before the new node reads the offsets out of ZK
109   protected val lock = new InterProcessSemaphoreV2(curator, lockPath, 1)
110 
111   /**
112    * Fetch Mode - Defaults to automatic
113    * @return
114    **/
115   def fetchMode():ConsumptionFetchMode = AUTOMATIC
116 
117   /**
118    * Override method to set functionality when consuming, to manage offsets be sure to
119    * set ackedOffset when account has executed its desired data processing successfully.
120    * @param messageResponse message received by worker on its assigned partition
121    */
122   def onReceive(messageResponse: MessageResponse) = {
123     ackedOffset = messageResponse.nextOffsetOfSet
124   }
125 
126 
127 
128   // A worker can be in one of 4 standard states:
129   // 1. Stopped  -  The worker has been created and told to process data, but it is not yet ready to read from kafka.
130   //                If told to start, the worker will obtain the distributed lock for this partition,
131   //                and read its initial state out of ZK
132   // 2. Starting -  The worker has been told to start and has acquired a lock, it will not try to read its initial
133   //                state from ZK and start consuming if successful
134   // 3. Consuming - The worker has fully started up and is reading data from kafka
135   startWith(Stopped, Unlocked)
136 
137   // These messages should always be responded to in all states
138   whenUnhandled {
139     case Event(Stop, aq: Acquired) =>
140       goto(Stopped) using Unlocked
141 
142     case Event(Stop, Unlocked) =>
143       stay()
144 
145     case Event(Start, aq: Acquired) =>
146       stay()
147   }
148 
149   onTransition {
150     case (Consuming | Starting) -> Stopped =>
151       log.info(s"$name: Transitioning state to Stopped")
152       stopWorker()
153       context.parent ! WorkerStopped(name)
154 
155     case Stopped -> Starting =>
156       log.debug(s"$name: Transitioning state to Starting")
157       offsetManager ! GetOffsetData(partitionName(assign))
158 
159     case Starting -> Consuming =>
160       log.info(s"$name: Transitioning state to Consuming")
161       context.parent ! KafkaHealthState(name, healthy = true, s"Worker started, no data yet", topic)
162 
163       fetchRequest(ackedOffset)
164       scheduler = scheduler :+ context.system.scheduler.schedule(zkCommitRate milliseconds,
165         zkCommitRate milliseconds, self, CommitOffset)
166   }
167 
168   when(Stopped) {
169     case Event(Start, Unlocked) =>
170       acquireLock() match {
171         case Some(aq) => goto(Starting) using aq
172         case None => stay()
173       }
174   }
175 
176   when(Starting, stateTimeout = 5 seconds) {
177     case Event(msg: OffsetDataResponse, aq: Acquired) =>
178       msg.data match {
179         case Left(data) =>
180           processOffsets(data)
181         case Right(ex) =>
182           log.error("Unable to load offset state")
183           goto(Stopped) using Unlocked
184       }
185 
186     case Event(StateTimeout, aq: Acquired) =>
187       goto(Stopped) using Unlocked
188   }
189 
190   when(Consuming) {
191     case Event(msg: FetchRes, aq: Acquired) =>
192       if (msg.messages.size > 0) {
193         context.parent ! KafkaHealthState(name, healthy = true, s"Successfully fetched ${msg.messages.size} events from kafka", topic)
194       }
195 
196       consume(msg)
197 
198       stay()
199 
200     case Event(msg: FetchErrorBadOffset, aq: Acquired) =>
201       log.warning(s"$name: Failed to fetch events from kafka. " +
202                   s"Offset out of range. Using offset ${msg.nextAvailableOffset}")
203       fetchRequest(msg.nextAvailableOffset)
204       stay()
205 
206     case Event(msg: FetchErrorPartitionNotAvailable, aq: Acquired) =>
207       context.parent ! KafkaHealthState(name, healthy = false, s"Failed to fetch events from kafka. Partition not available", topic)
208       log.error(s"$name: Unable to fetch events from kafka. Partition no available")
209       goto(Stopped) using Unlocked
210 
211     case Event(CommitOffset, aq: Acquired) =>
212       storeOffset()
213       stay()
214 
215     case Event(msg: OffsetDataResponse, aq: Acquired) =>
216       msg.data match {
217         case Left(data) =>
218           lastSentToStorage = extractOffset(data)
219           stay()
220         case Right(ex) =>
221           log.error("Unable to save offset state")
222           goto(Stopped) using Unlocked
223       }
224   }
225 
226   initialize()
227 
228   // Will only store if the ackedOffset has changed or force is true
229   protected def storeOffset(force: Boolean = false): Unit = {
230     if (force || (ackedOffset > 0 && lastSentToStorage != ackedOffset)) {
231        offsetManager ! StoreOffsetData(partitionName(assign), OffsetData(formatAckedOffset().getBytes(utf8)))
232     }
233   }
234 
235   override def postStop(): Unit = {
236     stopWorker()
237   }
238 
239   // Override to process initial offsets differently
240   protected def processOffsets(offsetData: OffsetData): State = {
241     val offset = extractOffset(offsetData)
242 
243     log.debug(s"$name has initialized with offset $offset")
244     ackedOffset = offset
245     lastSentToProxyOffset = offset
246     lastSentToStorage = offset
247 
248     goto(Consuming)
249   }
250 
251   // Can be overridden to read custom offsets
252   protected def extractOffset(offsetData:OffsetData):Long = {
253     val data = offsetData.asString()
254     if(data.isEmpty) 0L else data.toLong
255   }
256 
257   def stopWorker() = {
258     if(ackedOffset > 0) {
259       log.debug(s"Shutdown %s, storage offset=%d, proxy offset=%d, acked offset=%d"
260         .format(name, lastSentToStorage, lastSentToProxyOffset, ackedOffset))
261 
262       offsetManager ! StoreOffsetData(partitionName(assign), OffsetData(formatAckedOffset().getBytes(utf8)))
263     }
264     scheduler.foreach(_.cancel())
265     scheduler = List()
266 
267     stateData match {
268       case aq: Acquired =>
269         lock.returnLease(aq.lease)
270         log.debug(s"$name: Released lock")
271       case _ => // No lock
272     }
273     context.parent ! KafkaHealthState(name, healthy = true, null, null)
274   }
275 
276   def acquireLock(): Option[Acquired] = {
277     try {
278       val acquiredLease = lock.acquire()
279       log.debug(s"$name: Lock acquired")
280       Some(Acquired(acquiredLease))
281     } catch {
282       case e: Throwable =>
283         log.error(s"$name: Unable to obtain lock: ${e.getMessage} ")
284         None
285     }
286   }
287 
288   /**
289    * Takes the list of messages receive in the last fetch and sends them to
290    * the main worker method account(...) which should be overridden with the actual
291    * work we want to do.
292    * @param fetchRes Contains our next offset and the list of messages we just got
293    * @return Returns a request for more messages
294    */
295   def consume(fetchRes: FetchRes) = {
296 
297     lastSentToProxyOffset = fetchRes.nextOffset
298 
299     onReceive(MessageResponse(fetchRes.messages, fetchRes.nextOffset))
300 
301     //After consumption of messages automatically fetch the next
302     if(fetchMode() == AUTOMATIC) {
303       fetchRequest (fetchRes.nextOffset)
304     }
305   }
306 
307 
308   /**
309    * Fetch data from kafka
310    *
311    */
312   protected def fetchRequest(nextOffset: Long) = {
313     //log.debug(s"$name fetching offset $nextOffset")
314     kafkaProxy ! FetchReq(topic, partition, nextOffset, host)
315   }
