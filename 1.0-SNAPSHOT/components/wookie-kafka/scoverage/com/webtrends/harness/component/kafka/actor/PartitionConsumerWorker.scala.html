92   // Harness 3.0 does not provide distributed lock support. Need to pull the curator instance and create our own

95   val zkCommitRate = kafkaConfig.getLong("zk-offset-commit-rate-millis")

97   // Add scheduled jobs to this list, they will be cancelled on Stop

100   // The acknowledged offset
101   protected var ackedOffset = 0L
102 
103   // OVERRIDE to set custom offset writing
104   def formatAckedOffset() = ackedOffset.toString
105 
106   // Last offset that was sent over to kafka proxy
107   protected var lastSentToProxyOffset = 0L
108 
109   // Last offset stored

112   // Each partition worker needs to lock the partition it is working on to prevent data duplication.
113   // We need to ensure that the initial node completes before the new node reads the offsets out of ZK
114   protected val lock = new InterProcessSemaphoreV2(curator, lockPath, 1)

116   protected var consumer: Option[KafkaConsumer] = None

118   /**

121    **/

124   /**
125    * Override method to set functionality when consuming, to manage offsets be sure to
126    * set ackedOffset when account has executed its desired data processing successfully.
127    * @param messageResponse message received by worker on its assigned partition
128    */
129   def onReceive(messageResponse: MessageResponse) = {
130     ackedOffset = messageResponse.nextOffsetOfSet

134   // A worker can be in one of 4 standard states:
135   // 1. Stopped  -  The worker has been created and told to process data, but it is not yet ready to read from kafka.
136   //                If told to start, the worker will obtain the distributed lock for this partition,
137   //                and read its initial state out of ZK
138   // 2. Starting -  The worker has been told to start and has acquired a lock, it will not try to read its initial
139   //                state from ZK and start consuming if successful
140   // 3. Consuming - The worker has fully started up and is reading data from kafka
141   startWith(Stopped, Unlocked)
142 
143   // These messages should always be responded to in all states
144   whenUnhandled {
145     case Event(Stop, aq: Acquired) =>
146       goto(Stopped) using Unlocked
147 
148     case Event(Stop, Unlocked) =>
149       stay()
150 
151     case Event(Start, aq: Acquired) =>
152       stay()
153   }
154 
155   onTransition {
156     case (Consuming | Starting) -> Stopped =>
157       log.info(s"$name: Transitioning state to Stopped")
158       stopWorker()
159 
160     case Stopped -> Starting =>
161       log.debug(s"$name: Transitioning state to Starting")
162       offsetManager ! GetOffsetData(partitionName(assign))
163 
164     case Starting -> Consuming =>
165       fetchConsumer()
166       if (consumer.isDefined) {
167         context.parent ! KafkaHealthState(name, healthy = true, s"Worker started", topic)
168         fetchRequest(ackedOffset)
169         scheduler = scheduler :+ context.system.scheduler.schedule(zkCommitRate milliseconds,
170           zkCommitRate milliseconds, self, CommitOffset)
171       }
172       log.info(s"$name: We are transitioned to Consuming")
173   }
174 
175   protected def fetchConsumer() {
176     log.debug(s"$name: Consumer closed, fetching new one")
177     Try { consumer = Await.result(kafkaProxy.ask(FetchConsumer(host))(Timeout(consumerWait, TimeUnit.MILLISECONDS)), consumerWait milliseconds) match {
178       case Some(con) => Some(con.asInstanceOf[KafkaConsumer])
179       case None =>
180         log.error(s"No consumer found for $host")
181         context.parent ! KafkaHealthState(name, healthy = false, s"Could not get consumer", topic)
182         self ! Stop
183         None
184     }} recover {
185       case ex: Exception =>
186         log.error(s"Consumer request for $host timed out, stopping and will retry")
187         consumer = None
188         self ! Stop
189     }
190   }
191 
192   when(Stopped) {
193     case Event(Start, Unlocked) =>
194       acquireLock() match {
195         case Some(aq) => goto(Starting) using aq
196         case None => stay()
197       }
198 
199     case Event(OffsetDataResponse | CommitOffset | FetchRes, Unlocked) =>
200       stay()
201   }
202 
203   when(Starting, stateTimeout = 5 seconds) {
204     case Event(msg: OffsetDataResponse, aq: Acquired) =>
205       msg.data match {
206         case Left(data) =>
207           processOffsets(data)
208         case Right(ex) =>
209           log.error("Unable to load offset state")
210           goto(Stopped) using Unlocked
211       }
212 
213     case Event(StateTimeout, aq: Acquired) =>
214       goto(Stopped) using Unlocked
215   }
216 
217   when(Consuming) {
218     case Event(msg: FetchRes, aq: Acquired) =>
219       consume(msg)
220       stay()
221 
222     case Event(msg: FetchErrorBadOffset, aq: Acquired) =>
223       log.warning(s"$name: Failed to fetch events from kafka. " +
224                   s"Offset out of range. Using offset ${msg.nextAvailableOffset}")
225       fetchRequest(msg.nextAvailableOffset)
226       stay()
227 
228     case Event(msg: FetchErrorPartitionNotAvailable, aq: Acquired) =>
229       context.parent ! KafkaHealthState(name, healthy = false, s"Failed to fetch events from kafka. Partition not available", topic)
230       log.error(s"$name: Unable to fetch events from kafka. Partition no available")
231       goto(Stopped) using Unlocked
232 
233     case Event(CommitOffset, aq: Acquired) =>
234       storeOffsetAndUpdateHealth()
235       stay()
236 
237     case Event(msg: OffsetDataResponse, aq: Acquired) =>
238       msg.data match {
239         case Left(data) =>
240           lastSentToStorage = extractOffset(data)
241           stay()
242         case Right(ex) =>
243           log.error("Unable to save offset state")
244           goto(Stopped) using Unlocked
245       }
246   }

250   // Will only store if the ackedOffset has changed or force is true
251   protected def storeOffsetAndUpdateHealth(force: Boolean = false): Unit = {
252     if (force || (ackedOffset > 0 && lastSentToStorage != ackedOffset)) {
253       context.parent ! KafkaHealthState(name, healthy = true, s"Successfully fetched to $ackedOffset", topic)
254       offsetManager ! StoreOffsetData(partitionName(assign), OffsetData(formatAckedOffset().getBytes(utf8)))
255     }
256   }

258   override def postStop(): Unit = {
259     stopWorker()
260   }

262   // Override to process initial offsets differently
263   protected def processOffsets(offsetData: OffsetData): State = {
264     val offset = extractOffset(offsetData)
265 
266     log.debug(s"$name has initialized with offset $offset")
267     ackedOffset = offset
268     lastSentToProxyOffset = offset
269     lastSentToStorage = offset
270 
271     goto(Consuming)
272   }

274   // Can be overridden to read custom offsets
275   protected def extractOffset(offsetData:OffsetData):Long = {
276     val data = offsetData.asString()
277     if(data.isEmpty) 0L else data.toLong
278   }

280   def stopWorker() = {
281     if(ackedOffset > 0) {
282       log.debug(s"Shutdown %s, storage offset=%d, proxy offset=%d, acked offset=%d"
283         .format(name, lastSentToStorage, lastSentToProxyOffset, ackedOffset))
284 
285       offsetManager ! StoreOffsetData(partitionName(assign), OffsetData(formatAckedOffset().getBytes(utf8)))
286     }
287     scheduler.foreach(_.cancel())
288     scheduler = List()
289 
290     stateData match {
291       case aq: Acquired =>
292         lock.returnLease(aq.lease)
293         log.debug(s"$name: Released lock")
294       case _ => // No lock
295     }
296     context.parent ! KafkaHealthState(name, healthy = true, null, null)
297   }

299   def acquireLock(): Option[Acquired] = {
300     try {
301       val acquiredLease = lock.acquire(5, TimeUnit.SECONDS) match {
302         case null => throw new InterruptedException("Could not acquire lock before timeout")
303         case l: Lease => l
304       }
305       log.debug(s"$name: Lock acquired")
306       Some(Acquired(acquiredLease))
307     } catch {
308       case e: Throwable =>
309         log.error(s"$name: Unable to obtain lock: ${e.getMessage} ")
310         None
311     }
312   }

314   /**
315    * Takes the list of messages receive in the last fetch and sends them to
316    * the main worker method account(...) which should be overridden with the actual
317    * work we want to do.
318    * @param fetchRes Contains our next offset and the list of messages we just got
319    * @return Returns a request for more messages
320    */
321   def consume(fetchRes: FetchRes) = {
322     lastSentToProxyOffset = fetchRes.nextOffset
323     onReceive(MessageResponse(fetchRes.messages, fetchRes.nextOffset))
324 
325     // After consumption of messages automatically fetch the next
326     if(fetchMode() == AUTOMATIC) {
327       fetchRequest (fetchRes.nextOffset)
328     }
329   }
330 
331   /**
332    * Fetch data from kafka
333    */
334   protected def fetchRequest(nextOffset: Long) = {
335     //log.debug(s"$name fetching offset $nextOffset")
336     try {
337       if (consumer.forall(_.closed)) {
338         fetchConsumer()
339       }
340       if (consumer.isDefined) {
341         self ! fetchTopicPart(topic, partition, nextOffset, clientId)
342       }
343     } catch {
344       case ex: OffsetOutOfRangeException =>
345         self ! FetchErrorBadOffset(KafkaUtil.getDesiredAvailableOffset(consumer.get, topic, partition, nextOffset, clientId))
346 
347       case ex: InvalidMessageSizeException =>
348         log.error(s"Failed to fetch message for $topic:$partition from $host because message was too big for buffer, you need to increase the buffer size!")
349         self ! FetchErrorPartitionNotAvailable(nextOffset)
350 
351       // Catch all. Respond with a generic error message that will result in the workers waiting before retrying
352       // and kick off a refresh of our kafka info
353       case ex: Exception =>
354         log.error(s"Failed to fetch from [$host] [$topic] [$partition] [$nextOffset] [${ex.getClass.toString}] [${ex.getMessage}]")
355         kafkaProxy ! KafkaRefreshReq(light = true)
356         self ! FetchErrorPartitionNotAvailable(nextOffset)
357     }
358   }
359 
360   def fetchTopicPart(topic: String, part: Int, offset: Long, clientId: String): FetchRes = {
361     val fetch = new FetchRequestBuilder()
362       .clientId(clientId)
363       .addFetch(topic, part, offset, 150000)
364       .build()
365 
366     val fetchResponse = consumer.get.fetch(fetch)
367     if (fetchResponse.hasError) {
368       ErrorMapping.maybeThrowException(fetchResponse.errorCode(topic, part))
369     }
370 
371     val msgSet = fetchResponse.messageSet(topic, part)
372 
373     val msgSeq = KafkaMessageSet(msgSet, offset) // This is needed since if Kafka is compressing the messages, the fetch request will return an entire compressed block even if the requested offset isn't the beginning of the compressed block. Thus a message we saw previously may be returned again.
374 
375     FetchRes(msgSeq, if (msgSet.size > 0) msgSet.last.nextOffset else offset)
376   }
