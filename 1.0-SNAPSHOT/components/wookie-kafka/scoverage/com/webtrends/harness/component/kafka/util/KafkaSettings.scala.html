35 trait KafkaSettings extends ConfigHelper { this: Actor =>
36   @volatile
37   var kafkaConfig: Config = ConfigUtil.prepareSubConfig(renewableConfig, "wookie-kafka")
38   val hostname = InetAddress.getLocalHost.getHostName
39 
40   // Used for ZK paths
41   def pod = zkConf.get.dataCenter+"_"+zkConf.get.pod
42   def appRootPath = s"/$appName/$pod"
43 
44   def distributionRootPath = s"${appRootPath}/kafkaConsumerDistribution"
45 
46   val utf8 = StandardCharsets.UTF_8
47 
48   def distributorPaths = DistributorPaths(s"${distributionRootPath}/nodes",
49                                           s"${distributionRootPath}/assignments",
50                                           s"${distributionRootPath}/leader")
51 
52   def zkConf = renewableConfig.hasPath("wookie-zookeeper.quorum") match {
53     case true => Some(ZookeeperSettings(renewableConfig.getConfig("wookie-zookeeper")))
54     case false => None
55   }
56   def topicWorker = this.getClass.getClassLoader.loadClass(kafkaConfig.getString("worker-class"))
57   def leader = {
58     if (kafkaConfig.hasPath("leader-class")) {
59       this.getClass.getClassLoader.loadClass(kafkaConfig.getString("leader-class"))
60     } else {
61       classOf[KafkaConsumerCoordinator]
62     }
63   }
64   def appName = kafkaConfig.getString("app-name")
65 
66   def init: KafkaSettings = {
67     this
68   }
69 
70   def topicMap: Map[String, Config] = kafkaConfig.getConfigList("consumer.topics").toList.map{top =>
71     top.getString("name") -> top
72   }.toMap
73   
74   def kafkaSources: Map[String, List[String]] = kafkaConfig.getConfigList("consumer.kafka-hosts").toList.map { clusters =>
75     clusters.getString("id") -> clusters.getStringList("brokers").toList
76   }.toMap
77 
78   def assignmentName(spec:PartitionAssignment): String = {
79     s"${spec.host}-${spec.topic}-${spec.partition}"
80   }
81 
82   def partitionName(spec:PartitionAssignment): String = {
83     s"${spec.cluster}-${spec.topic}-${spec.partition}"
84   }
85 
86   override def renewConfiguration() = {
87     super.renewConfiguration()
88     kafkaConfig = ConfigUtil.prepareSubConfig(renewableConfig, "wookie-kafka")
89   }
90 }
