33   def getDesiredAvailableOffset(consumer: SimpleConsumer, topic: String, partition: Int, desiredStartOffset: Long, clientName: String): Long = {
34     val topicAndPartition: TopicAndPartition = new TopicAndPartition(topic, partition)
35 
36     val startOffsetRequest = OffsetRequest(
37       Map(topicAndPartition -> PartitionOffsetRequestInfo(OffsetRequest.EarliestTime, 1)),
38       replicaId = 0) //TODO: determine replicaId, if needed?
39 
40     val startResponse: OffsetResponse = consumer.getOffsetsBefore(startOffsetRequest)
41     if (startResponse.hasError) {
42       throw new IllegalStateException("Kafka response error getting earliest offsets" )
43     }
44     val startOffsets: Seq[Long] = startResponse.partitionErrorAndOffsets(topicAndPartition).offsets
45     if (startOffsets.length != 1) {
46       throw new IllegalStateException(s"Expect one earliest offset but got [${startOffsets.length}]")
47     }
48 
49 
50     val endOffsetRequest = OffsetRequest(
51       Map(topicAndPartition -> PartitionOffsetRequestInfo(OffsetRequest.LatestTime, 1)),
52       replicaId = 0)  //TODO: determine replicaId, if needed?
53 
54     val endResponse: OffsetResponse = consumer.getOffsetsBefore(endOffsetRequest)
55     if (endResponse.hasError) {
56       throw new IllegalStateException("Kafka response error getting latest offsets" ) //+ startResponse.error())
57     }
58     val endOffsets: Seq[Long] = endResponse.partitionErrorAndOffsets(topicAndPartition).offsets
59     if (endOffsets.length != 1) {
60       throw new IllegalStateException(s"Expect one latest offset but got [${endOffsets.length}]")
61     }
62 
63     val range:Array[Long] = new Array(2)
64 
65     range(0) = startOffsets(0)
66     range(1) = endOffsets(0)
67 
68     if (range(1) < range(0)) {
69       range(0) = range(1)
70     }
71     if (desiredStartOffset > range(0) && desiredStartOffset < range(1)) {
72       range(0) = desiredStartOffset
73     }
74 
75     range(0)
76 
77   }
