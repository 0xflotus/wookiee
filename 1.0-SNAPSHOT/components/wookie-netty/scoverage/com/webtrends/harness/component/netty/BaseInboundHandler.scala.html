38   override final def channelRead(ctx: ChannelHandlerContext, msg: AnyRef) {
39     var msgHandled = false
40 
41     try {
42     msg match {
43       // http requests
44       case req: FullHttpRequest => msgHandled = handleHttpRequest(ctx, req)
45       // websocket frames
46       case frame: WebSocketFrame => msgHandled = handleWebSocketFrame(ctx, frame)
47     }
48   } finally {
49       if (!msgHandled) {
50         ctx.fireChannelRead(msg)
51       } else {
52         ReferenceCountUtil.release(msg)
53       }
54     }
55   }

80   def sendContent(ctx: ChannelHandlerContext, req: FullHttpRequest, content: String, contentType: String) {
81     val buf = Unpooled.copiedBuffer(content, CharsetUtil.UTF_8)
82     val res = new DefaultFullHttpResponse(req.getProtocolVersion, OK, buf)
83     res.headers().set(HttpHeaders.Names.CONTENT_TYPE, contentType.concat("; charset=UTF-8"))
84     sendHttpResponse(ctx, req, res)
85   }

100   /**
101    * Sends an HttpResponse
102    * @param ctx ChannelHandlerContext
103    * @param req The request being responded to
104    * @param res The response to be sent
105    */
106   def sendHttpResponse(ctx: ChannelHandlerContext, req: FullHttpRequest, res: FullHttpResponse) {
107     val keepAlive = HttpHeaders.isKeepAlive(req)
108 
109     HttpHeaders.setContentLength(res, res.content().readableBytes())
110 
111     res.getStatus.code match {
112       case 200 =>
113       case _ =>
114         val buf = Unpooled.copiedBuffer(res.getStatus.toString, CharsetUtil.UTF_8)
115         res.content().writeBytes(buf)
116         buf.release()
117         HttpHeaders.setContentLength(res, res.content().readableBytes())
118     }
119 
120     val f = ctx.channel().writeAndFlush(res)
121     if (!keepAlive || res.getStatus.code() != 200) {
122       f.addListener(ChannelFutureListener.CLOSE)
123     }
124   }
