53   implicit val timeout = Timeout(settings.requestTimeout, TimeUnit.SECONDS)
54   implicit val formats = DefaultFormats + new EnumNameSerializer(ComponentState) ++ JodaTimeSerializers.all

57   override def handleHttpRequest(ctx: ChannelHandlerContext, req: FullHttpRequest): Boolean = {
58     import scala.concurrent.ExecutionContext.Implicits.global
59 
60     if (!req.getDecoderResult.isSuccess) {
61       sendHttpResponse(ctx, req, new DefaultFullHttpResponse(req.getProtocolVersion(), BAD_REQUEST))
62     }
63 
64     (req.getMethod, req.getUri) match {
65       case (GET, "/favicon.ico") =>
66         sendHttpResponse(ctx, req, new DefaultFullHttpResponse(req.getProtocolVersion(), NO_CONTENT))
67       case (GET, "/ping") =>
68         sendContent(ctx, req, "pong: ".concat(new DateTime(System.currentTimeMillis(), DateTimeZone.UTC).toString), "text/plain")
69       case (GET, "/metrics") =>
70         (worker ? GetSystemInfo("wookie-metrics", ComponentRequest(StatusRequest()))).mapTo[ComponentResponse[JValue]] onComplete {
71           case Success(s) => sendContent(ctx, req, JsonAST.compactRender(s.resp), "application/json")
72           case Failure(f) => sendFailure(ctx, req, f)
73         }
74       case (GET, "/services") =>
75         (worker ? GetServiceInfo()).mapTo[Seq[ServiceMetaData]] onComplete {
76           case Success(s) => sendContent(ctx, req, compactRender(decompose(s)), "application/json")
77           case Failure(f) => sendFailure(ctx, req, f)
78         }
79       case (GET, "/cluster") =>
80         (worker ? GetSystemInfo("wookie-cluster", ComponentRequest(ClusterState(), Some("cluster")))).mapTo[ComponentResponse[JValue]] onComplete {
81           case Success(s) => sendContent(ctx, req, JsonAST.compactRender(s.resp), "application/json")
82           case Failure(f) => sendFailure(ctx, req, f)
83         }
84       case (GET, "/cluster/discovery") =>
85         (worker ? GetSystemInfo("wookie-cluster", ComponentRequest(Subscriptions()))).mapTo[ComponentResponse[JValue]] onComplete {
86           case Success(s) => sendContent(ctx, req, JsonAST.compactRender(s.resp), "application/json")
87           case Failure(f) => sendFailure(ctx, req, f)
88         }
89       case (GET, "/healthcheck/full") =>
90         (worker ? GetHealth(HealthResponseType.FULL)).mapTo[ApplicationHealth] onComplete {
91           case Success(s) => sendContent(ctx, req, compactRender(decompose(s)), "application/json")
92           case Failure(f) => sendFailure(ctx, req, f)
93         }
94       case (GET, "/healthcheck/lb") =>
95         (worker ? GetHealth(HealthResponseType.LB)).mapTo[String] onComplete {
96           case Success(s) => sendContent(ctx, req, s, "text/plain")
97           case Failure(f) => sendFailure(ctx, req, f)
98         }
99       case (GET, "/healthcheck/nagios") =>
100         (worker ? GetHealth(HealthResponseType.NAGIOS)).mapTo[String] onComplete {
101           case Success(s) => sendContent(ctx, req, s, "text/plain")
102           case Failure(f) => sendFailure(ctx, req, f)
103         }
104       case (GET, settings.websocketPath) if settings.websocketPath.length > 1 =>
105         val wsPath = s"ws://${req.headers().get(HOST)}"
106         val wsFactory = new WebSocketServerHandshakerFactory(wsPath, null, false)
107         handshaker = Some(wsFactory.newHandshaker(req))
108         handshaker match {
109           case Some(hs) => hs.handshake(ctx.channel(), req)
110           case None => WebSocketServerHandshakerFactory.sendUnsupportedVersionResponse(ctx.channel())
111         }
112       case _ => staticFile(ctx, req)
113     }
114 
115     // All cases were handled
116     true
117   }
118 
119   def staticFile(ctx: ChannelHandlerContext, req: FullHttpRequest): Unit = {
120     if (settings.fileLocation.isEmpty) {
121       sendHttpResponse(ctx, req, new DefaultFullHttpResponse(req.getProtocolVersion, NOT_FOUND))
122       return
123     }
124 
125     if (req.getMethod != HttpMethod.GET) {
126       sendHttpResponse(ctx, req, new DefaultFullHttpResponse(req.getProtocolVersion, METHOD_NOT_ALLOWED))
127       return
128     }
129 
130     // get input stream from file or from jar
131     val url = new URL(s"file://${req.getUri}")
132     val is = try {
133 
134       // load static from file or jar
135       val inputStream = settings.fileSource match {
136         case "file" => new FileInputStream(new File(s"${settings.fileLocation}${url.getPath}"))
137         case "jar" => getClass.getResourceAsStream(s"${settings.fileLocation}${url.getPath}")
138       }
139 
140       if (inputStream == null)
141         throw new NullPointerException
142       inputStream
143     } catch {
144       case e: Throwable =>
145         sendHttpResponse(ctx, req, new DefaultFullHttpResponse(req.getProtocolVersion, NOT_FOUND))
146         return
147     }
148 
149     // write headers
150     val response = new DefaultHttpResponse(req.getProtocolVersion, OK)
151     val mimeTypesMap = new MimetypesFileTypeMap()
152     response.headers().set(CONTENT_TYPE, mimeTypesMap.getContentType(url.getPath))
153     response.headers().set(Names.TRANSFER_ENCODING, HttpHeaders.Values.CHUNKED)
154     ctx.write(response)
155 
156     // write stream
157     ctx.write(new HttpChunkedInput(new ChunkedStream(is)))
158     val future = ctx.writeAndFlush(LastHttpContent.EMPTY_LAST_CONTENT)
159     if (!HttpHeaders.isKeepAlive(req)) {
160       future.addListener(ChannelFutureListener.CLOSE)
161     }
162   }
163 }
