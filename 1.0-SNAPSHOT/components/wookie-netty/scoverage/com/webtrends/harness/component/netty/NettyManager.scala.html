35   /**
36    * We add super.receive because if you override the receive message from the component
37    * and then do not include super.receive it will not handle messages from the
38    * ComponentManager correctly and basically not start up properly
39    *
40    * @return
41    */
42   override def receive = super.receive orElse {
43     case NettyRunning => context.parent ! ComponentStarted(self.path.name)
44     case SystemReady =>
45   }
46 
47   /**
48    * Start function will start any child actors that will be managed by the ComponentManager
49    * @return
50    */
51   override def start = {
52     startNettyServer(serverSettings)
53   }
54 
55   /**
56    * Stop will execute any cleanup work to be done for the child actors
57    * if not necessary this can be deleted
58    * @return
59    */
60   override def stop = {
61     super.stop

63   }
64 }
