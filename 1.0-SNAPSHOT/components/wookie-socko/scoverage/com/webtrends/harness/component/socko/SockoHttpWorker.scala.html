49 class SockoHttpWorker extends HActor with ComponentHelper {
50   import context.dispatcher
51   implicit val timeout = Timeout(10 seconds)
52   val staticContent = ConfigUtil.getDefaultValue(SockoManager.KeyRootPaths, context.system.settings.config.getString, "")
53 
54   implicit val formats = DefaultFormats + new EnumNameSerializer(ComponentState) ++ JodaTimeSerializers.all
55 
56   val healthActor = context.actorSelection(HarnessConstants.HealthFullName)
57   val serviceActor = context.actorSelection(HarnessConstants.ServicesFullName)
58 
59   override def postRestart(reason: Throwable): Unit = {
60     super.postRestart(reason)
61     self ! HttpStartProcessing
62   }
63 
64   override def receive: Receive = {
65     case HttpStartProcessing =>
66       log.debug("Worker beginning to process")
67       context.become(running)
68   }
69 
70   def running:Receive = super.receive orElse {
71     case SockoWorkerMessage(event) => handleSockoRoute(event)
72   }
73 
74   /**
75    * Attempts to handle the Socko route
76    * @param event The http request
77    */
78   protected def handleSockoRoute(event: HttpRequestEvent) = {
79     var handled = true
80     event match {
81       case GET(Path("/favicon.ico")) =>
82         event.response.write(HttpResponseStatus.NO_CONTENT)
83       case GET(Path("/ping")) =>
84         event.response.write("pong: ".concat(new DateTime(System.currentTimeMillis(), DateTimeZone.UTC).toString))
85       case req =>
86         req match {
87           case GET(Path("/healthcheck/lb")) =>
88             (healthActor ? HealthRequest(HealthResponseType.LB)).mapTo[String] onComplete {
89               case Success(h) => event.response.write(h)
90               case Failure(f) => messageFailure(f, event)
91             }
92           case GET(Path("/healthcheck/nagios")) =>
93             (healthActor ? HealthRequest(HealthResponseType.NAGIOS)).mapTo[String] onComplete {
94               case Success(h) => event.response.write(h)
95               case Failure(f) => messageFailure(f, event)
96             }
97           case GET(Path("/healthcheck/full")) =>
98             (healthActor ? HealthRequest(HealthResponseType.FULL)).mapTo[ApplicationHealth] onComplete {
99               case Success(h) => event.response.write(compactRender(decompose(h)))
100               case Failure(f) => messageFailure(f, event)
101             }
102           case GET(Path("/metrics")) =>
103             componentRequest[StatusRequest, JValue]("wookie-metrics", ComponentRequest(StatusRequest())) onComplete {
104               case Success(s) => event.response.write(JsonAST.compactRender(s.resp))
105               case Failure(f) => messageFailure(f, event)
106             }
107           case GET(PathSegments("services" :: relativePath)) =>
108             if (relativePath.size == 0 || relativePath(0).equals("")) {
109               (serviceActor ? GetMetaData(None)).mapTo[Seq[ServiceMetaData]] onComplete {
110                 case Success(s) => event.response.write(compactRender(decompose(s)))
111                 case Failure(f) => messageFailure(f, event)
112               }
113             } else {
114               (serviceActor ? GetMetaDataByName(relativePath(0))).mapTo[Seq[ServiceMetaData]] onComplete {
115                 case Success(s) => event.response.write(compactRender(decompose(s)))
116                 case Failure(f) => messageFailure(f, event)
117               }
118             }
119           case GET(Path("/cluster")) =>
120             val req = ComponentRequest(ClusterState(), Some("cluster"))
121             componentRequest[ClusterState, JValue]("wookie-cluster", req) onComplete {
122               case Success(s) => event.response.write(compactRender(s.resp))
123               case Failure(f) => messageFailure(f, event)
124             }
125           case GET(Path("/cluster/discovery")) =>
126             componentRequest[Subscriptions, JValue]("wookie-cluster", ComponentRequest(Subscriptions())) onComplete {
127               case Success(s) => event.response.write(compactRender(s.resp))
128               case Failure(f) => messageFailure(f, event)
129             }
130           case _ =>
131             var found = false
132             SockoRouteManager.getHandlers(event.nettyHttpRequest.getMethod.toString.toUpperCase) match {
133               case Some(handlers) =>
134                 try {
135                   handlers.takeWhile { handler =>
136                     handler._2.matchSockoRoute(event) match {
137                       case Some(map) =>
138                         Future {
139                           handler._2.handleSockoRoute(event, map)
140                         }
141                         found = true
142                       case None =>
143                     }
144                     !found
145                   }
146                 } catch {
147                   case ce: SockoCommandException => event.response.write(ce.code, s"Cause - ${ce.getMessage}")
148                   case ex: Exception => event.response.write(HttpResponseStatus.INTERNAL_SERVER_ERROR, s"Cause - ${ex.getMessage}")
149                 }
150               case None =>
151             }
152             handled = found
153         }
154     }
155 
156     if (!handled) {
157       //if the static content handler was initialized then lets send the message out to the content handler
158       if (ConfigUtil.getDefaultValue(SockoManager.KeyRootPaths, context.system.settings.config.getString, "").nonEmpty) {
159         context.actorSelection(s"${HarnessConstants.ComponentFullName}/wookie-socko").resolveOne() onComplete {
160           case Success(s) => s ! SockoContentRequest(event)
161           case Failure(f) => event.response.write(HttpResponseStatus.NOT_FOUND)
162         }
163       } else {
164         event.response.write(HttpResponseStatus.NOT_FOUND)
165       }
166     }
167   }
168 
169   def messageFailure(f:Throwable, event:HttpRequestEvent) = {
170     f match {
171       case notFound:ComponentNotFoundException => event.response.write(HttpResponseStatus.NOT_FOUND, notFound.getMessage)
172       case t => event.response.write(HttpResponseStatus.INTERNAL_SERVER_ERROR, t.getMessage)
173     }
174   }
