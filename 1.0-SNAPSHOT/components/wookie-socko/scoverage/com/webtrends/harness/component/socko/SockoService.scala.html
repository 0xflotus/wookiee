36 trait SockoService extends Service {
37 
38   // list of currently held Socko Route handlers
39   protected val handlers = mutable.Map[String, SockoRouteHandler]()
40 
41   def addHandler[T<:SockoRouteHandler](handlerName:String, handler:SockoRouteHandler, methods:Seq[String]) : Future[SockoRouteHandler] = {
42     val p = Promise[SockoRouteHandler]
43     if (classOf[Command].isAssignableFrom(handler.getClass)) {
44       CommandManager.getCommand(handlerName) match {
45         case Some(ref) => p success handler
46         case None =>
47           addCommand(handlerName, handler.getClass.asInstanceOf[Class[Command]]) onComplete {
48             case Success(ref) =>
49               handlers += handlerName -> handler
50               SockoRouteManager.addHandlers(handlerName, handler, methods)
51               p success handler
52             case Failure(f) => p failure f
53           }
54       }
55     } else {
56       handlers += handlerName -> handler
57       SockoRouteManager.addHandlers(handlerName, handler, methods)
58       p success handler
59     }
60     p.future
61   }
62 
63   // To be defined in service actor
64   override def serviceReceive = {
65     // This is called by the harness itself in order to get internal details of the service
66     case GetMetaDetails =>
67       // Get the services meta data
68       sender() ! getMetaDetails
69   }: Receive
70 
71   /**
72    * The actor has been asked to respond with some additional meta information.
73    * @return An instance of ServiceMetaDetails
74    */
75   protected def getMetaDetails: ServiceMetaDetails = {
76     ServiceMetaDetails(handlers.size>0)
77   }
