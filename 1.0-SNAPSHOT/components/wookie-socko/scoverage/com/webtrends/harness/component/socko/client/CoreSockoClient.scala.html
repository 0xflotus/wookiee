59   private def getHeaderFields(conn:HttpURLConnection) : Map[String, String] = {

65   def service[T:Manifest](config:HttpConfig, method:String, path:String, body:Option[T], headers:Map[String, String]) = {
66     val urlConnection = new URL(config.fullPath(path)).openConnection().asInstanceOf[HttpURLConnection]
67     urlConnection.setRequestMethod(method)
68     headers foreach {
69       case (name, value) => urlConnection.setRequestProperty(name, value)
70     }
71     method match {
72       case HttpConstants.POST | HttpConstants.PUT =>
73         body match {
74           case Some(v) =>
75             urlConnection.setDoOutput(true)
76             val wr = new DataOutputStream(urlConnection.getOutputStream)
77             // TODO marshall this
78             wr.writeBytes(v.toString)
79             wr.flush()
80             wr.close()
81           case None => // ignore
82         }
83       case _ => // don't have to do anything special with the other ones
84     }
85 
86     try {
87       urlConnection.connect()
88       val responseCode = urlConnection.getResponseCode
89 
90       val response = try {
91         val in = new BufferedReader(new InputStreamReader(urlConnection.getInputStream))
92         var inputLine: String = null
93         val response = new StringBuffer()
94         while ((inputLine = in.readLine()) != null) {
95           response.append(inputLine)
96         }
97         in.close()
98         response.toString.getBytes
99       } catch {
100         case t: Throwable =>
101           log.debug(t, t.getMessage)
102           new Array[Byte](0)
103       }
104       sender ! HttpResp(response, getHeaderFields(urlConnection), responseCode)
105     } finally {
106       urlConnection.disconnect()
107     }
108   }

110   def ping(config:HttpConfig, timeoutValue:Int=1, pingPath:String="ping") : Future[ConnectionStatus] = {
111     implicit val timeout = Timeout(Duration(timeoutValue, "seconds"))
112     val p = Promise[ConnectionStatus]()
113     try {
114       val f = (self ? HttpGet(config, pingPath))(timeout)
115       f onComplete {
116         case Failure(f) =>
117           p success ConnectionStatus(false, "Could not connect to " + config.fullPath("ping") + " responded with exception " + f.getMessage)
118         case Success(s) =>
119           s.asInstanceOf[HttpResp].statusCode match {
120             case 200 => p success ConnectionStatus(true, "Connected successfully to " + config.fullPath(pingPath))
121             case _ => p success ConnectionStatus(false, "Could not connect to " + config.fullPath(pingPath))
122           }
123       }
124     } catch {
125       case e:Throwable =>
126         log.debug("Could not connect to " + config.fullPath(pingPath) + " responded with error " + e.getMessage)
127         p success ConnectionStatus(false, "Could not connect to " + config.fullPath(pingPath) + " responded with error " + e.getMessage)
128     }
129     p.future
130   }
