91   def marshallObject(obj: Any, respType: String = "json"): Array[Byte] = {
92     val realRespType = SockoUtils.normalizeContentType(respType)
93     if (realRespType == "application/json") {
94       obj match {
95         case s: String => s.getBytes
96         case _ => compactRender(decompose(obj)).getBytes
97       }
98     } else obj match {
99       case bytes: Array[Byte] =>
100         bytes
101       case _ =>
102         obj.toString.getBytes
103     }
104   }

141   protected def getRejectionHandler(event:HttpRequestEvent, e:Throwable) = {
142     log.debug(e.getMessage, e)
143     e match {
144       case ce: ParseException =>
145         event.response.write(HttpResponseStatus.BAD_REQUEST, s"Invalid JSON - ${ce.getMessage}")
146       case ce: SockoCommandException =>
147         ce.msg == null || ce.msg == "" match {
148           case true => event.response.write(ce.code)
149           case false => event.response.write(ce.code, ce.msg)
150         }
151       case ce:CommandException =>
152         event.response.write(HttpResponseStatus.BAD_REQUEST, s"Command Exception - ${ce.getMessage}\n\t${ce.toString}")
153       case arg:IllegalArgumentException =>
154         event.response.write(HttpResponseStatus.BAD_REQUEST, s"Illegal Arguments - ${arg.getMessage}\n\t${arg.toString}")
155       case me:MappingException =>
156         event.response.write(HttpResponseStatus.BAD_REQUEST, s"Mapping Exception - ${me.getMessage}\n\t${me.toString}")
157       case ex:Exception =>
158         event.response.write(HttpResponseStatus.INTERNAL_SERVER_ERROR, s"Server Error - ${ex.getMessage}\n\t${ex.toString}")
159     }
160   }

187   def innerExecute[T<:AnyRef:Manifest](bean:SockoCommandBean) = {
188     try {
189       applyQSMap(bean)
190       execute(Some(bean)).mapTo[CommandResponse[T]] onComplete {
191         case Success(resp) =>
192           resp.data match {
193             case Some(value) =>
194               val content = if (bean.event.endPoint.method.equalsIgnoreCase("head")) Array[Byte]() else marshallObject(value, resp.responseType)
195               bean.event.response.write(
196                 responseStatusCode,
197                 // just double check the end point method, and if it is head then simply remove any content from the response
198                 content,
199                 SockoUtils.normalizeContentType(resp.responseType),
200                 SockoUtils.toSockoHeaderFormat(bean.event.nettyHttpRequest.getMethod.name(), getResponseHeaders(bean.event.request.headers), content.length)
201               )
202             case None => bean.event.response.write(HttpResponseStatus.NO_CONTENT)
203           }
204         case Failure(f) => f match {
205           case ce: SockoCommandException => getRejectionHandler(bean.event, ce)
206           case _ => bean.event.response.write(HttpResponseStatus.INTERNAL_SERVER_ERROR, f.getMessage)
207         }
208       }
209     } catch {
210       case e:Throwable => getRejectionHandler(bean.event, e)
211     }
212   }

261   private[route] def addEntityToBean[T<:AnyRef:Manifest](bean:SockoCommandBean, obj:Array[Byte]) = {
262     val conType = bean.headers.get("Content-Type") match {
263       case null => "application/x-www-form-urlencoded"
264       case x: String => x
265     }
266 
267     var content = obj
268     val contentEncoding =  bean.headers.get("Content-Encoding")
269     contentEncoding match {
270       case "gzip" =>
271         content = Source.fromInputStream(new GZIPInputStream(new ByteArrayInputStream(content))).mkString.getBytes
272       case "deflate" =>
273         content = Source.fromInputStream(new InflaterInputStream(new ByteArrayInputStream(content))).mkString.getBytes
274       case _ =>
275     }
276 
277     bean.put(SockoCommandBean.Content, content)
278     unmarshall(content, conType) match {
279       case Some(v) => bean.appendMap(Map(CommandBean.KeyEntity -> v))
280       case None => //ignore
281     }
282   }
283 
284   def unmarshall[T<:AnyRef:Manifest](obj: Array[Byte], contentType: String = "application/json"): Option[T] = {
285     if (obj == null || obj.length == 0) {
286       None
287     } else {
288       val stringObj = new String(obj, StandardCharsets.UTF_8).trim()
289       if (stringObj.length == 0) {
290         None
291       } else if (contentType == "application/x-www-form-urlencoded" || contentType == "multipart/form-data") {
292         var jObj = JObject(Nil)
293         new QueryStringDecoder("?" + stringObj, CharsetUtil.UTF_8).parameters foreach {
294           e => jObj = jObj merge JObject(List(JField(e._1, JString(e._2.last))))
295         }
296         Some(jObj.extract[T])
297       } else if (contentType == "application/json") {
298         Some(JsonParser.parse(URLDecoder.decode(stringObj, "UTF-8")).extract[T])
299       } else {
300         val text = JString(URLDecoder.decode(stringObj, "UTF-8"))
301         Some(JObject(List(JField("content", text))).extract[T])
302       }
303     }
304   }
