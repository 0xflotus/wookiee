90   def marshallObject(obj: Any, respType: String = "json"): Array[Byte] = {
91     val realRespType = SockoUtils.normalizeContentType(respType)
92     if (realRespType == "application/json") {
93       obj match {
94         case s: String => s.getBytes
95         case _ => compactRender(decompose(obj)).getBytes
96       }
97     } else obj match {
98       case bytes: Array[Byte] =>
99         bytes
100       case _ =>
101         obj.toString.getBytes
102     }
103   }

140   protected def getRejectionHandler(event:HttpRequestEvent, e:Throwable) = {
141     log.debug(e.getMessage, e)
142     e match {
143       case ce: ParseException =>
144         event.response.write(HttpResponseStatus.BAD_REQUEST, s"Invalid JSON - ${ce.getMessage}")
145       case ce: SockoCommandException =>
146         ce.msg == null || ce.msg == "" match {
147           case true => event.response.write(ce.code)
148           case false => event.response.write(ce.code, ce.msg)
149         }
150       case ce:CommandException =>
151         event.response.write(HttpResponseStatus.BAD_REQUEST, s"Command Exception - ${ce.getMessage}\n\t${ce.toString}")
152       case arg:IllegalArgumentException =>
153         event.response.write(HttpResponseStatus.BAD_REQUEST, s"Illegal Arguments - ${arg.getMessage}\n\t${arg.toString}")
154       case me:MappingException =>
155         event.response.write(HttpResponseStatus.BAD_REQUEST, s"Mapping Exception - ${me.getMessage}\n\t${me.toString}")
156       case ex:Exception =>
157         event.response.write(HttpResponseStatus.INTERNAL_SERVER_ERROR, s"Server Error - ${ex.getMessage}\n\t${ex.toString}")
158     }
159   }

186   def innerExecute[T<:AnyRef:Manifest](bean:SockoCommandBean) = {
187     try {
188       applyQSMap(bean)
189       execute(Some(bean)).mapTo[CommandResponse[T]] onComplete {
190         case Success(resp) =>
191           resp.data match {
192             case Some(value) =>
193               val content = if (bean.event.endPoint.method.equalsIgnoreCase("head")) Array[Byte]() else marshallObject(value, resp.responseType)
194               bean.event.response.write(
195                 responseStatusCode,
196                 // just double check the end point method, and if it is head then simply remove any content from the response
197                 content,
198                 SockoUtils.normalizeContentType(resp.responseType),
199                 SockoUtils.toSockoHeaderFormat(bean.event.nettyHttpRequest.getMethod.name(), getResponseHeaders(bean.event.request.headers), content.length)
200               )
201             case None => bean.event.response.write(HttpResponseStatus.NO_CONTENT)
202           }
203         case Failure(f) => f match {
204           case ce: SockoCommandException => getRejectionHandler(bean.event, ce)
205           case _ => bean.event.response.write(HttpResponseStatus.INTERNAL_SERVER_ERROR, f.getMessage)
206         }
207       }
208     } catch {
209       case e:Throwable => getRejectionHandler(bean.event, e)
210     }
211   }

260   private[route] def addEntityToBean[T<:AnyRef:Manifest](bean:SockoCommandBean, obj:Array[Byte]) = {
261     val conType = bean.headers.get("Content-Type") match {
262       case null => "application/x-www-form-urlencoded"
263       case x: String => x
264     }
265 
266     var content = obj
267     val contentEncoding =  bean.headers.get("Content-Encoding")
268     contentEncoding match {
269       case "gzip" =>
270         content = Source.fromInputStream(new GZIPInputStream(new ByteArrayInputStream(content))).mkString.getBytes
271       case "deflate" =>
272         content = Source.fromInputStream(new InflaterInputStream(new ByteArrayInputStream(content))).mkString.getBytes
273       case _ =>
274     }
275 
276     bean.put(SockoCommandBean.Content, content)
277     unmarshall(content, conType) match {
278       case Some(v) => bean.appendMap(Map(CommandBean.KeyEntity -> v))
279       case None => //ignore
280     }
281   }
282 
283   def unmarshall[T<:AnyRef:Manifest](data: Array[Byte], contentTypeValue: String = "application/json"): Option[T] = {
284     if (data == null || data.length == 0) {
285       None
286     } else {
287 
288       val (contentType, charset) = parseContentType(contentTypeValue)
289 
290       contentType match {
291 
292         case "application/x-www-form-urlencoded" | "multipart/form-data" =>
293           val dataString = new String(data, charset).dropWhile( _ == "?")
294           val jObj = JObject(new QueryStringDecoder("?" + dataString, charset).parameters.map { kvp =>
295             JField(kvp._1, if (kvp._2 != null) new JString(kvp._2.last) else JNull)
296             }.toList)
297           Some(jObj.extract[T])
298 
299         case "application/json" =>
300           Some(JsonParser.parse(new String(data, "utf-8")).extract[T])
301 
302         case _ =>
303           val dataString = new String(data, charset)
304           Some(JObject(List(JField("content", new JString(dataString)))).extract[T])
305       }
306     }
307   }
308 
309   def parseContentType(ct: String): (String, Charset) = {
310 
311     val parts = ct.split(";").toList.map{part => part.toLowerCase.trim}
312 
313     val contentType = if (parts.length > 0 && !parts(0).isEmpty) parts(0) else "text/plain"
314 
315     val charSet = if (parts.length >= 2 && parts(1).startsWith("charset=")) {
316       Charset.forName(parts(1).split("=")(1))
317     }
318     else {
319       CharsetUtil.UTF_8
320     }
321 
322     (contentType, charSet)
323   }
