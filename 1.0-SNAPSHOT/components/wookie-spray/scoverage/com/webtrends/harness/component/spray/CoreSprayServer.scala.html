46 class CoreSprayServer(port:Int, settings:Option[ServerSettings]=None) extends HActor {
47   import context.system
48   import context.dispatcher
49 
50   implicit val timeout = Timeout(2 seconds)
51 
52   override def receive: Receive = binding
53 
54   def binding: Receive = {
55     case BindServer => bindHttpServer
56   }
57 
58   def running: Receive = health orElse {
59     case HttpStartProcessing =>
60       log.debug("Http processing starting.")
61       context.child(CoreSprayServer.ParamChildWorker).get ! Broadcast(HttpStartProcessing)
62     case HttpReloadRoutes =>
63       log.debug("Http reloading routes.")
64       context.child(CoreSprayServer.ParamChildWorker).get ! Broadcast(HttpReloadRoutes)
65     case ShutdownServer => shutdownHttpServer
66   }
67 
68   var metricsJob: Option[Cancellable] = None
69 
70   def httpServer = IO(UHttp)
71 
72   /**
73    * Bind to the http service
74    */
75   protected def bindHttpServer: Unit = {
76     // a running HttpServer can be bound, unbound and rebound
77     // initially to need to tell it where to bind to
78     log.info(s"Trying to bind to port ${port}")
79 
80     // Create and start the spray-can HttpServer, telling it that we want
81     // requests to be handled by the root service actor
82     if (context.child(CoreSprayServer.ParamChildWorker).isEmpty) {
83       (httpServer ? Http.Bind(listener = AkkaUtil.initActorFromConfig(Props[CoreSprayWorker], CoreSprayServer.ParamChildWorker),
84         interface = "0.0.0.0", port = port, settings = settings))
85         .onComplete {
86           case Failure(e) =>
87             log.error(s"Error trying to bind to the http port ${port}", e)
88             throw e
89           case Success(s) =>
90             context.become(running)
91             context.parent ! HttpRunning
92       }
93     }
94   }
95 
96   /**
97    * Unbind from the http service
98    */
99   protected def shutdownHttpServer: Unit = {
100 
101     metricsJob match {
102       case Some(job) =>
103         log.info("Cancelling the http server metrics job")
104         job.cancel
105       case None =>
106       // Do nothing
107     }
108 
109     log.info(s"Trying to shutdown the http server ${InetAddress.getLocalHost.getHostName} on port ${port}")
110     val f = httpServer ? Http.CloseAll
111     Try(Await.result(f, 1 second)).recover({
112       case e: Throwable =>
113         log.warning("we were unable to properly shutdown the HttpServer: {}", e.getMessage)
114     })
115 
116     log.info("Http server shutdown complete")
117     context.become(binding)
118   }
119 
120   /**
121    * This is the health of the current object, by default will be NORMAL
122    * In general this should be overridden to define the health of the current object
123    * For objects that simply manage other objects you shouldn't need to do anything
124    * else, as the health of the children components would be handled by their own
125    * CheckHealth function
126    *
127    * @return
128    */
129   override protected def getHealth: Future[HealthComponent] = {
130     val future = system.actorSelection(httpServer.path.toString.concat("/listener-0")) ? Http.GetStats
131     val p = Promise[HealthComponent]()
132 
133     future.onComplete {
134       case f: Failure[_] =>
135         p complete Success(HealthComponent(
136           "http-server",
137           ComponentState.CRITICAL,
138           "The internal http server is not running: " + f.exception.getMessage,
139           None)
140         )
141       case s: Success[_] =>
142         p complete Success(HealthComponent(
143           "http-server",
144           ComponentState.NORMAL,
145           s"The internal http server is running on port ${port}",
146           Some(s.value.asInstanceOf[Stats]))
147         )
148     }
149 
150     p.future
151   }
