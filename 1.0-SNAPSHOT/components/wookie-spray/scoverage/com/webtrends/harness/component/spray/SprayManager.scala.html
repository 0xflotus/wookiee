32 class SprayManager(name:String) extends Component(name)
33     with SprayServer
34     with SprayClient
35     with SprayWebSocketServer {
36 
37   implicit val spSettings = ServerSettings(config)
38   implicit val port = ConfigUtil.getDefaultValue(s"${name}.http-port", config.getInt, 8080)
39   implicit val websocketPort = ConfigUtil.getDefaultValue(s"${name}.websocket-port", config.getInt, 8081)
40   var rCount = new AtomicInteger(0)
41 
42   override protected def defaultChildName: Option[String] = Some(SprayServer.SprayServerName)
43 
44   override def receive = super.receive orElse {
45     case HttpRunning =>
46       checkRunning
47     case WebSocketRunning =>
48       checkRunning
49     case HttpReloadRoutes =>
50       httpServer match {
51         case Some(s) => s ! HttpReloadRoutes
52         case None => // ignore
53       }
54     case SystemReady =>
55       httpServer match {
56         case Some(s) => s ! HttpStartProcessing
57         case None => // ignore
58       }
59   }
60 
61   private def checkRunning: Unit = {
62     rCount.getAndIncrement()
63     if (rCount.get() == 2) {
64       context.parent ! ComponentStarted(self.path.name)
65     }
66   }
67 
68   override def start = {
69     startSprayServer(port, Some(spSettings))
70     startWebSocketServer(websocketPort, Some(spSettings))
71     // start the HttpClient actor
72     startSprayClient
73   }
74 
75   override def stop = {
76     super.stop
77     stopSprayServer
78     stopWebSocketServer
79   }
80 }
