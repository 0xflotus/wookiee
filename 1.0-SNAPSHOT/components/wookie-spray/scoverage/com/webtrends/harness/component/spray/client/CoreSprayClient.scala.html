50 class CoreSprayClient extends HActor with HttpLiftSupport {
51   import context.dispatcher
52   implicit val system:ActorSystem = context.system
53 
54   val pipeline:HttpRequest => Future[HttpResponse] = (
55     encode(Gzip)
56       ~> sendReceive(IO(UHttp))
57       ~> decode(Gzip)
58     )
59 
60   /**
61    * Will route messages to correct behavior, mainly will go to service
62    */
63   override def receive = super.receive orElse {
64     case HttpGet(config, path, headers) => service(config, GET, path, None, headers)
65     case HttpPost(config, path, body, headers) => service(config, POST, path, Some(body), headers)
66     case HttpPut(config, path, body, headers) => service(config, PUT, path, Some(body), headers)
67     case HttpDelete(config, path, headers) => service(config, DELETE, path, None, headers)
68     case HttpOptions(config, path, body, headers) => service(config, OPTIONS, path, Some(body), headers)
69     case HttpPatch(config, path, body, headers) => service(config, PATCH, path, Some(body), headers)
70     case HttpPing(config, timeout, path) => pipe(ping(config, timeout, path)) to sender
71   }
72 
73   /**
74    * Will service the request by routing
75    *
76    * @param config
77    * @param method
78    * @param path
79    * @param body
80    * @param headers
81    * @tparam T
82    */
83   // add marshallers based on the content-type in the headers
84   def service[T:Manifest](config:HttpConfig, method:HttpMethod, path:String, body:Option[T], headers:List[HttpHeader]) = {
85     implicit val HttpJsonMarshaller =
86       Marshaller.of[T](ContentTypes.`application/json`) { (value, contentType, ctx) =>
87         ctx.marshalTo(HttpEntity(contentType, compactRender(decompose(value))))
88       }
89     try {
90       val fullPath = config.fullPath(path)
91       val responseFuture = pipeline {
92         method match {
93           case GET =>
94             Get(fullPath).withHeaders(headers)
95           case POST =>
96             Post(fullPath, body).withHeaders(headers)
97           case PUT =>
98             Put(fullPath, body).withHeaders(headers)
99           case DELETE =>
100             Delete(fullPath).withHeaders(headers)
101           case OPTIONS =>
102             Options(fullPath, body).withHeaders(headers)
103           case PATCH =>
104             Patch(fullPath, body).withHeaders(headers)
105           case HEAD =>
106             Head(fullPath).withHeaders(headers)
107         }
108       }
109 
110       val caller = sender
111       responseFuture onComplete {
112         case Success(s) => caller ! HttpResp(s.entity, s.headers, s.status)
113         case Failure(f) =>
114           caller ! Status.Failure(HttpClientException("URL[%s] : %s".format(fullPath, f.getMessage), f))
115       }
116     } catch {
117       case NonFatal(e) =>
118         sender ! Status.Failure(HttpClientException(e.getMessage, e))
119         throw HttpClientException(e.getMessage, e)
120     }
121   }
122 
123   def shutdown = {
124     val f = IO(UHttp).ask(Http.CloseAll)(2 seconds)
125     Await.ready(f, 2 seconds)
126   }
127 
128   def ping(config:HttpConfig, timeoutValue:Int=1, pingPath:String="ping") : Future[ConnectionStatus] = {
129     implicit val timeout = Timeout(Duration(timeoutValue, "seconds"))
130     val p = Promise[ConnectionStatus]()
131     try {
132       val f = (self ? HttpGet(config, pingPath))(timeout)
133       f onComplete {
134         case Failure(f) =>
135           p success ConnectionStatus(false, "Could not connect to " + config.fullPath("ping") + " responded with exception " + f.getMessage)
136         case Success(s) =>
137           s.asInstanceOf[HttpResp].statusCode match {
138             case s:StatusCodes.Success => p success ConnectionStatus(true, "Connected successfully to " + config.fullPath(pingPath))
139             case _ => p success ConnectionStatus(false, "Could not connect to " + config.fullPath(pingPath))
140           }
141       }
142     } catch {
143       case e:Throwable =>
144         log.debug("Could not connect to " + config.fullPath(pingPath) + " responded with error " + e.getMessage)
145         p success ConnectionStatus(false, "Could not connect to " + config.fullPath(pingPath) + " responded with error " + e.getMessage)
146     }
147     p.future
148   }
