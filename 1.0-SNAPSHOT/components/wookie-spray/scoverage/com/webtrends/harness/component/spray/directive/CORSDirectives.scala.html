26 trait CORSDirectives extends BaseDirectives {
27   def respondWithCORSHeaders(origin: AllowedOrigins) =
28     respondWithHeaders(
29       `Access-Control-Allow-Origin`(origin),
30       `Access-Control-Allow-Credentials`(false))
31 
32   def corsFilter(allowedOrigins: AllowedOrigins)(route: Route) =
33     allowedOrigins match {
34       case AllOrigins =>
35         respondWithCORSHeaders(AllOrigins)(route)
36       case _ =>
37         optionalHeaderValueByName("Origin") {
38           case None => route
39           case Some(clientOrigin) =>
40             if (allowedOrigins == SomeOrigins(Seq(HttpOrigin(clientOrigin)))) {
41               respondWithCORSHeaders(allowedOrigins)(route)
42             } else {
43               complete(Forbidden, Nil, "Invalid origin") // Maybe, a Rejection will fit better
44             }
45         }
46     }
47 }
