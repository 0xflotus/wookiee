28 class EnumerationSerializer(enums: Enumeration*) extends Serializer[Enumeration#Value] {
29   val EnumerationClass = classOf[Enumeration#Value]
30   val formats = Serialization.formats(NoTypeHints)
31   def deserialize(implicit format: Formats): PartialFunction[(TypeInfo, JValue), Enumeration#Value] = {
32     case (TypeInfo(EnumerationClass, _), json) => json match {
33       case JString(value) => enums.find(_.values.exists(_.toString == value)).get.withName(value)
34       case value          => throw new MappingException("Can't convert " + value + " to " + EnumerationClass)
35     }
36   }
37 
38   def serialize(implicit format: Formats): PartialFunction[Any, JValue] = {
39     case i: Enumeration#Value => i.toString
40   }
