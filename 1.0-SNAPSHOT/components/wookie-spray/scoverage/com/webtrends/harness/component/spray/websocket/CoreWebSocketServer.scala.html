23 class CoreWebSocketServer(port:Int, settings:Option[ServerSettings]=None) extends HActor {
24   import context.system
25 
26   override def receive: Receive = binding
27 
28   def binding: Receive = {
29     case WebSocketBindServer => bindWebSocketServer()
30   }
31 
32   def running: Receive = health orElse {
33     // when a new connection comes in we register a WebSocketConnection actor as the per connection handler
34     case Http.Connected(remoteAddress, localAddress) =>
35       log.debug("WebSocket connected")
36       val serverConnection = sender()
37       val worker = context.actorOf(CoreWebSocketWorker.props(serverConnection))
38       serverConnection ! Http.Register(worker)
39     case WebSocketShutdownServer =>
40       shutdownWebSocketServer()
41     case x =>
42       log.debug("Unkown message: " + x)
43   }
44 
45   val server = IO(UHttp)
46 
47   def bindWebSocketServer(): Unit = {
48     import scala.concurrent.ExecutionContext.Implicits.global
49 
50     log.info(s"Trying to bind websocket server to port $port")
51 
52     (server ? Http.Bind(self, interface = "0.0.0.0", port = port, settings = settings)) onComplete {
53       case Failure(e) =>
54         log.error(s"Error trying to bind websocket server to port $port", e)
55         throw e
56       case Success(s) =>
57         context.become(running)
58         context.parent ! WebSocketRunning
59     }
60   }
61 
62   /**
63    * Unbind from the http service
64    */
65   protected def shutdownWebSocketServer(): Unit = {
66     log.info(s"Trying to shutdown the websocket server ${InetAddress.getLocalHost.getHostName} on port $port")
67     val f = server ? Http.CloseAll
68     Try(Await.result(f, 1 second)).recover({
69       case e: Throwable =>
70         log.warning("we were unable to properly shutdown the websocket server: {}", e.getMessage)
71     })
72 
73     log.info("WebSocket server shutdown complete")
74     context.become(binding)
75   }
76 
77   override protected def getHealth: Future[HealthComponent] = {
78     import scala.concurrent.ExecutionContext.Implicits.global
79 
80     val future = system.actorSelection(server.path.toString.concat("/listener-0")) ? Http.GetStats
81     val p = Promise[HealthComponent]()
82 
83     future.onComplete {
84       case f: Failure[_] =>
85         p complete Success(HealthComponent(
86           "websocket-server",
87           ComponentState.CRITICAL,
88           "The internal websocket server is not running: " + f.exception.getMessage,
89           None)
90         )
91       case s: Success[_] =>
92         p complete Success(HealthComponent(
93           "websocket-server",
94           ComponentState.NORMAL,
95           s"The internal websocket server is running on port $port",
96           Some(s.value.asInstanceOf[Stats]))
97         )
98     }
99 
100     p.future
101   }
102 }
