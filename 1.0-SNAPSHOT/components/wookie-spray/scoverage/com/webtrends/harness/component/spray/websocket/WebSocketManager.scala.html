37   def addWorker[T](path: String, props:Props): Unit = {
38     externalLogger.debug(s"new websocket worker [$path] registered with WebSocketManager")
39     workers += path -> props
40   }
41 
42   def removeCommand(path:String) : Boolean = {
43     workers.get(path) match {
44       case Some(n) =>
45         externalLogger.debug(s"Websocket worker [$path] removed from WebSocketManager")
46         workers -= path
47         true
48       case None => false
49     }
50   }
51 
52   def getWorker(path: String): Option[Props] = {
53     workers.get(path)
54   }
