39   /**
40    * Get the node/cluster base path
41    * @param config the system config
42    * @return the base path
43    */
44   def getBasePath(config: Config): String = {
45     val zookeeperSettings = ZookeeperSettings(config.getConfig("wookie-zookeeper"))
46     val basePath = if (config.hasPath("wookie-cluster.base-path")) {
47       config.getConfig("wookie-cluster").getString("base-path")
48     } else {
49       zookeeperSettings.basePath
50     }
51     s"$basePath/${zookeeperSettings.dataCenter}_${zookeeperSettings.pod}/${zookeeperSettings.version}"
52   }
53 }

55 trait NodeRegistration extends ZookeeperAdapter {
56   this: Actor with ActorLoggingAdapter =>
57 
58   implicit val timeout = Timeout(5 seconds)
59 
60   import context.dispatcher

62   val utf8 = Charset.forName("UTF-8")
63   val address = SystemExtension(context.system).address
64   val port = if (address.port.isDefined) address.port.get else context.system.settings.config.getInt("akka.remote.netty.tcp.port")
65 
66 
67   private def getAddress: String = {
68 
69     val host = if (address.host.isEmpty || address.host.get.equalsIgnoreCase("localhost") || address.host.get.equals("127.0.0.1")) {
70       InetAddress.getLocalHost.getCanonicalHostName
71     } else {
72       address.host.get
73     }
74 
75     s"${host}:${port}"
76   }
77 
78   def unregisterNode(curator: CuratorFramework, zookeeperSettings: ZookeeperSettings) = {
79 
80     val path = s"${NodeRegistration.getBasePath(context.system.settings.config)}/nodes/${getAddress}"
81     Try({
82       // Call Curator directly because this method is usually called after the actor's queue has been disabled
83       curator.delete.forPath(path)
84     }).recover({
85       case e: NoNodeException =>
86       // do nothing
87       case e: Throwable =>
88         log.warn(e, "The node {} could not be deleted", path)
89     })
90   }
91 
92   def registerNode(curator: CuratorFramework, zookeeperSettings: ZookeeperSettings, clusterEnabled: Boolean) {
93 
94     val add = getAddress
95     val path = s"${NodeRegistration.getBasePath(context.system.settings.config)}/nodes/${add}"
96 
97     // Delete the node first
98     deleteNode(path) onComplete {
99       case Success(_) => log.debug("The node {} was deleted", path)
100       case Failure(t) => log.error(t, "The node {} could not be deleted", path)
101     }
102 
103     log.info("Registering harness to path: " + path)
104     val json = compact(render(("address" -> add.toString) ~ ("cluster-enabled" -> clusterEnabled)))
105 
106     createNode(path, true, Some(json.getBytes(utf8))) onComplete {
107       case Success(_) => log.debug("The node {} was created", path)
108       case Failure(t) => log.error(t, "Failed to create node registration for {}", path)
109     }
110   }
