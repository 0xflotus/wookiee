149   private def getChildren(path: String, includeData: Boolean, namespace: Option[String]) = {
150     try {
151       val nodes = for {
152         child <- getClientContext(namespace).getChildren.forPath(path)
153         data = if (includeData) Some(getClientContext(namespace).getData.forPath(s"${path}/${child}")) else None
154       } yield (child, data)
155       sender() ! nodes
156     }
157     catch {
158       case e: Exception =>
159         log.error(e, "An error occurred trying to fetch children from the path {}", path)
160         sender() ! Status.Failure(e)
161     }
162   }

164   private def getData(path: String, namespace: Option[String]) = {
165     try {
166       sender() ! getClientContext(namespace).getData.forPath(path)
167     }
168     catch {
169       case nn: NoNodeException =>
170         log.error("No node found for path {}", path)
171         sender() ! Status.Failure(nn)
172       case e: Exception =>
173         log.error(e, "An error occurred trying to fetch data from the path {}", path)
174         sender() ! Status.Failure(e)
175     }
176   }

178   private def setData(path: String, data: Array[Byte], create: Boolean, ephemeral: Boolean, namespace: Option[String]) = {
179     def nodeCreate: String = {
180       try {
181         val mode = if (ephemeral) CreateMode.EPHEMERAL else CreateMode.PERSISTENT
182         getClientContext(namespace).create.creatingParentsIfNeeded.withMode(mode).forPath(path)
183       } catch {
184         case ne: NodeExistsException => path
185         case e: Exception =>
186           log.error(e, "An error occurred trying to create a node for the path {}", path)
187           path
188       }
189     }
190 
191     val dataPath = if (create) nodeCreate else path
192 
193     try {
194       getClientContext(namespace).setData.forPath(dataPath, data)
195       sender() ! data.length
196     }
197     catch {
198       case e: Exception =>
199         log.error(e, "An error occurred trying to set data for the path {}", path)
200         sender() ! Status.Failure(e)
201     }
202   }

262   protected def handleStateChange(state: ConnectionState) = {
263     log.info("Zookeeper connection state changed to {}", state.name)
264     currentState = state
265 
266     if (currentState == ConnectionState.RECONNECTED || currentState == ConnectionState.CONNECTED) {
267       unstashAll()
268       context.become(processing)
269     }
270     else {
271       context.become(initializing)
272     }
273 
274     // Now update all of the registered listeners
275     stateRegistrars foreach {
276       act => act ! ZookeeperStateEvent(currentState)
277     }
278 
279     state match {
280       case ConnectionState.RECONNECTED =>
281         // Re-register our self
282         self ! RegisterNode
283       case ConnectionState.CONNECTED =>
284         // Re-register our self
285         self ! RegisterNode
286       case _ => // Do nothing
287     }
288   }

290   private def registerForEvents(reg: RegisterZookeeperEvent) = {
291     reg.to match {
292       case ZookeeperStateEventRegistration(registrar) =>
293         stateRegistrars += registrar
294         // Send the current state of ZK to the registrar
295         registrar ! ZookeeperStateEvent(currentState)
296 
297       case ZookeeperChildEventRegistration(registrar, path, optNamespace) =>
298         // Update the child registrations
299         childRegistrars.get((path, optNamespace)).map {
300           entry =>
301             if (!entry.registrars.contains(registrar)) {
302               childRegistrars.updated((path, optNamespace), entry.copy(registrars = entry.registrars ++ Set(registrar)))
303             }
304         }.getOrElse {
305           Try({
306             // Adding a new cache
307             val cache = new PathChildrenCache(getClientContext(optNamespace), path, true)
308             cache.getListenable.addListener(this)
309             cache.start(StartMode.POST_INITIALIZED_EVENT)
310             childRegistrars += ((path, optNamespace) -> CacheEntry(cache, Set(registrar)))
311           }).recover({
312             case e: Exception => log.error(s"An error occurred trying to create a path listener for the path $path", e)
313           })
314         }
315 
316       case ZookeeperLeaderEventRegistration(registrar, path, optNamespace) =>
317         // Update the leadership registrations
318         leadershipRegistrars.get((path, optNamespace)).map {
319           entry =>
320             if (!entry.registrars.contains(registrar)) {
321               leadershipRegistrars.updated((path, optNamespace), entry.copy(registrars = entry.registrars ++ Set(registrar)))
322             }
323         }.getOrElse {
324           Try({
325             // Adding a new leader selector
326             val sel = new LeaderLatch(getClientContext(optNamespace), path)
327             sel.addListener(new LeaderListener(path, self, optNamespace))
328             sel.start
329             leadershipRegistrars += ((path, optNamespace) -> LeaderEntry(sel, Set(registrar)))
330           }).recover({
331             case e: Exception => log.error(s"An error occurred trying to create a leader selector for the path $path", e)
332           })
333         }
334     }
335 
336   }

338   private def unregisterForEvents(reg: UnregisterZookeeperEvent) = {
339     reg.to match {
340       case ZookeeperStateEventRegistration(registrar) => stateRegistrars -= registrar
341       case ZookeeperChildEventRegistration(registrar, path, optNamespace) =>
342         childRegistrars.get((path, optNamespace)).map {
343           entry =>
344             if (entry.registrars.contains(registrar)) {
345               val newEntry = ((path, optNamespace) -> entry.copy(registrars = entry.registrars.filterNot(_ == registrar)))
346               if (newEntry._2.registrars.isEmpty) {
347                 // No more registrars so we can shutdown the cache
348                 entry.cache.getListenable.removeListener(this)
349                 entry.cache.close
350                 val key = (path, optNamespace)
351                 childRegistrars -= key
352               }
353               else {
354                 childRegistrars += newEntry
355               }
356             }
357         }
358 
359       case ZookeeperLeaderEventRegistration(registrar, path, optNamespace) =>
360         leadershipRegistrars.get((path, optNamespace)).map {
361           entry =>
362             if (entry.registrars.contains(registrar)) {
363               val newEntry = ((path, optNamespace) -> entry.copy(registrars = entry.registrars.filterNot(_ == registrar)))
364               if (newEntry._2.registrars.isEmpty) {
365                 // No more registrars so we can shutdown the leader selector
366                 entry.leader.close
367                 val key = (path, optNamespace)
368                 leadershipRegistrars -= key
369               }
370               else {
371                 leadershipRegistrars += newEntry
372               }
373             }
374         }
375     }

389         false
390       case ConnectionState.RECONNECTED =>
391         true
392       case ConnectionState.CONNECTED =>

394     }

397     val msg = if (zookeeperState == ComponentState.NORMAL) {
398       s"We are currently connected to the zookeeper quorum on ${settings.quorum}"
399     }
400     else {
401       s"We are currently not able to connect to the zookeeper quorum on ${settings.quorum}"
402     }
403 
404     Future {
405       new HealthComponent("zookeeper", zookeeperState, details = msg, extra = None)
406     }

446   /**
447    * Curator handler for when a child node changes for a path that we are watching
448    * @param curator
449    * @param event
450    */
451   def childEvent(curator: CuratorFramework, event: PathChildrenCacheEvent): Unit = {
452     // Ignore any initialization events
453     if (event != null && (event.getType == PathChildrenCacheEvent.Type.CHILD_ADDED ||
454       event.getType == PathChildrenCacheEvent.Type.CHILD_REMOVED ||
455       event.getType == PathChildrenCacheEvent.Type.CHILD_UPDATED)) {
456 
457       val path = event.getData.getPath
458       log.debug(s"Zookeeper child event for path ${path}")
459 
460       // Cycle through the paths to determine if any start with this change
461       for (entry <- childRegistrars) {
462         if (path.startsWith(entry._1._1)) {
463           // Notify the registered listeners
464           for (act <- entry._2.registrars) act ! ZookeeperChildEvent(event)
465         }
466       }
467     }
468   }
