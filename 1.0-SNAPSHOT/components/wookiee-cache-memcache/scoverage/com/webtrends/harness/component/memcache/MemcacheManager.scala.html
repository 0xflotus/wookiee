33 /**
34  * MemcacheManager will allow users to send messages to create caches, get, and various other methods
35  * against Memcache. This code uses the Finagle library, so it's Futures and Promises are Finagle Futures and Promises
36  * which is not the same as Scala Future and Promises, although they work pretty much the same way.
37  */

60   /**
61    * Function that will convert Twitter Futures to Scala Futures, so we can respond to the user correctly
62    *
63    * @param twitterFuture
64    * @tparam A
65    * @return
66    */
67   def fromTwitter[A](twitterFuture: TwitterFuture[A]): Future[A] = {
68     val promise = Promise[A]()
69     twitterFuture respond {
70       case Return(a) => promise success a
71       case Throw(e)  => promise failure e
72     }
73     promise.future
74   }

76   /**
77    * Creates a cache with the given configuration

79    * @param config Configuration including namespace and setKey
80    * @return
81    */
82   override protected def createCache(config:CacheConfig) : Boolean = {
83     caches.get(config.namespace) match {
84       case Some(c) =>
85         if (c == null)
86           false
87         else
88           true
89       case None =>
90         val cache = new Memcache(build(config), config)
91         MemcacheManager.addCache(config.namespace, cache)
92         true
93     }
94   }

199   /**
200    * This builds the memcache client.
201    * In the future we will need to create more options for different types of clients, like Replication, Cluster and Ketama
202    * Initially will just use SimpleClient
203    *
204    * TODO: add timeouts for the connections
205    *
206    * @param config
207    * @return
208    */
209   private def build(config:CacheConfig) : Client = {
210     val serverList = config.getProperty(KeyServerList, None).get.asInstanceOf[String]
211     val concurrency = config.getProperty(KeyConcurrency, Some(3)).get.asInstanceOf[Int]
212     val timeout = config.getProperty(KeyTimeout, Some(1)).get.asInstanceOf[Int]
213 
214     KetamaClientBuilder()
215       .clientBuilder(ClientBuilder().hostConnectionLimit(concurrency).codec(Memcached()).failFast(false))
216       .failureAccrualParams(Int.MaxValue, Duration.Top)
217       .dest(serverList)
218       .build()
219       .asInstanceOf[PartitionedClient]
220   }

222   // todo check connection to memcache
223   override def checkHealth: Future[HealthComponent] = {
224     val p = Promise[HealthComponent]()
225     if (caches.isEmpty) {
226       p success HealthComponent(self.path.name, ComponentState.NORMAL, "Managing %d caches".format(0))
227     } else {
228       val future = Future.traverse(caches) {
229         case (k, v) => fromTwitter(v.checkHealth())
230       }.mapTo[Seq[CacheStatus]]
231       future onComplete {
232         case Success(result) =>
233           val comp = HealthComponent(self.path.name, ComponentState.NORMAL, "Managing %d caches".format(result.size))
234           result.zipWithIndex.foreach { case (status, i) =>
235             status match {
236               case CacheStatus(true, info) => comp.addComponent(HealthComponent(self.path.name + " Cache Connection " + i, ComponentState.NORMAL, info))
237               case CacheStatus(false, info) => comp.addComponent(HealthComponent(self.path.name + " Cache Connection " + i, ComponentState.CRITICAL, info))
238             }
239           }
240           p success comp
241         case Failure(f) => p success HealthComponent(self.path.name, ComponentState.CRITICAL, f.getMessage)
242       }
243     }
244     p.future
245   }
