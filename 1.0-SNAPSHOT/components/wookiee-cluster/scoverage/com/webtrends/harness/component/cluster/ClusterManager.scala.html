27 /**
28  * Important to note that if you use clustering you must not include the Zookeeper component
29  * as Clustering will start it up independently, and zookeeper config must be within the clustering config
30  */
31 class ClusterManager(name:String) extends Component(name)
32     with Clustering
33     with Zookeeper
34     with Messaging {
35 
36   private val akkaProvider = context.system.settings.config.getString("akka.actor.provider")
37   implicit val clusterSettings = ClusterSettings(ConfigUtil.prepareSubConfig(config, name), akkaProvider)
38   private val zookeeperConfig = ConfigUtil.prepareSubConfig(config, "wookiee-zookeeper")
39   implicit val zookeeperSettings = ZookeeperSettings(zookeeperConfig)
40 
41   override protected def defaultChildName: Option[String] = Some(Messaging.MessagingName)
42 
43   override def start = {
44     startZookeeper(true)
45     startClustering
46     startMessaging
47     super.start
48   }
49 
50   override def stop = {
51     super.stop
52     stopClustering
53   }
54 }
