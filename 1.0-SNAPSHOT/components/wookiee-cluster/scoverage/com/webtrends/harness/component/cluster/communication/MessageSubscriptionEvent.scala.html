28 object MessageSubscriptionEvent {
29 
30   /**
31    * Marker interface for message domain events.
32    */
33   sealed trait MessageSubscriptionEvent {
34     def topic: String
35 
36     def ref: ActorRef
37   }
38 
39   /**
40    * Subscription registration has been added. Published when a change has occured
41    * in the subscription information. This occurs from either internal subscribe/unsubscribe
42    * events or when an external system has changed.
43    */
44   @SerialVersionUID(1L) case class SubscriptionAddedEvent(val topic: String, ref: ActorRef) extends MessageSubscriptionEvent
45 
46   /**
47    * Subscription registration has been remove. Published when a change has occured
48    * in the subscription information. This occurs from either internal subscribe/unsubscribe
49    * events or when an external system has changed.
50    */
51   @SerialVersionUID(1L) case class SubscriptionRemovedEvent(val topic: String, ref: ActorRef) extends MessageSubscriptionEvent
52 
53   private[harness] object Internal {

55     @SerialVersionUID(1L) case class RegisterSubscriptionEvent(registrar: ActorRef, to: Class[_])
56 
57     @SerialVersionUID(1L) case class UnregisterSubscriptionEvent(registrar: ActorRef, to: Class[_])
58 
59   }
60 
61 }

63 trait MessageSubscriptionEventAdapter {

68   lazy val msgEventService = MessageService()

70   /**
71    * Register for subscription events. This is not used for maintaining

73    * @param registrar the actor that is to receive the events
74    * @param to the class to register for
75    */
76   def register(registrar: ActorRef, to: Class[_ <: MessageSubscriptionEvent]): Unit = msgEventService.register(registrar, to)
