40 class MessagingTopicActor(selfAddress: Address, trashInterval: FiniteDuration, seed: Set[Subscription]) extends Actor
41 with ActorLoggingAdapter {
42 
43   import context.dispatcher
44 
45   // The name of this actor is actually the topic
46   val topic = URLDecoder.decode(self.path.name, "utf-8")
47 
48   // Setup a recurring task that checks to see if we have no more subscribers.
49   // This helps to prevent the retention of resources that are not needed.
50   case object Sweep
51 
52   val sweepInterval: FiniteDuration = trashInterval / 2
53   val sweepTask = context.system.scheduler.schedule(sweepInterval, sweepInterval, self, Sweep)
54   var trashDeadline: Option[Deadline] = None
55 
56   // The list of subscribers
57   var registry = Map.empty[ActorRef, Boolean]
58 
59   // Establish the routing logic
60   val routingLogic = RoundRobinRoutingLogic()
61 
62   override def preStart: Unit = {
63     // Take the seed subscriptions and set them up
64     updateSubscriptions(seed)
65   }
66 
67   override def postStop(): Unit = {
68     sweepTask.cancel()
69     super.postStop()
70   }
71 
72   def receive = {
73     case UpdateSubscriptions(subscriptions) => updateSubscriptions(subscriptions)
74     case message@Subscribe(_, ref, localOnly) => subscribe(ref, localOnly)
75     case message@Unsubscribe(_, ref) => unsubscribe(ref, false)
76     case Terminated(ref) => unsubscribe(ref, true)
77     case Publish(_, message) => publish(message)
78     case Send(_, message) => send(message)
79     case Sweep if trashDeadline.isDefined && trashDeadline.get.isOverdue =>
80       log.debug("The actor [{}] is stopping because there are no more subscriptions and it was scheduled for deletion", self.path)
81       context stop self
82 
83   }
84 
85   /**
86    * Bulk update the subscriptions
87    * @param subscriptions
88    */
89   private def updateSubscriptions(subscriptions: Set[Subscription]): Unit = {
90     log.info("The actor [{}] is updating it's subscriptions: {}", self.path, subscriptions.map(_.subscriber.path).mkString(","))
91 
92     // Unwatch all of the entries that are not in the passed in set
93     val removals = registry.map(_._1).toSet &~ subscriptions.map(_.subscriber)
94     removals foreach { sub =>
95       context unwatch sub
96     }
97 
98     // Copy the subscriptions
99     registry = subscriptions.map(e => (e.subscriber -> e.localOnly)).toMap
100     if (registry.isEmpty) {
101       // If we have no more subscriptions then schedule the removal of this actor
102       trashDeadline = Some(Deadline.now + trashInterval)
103     }
104     else {
105       trashDeadline = None
106       registry foreach { sub =>
107         context watch sub._1
108       }
109     }
110   }
111 
112   /**
113    * Add the subscription
114    * @param ref the actor to subscribe
115    * @param localOnly
116    */
117   private def subscribe(ref: ActorRef, localOnly: Boolean): Unit = {
118     if (!registry.contains(ref)) {
119       log.info("The actor {} is subscribing to the topic {}", ref.path.toString, topic)
120       context watch ref
121       registry += (ref -> localOnly)
122       trashDeadline = None
123     }
124   }
125 
126   /**
127    * Remove the subscription
128    * @param ref the actor to unsubscribe
129    * @param terminated is the unsubscribe due to a termination
130    */
131   private def unsubscribe(ref: ActorRef, terminated: Boolean): Unit = {
132     if (registry.contains(ref)) {
133       if (terminated) {
134         log.info("The actor {} is un-subscribed to the topic {} since it has been terminated", ref.path.toString, topic)
135       }
136       else {
137         log.info("The actor {} is un-subscribing to the topic {}", ref.path.toString, topic)
138       }
139 
140       context unwatch ref
141       registry -= ref
142       if (registry.isEmpty) {
143         // If we have no more subscriptions then schedule the removal of this actor
144         trashDeadline = Some(Deadline.now + trashInterval)
145       }
146     }
147   }
148 
149   /**
150    * Forward the message to all subscribers
151    * @param message
152    */
153   private def publish(message: Any): Unit = {
154     for {
155       subscription <- registry
156       // Only send the message to internal actors or external ones that are registered for external publishes
157       if (subscription._1.path.address.host.isEmpty || subscription._1.path.address == selfAddress || !subscription._2)
158     } {
159       log.debug("Publishing message for '{}' to {}", topic, subscription._1.path)
160       subscription._1 forward message
161     }
162   }
163 
164   private def send(message: Any): Unit = {
165     val routees = (for {
166       sub <- registry
167     } yield ActorRefRoutee(sub._1)).toVector
168 
169     if (routees.nonEmpty) {
170       // Forward the message through our router
171       log.debug("Routing the message")
172       Router(routingLogic, routees).route(message, sender())
173     }
174     else {
175       log.warn("There are no subscribers available for the topic {}", topic)
176     }
177   }
178 }
