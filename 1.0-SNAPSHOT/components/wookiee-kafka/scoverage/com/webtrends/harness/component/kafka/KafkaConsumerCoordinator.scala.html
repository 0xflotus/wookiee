87   def initial:Receive = healthReceive orElse configReceive orElse {
88     case NodeData(nodeId, data) =>
89       val assignments = Serialization.read(data)(formats, manifest[List[PartitionAssignment]])
90       processAssignmentEvent(assignments)
91 
92     // ZK data has changed
93     case ZookeeperChildEvent(event) =>
94       // Node registration change
95       event.getType match {
96         // Node data has changed. If the node is local, notify the node actor
97         case CHILD_UPDATED | CHILD_ADDED | CHILD_REMOVED =>
98           val nodeId = event.getData.getPath.split("/").last
99           if (hostname == nodeId) {
100             log.debug(s"Node data for [$nodeId] has been updated. Notifying it")
101             self ! NodeData(nodeId, new String(event.getData.getData, utf8))
102           }
103 
104         case _ =>
105         // To remove a compiler warning
106       }
107 
108     case PrepareForShutdown =>
109       //Stop all workers
110       stopAllWorkers
111       context.become(awaitingShutdown)
112 
113     case BroadcastToWorkers(msg) =>
114       workers withFilter(_._2.started) foreach(_._2.actor.forward(msg))
115   }

124   }

127   def stopAllWorkers = {
128     log.info(s"Stopping ${workers.size} workers")
129     workers foreach(_._2.stopWorker())
130   }

132   override def renewConfiguration() = {

136   }

138   // Will stop all actors that are no longer needed and start any that are now needed
139   def stopUnneededWorkers(assigns: List[PartitionAssignment]) {
140     val specs = assigns.map(_.assignmentName)
141     workers foreach { worker =>
142       if (specs.contains(worker._1)) {
143         worker._2.startWorker()
144       } else {
145         worker._2.stopWorker()
146         workerKafkaHealth.remove(worker._1)
147       }
148     }
149   }
150 
151   def createWorkerIfNeeded(assign: PartitionAssignment) {
152     val spec = assign.assignmentName
153     if (!workers.contains(spec)) {
154       log.info(s"Creating worker ${assign.host}, ${assign.topic}_${assign.partition}")
155       val ref = context.actorOf(Props(topicWorker, kafkaProxy, assign, offsetManager).withDispatcher("worker-dispatcher"), spec)
156       workers.put(spec, new WorkerRef(true, ref))
157     }
158   }
159 
160   def processAssignmentEvent(assigns: List[PartitionAssignment]) {
161     log.debug(s"Processing ${assigns.length} assignments.")
162     stopUnneededWorkers(assigns)
163     assigns foreach { assign => createWorkerIfNeeded(assign) }
164   }
165 
166   // Used to keep track of workers and their states, will start on construction if start=true
167   class WorkerRef(start: Boolean, ref: ActorRef) {
168     var started = start
169     val actor = ref
170     if (started) {
171       startWorker()
172     }
173 
174     override def toString = s"WorkerRef(started=$started, actor=$actor)"
175 
176     def stopWorker() {
177       ref ! PartitionConsumerWorker.Stop
178       started = false
179     }
180 
181     def startWorker() {
182       ref ! Start
183       started = true
184     }
185   }
