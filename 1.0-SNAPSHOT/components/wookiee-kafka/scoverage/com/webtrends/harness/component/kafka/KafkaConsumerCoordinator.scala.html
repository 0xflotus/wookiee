83   def initial:Receive = healthReceive orElse configReceive orElse {
84     case NodeData(nodeId, data) =>
85       val assignments = Serialization.read(data)(formats, manifest[List[PartitionAssignment]])
86       sourceMonitor foreach (_ ! HostList(assignments.map(_.host)))
87       processAssignmentEvent(assignments)
88 
89     // ZK data has changed
90     case ZookeeperChildEvent(event) =>
91       // Node registration change
92       event.getType match {
93         // Node data has changed. If the node is local, notify the node actor
94         case CHILD_UPDATED | CHILD_ADDED | CHILD_REMOVED =>
95           val nodeId = event.getData.getPath.split("/").last
96           if (hostname == nodeId) {
97             log.debug(s"Node data for [$nodeId] has been updated. Notifying it")
98             self ! NodeData(nodeId, new String(event.getData.getData, utf8))
99           }
100 
101         case _ =>
102         // To remove a compiler warning
103       }
104 
105     case PrepareForShutdown =>
106       //Stop all workers
107       stopAllWorkers()
108       context.become(awaitingShutdown)
109 
110     case BroadcastToWorkers(msg) =>
111       workers withFilter(_._2.started) foreach(_._2.actor.forward(msg))
112   }

116    * @return

118   def awaitingShutdown:Receive = healthReceive orElse configReceive orElse {
119     case msg:Any => //Just ignore all other messages
120       log.debug(s"Waiting for shutdown ignoring $msg")
121   }

124   def stopAllWorkers() = {
125     log.info(s"Stopping ${workers.size} workers")
126     workers foreach(_._2.stopWorker())
127   }
128 
129   override def renewConfiguration() = {
130     super.renewConfiguration()
131     kafkaProxy ! ConfigChange()
132     renewTopicAgeThresholds()
133   }

135   // Will stop all actors that are no longer needed and start any that are now needed
136   def stopUnneededWorkers(assigns: List[PartitionAssignment]) {
137     val specs = assigns.map(_.assignmentName)
138     workers foreach { worker =>
139       if (specs.contains(worker._1)) {
140         worker._2.startWorker()
141       } else {
142         worker._2.stopWorker()
143         workerKafkaHealth.remove(worker._1)
144       }
145     }
146   }
147 
148   def createWorkerIfNeeded(assign: PartitionAssignment) {
149     val spec = assign.assignmentName
150     if (!workers.contains(spec)) {
151       log.info(s"Creating worker ${assign.host}, ${assign.topic}_${assign.partition}")
152       val ref = context.actorOf(Props(topicWorker, kafkaProxy, assign, offsetManager).withDispatcher("worker-dispatcher"), spec)
153       workers.put(spec, new WorkerRef(true, ref))
154     }
155   }
156 
157   def processAssignmentEvent(assigns: List[PartitionAssignment]) {
158     log.debug(s"Processing ${assigns.length} assignments.")
159     stopUnneededWorkers(assigns)
160     assigns foreach { assign => createWorkerIfNeeded(assign) }
161   }
162 
163   // Used to keep track of workers and their states, will start on construction if start=true
164   class WorkerRef(start: Boolean, ref: ActorRef) {
165     var started = start
166     val actor = ref
167     if (started) {
168       startWorker()
169     }
170 
171     override def toString = s"WorkerRef(started=$started, actor=$actor)"
172 
173     def stopWorker() {
174       ref ! PartitionConsumerWorker.Stop
175       started = false
176     }
177 
178     def startWorker() {
179       ref ! Start
180       started = true
181     }
182   }
