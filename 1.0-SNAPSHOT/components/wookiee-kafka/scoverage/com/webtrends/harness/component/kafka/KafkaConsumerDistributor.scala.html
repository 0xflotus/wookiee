62 class KafkaConsumerDistributor(sourceProxy: ActorRef) extends Actor
63   with ActorLoggingAdapter
64   with ZookeeperAdapter
65   with ZookeeperEventAdapter
66   with KafkaSettings
67   with Stash {
68 
69   import KafkaConsumerDistributor._
70 
71   implicit val timeout = Timeout(5 seconds)
72 
73   //Registration zk path
74   val zkNodePath = s"${distributorPaths.nodePath}/$hostname"
75   val hcName = "consumer-distributor"
76   var lastOutOfSyncUpdate = 0L
77   var assignmentLeader: Option[ActorRef] = None
78 
79 
80   override def preStart() {
81     // Register for ZooKeeper State events. When registering we will be sent a message
82     // for the current state
83     register(self, ZookeeperStateEventRegistration(self))
84     register(self, ZookeeperChildEventRegistration(self, distributorPaths.nodePath))
85     register(self, ZookeeperLeaderEventRegistration(self, distributorPaths.leaderPath))
86 
87     self ! RegisterSelf
88   }
89 
90   def shutdown() {
91     unregisterNode()
92     unregister(self, ZookeeperStateEventRegistration(self))
93     unregister(self, ZookeeperChildEventRegistration(self, distributorPaths.nodePath))
94     unregister(self, ZookeeperLeaderEventRegistration(self, distributorPaths.leaderPath))
95   }
96 
97   def receive = initializing
98 
99   /**
100    * Initializing the Distributor, if we can't then
101    * we will schedule a message to keep trying
102    * @return
103    */
104   def initializing: Receive = {
105     case RegisterSelf => registerNode()
106 
107     case NodeRegistrationResult(result) =>
108       //If we can't register, we keep trying
109       result match {
110         case true =>
111           log.info(s"$hostname registered!, moving to registered state")
112           context.become(registered)
113           unstashAll()
114         case false =>
115           context.system.scheduler.scheduleOnce(3 seconds,
116             self,
117             RegisterSelf)
118       }
119 
120     case CheckHealth => sender() ! HealthComponent(hcName,
121                                                     ComponentState.DEGRADED,
122                                                     "Distributor has not registered with ZK yet")
123 
124     case PrepareForShutdown => shutdown()
125 
126     case msg => stash()
127   }
128 
129   /**
130    * Once we are registered, we can accept other messages
131    * @return
132    */
133   def registered: Receive = {
134       // Our ZK state has changed
135       case ZookeeperStateEvent(state) =>
136         state match {
137           // If our ZK state is flaky, we may need to reestablish the local node registrations
138           case conn @ (CONNECTED | RECONNECTED) =>
139             log.info(s"Moving back to initializing due to Zk connection State =${conn} ")
140             context.become(initializing)
141             self ! RegisterSelf
142           case _ =>
143             log.debug(s"Zk state event=${state}")
144             relinquishLeadership()
145         }
146 
147       // ZK data has changed
148       case ZookeeperChildEvent(event) =>
149         handleChildEvent(event)
150 
151       // A change in leadership has occurred
152       case ZookeeperLeadershipEvent(leader) =>
153         if (leader) takeLeadership() else relinquishLeadership()
154 
155       case GetRegisteredNodes =>
156         getRegisteredNodes(sender())
157 
158       case GetNodeAssignments =>
159         getNodeAssignments(sender())
160 
161       case GetAssignmentLeader => sender ! AssignmentLeaderResult(assignmentLeader)
162 
163       case PrepareForShutdown => shutdown()
164 
165       case CheckHealth =>
166         val msg = assignmentLeader.map(_ => "LEADER - ")
167                                   .getOrElse("") + "Distributor Registered with Zk"
168 
169         sender() ! HealthComponent(hcName, ComponentState.NORMAL, msg)
170 
171       case msg => log.debug(s"Unknown message $msg")
172   }
173 
174   def registerNode() = {
175     log.debug(s"Registering node: $hostname")
176     // When the service is restarted, the ephemeral node from the previous process may not have been deleted yet.
177     // Make sure it has been deleted before attempting to add the new one.
178     deleteNode(zkNodePath).onComplete {
179      case _ =>
180        log.debug(s"Node deleted, $zkNodePath!")
181        setData(zkNodePath, hostname.getBytes(utf8), create = true, ephemeral = true).onComplete {
182          case Success(data) =>
183             log.info(s"Successfully initialized node registration for node [$hostname]")
184 
185             self ! NodeRegistrationResult(true)
186           case Failure(fail) =>
187             log.error(s"Unable to initialize node registration for node [$hostname]. ${fail.getMessage()}")
188             self  ! NodeRegistrationResult(false)
189         }
190     }
191   }
192 
193   def unregisterNode() = {
194     deleteNode(zkNodePath).onComplete {
195       case Success(path) =>
196         log.info(s"Successfully removed node registration for node [$hostname]")
197 
198       case Failure(fail) =>
199         log.error(s"Unable to remove node registration for node [$hostname]. ${fail.getMessage()}")
200     }
201   }
202 
203   def getRegisteredNodes(requestingActor: ActorRef) = {
204     getChildren(s"${distributorPaths.nodePath}", includeData = false).onComplete {
205       case Success(nodes) =>
206         requestingActor ! RegisteredNodes(Some(nodes.map{node => node._1}.toList))
207       case Failure(fail) =>
208         requestingActor ! RegisteredNodes(None)
209         log.error(s"Not requesting node data update because " +
210           s"I am unable to get list of nodes at ${distributorPaths.nodePath}. ${fail.getMessage()}")
211     }
212   }
213 
214   def getNodeAssignments(requestingActor: ActorRef) = {
215     log.debug(s"Fetching assigments for ${distributorPaths.assignmentPath}")
216     getChildren(s"${distributorPaths.assignmentPath}", includeData = true).onComplete {
217       case Success(nodes) =>
218         requestingActor ! NodeAssignments(Some(nodes.map { node => node._1 -> new String(node._2.get, utf8) }.toMap))
219       case Failure(fail) =>
220         log.error(s"Unable to get list of nodes at $appRootPath. ${fail.getMessage()}")
221         requestingActor ! NodeAssignments(None)
222     }
223   }
224 
225   def takeLeadership() = {
226     assignmentLeader.fold({
227       log.info(s"Taking leadership of ${distributorPaths.leaderPath}")
228       val leaderActor = context.actorOf(AssignmentDistributorLeader.props(sourceProxy), "assignment-leader")
229       assignmentLeader = Some(leaderActor)
230       leaderActor ! RefreshNodeAssignments
231     })( _ => log.info(s"Asked to take leadership when I am already the leader"))
232   }
233 
234   def relinquishLeadership() = {
235     assignmentLeader.foreach { actor =>
236       log.info(s"Relinquishing leadership of ${distributorPaths.leaderPath}")
237       context.stop(_)
238     }
239     assignmentLeader = None
240   }
241 
242   def handleChildEvent(event: PathChildrenCacheEvent) = {
243     val eventPath = event.getData.getPath
244 
245     // Node registration change
246     if (eventPath.startsWith(distributorPaths.nodePath)) {
247       event.getType match {
248         case CHILD_ADDED | CHILD_REMOVED =>
249           assignmentLeader.foreach { leader =>
250             log.info("Node child registration changed")
251             val now = System.currentTimeMillis()
252             // Limit churn caused by other nodes restarting or being in a wonky state.
253             if (now > lastOutOfSyncUpdate + 30000) {
254               lastOutOfSyncUpdate = now
255               leader ! RefreshNodeAssignments
256             }
257           }
258 
259         case _ =>
260         // To remove a compiler warning
261       }
262     }
263   }
264 }
