34 /**
35  * This class manages the creation of both the KafkaConsumerCoordinator (if 'consumer' is configured)
36  * and the KafkaWriter (if 'producer' is configured). One can retrieve either of these actors by
37  * making a call to GetCoordinator or GetWriters
38  */
39 object KafkaManager {
40   // User this to retrieve the coordinator in charge of managing Workers
41   case object GetCoordinator
42   //Will return the distributor
43   case object GetDistributor
44   // Kafka producer started up if 'producer' is configured
45   var producer: Option[ActorRef] = None
46 }

48 class KafkaManager(name: String) extends Component(name) with KafkaSettings {
49   import KafkaManager._
50   import context.dispatcher
51   // Consumer coordinator started up if 'consumer' is configured
52   var coordinator: Option[ActorRef] = None
53 
54   // Consumer proxy, started up if 'consumer' is configured. Used to retrieve topic
55   // information and the data to consume
56   var consumerManager: Option[ActorRef] = None
57 
58   //Distributor actor will start up if a 'consumer' is configured
59   var distributor: Option[ActorRef] = None
60 
61   var consumerManagerHealth: Option[HealthComponent] = None
62 
63   override def start = {
64     if (kafkaConfig.hasPath("producer")) {
65       startProducer()
66     }
67     super.start
68   }
69 
70   override def receive = super.receive orElse configReceive orElse {
71     // Use this call to get the Kafka Consumer Coordinator, if configured
72     case GetCoordinator =>
73       sender ! coordinator
74 
75     // We are loading the coordinator here because it depends on zookeeper component being loaded up which
76     // will have occurred earlier at the ComponentStart phase
77     case SystemReady =>
78       if (kafkaConfig.hasPath("consumer")) {
79         startCoordinator()
80       }
81 
82     case GetDistributor =>
83       sender ! distributor
84 
85     case PrepareForShutdown =>
86       coordinator.foreach(_ ! PrepareForShutdown)
87       distributor.foreach(_ ! PrepareForShutdown)
88 
89     case h: HealthComponent =>
90       consumerManagerHealth = Some(h)
91   }
92 
93   def startProducer() {
94     log.info("Starting producer as wookiee-kafka config contained 'producer' config")
95     val writerCount = Try { kafkaConfig.getInt("producer-count") } getOrElse 10
96     producer = Some(context.system.actorOf(Props[KafkaWriter], "producer"))
97   }
98 
99   override def checkHealth: Future[HealthComponent] = {
100     val p = Promise[HealthComponent]()
101     val children = Seq(distributor, coordinator, producer)
102 
103     getHealth.onComplete {
104       case Success(s) =>
105         val healthFutures = children.flatten map { ref =>
106           (ref ? CheckHealth).mapTo[HealthComponent] recover {
107             case ex: Exception => HealthComponent(ref.path.name, ComponentState.CRITICAL, s"Failure to get health of child component. ${ex.getMessage}")
108           }
109         }
110 
111         Future.sequence(healthFutures) onComplete {
112           case Failure(f) =>
113             log.debug(f, "Failed to retrieve health of children objects")
114             p success HealthComponent(s.name, ComponentState.CRITICAL, s"Failure to get health of child components. ${f.getMessage}")
115           case Success(healths) =>
116             healths foreach { it => s.addComponent(it) }
117             p success s
118         }
119       case Failure(f) =>
120         log.debug(f, "Failed to get health from component")
121         p success HealthComponent(self.path.toString, ComponentState.CRITICAL, f.getMessage)
122     }
123 
124     p.future
125   }
126 
127   override protected def getHealth: Future[HealthComponent] = {
128     val h = HealthComponent(self.path.toString, ComponentState.NORMAL, "Healthy")
129     consumerManagerHealth.foreach(h.addComponent)
130     Future.successful(h)
131   }
132 
133   /**
134    * Start up all necessary actors
135    */
136   def startCoordinator() {
137     if(coordinator.isEmpty) {
138       log.info(s"Starting coordinator class")
139       consumerManager = Some(context.actorOf(KafkaTopicManager.props(), "consumer-manager"))
140       coordinator = Some(context.actorOf(Props(leader, consumerManager.get), "consumer-coordinator"))
141       distributor = Some(context.actorOf(KafkaConsumerDistributor.props(consumerManager.get), "consumer-distributor"))
142     }
143   }
144 
145   override def renewConfiguration() {
146     log.info("Received config change message, checking hosts for changes...")
147     super.renewConfiguration()
148     if (coordinator.isDefined) {
149       coordinator.get ! ConfigChange()
150     }
151   }
152 
153   override def postStop() {
154     stop
155   }
156 
157   // Stops the coordinator, unregistering this node from zookeeper
158   override def stop {
159     context.children foreach { child ⇒
160       log.info(s"Stopping child [${child.path}]")
161       context.stop(child)
162     }
163     coordinator = None
164     consumerManager = None
165     distributor = None
166     producer = None
167   }
168 }
