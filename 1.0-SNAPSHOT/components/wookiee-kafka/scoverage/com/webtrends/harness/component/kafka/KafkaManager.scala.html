34 /**
35  * This class manages the creation of both the KafkaConsumerCoordinator (if 'consumer' is configured)
36  * and the KafkaWriter (if 'producer' is configured). One can retrieve either of these actors by
37  * making a call to GetCoordinator or GetWriters
38  */
39 object KafkaManager {
40   // User this to retrieve the coordinator in charge of managing Workers
41   case object GetCoordinator
42   //Will return the distributor
43   case object GetDistributor
44   // Kafka producer started up if 'producer' is configured
45   var producer: Option[ActorRef] = None
46 }

48 class KafkaManager(name: String) extends Component(name) with KafkaSettings {
49   import KafkaManager._
50 
51   import context.dispatcher
52   // Consumer coordinator started up if 'consumer' is configured
53   var coordinator: Option[ActorRef] = None
54 
55   // Consumer proxy, started up if 'consumer' is configured. Used to retrieve topic
56   // information and the data to consume
57   var consumerManager: Option[ActorRef] = None
58 
59   //Distributor actor will start up if a 'consumer' is configured
60   var distributor: Option[ActorRef] = None
61 
62   var consumerManagerHealth: Option[HealthComponent] = None
63 
64   override def receive = super.receive orElse configReceive orElse {
65     // Use this call to get the Kafka Consumer Coordinator, if configured
66     case GetCoordinator =>
67       sender ! coordinator
68 
69     case GetDistributor =>
70       sender ! distributor
71 
72     case SystemReady =>
73       if (kafkaConfig.hasPath("producer")) {
74         startProducer()
75       }
76       if (kafkaConfig.hasPath("consumer")) {
77         startCoordinator()
78       }
79     case PrepareForShutdown =>
80       coordinator.foreach(_ ! PrepareForShutdown)
81       distributor.foreach(_ ! PrepareForShutdown)
82 
83     case h: HealthComponent =>
84       consumerManagerHealth = Some(h)
85   }
86 
87   def startProducer() {
88     log.info("Starting producer as wookiee-kafka config contained 'producer' config")
89     producer = Some(context.actorOf(KafkaProducer.props(), "producer"))
90   }
91 
92   override def checkHealth: Future[HealthComponent] = {
93     val p = Promise[HealthComponent]()
94     val children = Seq(distributor, coordinator, producer)
95 
96     getHealth.onComplete {
97       case Success(s) =>
98         val healthFutures = children.flatten map { ref =>
99           (ref ? CheckHealth).mapTo[HealthComponent] recover {
100             case ex: Exception => HealthComponent(ref.path.name, ComponentState.CRITICAL, s"Failure to get health of child component. ${ex.getMessage}")
101           }
102         }
103 
104         Future.sequence(healthFutures) onComplete {
105           case Failure(f) =>
106             log.debug(f, "Failed to retrieve health of children objects")
107             p success HealthComponent(s.name, ComponentState.CRITICAL, s"Failure to get health of child components. ${f.getMessage}")
108           case Success(healths) =>
109             healths foreach { it => s.addComponent(it) }
110             p success s
111         }
112       case Failure(f) =>
113         log.debug(f, "Failed to get health from component")
114         p success HealthComponent(self.path.toString, ComponentState.CRITICAL, f.getMessage)
115     }
116 
117     p.future
118   }
119 
120   override protected def getHealth: Future[HealthComponent] = {
121     val h = HealthComponent(self.path.toString, ComponentState.NORMAL, "Healthy")
122     consumerManagerHealth.foreach(h.addComponent)
123     Future.successful(h)
124   }
125 
126   /**
127    * Start up all necessary actors
128    */
129   def startCoordinator() {
130     if(coordinator.isEmpty) {
131       log.info(s"Starting coordinator class")
132       consumerManager = Some(context.actorOf(KafkaConsumerManager.props(), "consumer-manager"))
133       coordinator = Some(context.actorOf(Props(leader, consumerManager.get), "consumer-coordinator"))
134       distributor = Some(context.actorOf(KafkaConsumerDistributor.props(consumerManager.get), "consumer-distributor"))
135     }
136   }
137 
138   override def renewConfiguration() {
139     log.info("Received config change message, checking hosts for changes...")
140     super.renewConfiguration()
141     if (coordinator.isDefined) {
142       coordinator.get ! ConfigChange()
143     }
144   }
145 
146   override def postStop() {
147     stop
148   }
149 
150   // Stops the coordinator, unregistering this node from zookeeper
151   override def stop {
152     context.children foreach { child ⇒
153       log.info(s"Stopping child [${child.path}]")
154       context.stop(child)
155     }
156     coordinator = None
157     consumerManager = None
158     distributor = None
159     producer = None
160   }
161 }
