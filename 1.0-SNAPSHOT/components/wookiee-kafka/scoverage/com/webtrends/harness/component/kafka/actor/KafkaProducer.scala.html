83   def receive:Receive = healthReceive orElse {
84     case EventToWrite(topic, event) => sendData(topic, Seq(event))
85 
86     case EventsToWrite(topic, events) => sendData(topic, events)
87 
88     case msg: EventsToWriteWithAck =>
89       sender() ! EventWriteAck(sendData(msg.topic, msg.events, msg.ackId))
90   }
91 
92   def sendData(topic: String, eventMessages: Seq[Array[Byte]], ackId: String = ""): Either[String, Throwable] ={
93     Try {
94       ensureProducer()
95       var bytes = 0L
96       val keyedMessages = eventMessages.withFilter(_.length > 0).map { it =>
97         bytes += it.length
98         keyedMessage(topic, it)
99       }
100 
101       // Only Send if we have >=1 valid messages
102       if(keyedMessages.nonEmpty) {
103         totalEvents.mark(keyedMessages.size)
104         totalBytesPerSecond.mark(bytes)
105         dataProducer.foreach(_.send(keyedMessages: _*))
106       }
107 
108       setHealth(ComponentState.NORMAL, "Messages sending successfully")
109       Left(ackId)
110     } recover {
111       case ex:Exception =>
112         log.error("Unable To write Event", ex)
113         setHealth(ComponentState.CRITICAL, "Last message failed to send")
114         resetProducer()
115         Right(ex)
116     } get
117   }
118 
119   private def keyedMessage(topic: String, value: Array[Byte]): KeyedMessage[String, Array[Byte]] = {
120     val keyedMessage =
121       if(evenPartitionDistribution)
122         new KeyedMessage[String, Array[Byte]](topic, partitionKey.toString, value)
123       else
124         new KeyedMessage[String, Array[Byte]](topic, value)
125 
126     partitionKey += 1
127     keyedMessage
128   }
129 
130   def ensureProducer() = {
131     if (dataProducer.isEmpty) {
132       resetProducer()
133     }
134   }
135 
136   def resetProducer() = {
137     if (dataProducer.isDefined) {
138       dataProducer.get.close()
139       dataProducer = None
140     }
141     dataProducer = Some(new Producer(new ProducerConfig(configToProps(kafkaConfig.getConfig("producer")))))
142     log.info("Kafka Producer Started")
143   }

145   def getKafkaNodes(config: Config): String = {
146     if (config.hasPath("metadata.broker.list")) {
147       return config.getString("metadata.broker.list")
148     }
149     if (!config.hasPath("broker.root")) {
150       throw new ExceptionInInitializerError("Must either set metadata.broker.list or broker.root in producer config")
151     }
152 
153     val brokerRoot = config.getString("broker.root")
154     val brokerPath = brokerRoot + "/brokers/ids"
155     val rawBrokerFuture = getChildren(brokerPath)
156     var brokers = ""
157     val rawBrokerData = Await.result(rawBrokerFuture, 10 seconds)
158     rawBrokerData foreach { data =>
159       val dataPath = brokerRoot + "/brokers/ids/" + data._1
160       val dataFut = Await.result(getData(dataPath), 10 seconds)
161 
162       val brokerData = Serialization.read(new String(dataFut, utf8))(formats, manifest[BrokerSpec])
163       if (!brokers.isEmpty) {
164         brokers = brokers + ","
165       }
166       brokers = brokers + brokerData.host + ":" + brokerData.port
167     }
168 
169     if (brokers.isEmpty) {
170       throw new ExceptionInInitializerError("No brokers found in ZK! Can't write to anywhere. Broker root: " + brokerRoot)
171     }
172     brokers
173   }

175   def configToProps(config: Config): Properties = {
176     val props: Properties = new Properties
177     import scala.collection.JavaConversions._
178     config.entrySet.foreach { entry =>
179       entry.getValue.valueType match {
180         case ConfigValueType.STRING =>
181           props.put(entry.getKey, config.getString(entry.getKey))
182         case ConfigValueType.NUMBER =>
183           props.put(entry.getKey, config.getNumber(entry.getKey).toString)
184         case _ =>
185         // Ignore other types
186       }
187     }
188     val brokerList = getKafkaNodes(kafkaConfig.getConfig("producer"))
189     props.put("key.serializer.class","kafka.serializer.StringEncoder")
190     props.put("metadata.broker.list", brokerList)
191     props
192   }
