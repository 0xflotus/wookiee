50 class KafkaTopicManager(sourceMonitor: Option[ActorRef]) extends Actor with KafkaSettings
51 with ActorLoggingAdapter with ZookeeperAdapter with ZookeeperEventAdapter {
52   import KafkaTopicManager._
53 
54   val actorName = "Kafka Topic Manager"
55 
56   // Holder of consumers connected to each kafka broker
57   val consumersByHost = new mutable.HashMap[String, SimpleConsumer]()
58   var downSources = Set[String]()
59 
60   context.parent ! HealthComponent(actorName, ComponentState.NORMAL, "Proxy has been started")
61 
62   def receive: Receive = configReceive orElse {
63     case TopicPartitionReq =>
64       sender ! TopicPartitionResp(getPartitionLeaders)
65 
66     case msg: DownSources => downSources = msg.sources
67   }
68 
69   def getPartitionLeaders: Set[PartitionAssignment] = {
70     val partitionsByTopic = new mutable.HashSet[PartitionAssignment]()
71     val topicMetaRequest = new TopicMetadataRequest(versionId = 1, correlationId = 0, clientId = clientId, topics = Seq())
72 
73     // Get our partition meta data for the configured topics
74     val processedClusters = new mutable.HashSet[String]()
75     val brokers = kafkaSources
76     sourceMonitor foreach(_ ! HostList(brokers.values.map(_.host).toList))
77     for (bro <- brokers.values
78          if !processedClusters.contains(bro.cluster)) {
79       try {
80         val consumer = consumersByHost.getOrElseUpdate(bro.host, new SimpleConsumer(bro.host, bro.port, 15000, bufferSize, clientId))
81         val topicsMetaResp = consumer.send(topicMetaRequest)
82         for ( topicMeta <- topicsMetaResp.topicsMetadata.filter { meta => topicMap.keys.toList.contains(meta.topic) };
83               partMeta <- topicMeta.partitionsMetadata )
84           yield {
85             partMeta.leader match {
86               case Some(broker) =>
87                 log.debug(s"Leader found for topic [${topicMeta.topic}:${partMeta.partitionId}]: ${broker.host}")
88                 partitionsByTopic.add(PartitionAssignment(topicMeta.topic, partMeta.partitionId, brokers(broker.host).cluster, broker.host))
89               case None =>
90                 log.error(s"No leader found for topic [${topicMeta.topic}:${partMeta.partitionId}]")
91             }
92             processedClusters.add(bro.cluster)
93           }
94       } catch {
95         case e: Throwable =>
96           log.error(s"Unable to get topic meta data from ${bro.host}, will retry soon", e)
97           consumersByHost.remove(bro.host).foreach(_.close())
98       }
99     }
100 
101     val unprocClusters = brokers.filter(it => !processedClusters.contains(it._2.cluster)).map(_._2).toSet
102     if (unprocClusters.nonEmpty) {
103       log.warn(s"Some brokers despondent: ${unprocClusters.map(_.cluster).mkString(",")}. Remaining brokers will start their workers.")
104       val okayAndDown = unprocClusters.partition(it => downSources.contains(it.host))
105       context.parent ! HealthComponent(actorName, if (okayAndDown._2.size > 0) ComponentState.DEGRADED else ComponentState.NORMAL,
106         s"Despondent Clusters: [${okayAndDown._2.map(_.cluster).mkString(",")}], Scheduled Downtime: [${okayAndDown._1.map(_.cluster).mkString(",")}]")
107     } else {
108       log.debug("Successfully processed brokers {}", brokers.toString())
109       context.parent ! HealthComponent(actorName, ComponentState.NORMAL, "Successfully fetched broker data")
110     }
111     partitionsByTopic.toSet[PartitionAssignment]
112   }
