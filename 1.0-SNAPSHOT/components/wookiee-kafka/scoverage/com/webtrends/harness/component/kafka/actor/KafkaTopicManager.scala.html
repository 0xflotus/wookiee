50 class KafkaTopicManager extends Actor with KafkaSettings
51 with ActorLoggingAdapter with ZookeeperAdapter with ZookeeperEventAdapter {
52   import KafkaTopicManager._
53 
54   val actorName = "Kafka Topic Manager"
55 
56   // Holder of consumers connected to each kafka broker
57   val consumersByHost = new mutable.HashMap[String, SimpleConsumer]()
58 
59   context.parent ! HealthComponent(actorName, ComponentState.NORMAL, "Proxy has been started")
60 
61   def receive: Receive = configReceive orElse {
62     case TopicPartitionReq =>
63       sender ! TopicPartitionResp(getPartitionLeaders)
64   }
65 
66   def getPartitionLeaders: Set[PartitionAssignment] = {
67     val partitionsByTopic = new mutable.HashSet[PartitionAssignment]()
68     val topicMetaRequest = new TopicMetadataRequest(versionId = 1, correlationId = 0, clientId = clientId, topics = Seq())
69 
70     // Get our partition meta data for the configured topics
71     val processedClusters = new mutable.HashSet[String]()
72     val brokers = kafkaSources
73     for (bro <- brokers.values
74          if !processedClusters.contains(bro.cluster)) {
75       try {
76         if (!consumersByHost.contains(bro.host)) {
77           consumersByHost.put(bro.host, new SimpleConsumer(bro.host, bro.port, 15000, bufferSize, clientId))
78         }
79         val consumer = consumersByHost(bro.host)
80         val topicsMetaResp = consumer.send(topicMetaRequest)
81         for ( topicMeta <- topicsMetaResp.topicsMetadata.filter { meta => topicMap.keys.toList.contains(meta.topic) };
82               partMeta <- topicMeta.partitionsMetadata )
83           yield {
84             partMeta.leader match {
85               case Some(broker) =>
86                 log.debug(s"Leader found for topic [${topicMeta.topic}:${partMeta.partitionId}]: ${broker.host}")
87                 partitionsByTopic.add(PartitionAssignment(topicMeta.topic, partMeta.partitionId, brokers(broker.host).cluster, broker.host))
88               case None =>
89                 log.error(s"No leader found for topic [${topicMeta.topic}:${partMeta.partitionId}]")
90             }
91             processedClusters.add(bro.cluster)
92           }
93       } catch {
94         case e: Throwable =>
95           log.error(s"Unable to get topic meta data from ${bro.host}, will retry soon", e)
96           consumersByHost.remove(bro.host).foreach(_.close())
97       }
98     }
99 
100     val unprocClusters = brokers.filter(it => !processedClusters.contains(it._2.cluster)).map(_._2.cluster).toSet
101     if (unprocClusters.nonEmpty) {
102       log.warn(s"Some brokers despondent: ${unprocClusters.mkString(",")}. Remaining brokers will start their workers.")
103       context.parent ! HealthComponent(actorName, ComponentState.DEGRADED, s"Brokers despondent: ${unprocClusters.mkString(",")}.")
104     } else {
105       log.debug("Successfully processed brokers {}", brokers.toString())
106       context.parent ! HealthComponent(actorName, ComponentState.NORMAL, "Successfully fetched broker data")
107     }
108     partitionsByTopic.toSet[PartitionAssignment]
109   }
110 }
