43 class KafkaWriter extends Actor
44   with ActorLoggingAdapter with KafkaWriterHealthCheck with KafkaSettings {
45   import KafkaWriter._
46 
47   lazy val totalEvents = Meter("total-events-per-second")
48   lazy val totalBytesPerSecond = Meter("total-bytes-per-second")
49 
50   val dataProducer = new KafkaProducer[String, Array[Byte]](KafkaUtil.configToProps(kafkaConfig.getConfig("producer")))
51   var partitionKey = 0L
52   val evenPartitionDistribution = Try { kafkaConfig.getBoolean("even-partition-distribution") } getOrElse true
53 
54   override def postStop() = {
55     log.info("Stopping Kafka Writer")
56     super.postStop()
57     dataProducer.close()
58   }
59 
60   def receive:Receive = healthReceive orElse {
61     case EventToWrite(topic, event) => sendData(topic, Seq(event))
62 
63     case EventsToWrite(topic, events) => sendData(topic, events)
64 
65     case msg: EventsToWriteWithAck =>
66       sender() ! EventWriteAck(sendData(msg.topic, msg.events, msg.ackId))
67   }
68 
69   def sendData(topic: String, eventMessages: Seq[Array[Byte]], ackId: String = ""): Either[String, Throwable] ={
70     Try {
71       var bytes = 0L
72       val keyedMessages = eventMessages.withFilter(_.length > 0).map { it =>
73         bytes += it.length
74         val message = keyedMessage(topic, it)
75 
76         // If the Kafka target is down completely, the Kafka client provides no feedback.
77         // The callback is never called and a get on the returned java Future will block indefinitely.
78         // Handling this by waiting on the
79         val productionFuture = dataProducer.send(message)
80         monitorSendHealth(productionFuture)
81 
82         message
83       }
84 
85       if (keyedMessages.nonEmpty) {
86         totalEvents.mark(keyedMessages.size)
87         totalBytesPerSecond.mark(bytes)
88       }
89 
90       if (currentHealth.isEmpty || currentHealth.get.state == ComponentState.NORMAL) {
91         Left(ackId)
92       }
93       else {
94         Right(new IllegalStateException("Producer is not healthy"))
95       }
96     } recover {
97       case ex:Exception =>
98         log.error("Unable To write Event", ex)
99         setHealth(ComponentState.CRITICAL, "Last message failed to send")
100         Right(ex)
101     } get
102   }
103 
104   private def keyedMessage(topic: String, value: Array[Byte]): ProducerRecord[String, Array[Byte]] = {
105     val keyedMessage =
106       if(evenPartitionDistribution)
107         new ProducerRecord[String, Array[Byte]](topic, partitionKey.toString, value)
108       else
109         new ProducerRecord[String, Array[Byte]](topic, value)
110 
111     partitionKey += 1
112     keyedMessage
113   }
114 }
