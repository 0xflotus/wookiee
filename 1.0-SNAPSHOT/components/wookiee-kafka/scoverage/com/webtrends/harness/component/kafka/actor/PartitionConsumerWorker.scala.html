74 class PartitionConsumerWorker(kafkaProxy: ActorRef, assign: PartitionAssignment, offsetManager: ActorRef)

80   val topic = assign.topic

83   val name = assign.assignmentName
84   val lockPath = s"$appRootPath/locks/${assign.cluster}_${topic}_$partition"

86   // Harness 3.0 does not provide distributed lock support. Need to pull the curator instance and create our own

91   // Add scheduled jobs to this list, they will be cancelled on Stop

94   // The acknowledged offset
95   protected var ackedOffset = 0L
96 
97   // OVERRIDE to set custom offset writing
98   def formatAckedOffset() = ackedOffset.toString

100   // Last offset that was sent over to kafka proxy
101   protected var lastSentToProxyOffset = 0L

106   // Each partition worker needs to lock the partition it is working on to prevent data duplication.
107   // We need to ensure that the initial node completes before the new node reads the offsets out of ZK
108   protected val lock = new InterProcessSemaphoreV2(curator, lockPath, 1)

110   protected var consumer: Option[SimpleConsumer] = None

112   /**
113    * Fetch Mode - Defaults to automatic
114    * @return
115    **/
116   def fetchMode():ConsumptionFetchMode = AUTOMATIC

118   /**
119    * Override method to set functionality when consuming, to manage offsets be sure to
120    * set ackedOffset when account has executed its desired data processing successfully.
121    * @param messageResponse message received by worker on its assigned partition
122    */
123   def onReceive(messageResponse: MessageResponse) = {
124     ackedOffset = messageResponse.nextOffsetOfSet
125   }

128   // A worker can be in one of 4 standard states:
129   // 1. Stopped  -  The worker has been created and told to process data, but it is not yet ready to read from kafka.
130   //                If told to start, the worker will obtain the distributed lock for this partition,
131   //                and read its initial state out of ZK
132   // 2. Starting -  The worker has been told to start and has acquired a lock, it will not try to read its initial
133   //                state from ZK and start consuming if successful
134   // 3. Consuming - The worker has fully started up and is reading data from kafka
135   startWith(Stopped, Unlocked)
136 
137   // These messages should always be responded to in all states
138   whenUnhandled {
139     case Event(Stop, aq: Acquired) =>
140       goto(Stopped) using Unlocked
141 
142     case Event(Stop, Unlocked) =>
143       stay()
144 
145     case Event(Start, aq: Acquired) =>
146       stay()
147   }
148 
149   onTransition {
150     case (Consuming | Starting) -> Stopped =>
151       log.info(s"$name: Transitioning state to Stopped")
152       stopWorker()
153 
154     case Stopped -> Starting =>
155       log.debug(s"$name: Transitioning state to Starting")
156       offsetManager ! GetOffsetData(partitionName(assign))
157 
158     case Starting -> Consuming =>
159       consumer = Some(new SimpleConsumer(host, Try { kafkaSources(host).port } getOrElse 9092, 15000, bufferSize, clientId))
160       context.parent ! KafkaHealthState(name, healthy = true, s"Worker started", topic)
161       fetchRequest(ackedOffset)
162       scheduler = scheduler :+ context.system.scheduler.schedule(zkCommitRate milliseconds,
163         zkCommitRate milliseconds, self, CommitOffset)
164       log.info(s"$name: We are transitioned to Consuming")
165   }
166 
167   when(Stopped) {
168     case Event(Start, Unlocked) =>
169       acquireLock() match {
170         case Some(aq) => goto(Starting) using aq
171         case None => stay()
172       }
173 
174     case Event(CommitOffset, Unlocked) =>
175       stay()
176 
177     case Event(msg: OffsetDataResponse, Unlocked) =>
178       stay()
179 
180     case Event(msg: FetchRes, Unlocked) =>
181       stay()
182   }
183 
184   when(Starting, stateTimeout = offsetGetExpiration) {
185     case Event(msg: OffsetDataResponse, aq: Acquired) =>
186       msg.data match {
187         case Left(data) =>
188           processOffsets(data)
189         case Right(ex) =>
190           log.error("Unable to load offset state")
191           goto(Stopped) using Unlocked
192       }
193 
194     case Event(StateTimeout, aq: Acquired) =>
195       goto(Stopped) using Unlocked
196   }
197 
198   when(Consuming) {
199     case Event(msg: FetchRes, aq: Acquired) =>
200       consume(msg)
201       stay()
202 
203     case Event(msg: FetchErrorBadOffset, aq: Acquired) =>
204       log.warning(s"$name: Failed to fetch events from kafka. " +
205         s"Offset out of range. Using offset ${msg.nextAvailableOffset}")
206       fetchRequest(msg.nextAvailableOffset)
207       stay()
208 
209     case Event(msg: FetchErrorPartitionNotAvailable, aq: Acquired) =>
210       context.parent ! KafkaHealthState(name, healthy = false, s"Failed to fetch events from kafka. Partition not available", topic)
211       log.error(s"$name: Unable to fetch events from kafka. Partition not available")
212       goto(Stopped) using Unlocked
213 
214     case Event(CommitOffset, aq: Acquired) =>
215       storeOffsetAndUpdateHealth()
216       stay()
217 
218     case Event(msg: OffsetDataResponse, aq: Acquired) =>
219       msg.data match {
220         case Left(data) =>
221           lastSentToStorage = extractOffset(data)
222           stay()
223         case Right(ex) =>
224           log.error("Unable to save offset state")
225           goto(Stopped) using Unlocked
226       }
227   }

231   // Will only store if the ackedOffset has changed or force is true
232   protected def storeOffsetAndUpdateHealth(force: Boolean = false): Unit = {
233     if (force || (ackedOffset > 0 && lastSentToStorage != ackedOffset)) {
234       context.parent ! KafkaHealthState(name, healthy = true, s"Successfully fetched to $ackedOffset", topic)
235       offsetManager ! StoreOffsetData(partitionName(assign), OffsetData(formatAckedOffset().getBytes(utf8)))
236     }
237   }

239   override def postStop(): Unit = {
240     stopWorker()
241   }

243   // Override to process initial offsets differently
244   protected def processOffsets(offsetData: OffsetData): State = {
245     val offset = extractOffset(offsetData)
246 
247     log.debug(s"$name has initialized with offset $offset")
248     ackedOffset = offset
249     lastSentToProxyOffset = offset
250     lastSentToStorage = offset
251 
252     goto(Consuming)
253   }
254 
255   // Can be overridden to read custom offsets
256   protected def extractOffset(offsetData:OffsetData):Long = {
257     val data = offsetData.asString()
258     if(data.isEmpty) 0L else data.toLong
259   }

261   def stopWorker() = {
262     if(ackedOffset > 0) {
263       log.debug(s"Shutdown %s, storage offset=%d, proxy offset=%d, acked offset=%d"
264         .format(name, lastSentToStorage, lastSentToProxyOffset, ackedOffset))
265 
266       offsetManager ! StoreOffsetData(partitionName(assign), OffsetData(formatAckedOffset().getBytes(utf8)))
267     }
268     scheduler.foreach(_.cancel())
269     scheduler = List()
270 
271     stateData match {
272       case aq: Acquired =>
273         lock.returnLease(aq.lease)
274         log.debug(s"$name: Released lock")
275       case _ => // No lock
276     }
277     consumer.foreach(_.close())
278     consumer = None
279     context.parent ! KafkaHealthState(name, healthy = true, "", "")
280   }

282   def acquireLock(): Option[Acquired] = {
283     try {
284       val acquiredLease = lock.acquire(5, TimeUnit.SECONDS) match {
285         case null => throw new InterruptedException("Could not acquire lock before timeout")
286         case l: Lease => l
287       }
288       log.debug(s"$name: Lock acquired")
289       Some(Acquired(acquiredLease))
290     } catch {
291       case e: Throwable =>
292         log.error(s"$name: Unable to obtain lock: ${e.getMessage} ")
293         None
294     }
295   }

297   /**
298    * Takes the list of messages receive in the last fetch and sends them to
299    * the main worker method account(...) which should be overridden with the actual
300    * work we want to do.
301    * @param fetchRes Contains our next offset and the list of messages we just got
302    * @return Returns a request for more messages
303    */
304   def consume(fetchRes: FetchRes) = {
305     lastSentToProxyOffset = fetchRes.nextOffset
306     onReceive(MessageResponse(fetchRes.messages, fetchRes.nextOffset))
307 
308     // After consumption of messages automatically fetch the next
309     if(fetchMode() == AUTOMATIC) {
310       fetchRequest (fetchRes.nextOffset)
311     }
312   }
313 
314   /**
315    * Fetch data from kafka
316    */
317   protected def fetchRequest(nextOffset: Long) = {
318     try {
319       self ! fetchTopicPart(topic, partition, nextOffset, clientId)
320     } catch {
321       case ex: OffsetOutOfRangeException =>
322         self ! FetchErrorBadOffset(KafkaUtil.getDesiredAvailableOffset(consumer.get, topic, partition, nextOffset, clientId))
323 
324       case ex: InvalidMessageSizeException =>
325         log.error(s"Failed to fetch message for $topic:$partition from $host because message was too big for buffer, you need to increase the buffer size!")
326         self ! FetchErrorPartitionNotAvailable(nextOffset)
327 
328       // Catch all. Respond with a generic error message that will result in the workers waiting before retrying
329       // and kick off a refresh of our kafka info
330       case ex: Exception =>
331         log.error(s"Failed to fetch from [$host] [$topic] [$partition] [$nextOffset] [${ex.getClass.toString}] [${ex.getMessage}]")
332         self ! FetchErrorPartitionNotAvailable(nextOffset)
333     }
334   }
335 
336   def fetchTopicPart(topic: String, part: Int, offset: Long, clientId: String): FetchRes = {
337     val fetch = new FetchRequestBuilder()
338       .clientId(clientId)
339       .addFetch(topic, part, offset, 150000)
340       .build()
341 
342     val fetchResponse = consumer.get.fetch(fetch)
343     if (fetchResponse.hasError) {
344       ErrorMapping.maybeThrowException(fetchResponse.errorCode(topic, part))
345     }
346 
347     val msgSet = fetchResponse.messageSet(topic, part)
348     val msgSeq = KafkaMessageSet(msgSet, offset) // This is needed since if Kafka is compressing the messages, the fetch request will return an entire compressed block even if the requested offset isn't the beginning of the compressed block. Thus a message we saw previously may be returned again.
349     FetchRes(msgSeq, if (msgSet.size > 0) msgSet.last.nextOffset else offset)
350   }
