45  * Automatic - Automatically ask Kafka for more messages

74 class PartitionConsumerWorker(kafkaProxy: ActorRef, assign: PartitionAssignment, offsetManager: ActorRef)
75   extends LoggingFSM[State, Lock] with Actor with ZookeeperAdapter with KafkaSettings {
76   implicit val timeout = Timeout(5000, TimeUnit.MILLISECONDS)

80   val topic = assign.topic
81   val partition = assign.partition
82   val host = assign.host
83   val name = assign.assignmentName
84   val lockPath = s"$appRootPath/locks/${assign.cluster}_${topic}_$partition"

86   // Harness 3.0 does not provide distributed lock support. Need to pull the curator instance and create our own
87   val curator = Curator(zkConf.get).client
88 
89   val zkCommitRate = kafkaConfig.getLong("zk-offset-commit-rate-millis")

91   // Add scheduled jobs to this list, they will be cancelled on Stop
92   var scheduler: List[Cancellable] = List.empty
93 
94   // The acknowledged offset
95   protected var ackedOffset = 0L
96 
97   // OVERRIDE to set custom offset writing
98   def formatAckedOffset() = ackedOffset.toString
99 
100   // Last offset that was sent over to kafka proxy
101   protected var lastSentToProxyOffset = 0L
102 
103   // Last offset stored
104   protected var lastSentToStorage = 0L
105 
106   // Each partition worker needs to lock the partition it is working on to prevent data duplication.
107   // We need to ensure that the initial node completes before the new node reads the offsets out of ZK
108   protected val lock = new InterProcessSemaphoreV2(curator, lockPath, 1)
109 
110   protected var consumer: Option[SimpleConsumer] = None
111 
112   /**
113    * Fetch Mode - Defaults to automatic
114    * @return
115    **/
116   def fetchMode():ConsumptionFetchMode = AUTOMATIC
117 
118   /**
119    * Override method to set functionality when consuming, to manage offsets be sure to
120    * set ackedOffset when account has executed its desired data processing successfully.
121    * @param messageResponse message received by worker on its assigned partition
122    */
123   def onReceive(messageResponse: MessageResponse) = {
124     ackedOffset = messageResponse.nextOffsetOfSet
125   }

128   // A worker can be in one of 4 standard states:
129   // 1. Stopped  -  The worker has been created and told to process data, but it is not yet ready to read from kafka.
130   //                If told to start, the worker will obtain the distributed lock for this partition,
131   //                and read its initial state out of ZK
132   // 2. Starting -  The worker has been told to start and has acquired a lock, it will not try to read its initial
133   //                state from ZK and start consuming if successful
134   // 3. Consuming - The worker has fully started up and is reading data from kafka
135   startWith(Stopped, Unlocked)
136 
137   // These messages should always be responded to in all states
138   whenUnhandled {
139     case Event(Stop, aq: Acquired) =>
140       goto(Stopped) using Unlocked
141 
142     case Event(Stop, Unlocked) =>
143       stay()
144 
145     case Event(Start, aq: Acquired) =>
146       stay()
147   }
148 
149   onTransition {
150     case Starting -> Stopped =>
151       log.info(s"$name: Transitioning from Starting to Stopped")
152       stopWorker(write = false)
153 
154     case Consuming -> Stopped =>
155       log.info(s"$name: Transitioning from Consuming to Stopped")
156       stopWorker()
157 
158     case Stopped -> Starting =>
159       log.info(s"$name: Transitioning from Stopped to Starting")
160       offsetManager ! GetOffsetData(partitionName(assign))
161 
162     case Starting -> Consuming =>
163       consumer = Some(new SimpleConsumer(host, Try { kafkaSources(host).port } getOrElse 9092, 15000, bufferSize, clientId))
164       context.parent ! KafkaHealthState(name, healthy = true, s"Worker started", topic)
165       fetchRequest(ackedOffset)
166       scheduler = scheduler :+ context.system.scheduler.schedule(zkCommitRate milliseconds,
167         zkCommitRate milliseconds, self, CommitOffset)
168       log.info(s"$name: We are transitioned to Consuming")
169   }
170 
171   when(Stopped) {
172     case Event(Start, Unlocked) =>
173       acquireLock() match {
174         case Some(aq) => goto(Starting) using aq
175         case None => stay()
176       }
177 
178     case Event(CommitOffset, Unlocked) =>
179       stay()
180 
181     case Event(msg: OffsetDataResponse, Unlocked) =>
182       stay()
183 
184     case Event(msg: FetchRes, Unlocked) =>
185       stay()
186   }
187 
188   when(Starting, stateTimeout = offsetGetExpiration) {
189     case Event(msg: OffsetDataResponse, aq: Acquired) =>
190       msg.data match {
191         case Left(data) =>
192           processOffsets(data)
193         case Right(ex) =>
194           log.error("Unable to load offset state")
195           goto(Stopped) using Unlocked
196       }
197 
198     case Event(StateTimeout, aq: Acquired) =>
199       goto(Stopped) using Unlocked
200   }
201 
202   when(Consuming) {
203     case Event(msg: FetchRes, aq: Acquired) =>
204       consume(msg)
205       stay()
206 
207     case Event(msg: FetchErrorBadOffset, aq: Acquired) =>
208       log.warning(s"$name: Failed to fetch events from kafka. " +
209         s"Offset out of range. Using offset ${msg.nextAvailableOffset}")
210       fetchRequest(msg.nextAvailableOffset)
211       stay()
212 
213     case Event(msg: FetchErrorPartitionNotAvailable, aq: Acquired) =>
214       context.parent ! KafkaHealthState(name, healthy = false, s"Failed to fetch events from kafka. Partition not available", topic)
215       log.error(s"$name: Unable to fetch events from kafka. Partition not available")
216       goto(Stopped) using Unlocked
217 
218     case Event(CommitOffset, aq: Acquired) =>
219       storeOffsetAndUpdateHealth()
220       stay()
221 
222     case Event(msg: OffsetDataResponse, aq: Acquired) =>
223       msg.data match {
224         case Left(data) =>
225           lastSentToStorage = extractOffset(data)
226           stay()
227         case Right(ex) =>
228           log.error("Unable to save offset state")
229           goto(Stopped) using Unlocked
230       }
231   }

236   override def postStop(): Unit = {
237     stopWorker(stateName == Consuming)
238   }

240   // Will only store if the ackedOffset has changed or force is true
241   protected def storeOffsetAndUpdateHealth(force: Boolean = false): Unit = {
242     if (force || (ackedOffset > 0 && lastSentToStorage != ackedOffset)) {
243       context.parent ! KafkaHealthState(name, healthy = true, s"Successfully fetched to $ackedOffset", topic)
244       offsetManager ! StoreOffsetData(partitionName(assign), OffsetData(formatAckedOffset().getBytes(utf8)))
245     }
246   }
247 
248   // Override to process initial offsets differently
249   protected def processOffsets(offsetData: OffsetData): State = {
250     val offset = extractOffset(offsetData)
251 
252     log.debug(s"$name has initialized with offset $offset")
253     ackedOffset = offset
254     lastSentToProxyOffset = offset
255     lastSentToStorage = offset
256 
257     goto(Consuming)
258   }
259 
260   // Can be overridden to read custom offsets
261   protected def extractOffset(offsetData:OffsetData):Long = {
262     val data = offsetData.asString()
263     if(data.isEmpty) 0L else data.toLong
264   }
265 
266   def stopWorker(write: Boolean = true) = {
267     if (ackedOffset > 0 && write) {
268       log.debug(s"Shutdown %s, storage offset=%d, proxy offset=%d, acked offset=%d"
269         .format(name, lastSentToStorage, lastSentToProxyOffset, ackedOffset))
270 
271       offsetManager ! StoreOffsetData(partitionName(assign), OffsetData(formatAckedOffset().getBytes(utf8)))
272     }
273     scheduler.foreach(_.cancel())
274     scheduler = List()
275 
276     stateData match {
277       case aq: Acquired =>
278         lock.returnLease(aq.lease)
279         log.debug(s"$name: Released lock")
280       case _ => // No lock
281     }
282     consumer.foreach(_.close())
283     consumer = None
284     context.parent ! KafkaHealthState(name, healthy = true, "", "")
285     ackedOffset = 0L
286   }

288   def acquireLock(): Option[Acquired] = {
289     try {
290       val acquiredLease = lock.acquire(5, TimeUnit.SECONDS) match {
291         case null => throw new InterruptedException("Could not acquire lock before timeout")
292         case l: Lease => l
293       }
294       log.debug(s"$name: Lock acquired")
295       Some(Acquired(acquiredLease))
296     } catch {
297       case e: Throwable =>
298         log.error(s"$name: Unable to obtain lock: ${e.getMessage} ")
299         None
300     }
301   }
302 
303   /**
304    * Takes the list of messages receive in the last fetch and sends them to
305    * the main worker method account(...) which should be overridden with the actual
306    * work we want to do.
307    * @param fetchRes Contains our next offset and the list of messages we just got
308    * @return Returns a request for more messages
309    */
310   def consume(fetchRes: FetchRes) = {
311     lastSentToProxyOffset = fetchRes.nextOffset
312     onReceive(MessageResponse(fetchRes.messages, fetchRes.nextOffset))
313 
314     // After consumption of messages automatically fetch the next
315     if(fetchMode() == AUTOMATIC) {
316       fetchRequest (fetchRes.nextOffset)
317     }
318   }
319 
320   /**
321    * Fetch data from kafka
322    */
323   protected def fetchRequest(nextOffset: Long) = {
324     try {
325       self ! fetchTopicPart(topic, partition, nextOffset, clientId)
326     } catch {
327       case ex: OffsetOutOfRangeException =>
328         self ! FetchErrorBadOffset(KafkaUtil.getDesiredAvailableOffset(consumer.get, topic, partition, nextOffset, clientId))
329 
330       case ex: InvalidMessageSizeException =>
331         log.error(s"Failed to fetch message for $topic:$partition from $host because message was too big for buffer, you need to increase the buffer size!")
332         self ! FetchErrorPartitionNotAvailable(nextOffset)
333 
334       // Catch all. Respond with a generic error message that will result in the workers waiting before retrying
335       // and kick off a refresh of our kafka info
336       case ex: Exception =>
337         log.error(s"Failed to fetch from [$host] [$topic] [$partition] [$nextOffset] [${ex.getClass.toString}] [${ex.getMessage}]")
338         self ! FetchErrorPartitionNotAvailable(nextOffset)
339     }
340   }
341 
342   def fetchTopicPart(topic: String, part: Int, offset: Long, clientId: String): FetchRes = {
343     val fetch = new FetchRequestBuilder()
344       .clientId(clientId)
345       .addFetch(topic, part, offset, 150000)
346       .build()
347 
348     val fetchResponse = consumer.get.fetch(fetch)
349     if (fetchResponse.hasError) {
350       ErrorMapping.maybeThrowException(fetchResponse.errorCode(topic, part))
351     }
352 
353     val msgSet = fetchResponse.messageSet(topic, part)
354     val msgSeq = KafkaMessageSet(msgSet, offset) // This is needed since if Kafka is compressing the messages, the fetch request will return an entire compressed block even if the requested offset isn't the beginning of the compressed block. Thus a message we saw previously may be returned again.
355     FetchRes(msgSeq, if (msgSet.size > 0) msgSet.last.nextOffset else offset)
356   }
