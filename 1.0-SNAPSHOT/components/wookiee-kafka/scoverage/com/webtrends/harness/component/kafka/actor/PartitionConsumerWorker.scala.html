77 class PartitionConsumerWorker(kafkaProxy: ActorRef, assign: PartitionAssignment, offsetManager: ActorRef)
78   extends LoggingFSM[State, Lock] with Actor with ZookeeperAdapter with KafkaSettings {

85   val topic = assign.topic
86   val partition = assign.partition
87   val host = assign.host
88   val cluster = assign.cluster
89   val name = assign.assignmentName
90   val consumerWait = Try { kafkaConfig.getLong("consumer-wait-millis") } getOrElse 20000L
91   val lockPath = s"$appRootPath/locks/${cluster}_${topic}_$partition"

93   // Harness 3.0 does not provide distributed lock support. Need to pull the curator instance and create our own
94   val curator = Curator(zkConf.get).client

96   val zkCommitRate = kafkaConfig.getLong("zk-offset-commit-rate-millis")

98   // Add scheduled jobs to this list, they will be cancelled on Stop
99   var scheduler: List[Cancellable] = List.empty
100 
101   // The acknowledged offset
102   protected var ackedOffset = 0L
103 
104   // OVERRIDE to set custom offset writing
105   def formatAckedOffset() = ackedOffset.toString
106 
107   // Last offset that was sent over to kafka proxy
108   protected var lastSentToProxyOffset = 0L
109 
110   // Last offset stored
111   protected var lastSentToStorage = 0L
112 
113   // Each partition worker needs to lock the partition it is working on to prevent data duplication.
114   // We need to ensure that the initial node completes before the new node reads the offsets out of ZK
115   protected val lock = new InterProcessSemaphoreV2(curator, lockPath, 1)
116 
117   protected var consumer: Option[SimpleConsumer] = None
118 
119   /**
120    * Fetch Mode - Defaults to automatic
121    * @return
122    **/
123   def fetchMode():ConsumptionFetchMode = AUTOMATIC
124 
125   /**
126    * Override method to set functionality when consuming, to manage offsets be sure to
127    * set ackedOffset when account has executed its desired data processing successfully.
128    * @param messageResponse message received by worker on its assigned partition
129    */
130   def onReceive(messageResponse: MessageResponse) = {
131     ackedOffset = messageResponse.nextOffsetOfSet
132   }

135   // A worker can be in one of 4 standard states:
136   // 1. Stopped  -  The worker has been created and told to process data, but it is not yet ready to read from kafka.
137   //                If told to start, the worker will obtain the distributed lock for this partition,
138   //                and read its initial state out of ZK
139   // 2. Starting -  The worker has been told to start and has acquired a lock, it will not try to read its initial
140   //                state from ZK and start consuming if successful
141   // 3. Consuming - The worker has fully started up and is reading data from kafka
142   startWith(Stopped, Unlocked)
143 
144   // These messages should always be responded to in all states
145   whenUnhandled {
146     case Event(Stop, aq: Acquired) =>
147       goto(Stopped) using Unlocked
148 
149     case Event(Stop, Unlocked) =>
150       stay()
151 
152     case Event(Start, aq: Acquired) =>
153       stay()
154   }
155 
156   onTransition {
157     case (Consuming | Starting) -> Stopped =>
158       log.info(s"$name: Transitioning state to Stopped")
159       stopWorker()
160 
161     case Stopped -> Starting =>
162       log.debug(s"$name: Transitioning state to Starting")
163       offsetManager ! GetOffsetData(partitionName(assign))
164 
165     case Starting -> Consuming =>
166       fetchConsumer()
167       if (consumer.isDefined) {
168         context.parent ! KafkaHealthState(name, healthy = true, s"Worker started", topic)
169         fetchRequest(ackedOffset)
170         scheduler = scheduler :+ context.system.scheduler.schedule(zkCommitRate milliseconds,
171           zkCommitRate milliseconds, self, CommitOffset)
172         log.info(s"$name: We are transitioned to Consuming")
173       }
174   }
175 
176   protected def fetchConsumer() {
177     log.debug(s"$name: Consumer closed, fetching new one")
178     Try { consumer = Await.result(kafkaProxy.ask(FetchConsumer(host))(Timeout(consumerWait, TimeUnit.MILLISECONDS)), consumerWait milliseconds) match {
179       case Some(con) => Some(con.asInstanceOf[SimpleConsumer])
180       case None =>
181         log.error(s"No consumer found for $host")
182         context.parent ! KafkaHealthState(name, healthy = false, s"Could not get consumer", topic)
183         self ! Stop
184         None
185     }} recover {
186       case ex: Exception =>
187         log.error(s"Consumer request for $host timed out, stopping and will retry")
188         consumer = None
189         self ! Stop
190     }
191   }
192 
193   when(Stopped) {
194     case Event(Start, Unlocked) =>
195       acquireLock() match {
196         case Some(aq) => goto(Starting) using aq
197         case None => stay()
198       }
199 
200     case Event(CommitOffset, Unlocked) =>
201       stay()
202 
203     case Event(msg: OffsetDataResponse, Unlocked) =>
204       stay()
205 
206     case Event(msg: FetchRes, Unlocked) =>
207       stay()
208   }
209 
210   when(Starting, stateTimeout = 5 seconds) {
211     case Event(msg: OffsetDataResponse, aq: Acquired) =>
212       msg.data match {
213         case Left(data) =>
214           processOffsets(data)
215         case Right(ex) =>
216           log.error("Unable to load offset state")
217           goto(Stopped) using Unlocked
218       }
219 
220     case Event(StateTimeout, aq: Acquired) =>
221       goto(Stopped) using Unlocked
222   }
223 
224   when(Consuming) {
225     case Event(msg: FetchRes, aq: Acquired) =>
226       consume(msg)
227       stay()
228 
229     case Event(msg: FetchErrorBadOffset, aq: Acquired) =>
230       log.warning(s"$name: Failed to fetch events from kafka. " +
231         s"Offset out of range. Using offset ${msg.nextAvailableOffset}")
232       fetchRequest(msg.nextAvailableOffset)
233       stay()
234 
235     case Event(msg: FetchErrorPartitionNotAvailable, aq: Acquired) =>
236       context.parent ! KafkaHealthState(name, healthy = false, s"Failed to fetch events from kafka. Partition not available", topic)
237       log.error(s"$name: Unable to fetch events from kafka. Partition not available")
238       goto(Stopped) using Unlocked
239 
240     case Event(CommitOffset, aq: Acquired) =>
241       storeOffsetAndUpdateHealth()
242       stay()
243 
244     case Event(msg: OffsetDataResponse, aq: Acquired) =>
245       msg.data match {
246         case Left(data) =>
247           lastSentToStorage = extractOffset(data)
248           stay()
249         case Right(ex) =>
250           log.error("Unable to save offset state")
251           goto(Stopped) using Unlocked
252       }
253   }

257   // Will only store if the ackedOffset has changed or force is true
258   protected def storeOffsetAndUpdateHealth(force: Boolean = false): Unit = {
259     if (force || (ackedOffset > 0 && lastSentToStorage != ackedOffset)) {
260       context.parent ! KafkaHealthState(name, healthy = true, s"Successfully fetched to $ackedOffset", topic)
261       offsetManager ! StoreOffsetData(partitionName(assign), OffsetData(formatAckedOffset().getBytes(utf8)))
262     }
263   }
264 
265   override def postStop(): Unit = {
266     stopWorker()
267   }
268 
269   // Override to process initial offsets differently
270   protected def processOffsets(offsetData: OffsetData): State = {
271     val offset = extractOffset(offsetData)
272 
273     log.debug(s"$name has initialized with offset $offset")
274     ackedOffset = offset
275     lastSentToProxyOffset = offset
276     lastSentToStorage = offset
277 
278     goto(Consuming)
279   }
280 
281   // Can be overridden to read custom offsets
282   protected def extractOffset(offsetData:OffsetData):Long = {
283     val data = offsetData.asString()
284     if(data.isEmpty) 0L else data.toLong
285   }
286 
287   def stopWorker() = {
288     if(ackedOffset > 0) {
289       log.debug(s"Shutdown %s, storage offset=%d, proxy offset=%d, acked offset=%d"
290         .format(name, lastSentToStorage, lastSentToProxyOffset, ackedOffset))
291 
292       offsetManager ! StoreOffsetData(partitionName(assign), OffsetData(formatAckedOffset().getBytes(utf8)))
293     }
294     scheduler.foreach(_.cancel())
295     scheduler = List()
296 
297     stateData match {
298       case aq: Acquired =>
299         lock.returnLease(aq.lease)
300         log.debug(s"$name: Released lock")
301       case _ => // No lock
302     }
303     context.parent ! KafkaHealthState(name, healthy = true, "", "")
304   }
305 
306   def acquireLock(): Option[Acquired] = {
307     try {
308       val acquiredLease = lock.acquire(5, TimeUnit.SECONDS) match {
309         case null => throw new InterruptedException("Could not acquire lock before timeout")
310         case l: Lease => l
311       }
312       log.debug(s"$name: Lock acquired")
313       Some(Acquired(acquiredLease))
314     } catch {
315       case e: Throwable =>
316         log.error(s"$name: Unable to obtain lock: ${e.getMessage} ")
317         None
318     }
319   }
320 
321   /**
322    * Takes the list of messages receive in the last fetch and sends them to
323    * the main worker method account(...) which should be overridden with the actual
324    * work we want to do.
325    * @param fetchRes Contains our next offset and the list of messages we just got
326    * @return Returns a request for more messages
327    */
328   def consume(fetchRes: FetchRes) = {
329     lastSentToProxyOffset = fetchRes.nextOffset
330     onReceive(MessageResponse(fetchRes.messages, fetchRes.nextOffset))
331 
332     // After consumption of messages automatically fetch the next
333     if(fetchMode() == AUTOMATIC) {
334       fetchRequest (fetchRes.nextOffset)
335     }
336   }
337 
338   /**
339    * Fetch data from kafka
340    */
341   protected def fetchRequest(nextOffset: Long) = {
342     try {
343       self ! fetchTopicPart(topic, partition, nextOffset, clientId)
344     } catch {
345       case ex: OffsetOutOfRangeException =>
346         self ! FetchErrorBadOffset(KafkaUtil.getDesiredAvailableOffset(consumer.get, topic, partition, nextOffset, clientId))
347 
348       case ex: InvalidMessageSizeException =>
349         log.error(s"Failed to fetch message for $topic:$partition from $host because message was too big for buffer, you need to increase the buffer size!")
350         self ! FetchErrorPartitionNotAvailable(nextOffset)
351 
352       // Catch all. Respond with a generic error message that will result in the workers waiting before retrying
353       // and kick off a refresh of our kafka info
354       case ex: Exception =>
355         log.error(s"Failed to fetch from [$host] [$topic] [$partition] [$nextOffset] [${ex.getClass.toString}] [${ex.getMessage}]")
356         self ! FetchErrorPartitionNotAvailable(nextOffset)
357     }
358   }
359 
360   def fetchTopicPart(topic: String, part: Int, offset: Long, clientId: String): FetchRes = {
361     val fetch = new FetchRequestBuilder()
362       .clientId(clientId)
363       .addFetch(topic, part, offset, 150000)
364       .build()
365 
366     val fetchResponse = consumer.get.fetch(fetch)
367     if (fetchResponse.hasError) {
368       ErrorMapping.maybeThrowException(fetchResponse.errorCode(topic, part))
369     }
370 
371     val msgSet = fetchResponse.messageSet(topic, part)
372     val msgSeq = KafkaMessageSet(msgSet, offset) // This is needed since if Kafka is compressing the messages, the fetch request will return an entire compressed block even if the requested offset isn't the beginning of the compressed block. Thus a message we saw previously may be returned again.
373     FetchRes(msgSeq, if (msgSet.size > 0) msgSet.last.nextOffset else offset)
374   }
