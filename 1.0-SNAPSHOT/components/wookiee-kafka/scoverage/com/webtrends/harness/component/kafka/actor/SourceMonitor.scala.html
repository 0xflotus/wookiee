14 /**
15  * Allows one to setup monitoring on any source nodes to watch for scheduled downtime and put workers into a NORMAL state
16  * instead of DEGRADED when it is expected for them to go down
17  */
18 case class HostList(hosts: List[String])
19 
20 class SourceMonitor extends HActor with KafkaSettings {
21   val nagiosServer = Try { kafkaConfig.getString("nagios-host") } getOrElse "http://nagios:8080/host/"
22 
23   override def receive: Receive = {
24     case HostList(hosts) => sender ! checkHosts(hosts)
25   }
26 
27   def checkHosts(assignments: List[String]): DownSources = {
28     val checkedHosts = mutable.HashSet[String]()
29     val downHosts = mutable.HashSet[String]()
30     assignments foreach { assignment =>
31       if (!checkedHosts.contains(assignment)) {
32         checkedHosts.add(assignment)
33         if (!hostIsGreen(assignment.split('.')(0))) downHosts.add(assignment)
34       }
35     }
36     DownSources(downHosts.toSet)
37   }
38 
39   def hostIsGreen(host: String): Boolean = {
40     try {
41       val response = getRestContent(s"$nagiosServer$host")
42       response match {
43         case Some(resp) =>
44           val parsed = JSON.parseRaw(resp).get.asInstanceOf[JSONObject].obj("content").asInstanceOf[JSONObject]
45           parsed.obj("scheduled_downtime_depth") == "0"
46         case None => true // If nagios not responding, assume no problems in source, so we can go CRITICAL
47       }
48     } catch {
49       case e: NoSuchElementException => true // In the case where the host is not using this service
50       case e: Exception =>
51         log.error(e, s"Error parsing nagios response for host $host")
52         true
53     }
54   }
55 
56   // Returns the text content from a REST URL. Returns a None if there is a problem.
57   def getRestContent(url:String): Option[String] = {
58     var content: Option[String] = None
59     val httpClient = HttpClientBuilder.create().build()
60     try {
61       val httpResponse = httpClient.execute(new HttpGet(url))
62       val entity = httpResponse.getEntity
63 
64       if (entity != null) {
65         val inputStream = entity.getContent
66         content = Some(Source.fromInputStream(inputStream).getLines().mkString)
67         inputStream.close()
68       }
69     } finally {
70       httpClient.close()
71     }
72     content
73   }
