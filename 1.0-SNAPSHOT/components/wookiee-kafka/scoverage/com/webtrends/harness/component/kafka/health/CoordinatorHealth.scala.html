40 trait CoordinatorHealth { this: KafkaConsumerCoordinator =>
41 
42   // We need to keep track of the health of our partition workers. Each worker will notify the leader
43   // of the status when a kafka fetch is performed. We can not have each individual worker
44   // keep track of its own health because asking for that health will timeout if the worker is processing
45   // a message which takes a significant amount of time
46   val workerKafkaHealth = mutable.Map[String, (WorkerHealthState, Long, String)]()
47   val workerZKHealth = mutable.Map[String, WorkerHealthState]()
48   // Holds millis allowed before we say that a worker is starved
49   var topicAgeThresholds = mutable.Map[String, Long]()
50   var downSources = Set[String]()
51 
52   // We need to report our health status as critical if the events being processed are too old.
53   // There are a few scenarios where this could occur:
54   //
55   // 1. There is an issue copying data from one or more servers (or whole data centers.)
56   //
57   // 2. We are backed up. This is the scenario where harness is down for a while or it can't process data fast enough.
58   //    Its expected that all partitions will be backed up in this case, but its possible that only some partitions are
59   //
60   // To address these scenario we will be storing and reporting the partitions per topic and collection server that have
61   // old events.
62   //
63   // If we are in scenario 1, we would see the troubled server(s) listed with all partitions called out.
64   //
65   // If we are in scenario 2, we will see all servers listed with the backed up partition(s) listed.
66   val criticalPartsByTopicAndServer = mutable.Map[String, mutable.Map[String, Set[Int]]]()
67 
68   def healthReceive: Receive = {
69     case CheckHealth =>
70       try {
71         sender ! aggregateHealthStates()
72       } catch {
73         case ex: Exception => sender ! HealthComponent("Coordinator Health",
74           ComponentState.CRITICAL, s"Could not get health: ${ex.getMessage}")
75       }
76 
77     case msg: KafkaHealthState =>
78       if (msg.topic.isEmpty && workers.contains(msg.name)) {
79         workers(msg.name).started = false
80       } else {
81         workerKafkaHealth.put(msg.name, (msg, System.currentTimeMillis(), msg.topic))
82       }
83 
84     case msg: ZKHealthState =>
85       workerZKHealth.put(msg.name, msg)
86 
87     case msg: DownSources => downSources = msg.sources
88   }

90   def eventAgeHealthByServer(): List[HealthComponent]  = {
91     (for ((topic, partitionsByServer) <- criticalPartsByTopicAndServer) yield {
92       val serverHealths = for ( (server, partitions) <- partitionsByServer) yield {
93         HealthComponent(name = s"$server",
94           state = ComponentState.CRITICAL,
95           details = s"Lagging events found in partition(s) ${partitions.toList.sorted.mkString(",")}")
96       }
97 
98       HealthComponent(name = s"$topic: Servers exceeding age threshold",
99         state = ComponentState.CRITICAL,
100         details = s"Data from ${serverHealths.size} server(s) exceeded the age threshold",
101         components = serverHealths.to[List]
102       )
103     }).to[List]
104   }
105 
106   def renewTopicAgeThresholds() {
107     topicAgeThresholds.clear()
108     topicMap foreach { case (s: String, c: Config) =>
109       topicAgeThresholds(s) = c.hasPath("event-age-threshold-seconds") match {
110         case true => c.getLong("event-age-threshold-seconds") * 1000
111         case false => 0
112       }
113     }
114   }

116   def listActiveWorkers: List[String] = {
117     workers.filter(entry => entry._2.started).map(entry => entry._1).toList
118   }
119 
120   def aggregateHealthStates(): HealthComponent = {
121     val workerList = listActiveWorkers
122     mergeHealths(name = "Coordinator Health",
123       desc = "Aggregate coordinator health",
124       subComponents = List(
125         collapseWithTimeCheck(workerKafkaHealth.toMap, "Kafka Health"),
126         collapseHealthStates(workerZKHealth.toMap, "Zookeeper Health"),
127         HealthComponent("Active Worker List", ComponentState.NORMAL, s"Workers = ${workerList.size}", Some(workerList.mkString(", "))))
128         ++ eventAgeHealthByServer()
129     )
130   }

132   def collapseWithTimeCheck(healths: Map[String, (WorkerHealthState, Long, String)], label: String): HealthComponent = {
133     val filteredHealths = healths.filter { h =>
134       workers.contains(h._1) && (workers(h._1).started || (!workers(h._1).started && !h._2._1.healthy))
135     }
136 
137     var unHealthy = filteredHealths.filter { h => !h._2._1.healthy && !downSources.contains(getAssignmentHost(h._2._1.name)) }.map { h: (String, (WorkerHealthState, Long, String)) =>
138       HealthComponent(name = h._2._1.name,
139         state = ComponentState.CRITICAL,
140         details = h._2._1.status)
141     }.toList
142     val time = System.currentTimeMillis()
143     unHealthy = unHealthy ++ filteredHealths.filter { h =>
144       h._2._1.healthy && topicAgeThresholds(h._2._3) > 0 && (time - h._2._2) > topicAgeThresholds(h._2._3) && !downSources.contains(getAssignmentHost(h._2._1.name))
145     }.map {
146       h: (String, (WorkerHealthState, Long, String)) =>
147         HealthComponent(name = h._2._1.name,
148           state = ComponentState.DEGRADED,
149           details = s"No events in ${(time - h._2._2) / 1000} seconds")
150     }.toList
151     aggregatedComponent(filteredHealths.toMap, label, unHealthy)
152   }
153 
154   def collapseHealthStates(healths: Map[String, WorkerHealthState], label: String): HealthComponent = {
155     val unHealthy = healths.filterNot { h => h._2.healthy }.map { h: (String, WorkerHealthState) =>
156       HealthComponent(name = h._2.name,
157         state = ComponentState.CRITICAL,
158         details = h._2.status)
159     }.toList
160     aggregatedComponent(healths.toMap, label, unHealthy)
161   }
162 
163   def aggregatedComponent(healths: Map[String, Any], label: String, unHealthy: List[HealthComponent]): HealthComponent = {
164     HealthComponent(name = label,
165       state = if (unHealthy.isEmpty) ComponentState.NORMAL else ComponentState.DEGRADED,
166       details = s"${healths.size - unHealthy.size}/${healths.size} workers are healthy",
167       components = unHealthy)
168   }
169 
170   def mergeHealths(name: String, desc: String = "", subComponents: List[HealthComponent]): HealthComponent = {
171     def worstState(): ComponentState = {
172       val states = subComponents.map { subComp => subComp.state }
173       states.sortWith{(lt: ComponentState, gt: ComponentState) =>
174         if (lt == ComponentState.CRITICAL) true
175         else if (lt == ComponentState.DEGRADED && gt == ComponentState.NORMAL) true
176         else false
177       }.head
178     }
179 
180     health.HealthComponent(name = name,
181       state = worstState(),
182       details = desc,
183       components = subComponents.to[List])
184   }
