39 trait CoordinatorHealth { this: KafkaConsumerCoordinator =>
40 
41   // We need to keep track of the health of our partition workers. Each worker will notify the leader
42   // of the status when a kafka fetch is performed. We can not have each individual worker
43   // keep track of its own health because asking for that health will timeout if the worker is processing
44   // a message which takes a significant amount of time
45   val workerKafkaHealth = mutable.Map[String, (WorkerHealthState, Long, String)]()
46   val workerZKHealth = mutable.Map[String, WorkerHealthState]()
47   // Holds millis allowed before we say that a worker is starved
48   var topicAgeThresholds = mutable.Map[String, Long]()
49 
50   // We need to report our health status as critical if the events being processed are too old.
51   // There are a few scenarios where this could occur:
52   //
53   // 1. There is an issue copying data from one or more servers (or whole data centers.)
54   //
55   // 2. We are backed up. This is the scenario where harness is down for a while or it can't process data fast enough.
56   //    Its expected that all partitions will be backed up in this case, but its possible that only some partitions are
57   //
58   // To address these scenario we will be storing and reporting the partitions per topic and collection server that have
59   // old events.
60   //
61   // If we are in scenario 1, we would see the troubled server(s) listed with all partitions called out.
62   //
63   // If we are in scenario 2, we will see all servers listed with the backed up partition(s) listed.
64   val criticalPartsByTopicAndServer = mutable.Map[String, mutable.Map[String, Set[Int]]]()
65 
66   def healthReceive: Receive = {
67     case CheckHealth =>
68       try {
69         sender ! aggregateHealthStates()
70       } catch {
71         case ex: Exception => sender ! HealthComponent("Coordinator Health",
72           ComponentState.CRITICAL, s"Could not get health: ${ex.getMessage}")
73       }
74 
75     case msg: KafkaHealthState =>
76       if (msg.topic.isEmpty && workers.contains(msg.name)) {
77         workers(msg.name).started = false
78       } else {
79         workerKafkaHealth.put(msg.name, (msg, System.currentTimeMillis(), msg.topic))
80       }
81 
82     case msg: ZKHealthState =>
83       workerZKHealth.put(msg.name, msg)
84   }
85 
86   def eventAgeHealthByServer(): List[HealthComponent]  = {
87 
88     (for ((topic, partitionsByServer) <- criticalPartsByTopicAndServer) yield {
89 
90       val serverHealths = for ( (server, partitions) <- partitionsByServer) yield {
91         HealthComponent(name = s"$server",
92           state = ComponentState.CRITICAL,
93           details = s"Lagging events found in partition(s) ${partitions.toList.sorted.mkString(",")}")
94       }
95 
96       HealthComponent(name = s"$topic: Servers exceeding age threshold",
97         state = ComponentState.CRITICAL,
98         details = s"Data from ${serverHealths.size} server(s) exceeded the age threshold",
99         components = serverHealths.to[List]
100       )
101     }).to[List]
102   }
103 
104   def renewTopicAgeThresholds() {
105     topicAgeThresholds.clear()
106     topicMap foreach { case (s: String, c: Config) =>
107       topicAgeThresholds(s) = c.hasPath("event-age-threshold-seconds") match {
108         case true => c.getLong("event-age-threshold-seconds") * 1000
109         case false => 0
110       }
111     }
112   }

114   def listActiveWorkers: List[String] = {
115     workers.filter(entry => entry._2.started).map(entry => entry._1).toList
116   }
117 
118   def aggregateHealthStates(): HealthComponent = {
119     val workerList = listActiveWorkers
120     mergeHealths(name = "Coordinator Health",
121       desc = "Aggregate coordinator health",
122       subComponents = List(
123         collapseWithTimeCheck(workerKafkaHealth.toMap, "Kafka Health"),
124         collapseHealthStates(workerZKHealth.toMap, "Zookeeper Health"),
125         HealthComponent("Active Worker List", ComponentState.NORMAL, s"Workers = ${workerList.size}", Some(workerList.mkString(", "))))
126         ++ eventAgeHealthByServer()
127     )
128   }

130   def collapseWithTimeCheck(healths: Map[String, (WorkerHealthState, Long, String)], label: String): HealthComponent = {
131     val filteredHealths = healths.filter { h =>
132       workers.contains(h._1) && (workers(h._1).started || (!workers(h._1).started && !h._2._1.healthy))
133     }
134 
135     var unHealthy = filteredHealths.filterNot { h => h._2._1.healthy }.map { h: (String, (WorkerHealthState, Long, String)) =>
136       HealthComponent(name = h._2._1.name,
137         state = ComponentState.CRITICAL,
138         details = h._2._1.status)
139     }.toList
140     val time = System.currentTimeMillis()
141     unHealthy = unHealthy ++ filteredHealths.filter { h =>
142       h._2._1.healthy && topicAgeThresholds(h._2._3) > 0 && (time - h._2._2) > topicAgeThresholds(h._2._3) }.map {
143       h: (String, (WorkerHealthState, Long, String)) =>
144         HealthComponent(name = h._2._1.name,
145           state = ComponentState.DEGRADED,
146           details = s"No events in ${(time - h._2._2) / 1000} seconds")
147     }.toList
148     aggregatedComponent(filteredHealths.toMap, label, unHealthy)
149   }
150 
151   def collapseHealthStates(healths: Map[String, WorkerHealthState], label: String): HealthComponent = {
152     val unHealthy = healths.filterNot { h => h._2.healthy }.map { h: (String, WorkerHealthState) =>
153       HealthComponent(name = h._2.name,
154         state = ComponentState.CRITICAL,
155         details = h._2.status)
156     }.toList
157     aggregatedComponent(healths.toMap, label, unHealthy)
158   }
159 
160   def aggregatedComponent(healths: Map[String, Any], label: String, unHealthy: List[HealthComponent]): HealthComponent = {
161     HealthComponent(name = label,
162       state = if (unHealthy.isEmpty) ComponentState.NORMAL else ComponentState.DEGRADED,
163       details = s"${healths.size - unHealthy.size}/${healths.size} workers are healthy",
164       components = unHealthy)
165   }
166 
167   def mergeHealths(name: String, desc: String = "", subComponents: List[HealthComponent]): HealthComponent = {
168 
169     def worstState(): ComponentState = {
170       val states = subComponents.map { subComp => subComp.state }
171       states.sortWith{(lt: ComponentState, gt: ComponentState) =>
172         if (lt == ComponentState.CRITICAL) true
173         else if (lt == ComponentState.DEGRADED && gt == ComponentState.NORMAL) true
174         else false
175       }(0)
176     }
177 
178     health.HealthComponent(name = name,
179       state = worstState(),
180       details = desc,
181       components = subComponents.to[List])
182   }
