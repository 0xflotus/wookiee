36 trait KafkaWriterHealthCheck { this: KafkaWriter =>
37 
38   val healthName = "Kafka Writer"
39 
40   var currentHealth: Option[HealthComponent] = None
41 
42   var inProcessSend: Option[java.util.concurrent.Future[RecordMetadata]] = None
43 
44   // Amount of time we allow an async produce to run for before we consider it unhealthy
45   lazy val maxSendTimeMillis = Try {
46     kafkaConfig.getConfig("producer").getInt("linger.ms") + 2000
47   }.getOrElse(5000)
48 
49   // Schedule a regular send to detect changes even if no data is flowing through
50   val healthMessage = EventToWrite("producer_health", "Healthy".getBytes("utf-8"))
51   val scheduledHealthProduce = context.system.scheduler.schedule(0 seconds, 5 seconds, self, healthMessage)(scala.concurrent.ExecutionContext.Implicits.global)
52 
53   def healthReceive: Receive = {
54     case hc: HealthComponent =>
55       currentHealth = Some(hc)
56 
57     case CheckHealth =>
58       sender ! currentHealth.getOrElse(HealthComponent(healthName, ComponentState.NORMAL, "No data has been written yet"))
59   }
60 
61   def setHealth(componentState: ComponentState, details: String) {
62     currentHealth = Some(HealthComponent("Kafka Writer", componentState, details))
63   }
64 
65   def stopHealthCheck: Unit = {
66     scheduledHealthProduce.cancel()
67     currentHealth = None
68   }
69 
70   // The Kafka API makes it difficult to identify the target kafka server being down. A provided callback will not
71   // be called and calling get on the returned java Future will block indefinitely.
72   //
73   // Addressing this by wrapping the returned java Future and into a scala Future and blocking with defined timeout.
74   // If it completes successfully, and we don't have a normal health, set it to NORMAL
75   // If it throws an exception or times out, set our health to CRITICAL
76   //
77   // To limit performance impact, we are only monitoring a single send at at ime
78   def monitorSendHealth(sendFuture: java.util.concurrent.Future[RecordMetadata]) {
79     inProcessSend match {
80       case Some(f) => // Don't do anything, already monitoring a send
81       case None =>
82         inProcessSend = Some(sendFuture)
83         Future {
84           blocking {
85             try {
86               sendFuture.get(maxSendTimeMillis, TimeUnit.MILLISECONDS)
87               if (currentHealth.isEmpty || currentHealth.get.state != ComponentState.NORMAL) {
88                 self ! HealthComponent(healthName, ComponentState.NORMAL, "Write successful")
89               }
90             }
91             catch {
92               case ex: Exception =>
93                 log.error("Unable to produce data. Marking producer as unhealthy", ex)
94                 self ! HealthComponent(healthName, ComponentState.CRITICAL, ex.getMessage)
95             }
96             inProcessSend = None // Note that this may be executed in a different thread than the current message being processed
97           }
98         }(scala.concurrent.ExecutionContext.Implicits.global)
99     }
100   }
101 }
