36 trait KafkaWriterHealthCheck { this: KafkaWriter =>
37 
38   val healthName = "Kafka Writer"
39 
40   var currentHealth: Option[HealthComponent] = None
41 
42   var inProcessSend: Option[java.util.concurrent.Future[RecordMetadata]] = None
43 
44   // Amount of time we allow an async produce to run for before we consider it unhealthy
45   // We need to allow the entire linger time since the production may not even start until this time has passed
46   // plus some overhead for the production to actually occur.
47   lazy val maxSendTimeMillis = Try { kafkaConfig.getConfig("producer").getInt("linger.ms") }.getOrElse(5000) +
48     Try { kafkaConfig.getConfig("producer").getInt("produce-timeout-millis") }.getOrElse(10000)
49 
50   // Schedule a regular send to detect changes even if no data is flowing through
51   val healthMessage = EventToWrite("producer_health", "Healthy".getBytes("utf-8"))
52   val scheduledHealthProduce = context.system.scheduler.schedule(0 seconds, 5 seconds, self, healthMessage)(scala.concurrent.ExecutionContext.Implicits.global)
53 
54   def healthReceive: Receive = {
55     case hc: HealthComponent =>
56       currentHealth = Some(hc)
57 
58     case CheckHealth =>
59       sender ! currentHealth.getOrElse(HealthComponent(healthName, ComponentState.NORMAL, "No data has been written yet"))
60   }
61 
62   def setHealth(componentState: ComponentState, details: String) {
63     currentHealth = Some(HealthComponent("Kafka Writer", componentState, details))
64   }
65 
66   def stopHealthCheck: Unit = {
67     scheduledHealthProduce.cancel()
68     currentHealth = None
69   }
70 
71   // The Kafka API makes it difficult to identify the target kafka server being down. A provided callback will not
72   // be called and calling get on the returned java Future will block indefinitely.
73   //
74   // Addressing this by wrapping the returned java Future and into a scala Future and blocking with defined timeout.
75   // If it completes successfully, and we don't have a normal health, set it to NORMAL
76   // If it throws an exception or times out, set our health to CRITICAL
77   //
78   // To limit performance impact, we are only monitoring a single send at at ime
79   def monitorSendHealth(sendFuture: java.util.concurrent.Future[RecordMetadata]) {
80     inProcessSend match {
81       case Some(f) => // Don't do anything, already monitoring a send
82       case None =>
83         inProcessSend = Some(sendFuture)
84         Future {
85           blocking {
86             try {
87               sendFuture.get(maxSendTimeMillis, TimeUnit.MILLISECONDS)
88               if (currentHealth.isEmpty || currentHealth.get.state != ComponentState.NORMAL) {
89                 self ! HealthComponent(healthName, ComponentState.NORMAL, "Write successful")
90               }
91             }
92             catch {
93               case ex: Exception =>
94                 log.error("Unable to produce data. Marking producer as unhealthy", ex)
95                 self ! HealthComponent(healthName, ComponentState.CRITICAL, ex.getMessage)
96             }
97             inProcessSend = None // Note that this may be executed in a different thread than the current message being processed
98           }
99         }(scala.concurrent.ExecutionContext.Implicits.global)
100     }
101   }
102 }
