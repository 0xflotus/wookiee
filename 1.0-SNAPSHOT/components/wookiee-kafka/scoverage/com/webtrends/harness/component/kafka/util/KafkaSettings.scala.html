35 trait KafkaSettings extends ConfigHelper { this: Actor =>
36   @volatile

39   val clientId = s"kk_$hostname"

41   // Used for ZK paths
42   def pod = zkConf.get.dataCenter+"_"+zkConf.get.pod
43   def appRootPath = s"/$appName/$pod"
44 
45   def distributionRootPath = s"$appRootPath/kafkaConsumerDistribution"

47   val utf8 = StandardCharsets.UTF_8

49   def distributorPaths = DistributorPaths(s"$distributionRootPath/nodes",
50                                           s"$distributionRootPath/assignments",

53   def zkConf = renewableConfig.hasPath("wookiee-zookeeper.quorum") match {
54     case true => Some(ZookeeperSettings(renewableConfig.getConfig("wookiee-zookeeper")))
55     case false => None
56   }
57   def topicWorker = this.getClass.getClassLoader.loadClass(kafkaConfig.getString("worker-class"))
58   def leader = {
59     if (kafkaConfig.hasPath("leader-class")) {
60       this.getClass.getClassLoader.loadClass(kafkaConfig.getString("leader-class"))
61     } else {
62       classOf[KafkaConsumerCoordinator]
63     }
64   }
65   def appName = kafkaConfig.getString("app-name")
66 
67   def init: KafkaSettings = {
68     this
69   }
70 
71   def topicMap: Map[String, Config] = kafkaConfig.getConfigList("consumer.topics").toList.map{top =>
72     top.getString("name") -> top
73   }.toMap
74   
75   def kafkaSources: Map[String, List[String]] = kafkaConfig.getConfigList("consumer.kafka-hosts").toList.map { clusters =>
76     clusters.getString("id") -> clusters.getStringList("brokers").toList
77   }.toMap
78 
79   def assignmentName(spec:PartitionAssignment): String = {
80     s"${spec.host}-${spec.topic}-${spec.partition}"
81   }
82 
83   def partitionName(spec:PartitionAssignment): String = {
84     s"${spec.cluster}-${spec.topic}-${spec.partition}"
85   }
86 
87   override def renewConfiguration() = {
88     super.renewConfiguration()
89     kafkaConfig = ConfigUtil.prepareSubConfig(renewableConfig, "wookiee-kafka")
90   }
