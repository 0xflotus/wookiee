35   def getDesiredAvailableOffset(consumer: SimpleConsumer, topic: String, partition: Int, desiredStartOffset: Long, clientName: String): Long = {
36     val topicAndPartition: TopicAndPartition = new TopicAndPartition(topic, partition)
37 
38     val startOffsetRequest = OffsetRequest(
39       Map(topicAndPartition -> PartitionOffsetRequestInfo(OffsetRequest.EarliestTime, 1)),
40       replicaId = 0) //TODO: determine replicaId, if needed?
41 
42     val startResponse: OffsetResponse = consumer.getOffsetsBefore(startOffsetRequest)
43     if (startResponse.hasError) {
44       throw new IllegalStateException("Kafka response error getting earliest offsets" )
45     }
46     val startOffsets: Seq[Long] = startResponse.partitionErrorAndOffsets(topicAndPartition).offsets
47     if (startOffsets.length != 1) {
48       throw new IllegalStateException(s"Expect one earliest offset but got [${startOffsets.length}]")
49     }
50 
51 
52     val endOffsetRequest = OffsetRequest(
53       Map(topicAndPartition -> PartitionOffsetRequestInfo(OffsetRequest.LatestTime, 1)),
54       replicaId = 0)  //TODO: determine replicaId, if needed?
55 
56     val endResponse: OffsetResponse = consumer.getOffsetsBefore(endOffsetRequest)
57     if (endResponse.hasError) {
58       throw new IllegalStateException("Kafka response error getting latest offsets" ) //+ startResponse.error())
59     }
60     val endOffsets: Seq[Long] = endResponse.partitionErrorAndOffsets(topicAndPartition).offsets
61     if (endOffsets.length != 1) {
62       throw new IllegalStateException(s"Expect one latest offset but got [${endOffsets.length}]")
63     }
64 
65     val range:Array[Long] = new Array(2)
66 
67     range(0) = startOffsets.head
68     range(1) = endOffsets.head
69 
70     if (range(1) < range(0)) {
71       range(0) = range(1)
72     }
73     if (desiredStartOffset > range(0) && desiredStartOffset < range(1)) {
74       range(0) = desiredStartOffset
75     }
76 
77     range(0)
78 
79   }

118   def configToProps(config: Config): Properties = {
119     val props: Properties = new Properties
120     import scala.collection.JavaConversions._
121     config.entrySet.foreach { entry =>
122       entry.getValue.valueType match {
123         case ConfigValueType.STRING =>
124           props.put(entry.getKey, config.getString(entry.getKey))
125         case ConfigValueType.NUMBER =>
126           props.put(entry.getKey, config.getNumber(entry.getKey).toString)
127         case _ =>
128         // Ignore other types
129       }
130     }
131     props.put("key.serializer","org.apache.kafka.common.serialization.StringSerializer")
132     props.put("value.serializer","org.apache.kafka.common.serialization.ByteArraySerializer")
133     props
134   }
