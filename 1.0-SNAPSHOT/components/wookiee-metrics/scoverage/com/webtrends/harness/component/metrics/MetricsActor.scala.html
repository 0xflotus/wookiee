45 class MetricsActor(settings:MonitoringSettings) extends Actor with ActorLoggingAdapter with ActorHealth {
46   import context.dispatcher

48   private[metrics] var jmxReporter: Option[JmxReporter] = None
49   private[metrics] var graphiteReporter: Option[GraphiteReporter] = None
50   private[metrics] var jvmGraphiteReporter: Option[GraphiteReporter] = None

52   def receive = health orElse {
53     case StatusRequest(format) =>
54       val metrics = new MetricsWriter().getMetrics(settings.GraphiteIncludeVMMetrics)
55       format match {
56         case "string" =>
57           log.debug("Fetching the metrics in string format")
58           sender ! compact(render(metrics))
59         case _ =>
60           log.debug("Fetching the metrics in json format")
61           sender ! metrics
62       }
63 
64     case GetMetric(name) =>
65       log.debug(s"Fetching metric $name")
66       val reg = MetricBuilder.registry
67       val i = 0
68 
69     case RemoveMetric(metric) => MetricBuilder.remove(metric)
70 
71     case o: MetricObservation => o match {
72       case o: CounterObservation =>
73         MetricBuilder(o).inc(o.delta)
74       case o: GaugeObservation =>
75         MetricBuilder(o).setValue(o.value)
76       case o: HistogramObservation =>
77         MetricBuilder(o).update(o.value)
78       case o: MeterObservation =>
79         MetricBuilder(o).mark(o.events)
80       case o: TimerObservation =>
81         MetricBuilder(o).update(o.elapsed, o.unit)
82     }
83     case _ => // Just eat it
84   }
85 
86   override def preStart() {
87     // Subscribe to events
88     MetricsEventBus.subscribe(self)
89 
90     if (settings.GraphiteEnabled) {
91       graphiteReporter = Some(GraphiteReporter.forRegistry(MetricBuilder())
92         .prefixedWith("%s.%s.%s".format(settings.MetricPrefix,
93         InetAddress.getLocalHost.getHostName.split("\\.")(0),
94         settings.ApplicationName.replace(' ', '_').toLowerCase))
95         .build(new Graphite(new InetSocketAddress(settings.GraphiteHost,
96         settings.GraphitePort))))
97 
98       jvmGraphiteReporter = Some(GraphiteReporter.forRegistry(MetricBuilder.jvmRegistry)
99         .prefixedWith("%s.%s.%s.jvm".format(settings.MetricPrefix,
100         InetAddress.getLocalHost.getHostName.split("\\.")(0),
101         settings.ApplicationName.replace(' ', '_').toLowerCase))
102         .build(new Graphite(new InetSocketAddress(settings.GraphiteHost,
103         settings.GraphitePort))))
104 
105       graphiteReporter.get.start(settings.GraphiteInterval, TimeUnit.MINUTES)
106       jvmGraphiteReporter.get.start(settings.GraphiteInterval, TimeUnit.MINUTES)
107     }
108 
109     // Start the JMX engine
110     MetricBuilder.registerJvmMetrics
111 
112     if (settings.JmxEnabled) {
113       jmxReporter = Some(JmxReporter.forRegistry(MetricBuilder()).build)
114       jmxReporter.get.start
115     }
116 
117     log.info("Metrics manager started: {}", context.self.path)
118   }
119 
120   override def postStop() = {
121     // Un-subscribe to events
122     MetricsEventBus.unsubscribe(self)
123 
124     graphiteReporter match {
125       case Some(reporter) =>
126         // Flush our data
127         reporter.asInstanceOf[ScheduledReporter].report
128         reporter.close
129         graphiteReporter = None
130       case _ =>
131       // do nothing
132     }
133     jvmGraphiteReporter match {
134       case Some(reporter) =>
135         // Flush our data
136         reporter.asInstanceOf[ScheduledReporter].report
137         reporter.close
138         jvmGraphiteReporter = None
139       case _ =>
140       // do nothing
141     }
142 
143     jmxReporter match {
144       case Some(reporter) =>
145         reporter.stop
146         jmxReporter = None
147       case _ =>
148       // do nothing
149     }
150 
151     MetricBuilder.registry.removeMatching(MetricFilter.ALL)
152     MetricBuilder.jvmRegistry.removeMatching(MetricFilter.ALL)
153     log.info("Metrics manager stopped: {}", context.self.path)
154   }

156   /**
157    * This is the health of the current object, by default will be NORMAL
158    * In general this should be overridden to define the health of the current object
159    * For objects that simply manage other objects you shouldn't need to do anything
160    * else, as the health of the children components would be handled by their own
161    * CheckHealth function
162    *
163    * @return
164    */
165   override protected def getHealth: Future[HealthComponent] = {
166     Future {
167       Try({
168         log.debug("MetricsActor health requested")
169         settings.GraphiteEnabled match {
170           case true =>
171             HealthComponent(Metrics.MetricsName, ComponentState.NORMAL, "Currently sending metrics to graphite at %s:%d"
172               .format(settings.GraphiteHost, settings.GraphitePort))
173           case false =>
174             HealthComponent(Metrics.MetricsName, ComponentState.NORMAL, "Currently not sending metrics to graphite")
175         }
176       }).recover({
177         case e: Exception => HealthComponent(Metrics.MetricsName, ComponentState.CRITICAL, "An error occurred checking the metrics health: ".concat(e.getMessage))
178       }).get
179     }
180   }
