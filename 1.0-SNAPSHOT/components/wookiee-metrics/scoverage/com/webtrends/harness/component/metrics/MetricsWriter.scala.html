97         if (Hist.isInstance(metric)) {
98           processHistogram(metric.asInstanceOf[Histogram])
99         }
100         else if (Count.isInstance(metric)) {
101           processCounter(metric.asInstanceOf[Counter])
102         }
103         else if (Mtr.isInstance(metric)) {
104           processMeter(metric.asInstanceOf[Meter])
105         }
106         else if (Gau.isInstance(metric)) {
107           processGauge(metric.asInstanceOf[Gauge[_]])
108         }
109         else if (UGau.isInstance(metric)) {
110           processGauge(metric.asInstanceOf[UpdatableGauge[_]])
111         }
112         else {
113           processTimer(metric.asInstanceOf[Timer])
114         }
115     }
116   }
