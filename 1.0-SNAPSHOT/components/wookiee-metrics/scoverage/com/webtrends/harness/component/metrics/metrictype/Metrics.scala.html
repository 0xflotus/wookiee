92 object Metric {
93 
94   /**
95    * Concatenates elements to form a dotted name, eliding any null values or empty strings.
96    *
97    * @param name     the first element of the name
98    * @param names    the remaining elements of the name
99    * @return { @code name} and { @code names} concatenated by periods
100    */
101   def name(name: String, names: String*): String = {
102     val builder: StringBuilder = new StringBuilder
103     append(builder, name)
104     if (names != null) {
105       for (s <- names) {
106         append(builder, s)
107       }
108     }
109     builder.toString
110   }
111 
112   private def append(builder: StringBuilder, part: String) {
113     if (part != null && !part.isEmpty) {
114       if (builder.length > 0) {
115         builder.append('.')
116       }
117       builder.append(part)
118     }
119   }
