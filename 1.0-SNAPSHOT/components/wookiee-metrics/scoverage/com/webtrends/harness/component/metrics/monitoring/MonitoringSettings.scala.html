21 import com.typesafe.config.{ConfigFactory, Config}

23 class MonitoringSettings (config: Config = ConfigFactory.load) {
24 
25     // What is the application name (used for persisting metrics)
26     var ApplicationName         = config getString       "application-name"
27     //Is the application reporting through the JMX interface
28     var JmxEnabled              = config getBoolean      "jmx.enabled"
29     // If JMX is enabled then which port is it running on
30     var JmxPort                 = config getInt          "jmx.port"
31     // The prefix to append metrics being sent to graphite
32     var MetricPrefix            = config getString       "metric-prefix"
33     // Should the application pump metrics directly to graphite
34     var GraphiteEnabled         = config getBoolean      "graphite.enabled"
35     // What is the fqdn for the graphite server
36     var GraphiteHost            = config getString       "graphite.host"
37     // What port is graphite listening on
38     var GraphitePort            = config getInt          "graphite.port"
39     // How often (minutes) should we flush metrics to graphite
40     var GraphiteInterval        = config getInt          "graphite.interval"
41     // Should we include the JVM metrics when sending to graphite
42     var GraphiteIncludeVMMetrics = config getBoolean      "graphite.vmmetrics"
43     // This is a regular expression for which metrics should be sent on to graphite. All metrics are still exposed via JMX or the metrics endpoint
44     var GraphiteRegEx           = config getString       "graphite.regex"
45 
46     require(ApplicationName   != "", "application-name must be set")
47     require(MetricPrefix      != "", "metric-prefix must be set")
48 }
