41 class CoreNettyServer(settings:NettyServerSettings) extends HActor {
42   var server: Option[SocketServer] = None
43   var worker: Option[ActorRef] = None
44 
45   override def receive: Receive = starting
46 
47   def starting: Receive = {
48     case StartServer => startServer
49   }
50 
51   def running: Receive = health orElse {
52     case ShutdownServer => shutdownServer
53   }
54 
55   protected def startServer: Unit = {
56     server = Some(SocketServer(settings, this))
57     worker = Some(context.actorOf(FromConfig.props(Props[CoreNettyWorker]), "netty-worker"))
58     context.become(running)
59     context.parent ! NettyRunning
60     log.info(s"Netty Http server bound to port ${settings.port}")
61   }
62 
63   protected def shutdownServer: Unit = {
64     server match {
65       case Some(s) =>
66         log.info(s"Trying to shutdown the netty server ${InetAddress.getLocalHost.getHostName} on port ${settings.port}")
67         s.close()
68       case None => // do nothing
69     }
70   }
71 
72   /**
73    * Fetch routes from all registered services and concatenate with our default ones that
74    * are defined below.
75    */
76   def getHandlers: Map[String, ChannelHandler] = {
77     val serviceHandlers = HandlerManager.getHandlers.filter (r => ! r.equals (Map.empty) )
78     serviceHandlers ++ getBaseHandlers
79   }
80 
81   private def getBaseHandlers = {
82     Map[String, ChannelHandler]("core-http-handler" -> new CoreHttpHandler(settings, worker.get) with WebSocket)
83   }
84 
85   override protected def getHealth: Future[HealthComponent] = {
86     import scala.concurrent.ExecutionContext.Implicits.global
87     
88     Future {
89       server match {
90         case None =>
91           HealthComponent(
92             "netty-server",
93             ComponentState.CRITICAL,
94             "The internal netty server is not running",
95             None)
96         case Some(s) =>
97           HealthComponent(
98             "netty-server",
99             ComponentState.NORMAL,
100             s"The internal netty server is running on port ${settings.port}",
101             None)
102       }
103     }
104   }
105 }
