48   /**
49    * Establish our routes and other receive handlers
50    */
51   override def receive = super.receive orElse {
52     case GetSystemInfo(name, msg) => respondToComponentRequest(name, msg)
53     case GetHealth(msg) => respondToHealthRequest(msg)
54     case GetServiceInfo(name) => respondToServiceRequest(name)
55   }
56 
57   def respondToServiceRequest(name:Option[String]) = {
58     val caller = sender()
59     val msg = name match {
60       case Some(n) => GetMetaDataByName(n)
61       case None => GetMetaData(None)
62     }
63     (serviceActor ? msg) onComplete {
64       case Success(resp) => caller ! resp
65       case Failure(f) => caller ! Status.Failure(f)
66     }
67   }
68 
69   def respondToComponentRequest[T, M](name:String, msg:ComponentRequest[T]): Unit = {
70     val caller = sender()
71     componentRequest[T, M](name, msg) onComplete {
72       case Success(resp) => caller ! resp
73       case Failure(f) => caller ! Status.Failure(f)
74     }
75   }
76 
77   def respondToHealthRequest(msg:HealthResponseType): Unit = {
78     val caller = sender()
79     (healthActor ? HealthRequest(msg)) onComplete {
80       case Success(resp) => caller ! resp
81       case Failure(f) => caller ! Status.Failure(f)
82     }
83   }
