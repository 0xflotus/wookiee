28 trait WebSocket extends BaseInboundHandler {
29 
30   def WebSocketFrame(ctx: ChannelHandlerContext, frame: TextWebSocketFrame): Unit = {
31     val request = frame.text()
32 
33     log.info(s"request: $request")
34     ctx.writeAndFlush(new TextWebSocketFrame(request.toString.toUpperCase))
35   }
36 
37   override def handleWebSocketFrame(ctx: ChannelHandlerContext, frame: WebSocketFrame): Boolean = {
38     var msgHandled = true
39     try {
40       frame match {
41         case f: CloseWebSocketFrame =>
42           handshaker match {
43             case Some(hs) => hs.close(ctx.channel(), f.retain())
44             case None => msgHandled = false
45           }
46         case f: PingWebSocketFrame => ctx.channel().writeAndFlush(new PongWebSocketFrame(f.content().retain()))
47         case f: TextWebSocketFrame => WebSocketFrame(ctx, f)
48         case _ => throw new UnsupportedOperationException(
49           String.format("%s frame type not supported", frame.getClass.getName)
50         )
51       }
52     } catch {
53       case e: Throwable =>
54         log.warn("exception in handleWebSocketFrame", e)
55         sendError(ctx, s"error: ${e.getMessage}")
56     }
57     
58     msgHandled
59   }
60 
61   def sendError(ctx: ChannelHandlerContext, msg: String) {
62     ctx.channel.write(new TextWebSocketFrame(msg))
63     ctx.channel.flush()
64     ctx.flush()
65   }
66 }
