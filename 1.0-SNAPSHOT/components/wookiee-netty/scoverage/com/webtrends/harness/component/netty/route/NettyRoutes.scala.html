19 private[route] class NettyHandler(commandRef:ActorRef, paths:Map[String, String], method:HttpMethod) extends BaseInboundHandler {
20   /**
21    * Override this function in order to process http requests.
22    * @param ctx
23    * @param req
24    * @return true if this function handled the request; otherwise, false
25    */
26   override def handleHttpRequest(ctx: ChannelHandlerContext, req: FullHttpRequest): Boolean = {
27     (req.getMethod, req.getUri) match {
28       case (this.method, path) =>
29         paths.foreach {
30           case (name, testPath) =>
31             Command.matchPath(testPath, path.substring(1)) match {
32               case Some(bean) =>
33                 commandRef ! HandleHttpRequest(ctx, req, Some(bean))
34                 return true
35               case None => // ignore and go on to the next path to match
36             }
37         }
38 
39       case _ => // ignore and we will leave this as unhandled
40     }
41     return false
42   }
43 }
