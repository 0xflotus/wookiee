33 object SocketServer {
34   def apply(settings: NettyServerSettings, nettyServer: CoreNettyServer): SocketServer = {
35     val bossGroup = new NioEventLoopGroup(1)
36     val workerGroup = new NioEventLoopGroup()
37 
38     try {
39       val b = new ServerBootstrap()
40         .group(bossGroup,workerGroup)
41         .channel(classOf[NioServerSocketChannel])
42         .childOption[java.lang.Boolean](ChannelOption.TCP_NODELAY, settings.tcpNoDelay)
43         .childOption[java.lang.Boolean](ChannelOption.SO_KEEPALIVE, settings.tcpKeepAlive)
44         .handler(new LoggingHandler(LogLevel.INFO))
45       if ( settings.tcpSndBufferSize > 0 ) {
46         b.childOption[java.lang.Integer](ChannelOption.SO_SNDBUF, settings.tcpSndBufferSize.toInt)
47       }
48       if ( settings.tcpRcvBufferSize > 0 ) {
49         b.childOption[java.lang.Integer](ChannelOption.SO_RCVBUF, settings.tcpRcvBufferSize.toInt)
50       }
51 
52       b.childHandler(new SocketServerInitializer(settings, nettyServer))
53 
54       new SocketServer(b.bind(settings.port).sync().channel(), bossGroup, workerGroup)
55     } catch {
56       case e: Throwable =>
57         bossGroup.shutdownGracefully()
58         workerGroup.shutdownGracefully()
59         throw new RuntimeException(s"unable to bootstrap ${getClass.getName} ${e}")
60     }
61   }
62 }
