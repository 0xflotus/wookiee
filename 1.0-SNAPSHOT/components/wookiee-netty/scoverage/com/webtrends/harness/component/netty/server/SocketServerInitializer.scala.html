35 class SocketServerInitializer(settings: NettyServerSettings, baseServer: CoreNettyServer) extends ChannelInitializer[SocketChannel] {
36   override def initChannel(ch: SocketChannel) {
37     val pipeline = ch.pipeline()
38     pipeline.addLast(new HttpServerCodec())
39     pipeline.addLast(new HttpObjectAggregator(65536))
40     pipeline.addLast(new ChunkedWriteHandler)
41     pipeline.addLast(new IdleStateHandler(settings.idleTimeout, 0, 0, TimeUnit.SECONDS))
42     pipeline.addLast(new TimeoutHandler())
43     baseServer.getHandlers.values.foreach(h => pipeline.addLast(h))
44   }
45 }
