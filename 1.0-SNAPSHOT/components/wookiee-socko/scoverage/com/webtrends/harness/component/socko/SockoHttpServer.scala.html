42 class SockoHttpServer extends HActor with ComponentHelper {
43   val config = SockoServerSettings.get(context.system)
44   val routees = Try { context.system.settings.config.getInt(SockoManager.KeyNumHandlerRoutees) } getOrElse 5
45   val router = AkkaUtil.initActorFromConfig(Props[SockoHttpWorker], SockoHttpServer.ParamWorker, routees)
46   var webServer:Option[WebServer] = None
47 
48   override def receive:Receive = binding
49 
50   def binding:Receive = {
51     case BindServer => bindHttpServer
52     case HttpStartProcessing => context.become(running)
53   }
54 
55   def running:Receive = health orElse {
56     case HttpStartProcessing =>
57       log.debug("Http processing starting.")
58       context.child(SockoHttpServer.ParamWorker).get ! Broadcast(HttpStartProcessing)
59     case ShutdownServer => shutdownHttpServer
60   }
61 
62   /**
63    * Base routes for the harness
64    *
65    */
66   val routes = Routes({
67     case HttpRequest(request) => router ! SockoWorkerMessage(request)
68   })
69 
70   protected def bindHttpServer : Unit = {
71     log.info(s"Trying to bind to port ${config.port}")
72 
73     webServer = Some(new WebServer(config, context.system, {server => new SockoRequestHandler(server, routes)}))
74     Runtime.getRuntime.addShutdownHook(new Thread {
75       override def run: Unit = {
76         shutdownHttpServer
77       }
78     })
79     Try({
80       webServer.get.start()
81       context.become(running)
82       context.parent ! HttpRunning
83       log.info(s"Bound to port ${config.port}")
84     }).recover({
85       case e:Throwable =>
86         log.error(s"Error trying to bind to the http port ${config.port}", e)
87         throw e
88     })
89   }
90 
91   /**
92    * Unbind from the http service
93    */
94   protected def shutdownHttpServer: Unit = {
95     log.info(s"Trying to shutdown the http server ${config.hostname} on port ${config.port}")
96     webServer match {
97       case Some(ws) =>
98         Try(ws.stop()).recover({
99           case e: Throwable =>
100             log.warning("we were unable to properly shutdown the HttpServer: {}", e.getMessage)
101         })
102       case None => //ignore
103     }
104     log.info("Http server shutdown complete")
105     if (context != null) {
106       context.become(binding)
107     }
108   }
109 
110   // This should probably be overriden to get some custom information about the health of this actor
111   override protected def getHealth: Future[HealthComponent] = {
112     val p = Promise[HealthComponent]
113     webServer match {
114       case Some(s) => p success HealthComponent("socko", ComponentState.NORMAL, s"Web server up and running on port ${config.port}")
115       case None => p success HealthComponent("socko", ComponentState.CRITICAL, "Web server not started")
116     }
117     p.future
118   }
119 }
