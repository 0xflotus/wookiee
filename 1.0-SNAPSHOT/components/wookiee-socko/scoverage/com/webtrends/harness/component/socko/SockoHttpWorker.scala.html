42 class SockoHttpWorker extends HActor with ComponentHelper {
43   import context.dispatcher

58     case HttpStartProcessing =>
59       log.debug("Worker beginning to process")

63   def running:Receive = super.receive orElse {
64     case SockoWorkerMessage(event) => handleSockoRoute(event)
65   }

67   /**
68    * Attempts to handle the Socko route
69    * @param event The http request
70    */
71   protected def handleSockoRoute(event: HttpRequestEvent) = {
72     event match {
73       case GET(Path("/favicon.ico")) =>
74         event.response.write(HttpResponseStatus.NO_CONTENT)
75       case GET(Path("/ping")) =>
76         event.response.write("pong: ".concat(new DateTime(System.currentTimeMillis(), DateTimeZone.UTC).toString))
77       case GET(Path("/healthcheck/lb")) =>
78         (healthActor ? HealthRequest(HealthResponseType.LB)).mapTo[String] onComplete {
79           case Success(h) => event.response.write(h)
80           case Failure(f) => messageFailure(f, event)
81         }
82       case GET(Path("/healthcheck/nagios")) =>
83         (healthActor ? HealthRequest(HealthResponseType.NAGIOS)).mapTo[String] onComplete {
84           case Success(h) => event.response.write(h)
85           case Failure(f) => messageFailure(f, event)
86         }
87       case _ =>
88         val allHandlers = SockoRouteManager.getHandlers(event.nettyHttpRequest.getMethod.toString.toUpperCase)
89         val matchedHandler = allHandlers match {
90           case Some(handlers) =>
91             try {
92               handlers.find { handler =>
93                 handler._2.matchSockoRoute(event) match {
94                   case Some(map) =>
95                     Future {
96                       handler._2.handleSockoRoute(event, map)
97                     }
98                     true
99                   case None => false
100                 }
101               }
102             } catch {
103               case ex: Exception =>
104                 log.error(s"Error matching path ${event.endPoint.path}", ex)
105                 None
106             }
107           case None => None
108         }
109 
110         matchedHandler match {
111           case Some(h) => // Handler already called
112           case None => // Fallback to static content if initialized
113             if (ConfigUtil.getDefaultValue(SockoManager.KeyRootPaths, context.system.settings.config.getString, "").nonEmpty) {
114               context.actorSelection(s"${HarnessConstants.ComponentFullName}/harness-socko").resolveOne() onComplete {
115                 case Success(s) => s ! SockoContentRequest(event)
116                 case Failure(f) => event.response.write(HttpResponseStatus.NOT_FOUND)
117               }
118             } else {
119               event.response.write(HttpResponseStatus.NOT_FOUND)
120             }
121         }
122     }
123   }
124 
125 
126   def messageFailure(f:Throwable, event:HttpRequestEvent) = {
127     f match {
128       case notFound:ComponentNotFoundException => event.response.write(HttpResponseStatus.NOT_FOUND, notFound.getMessage)
129       case t => event.response.write(HttpResponseStatus.INTERNAL_SERVER_ERROR, t.getMessage)
130     }
131   }
