38 class SockoManager(name:String) extends Component(name) with Socko with SockoClient {
39 
40   val staticContentHandlerConfig =
41     if (config.hasPath(SockoManager.KeyRootPaths)) {
42       val rootPaths = config.getStringList(SockoManager.KeyRootPaths).asScala
43       if (rootPaths.length > 0) {
44         Some(StaticContentHandlerConfig(
45           rootFilePaths = rootPaths,
46           serverCacheMaxFileSize = ConfigUtil.getDefaultValue(SockoManager.KeyServerCacheMaxFileSize, config.getInt, 0),
47           serverCacheTimeoutSeconds = ConfigUtil.getDefaultValue(SockoManager.KeyServerCacheTimeoutSeconds, config.getInt, 0),
48           browserCacheTimeoutSeconds = ConfigUtil.getDefaultValue(SockoManager.KeyBrowserCacheTimeoutSeconds, config.getInt, 0)
49         ))
50       } else {
51         None
52       }
53     } else {
54       None
55     }
56 
57   /**
58    * We add super.receive because if you override the receive message from the component
59    * and then do not include super.receive it will not handle messages from the
60    * ComponentManager correctly and basically not start up properly
61    *
62    * @return
63    */
64   override def receive = super.receive orElse {
65     case HttpRunning => context.parent ! ComponentStarted(self.path.name)
66     case AddSockoHandler(cName, handler, method) => SockoRouteManager.addHandler(cName, handler, method)
67     case SockoContentRequest(event) => handleStaticContentRequest(event)
68     case SystemReady =>
69       sockoServer match {
70         case Some(s) => s ! HttpStartProcessing
71         case None => // ignore
72       }
73   }
74 
75   /**
76    * Start function will start any child actors that will be managed by the ComponentManager
77    * @return
78    */
79   override def start = {
80     // start the static file handler if the rootpaths have been set
81     staticContentHandlerConfig match {
82       case Some(c) => AkkaUtil.initActorFromConfig(Props(classOf[StaticContentHandler], c), SockoManager.StaticFileHandlerName)
83       case None => log.debug("Static Content Handler not enabled.")
84     }
85     startSocko
86     //startSockoClient
87   }
88 
89   /**
90    * Stop will execute any cleanup work to be done for the child actors
91    * if not necessary this can be deleted
92    * @return
93    */
94   override def stop = {
95     super.stop
96     stopSocko
97   }
98 
99   def handleStaticContentRequest(event:HttpRequestEvent) = {
100     context.child(SockoManager.StaticFileHandlerName) match {
101       case Some(s) =>
102         ConfigUtil.getDefaultValue(SockoManager.KeyFileType, config.getString, "file") match {
103           case "jar" =>
104             s ! StaticResourceRequest(event, staticContentHandlerConfig.get.rootFilePaths(0))
105           case _ =>
106             // check if file exists first otherwise we get a FileNotFoundException
107             val f = new File(staticContentHandlerConfig.get.rootFilePaths(0) + "/" + event.endPoint.path)
108             if (f.exists()) {
109               s ! StaticFileRequest(event, f)
110             }
111         }
112 
113       case None => event.response.write(HttpResponseStatus.NOT_FOUND)
114     }
115   }
116 }
