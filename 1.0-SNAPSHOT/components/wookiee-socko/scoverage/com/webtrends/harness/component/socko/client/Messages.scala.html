46 /**
47  * Config for any http request
48  *
49  * @param timeout default 5, implicitly set
50  * @param server server that you are posting the request too
51  * @param port port for that server, defaults to 80
52  * @param useSSL default false, if need ssl set to true
53  */
54 class HttpConfig(server:String, port:Int=80, useSSL:Boolean=false, timeout:Int=5) {
55   val pathString = "%s://%s:%d/%s"
56 
57   def fullPath(path:String) : String = {
58     val protocol = if (useSSL) "https" else "http"
59     pathString.format(protocol, server, port, path)
60   }
61 
62   def getTimeout = Timeout(Duration(timeout, "seconds"))
63 }
