27 object SockoRouteManager {
28   val externalLogger = LoggerFactory.getLogger(this.getClass)
29   private val handlerMap = new mutable.LinkedHashMap[String, List[(String, SockoRouteHandler)]]
30 
31   def addHandler(name:String, handler:SockoRouteHandler, method: String)(implicit context:ActorContext): Unit = {
32     addHandlers(name, handler, Seq(method))(context)
33   }
34 
35   def addHandlers(name:String, handler:SockoRouteHandler, methods: Seq[String])(implicit context:ActorContext) {
36     methods foreach { method =>
37       handlerMap.getOrElseUpdate(method.toUpperCase, List()).forall { it => it._1 != name } match {
38         case true => handlerMap.put(method.toUpperCase, handlerMap(method.toUpperCase) :+ (name, handler))
39         case false => externalLogger.debug(s"Socko Route Handler $name already added, discarding")
40       }
41     }
42   }
43 
44   def getHandler(name:String): Option[SockoRouteHandler] = {
45     handlerMap.values.foreach(it => it.foreach(handler => if (handler._1 == name) return Some(handler._2)))
46     None
47   }
48 
49   def getHandlers(method: String): Option[List[(String, SockoRouteHandler)]] = {
50     handlerMap.get(method.toUpperCase)
51   }
