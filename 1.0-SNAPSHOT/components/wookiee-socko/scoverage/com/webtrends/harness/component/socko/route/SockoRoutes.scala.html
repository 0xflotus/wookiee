88   def marshallObject(obj: Any, respType: String = "json"): Array[Byte] = {
89     val realRespType = SockoUtils.normalizeContentType(respType)
90     if (realRespType == "application/json") {
91       obj match {
92         case s: String => s.getBytes

95     } else obj match {
96       case bytes: Array[Byte] =>

98       case _ =>

194   def innerExecute[T<:AnyRef:Manifest](bean:SockoCommandBean) = {
195     try {
196       applyQSMap(bean)
197       execute(Some(bean)).mapTo[CommandResponse[T]] onComplete {
198         case Success(resp) =>
199           resp.data match {
200             case Some(value) =>
201               val content = if (bean.event.endPoint.method.equalsIgnoreCase("head")) Array[Byte]() else marshallObject(value, resp.responseType)
202               bean.event.response.write(
203                 responseStatusCode,
204                 // just double check the end point method, and if it is head then simply remove any content from the response
205                 content,
206                 SockoUtils.normalizeContentType(resp.responseType),
207                 SockoUtils.toSockoHeaderFormat(bean.event.nettyHttpRequest.getMethod.name(), getResponseHeaders(bean.event.request.headers), content.length)
208               )
209             case None => bean.event.response.write(HttpResponseStatus.NO_CONTENT)
210           }
211         case Failure(f) => f match {
212           case ce: SockoCommandException => getRejectionHandler(bean.event, ce)
213           case _ => bean.event.response.write(HttpResponseStatus.INTERNAL_SERVER_ERROR, f.getMessage)
214         }
215       }
216     } catch {
217       case e:Throwable => getRejectionHandler(bean.event, e)
218     }
219   }

268   private[route] def addEntityToBean[T<:AnyRef:Manifest](bean:SockoCommandBean, obj:Array[Byte]) = {
269     val conType = bean.headers.get("Content-Type") match {
270       case null => "application/x-www-form-urlencoded"
271       case x: String => x
272     }
273 
274     var content = obj
275     val contentEncoding =  bean.headers.get("Content-Encoding")
276     contentEncoding match {
277       case "gzip" =>
278         content = Source.fromInputStream(new GZIPInputStream(new ByteArrayInputStream(content))).mkString.getBytes
279       case "deflate" =>
280         content = Source.fromInputStream(new InflaterInputStream(new ByteArrayInputStream(content))).mkString.getBytes
281       case _ =>
282     }
283 
284     bean.put(SockoCommandBean.Content, content)
285     unmarshall(content, conType) match {
286       case Some(v) => bean.appendMap(Map(CommandBean.KeyEntity -> v))
287       case None => //ignore
288     }
289   }
290 
291   def unmarshall[T<:AnyRef:Manifest](data: Array[Byte], contentTypeValue: String = "application/json"): Option[T] = {
292     if (data == null || data.length == 0) {
293       None
294     } else {
295 
296       val (contentType, charset) = parseContentType(contentTypeValue)
297 
298       contentType match {
299 
300         case "application/x-www-form-urlencoded" | "multipart/form-data" =>
301           val dataString = new String(data, charset).dropWhile( _ == '?')
302           val jObj = JObject(new QueryStringDecoder("?" + dataString, charset).parameters.map { kvp =>
303             JField(kvp._1, if (kvp._2 != null) new JString(kvp._2.last) else JNull)
304             }.toList)
305           Some(jObj.extract[T])
306 
307         case "application/json" =>
308           Some(JsonParser.parse(new String(data, "utf-8")).extract[T])
309 
310         case _ =>
311           val dataString = new String(data, charset)
312           Some(JObject(List(JField("content", new JString(dataString)))).extract[T])
313       }
314     }
315   }
316 
317   def parseContentType(ct: String): (String, Charset) = {
318 
319     val parts = ct.split(";").toList.map{part => part.toLowerCase.trim}
320 
321     val contentType = if (parts.length > 0 && !parts(0).isEmpty) parts(0) else "text/plain"
322 
323     val charSet = if (parts.length >= 2 && parts(1).startsWith("charset=")) {
324       Charset.forName(parts(1).split("=")(1))
325     }
326     else {
327       CharsetUtil.UTF_8
328     }
329 
330     (contentType, charSet)
331   }
