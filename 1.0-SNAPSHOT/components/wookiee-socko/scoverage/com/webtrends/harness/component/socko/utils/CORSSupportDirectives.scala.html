26 trait CORSSupportDirectives extends EntityRoutes with Command {
27   /**
28    * Will add the CORS specific headers to the response if 'origin' is present on the request
29    */
30   override def getResponseHeaders(headers: ImmutableHttpHeaders): Map[String, List[SockoHeader]] = {
31     headers.get("Origin") match {
32       case None => super.getResponseHeaders(headers)
33       case Some(s) =>
34         Map("all" -> List(SockoHeader("Access-Control-Allow-Origin", if (s == "null") "*" else s"1.1://$s"), SockoHeader("Access-Control-Allow-Credentials", "true")))
35     }
36   }
37 }
