27 object SockoUtils {
28 
29   val KeyAllHeaders = "all"
30 
31   /**
32    * Based on the method this function will filter out only the required headers from the map
33    *
34    * @param method the method for the request
35    * @param headers the full list of headers
36    * @return
37    */
38   def toSockoHeaderFormat(method:String, headers:Map[String, List[SockoHeader]], size: Int = 0) : Map[String, String] = {
39     val retHeaders = (headers filter {
40       x => x._1.equalsIgnoreCase(method) || x._1 == KeyAllHeaders
41     } values).flatten.toList
42     toSockoHeaderFormat(retHeaders :+ SockoHeader("Content-Length", size + ""))
43   }
44 
45   /**
46    * Takes a list of SockoHeaders and converts it to a Map[String, String], which is what Socko expects when
47    * writing out the response
48    *
49    * @param headers the list of SockoHeaders
50    * @return
51    */
52   def toSockoHeaderFormat(headers:List[SockoHeader]) : Map[String, String] = (headers map { h => h.headerName -> h.headerValue }).toMap
53 
54   /**
55    * Does a little normalization of content type strings. So will convert stuff like JSON to application/json.
56    *
57    * @param contentType The content type you want normalized
58    * @return
59    */
60   def normalizeContentType(contentType:String) : String = {
61     // this should probably get a list of content types from a file and then normalize based on the file
62     if (contentType.equalsIgnoreCase("json")) {
63       "application/json"
64     } else if (contentType.equalsIgnoreCase("text")) {
65       "text/plain"
66     } else if (contentType.equalsIgnoreCase("html")) {
67       "text/html"
68     } else {
69       contentType
70     }
71   }
