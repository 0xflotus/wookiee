52 class CoreSprayClient extends HActor with HttpLiftSupport {
53   import context.dispatcher
54   implicit val system:ActorSystem = context.system
55 
56   // 60 seconds is the default that spray uses, so if the config value is not set we won't change the default
57   implicit val futureTimeout = ConfigUtil.getDefaultTimeout(context.system.settings.config, SprayManager.KeyHttpClientTimeout, 60 seconds)
58 
59   val pipeline:HttpRequest => Future[HttpResponse] = (
60     encode(Gzip)
61       ~> sendReceive(IO(UHttp))
62       ~> decode(Gzip)
63     )
64 
65   /**
66    * Will route messages to correct behavior, mainly will go to service
67    */
68   override def receive = super.receive orElse {
69     case HttpGet(config, path, headers) => service(config, GET, path, None, headers)
70     case HttpPost(config, path, body, headers) => service(config, POST, path, Some(body), headers)
71     case HttpPut(config, path, body, headers) => service(config, PUT, path, Some(body), headers)
72     case HttpDelete(config, path, headers) => service(config, DELETE, path, None, headers)
73     case HttpOptions(config, path, body, headers) => service(config, OPTIONS, path, Some(body), headers)
74     case HttpPatch(config, path, body, headers) => service(config, PATCH, path, Some(body), headers)
75     case HttpPing(config, timeout, path) => pipe(ping(config, timeout, path)) to sender
76   }
77 
78   /**
79    * Will service the request by routing
80    *
81    * @param config
82    * @param method
83    * @param path
84    * @param body
85    * @param headers
86    * @tparam T
87    */
88   // add marshallers based on the content-type in the headers
89   def service[T:Manifest](config:HttpConfig, method:HttpMethod, path:String, body:Option[T], headers:List[HttpHeader]) = {
90     implicit val HttpJsonMarshaller =
91       Marshaller.of[T](ContentTypes.`application/json`) { (value, contentType, ctx) =>
92         ctx.marshalTo(HttpEntity(contentType, compactRender(decompose(value))))
93       }
94     try {
95       val fullPath = config.fullPath(path)
96       val responseFuture = pipeline {
97         method match {
98           case GET =>
99             Get(fullPath).withHeaders(headers)
100           case POST =>
101             Post(fullPath, body).withHeaders(headers)
102           case PUT =>
103             Put(fullPath, body).withHeaders(headers)
104           case DELETE =>
105             Delete(fullPath).withHeaders(headers)
106           case OPTIONS =>
107             Options(fullPath, body).withHeaders(headers)
108           case PATCH =>
109             Patch(fullPath, body).withHeaders(headers)
110           case HEAD =>
111             Head(fullPath).withHeaders(headers)
112         }
113       }
114 
115       val caller = sender
116       responseFuture onComplete {
117         case Success(s) => caller ! HttpResp(s.entity, s.headers, s.status)
118         case Failure(f) =>
119           caller ! Status.Failure(HttpClientException("URL[%s] : %s".format(fullPath, f.getMessage), f))
120       }
121     } catch {
122       case NonFatal(e) =>
123         sender ! Status.Failure(HttpClientException(e.getMessage, e))
124         throw HttpClientException(e.getMessage, e)
125     }
126   }
127 
128   def shutdown = {
129     val f = IO(UHttp).ask(Http.CloseAll)(2 seconds)
130     Await.ready(f, 2 seconds)
131   }
132 
133   def ping(config:HttpConfig, timeoutValue:Int=1, pingPath:String="ping") : Future[ConnectionStatus] = {
134     implicit val timeout = Timeout(Duration(timeoutValue, "seconds"))
135     val p = Promise[ConnectionStatus]()
136     try {
137       val f = (self ? HttpGet(config, pingPath))(timeout)
138       f onComplete {
139         case Failure(f) =>
140           p success ConnectionStatus(false, "Could not connect to " + config.fullPath("ping") + " responded with exception " + f.getMessage)
141         case Success(s) =>
142           s.asInstanceOf[HttpResp].statusCode match {
143             case s:StatusCodes.Success => p success ConnectionStatus(true, "Connected successfully to " + config.fullPath(pingPath))
144             case _ => p success ConnectionStatus(false, "Could not connect to " + config.fullPath(pingPath))
145           }
146       }
147     } catch {
148       case e:Throwable =>
149         log.debug("Could not connect to " + config.fullPath(pingPath) + " responded with error " + e.getMessage)
150         p success ConnectionStatus(false, "Could not connect to " + config.fullPath(pingPath) + " responded with error " + e.getMessage)
151     }
152     p.future
153   }
