39 /**
40  * Config for any http request
41  *
42  * @param timeout default 5, implicitly set
43  * @param server server that you are posting the request too
44  * @param port port for that server, defaults to 80
45  * @param useSSL default false, if need ssl set to true
46  */
47 class HttpConfig(server:String, port:Int=80, useSSL:Boolean=false, timeout:Int=5) {
48   val pathString = "%s://%s:%d/%s"
49 
50   def fullPath(path:String) : String = {
51     val protocol = if (useSSL) "https" else "http"
52     pathString.format(protocol, server, port, path)
53   }
54 
55   def getTimeout = Timeout(Duration(timeout, "seconds"))
56 }
