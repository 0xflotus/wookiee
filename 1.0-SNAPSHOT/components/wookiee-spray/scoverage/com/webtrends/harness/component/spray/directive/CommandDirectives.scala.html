29 trait CommandDirectives extends BaseDirectives {
30 
31   /**
32    * Matches a string path like '/test/ping' to the current uri and returns
33    * a command bean if matched. Will also extract segments (designated with the
34    * first char value '$' in the test path), and place those values into the
35    * command bean with the key's defined in the path. So path variable $var1
36    * would be the key "var1"
37    *
38    * @param path the path that the user is trying to match.
39    * @return a commandbean containing any segments extracted from the url
40    */
41   def commandPath(path:String) : Directive1[CommandBean] = {
42     extract(_.request.uri.path.toString()) flatMap {
43       pa => Command.matchPath(path, pa) match {
44         case Some(s) => provide(s)
45         case None => reject
46       }
47     }
48   }
49 
50   /**
51    * Exactly like commandPath except it will evaluate over a set of paths and return
52    * the first matched path. There is no ordering, it is expected that each path in the
53    * set are mutually exclusive.
54    *
55    * @param paths The set of paths that a user wishes to match.
56    * @return
57    */
58   def commandPaths(paths:Map[String, String]) : Directive1[CommandBean] = {
59     extract(_.request.uri.path.toString()) flatMap {
60       pa =>
61         var matchFound = false
62         val retBean = new CommandBean()
63         paths.toStream.takeWhile(_ => !matchFound) foreach {
64           path => Command.matchPath(path._2, pa) match {
65             case Some(b) =>
66               b.addValue(CommandBean.KeyPath, path._1)
67               matchFound = true
68               retBean.appendMap(b.toMap)
69             case None => //ignore and continue
70           }
71         }
72         if (matchFound) {
73           provide(retBean)
74         } else {
75           reject
76         }
77     }
78   }
79 
80   /**
81    * Takes a map that adds specific headers to specific requests http method types
82    *
83    * @param headers map of headers http method -> list of headers
84    * @return
85    */
86   def mapHeaders(headers:Map[String, List[HttpHeader]]) : Directive0 = {
87     extract(_.request.method) flatMap {
88       method =>
89         val retHeaders = (headers filter {
90           x => x._1.equalsIgnoreCase(method.name) || x._1 == CommandDirectives.KeyAllHeaders
91         } values).flatten.toList
92         respondWithHeaders(retHeaders) hflatMap {
93           case _ => pass
94         }
95     }
96   }
97 }
