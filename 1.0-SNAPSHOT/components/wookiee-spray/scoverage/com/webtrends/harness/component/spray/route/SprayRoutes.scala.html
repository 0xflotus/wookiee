88   protected def innerExecute[T<:AnyRef:Manifest](bean:Option[CommandBean]=None) = {
89     parameterMap {
90       params =>
91         val updatedBean = bean match {
92           case Some(b) => b.appendMap(params); b
93           case None => CommandBean(params)
94         }
95         onComplete[CommandResponse[T]](execute(Some(updatedBean)).mapTo[CommandResponse[T]]) {
96           case Success(s) =>
97             s.data match {
98               case Some(data) =>
99                 val media = MediaTypes.forExtension(s.responseType) match {
100                   case Some(m) => m
101                   case None =>
102                     val mt = Try({
103                       val rt = s.responseType.split("/")
104                       MediaTypes.getForKey((rt(0), rt(1))).get
105                     }) recover {
106                       case _ => MediaTypes.`application/json`
107                     }
108                     mt getOrElse MediaTypes.`application/json`
109                 }
110                 respondWithMediaType(media) {
111                   complete { responseStatusCode -> data }
112                 }
113               case None =>
114                 complete(StatusCodes.NoContent)
115             }
116           case Failure(f) => throw f
117         }
118     }
119   }

134     }

240   /**
241    * Function will return all the allowed headers for the command. Basically if you mixin a trait like
242    * SprayGet, it will add the Get method to the allow header. This method should be overridden if you
243    * are using the SprayCustom trait as then you would be defining your own routes and methods.
244    *
245    * @return
246    */
247   private def getMethods = {
248     def getMethod[I<:SprayRoutes](klass:Class[I], method:HttpMethod) : Option[HttpMethod] = {
249       klass.isAssignableFrom(this.getClass) match {
250         case true => Some(method)
251         case false => None
252       }
253     }
254 
255     Seq[Option[HttpMethod]] (
256       getMethod(classOf[SprayGet], HttpMethods.GET),
257       getMethod(classOf[SprayHead], HttpMethods.HEAD),
258       getMethod(classOf[SprayPatch], HttpMethods.PATCH),
259       getMethod(classOf[SprayPut], HttpMethods.PUT),
260       getMethod(classOf[SprayPost], HttpMethods.POST),
261       getMethod(classOf[SprayOptions], HttpMethods.OPTIONS),
262       getMethod(classOf[SprayDelete], HttpMethods.DELETE)
263     ).flatten
264   }
