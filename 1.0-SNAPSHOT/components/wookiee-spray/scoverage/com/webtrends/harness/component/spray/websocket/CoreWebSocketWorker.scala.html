16   override def receive = health orElse handshaking orElse closeLogic
17 
18   var worker: Option[ActorRef] = None

20   override def handshaking: Receive = {
21     // when a client request for upgrading to websocket comes in, we send
22     // UHttp.Upgrade to upgrade to websocket pipelines with an accepting response.
23     case websocket.HandshakeRequest(state) =>
24       // Check to see if the uri has a registered worker
25       // if not close the connection
26       val path = state.request.uri.path.tail
27       val props = WebSocketManager.getWorker(path.toString())
28       props match {
29         case Some(p) =>
30           worker = Some(context.actorOf(p, URLEncoder.encode(path.toString(), "UTF-8")))
31           state match {
32             case wsFailure: websocket.HandshakeFailure => sender() ! wsFailure.response
33             case wsContext: websocket.HandshakeContext => sender() ! UHttp.UpgradeServer(websocket.pipelineStage(self, wsContext), wsContext.response)
34           }
35         case None =>
36           log.info(s"Received unsupported websocket connection request on [$path]")
37           context.stop(self)
38       }
39 
40     // upgraded successfully
41     case UHttp.Upgraded =>
42       context.become(health orElse businessLogic orElse closeLogic)
43       self ! websocket.UpgradedToWebSocket // notify Upgraded to WebSocket protocol
44   }
45 
46   def businessLogic: Receive = {
47     case Push(msg) =>
48       send(TextFrame(msg))
49     case ev: Http.ConnectionClosed =>
50       worker match {
51         case Some(w) =>
52           w ! ev
53         case None =>  // Error
54       }
55       context.stop(self)
56     case x =>
57       worker match {
58         case Some(w) =>
59           w.tell(x, sender)
60         case None =>   // Error
61       }
62   }
63 }
