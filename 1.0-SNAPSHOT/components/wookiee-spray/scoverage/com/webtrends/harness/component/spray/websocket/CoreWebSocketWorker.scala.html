19   override def receive = health orElse handshaking orElse closeLogic

21   var worker: Option[ActorRef] = None
22   var bean: Option[CommandBean] = None

24   override def handshaking: Receive = {
25     // when a client request for upgrading to websocket comes in, we send
26     // UHttp.Upgrade to upgrade to websocket pipelines with an accepting response.
27     case websocket.HandshakeRequest(state) =>
28       // Check to see if the uri has a registered worker
29       // if not close the connection
30       val path = state.request.uri.path
31       WebSocketManager.getWorker(path.toString()) match {
32         case Some((props, b)) =>
33           bean = b
34           worker = Some(context.actorOf(props, props.actorClass().getSimpleName))
35           state match {
36             case wsFailure: websocket.HandshakeFailure => sender() ! wsFailure.response
37             case wsContext: websocket.HandshakeContext => sender() ! UHttp.UpgradeServer(websocket.pipelineStage(self, wsContext), wsContext.response)
38           }
39         case None =>
40           log.info(s"Received unsupported websocket connection request on [$path]")
41           context.stop(self)
42       }
43 
44     // upgraded successfully
45     case UHttp.Upgraded =>
46     worker match {
47       case Some(w) =>
48         w.ask(SetBean(bean)) onComplete {
49           case Success (s) =>
50             context.become(health orElse businessLogic orElse closeLogic)
51             self ! websocket.UpgradedToWebSocket // notify Upgraded to WebSocket protocol
52           case Failure (f) =>
53             context.stop (self)
54         }
55       case None =>
56         log.error(s"Websocket upgraded with no worker, closing connection")
57         context.stop(self)
58     }
59   }
60 
61   def businessLogic: Receive = {
62     case Push(msg) =>
63       send(TextFrame(msg))
64     case ev: Http.ConnectionClosed =>
65       worker match {
66         case Some(w) =>
67           w ! ev
68         case None =>  // Error
69       }
70       context.stop(self)
71     case x =>
72       worker match {
73         case Some(w) =>
74           w.tell(x, sender)
75         case None =>   // Error
76       }
77   }
78 }
