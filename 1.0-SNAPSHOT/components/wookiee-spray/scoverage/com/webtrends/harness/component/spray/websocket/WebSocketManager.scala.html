39     externalLogger.debug(s"new websocket worker [$path] registered with WebSocketManager")
40     workers += path -> props
41   }

43   def removeCommand(path:String) : Boolean = {
44     workers.get(path) match {
45       case Some(n) =>
46         externalLogger.debug(s"Websocket worker [$path] removed from WebSocketManager")
47         workers -= path
48         true
49       case None => false
50     }
51   }

53   /**
54    * Gets a tuple containing the WebSocketWorker props and CommandBean for the given path.
55    *
56    * @param path The URI to match against
57    * @return If a match is found, a tuple contain the props and an optional CommandBean containing any parameters that
58    *         were part of the URI; otherwise None
59    */
60   def getWorker(path: String): Option[(Props, Option[CommandBean])] = {
61     workers.find({case (key,props) => matchPath(key,path)}) match {
62       case Some((key,props)) => Some(props, Command.matchPath(key, path))
63       case None => None
64     }
65   }
66 
67   private def matchPath(test: String, path:String): Boolean = {
68     Command.matchPath(test, path) match {
69       case Some(b) => true
70       case None => false
71     }
72   }
