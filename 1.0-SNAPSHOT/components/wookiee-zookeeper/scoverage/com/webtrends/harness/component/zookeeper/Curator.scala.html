31 private[zookeeper] class Curator(settings: ZookeeperSettings) extends LoggingAdapter {
32 
33   private[zookeeper] case class ProviderKey(basePath:String, key:String)
34 
35   // We manage an internal client since the CuratorFramework will not allow re-use of the client
36   // after it has been closed
37   private var internalClient: Option[CuratorFramework] = None
38   def client = internalClient.get
39 
40   // list of all discovery services by basepath
41   private val discoveries = mutable.Map[String, ServiceDiscovery[Void]]()
42   // we manage an internal list of all discoverable provider
43   private val providers = mutable.Map[ProviderKey, ServiceProvider[Void]]()
44 
45   private[zookeeper] def createClient: CuratorFramework = {
46     if (internalClient.isEmpty) {
47     internalClient = Some(CuratorFrameworkFactory.newClient(
48       settings.quorum,
49       settings.sessionTimeout.intValue,
50       settings.connectionTimeout.intValue, new RetryNTimes(
51         settings.retryCount,
52         settings.retrySleep.intValue)))
53 
54     }
55 
56     internalClient.get
57   }
58 
59   private[zookeeper] def start(listener: Option[ConnectionStateListener] = None): Unit = {
60     if (internalClient.isEmpty) {
61       createClient
62     }
63 
64     client.synchronized {
65       if (client.getState == CuratorFrameworkState.LATENT) {
66         log.info("Starting curator with quorum " + settings.quorum)
67         if (listener.isDefined) {
68           client.getConnectionStateListenable.addListener(listener.get)
69         }
70 
71         client.start
72         client.getZookeeperClient.blockUntilConnectedOrTimedOut
73       }
74     }
75   }
76 
77   private[zookeeper] def stop: Unit = {
78     log.debug("Stopping curator")
79     internalClient match {
80       case Some(c) if c.getState == CuratorFrameworkState.STARTED =>
81         c.close()
82         internalClient = None
83       case _ =>
84     }
85     discoveries.foreach(x => x._2.close())
86   }
87 
88   def discovery(basePath:String): ServiceDiscovery[Void] = {
89     val bp = getBasePath(basePath)
90     if (discoveries.contains(bp)) {
91       discoveries(bp)
92     } else {
93       val discovery = ServiceDiscoveryBuilder.builder(classOf[Void])
94         .client(client)
95         .basePath(bp)
96         .build()
97       discovery.start()
98       discoveries.put(bp, discovery)
99       discovery
100     }
101   }
102 
103   def createServiceProvider(basePath:String, name:String) : ServiceProvider[Void] = {
104     val bp = getBasePath(basePath)
105     val key = ProviderKey(bp, name)
106     if (providers.contains(key)) {
107       providers(key)
108     } else {
109       val provider = discovery(bp).serviceProviderBuilder().serviceName(name).build()
110       provider.start()
111       providers.put(key, provider)
112       provider
113     }
114   }
115 
116   def getServiceProviderDetails(name:Option[String]=None) : Map[ProviderKey, Iterable[ServiceInstance[Void]]] = {
117     val filteredMap = name match {
118       case Some(n) => providers.filter(k => k._1.equals(n))
119       case None => providers
120     }
121     filteredMap.map {
122       k => k._1 -> collectionAsScalaIterable(k._2.getAllInstances)
123     }.toMap
124   }
125 
126   def registerService(basePath:String, instance:ServiceInstance[Void]) = {
127     val bp = getBasePath(basePath)
128     discovery(bp).registerService(instance)
129     // create a provider for the service if one has not already been created for it
130     createServiceProvider(bp, instance.getName)
131   }
132 
133   private def getBasePath(basePath:String) : String = s"/${settings.dataCenter}/${settings.pod}${basePath}"
134 }

136 /**
137  * This allows access to a single instance of the curator client per quorum when an application
138  * needs direct access. It is still preferred for one to use the ZookeeperAdapter
139  * and ZookeeperEventAdapter traits.
140  */
141 object Curator {
142 
143   private val curators = mutable.Map.empty[String, Curator]
144 
145   /**
146    * Return the instance of the Curator client object. This is usually used after the harness has
147    * called the apply method that takes the required settings.
148    * @return the singleton instance of the Curator
149    */
150   def apply(settings: ZookeeperSettings): Curator = {
151     curators.synchronized {
152       curators.getOrElseUpdate(settings.quorum, new Curator(settings))
153     }
154   }
