29 private[zookeeper] class Curator(settings: ZookeeperSettings) extends LoggingAdapter {
30 
31   // We manage an internal client since the CuratorFramework will not allow re-use of the client
32   // after it has been closed
33   private var internalClient: Option[CuratorFramework] = None
34 
35   def client = internalClient.get
36 
37   private[zookeeper] def createClient: CuratorFramework = {
38     if (internalClient.isEmpty) {
39     internalClient = Some(CuratorFrameworkFactory.newClient(
40       settings.quorum,
41       settings.sessionTimeout.intValue,
42       settings.connectionTimeout.intValue, new RetryNTimes(
43         settings.retryCount,
44         settings.retrySleep.intValue)))
45 
46     }
47 
48     internalClient.get
49   }
50 
51   private[zookeeper] def start(listener: Option[ConnectionStateListener] = None): Unit = {
52     if (internalClient.isEmpty) {
53       createClient
54     }
55 
56     client.synchronized {
57       if (client.getState == CuratorFrameworkState.LATENT) {
58         log.info("Starting curator with quorum " + settings.quorum)
59         if (listener.isDefined) {
60           client.getConnectionStateListenable.addListener(listener.get)
61         }
62 
63         client.start
64         client.getZookeeperClient.blockUntilConnectedOrTimedOut
65       }
66     }
67   }
68 
69   private[zookeeper] def stop: Unit = {
70     log.debug("Stopping curator")
71     client.close
72     internalClient = None
73   }
74 }

76 /**
77  * This allows access to a single instance of the curator client per quorum when an application
78  * needs direct access. It is still preferred for one to use the ZookeeperAdapter
79  * and ZookeeperEventAdapter traits.
80  */
81 object Curator {
82 
83   private val curators = mutable.Map.empty[String, Curator]
84 
85   /**
86    * Return the instance of the Curator client object. This is usually used after the harness has
87    * called the apply method that takes the required settings.
88    * @return the singleton instance of the Curator
89    */
90   def apply(settings: ZookeeperSettings): Curator = {
91     curators.synchronized {
92       curators.getOrElseUpdate(settings.quorum, new Curator(settings))
93     }
94   }
95
