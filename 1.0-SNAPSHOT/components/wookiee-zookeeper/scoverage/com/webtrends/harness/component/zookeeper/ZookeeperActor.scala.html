147   private def getChildren(path: String, includeData: Boolean, namespace: Option[String]) = {
148     try {
149       val nodes = for {
150         child <- getClientContext(namespace).getChildren.forPath(path)
151         data = if (includeData) Some(getClientContext(namespace).getData.forPath(s"${path}/${child}")) else None
152       } yield (child, data)
153       sender() ! nodes
154     }
155     catch {
156       case e: Exception =>
157         log.error(e, "An error occurred trying to fetch children from the path {}", path)
158         sender() ! Status.Failure(e)
159     }
160   }

162   private def getData(path: String, namespace: Option[String]) = {
163     try {
164       sender() ! getClientContext(namespace).getData.forPath(path)
165     }
166     catch {
167       case nn: NoNodeException =>
168         log.error("No node found for path {}", path)
169         sender() ! Status.Failure(nn)
170       case e: Exception =>
171         log.error(e, "An error occurred trying to fetch data from the path {}", path)
172         sender() ! Status.Failure(e)
173     }

176   private def setData(path: String, data: Array[Byte], create: Boolean, ephemeral: Boolean, namespace: Option[String], async: Boolean) = {
177     def nodeCreate: String = {
178       try {
179         val mode = if (ephemeral) CreateMode.EPHEMERAL else CreateMode.PERSISTENT
180         getClientContext(namespace).create.creatingParentsIfNeeded.withMode(mode).forPath(path)
181       } catch {
182         case ne: NodeExistsException => path
183         case e: Exception =>
184           log.error(e, "An error occurred trying to create a node for the path {}", path)
185           path
186       }
187     }
188 
189     val dataPath = if (create) nodeCreate else path
190 
191     try {
192       if (async) {
193         getClientContext(namespace).setData().inBackground(callback).forPath(dataPath, data)
194       } else {
195         getClientContext(namespace).setData().forPath(dataPath, data)
196         sender() ! data.length
197       }
198     } catch {
199       case e: Exception =>
200         log.error(e, "An error occurred trying to set data for the path {}", path)
201         if (!async) sender() ! Status.Failure(e)
202     }
203   }

263   protected def handleStateChange(state: ConnectionState) = {
264     log.info("Zookeeper connection state changed to {}", state.name)
265     currentState = state
266 
267     if (currentState == ConnectionState.RECONNECTED || currentState == ConnectionState.CONNECTED) {
268       unstashAll()
269       context.become(processing)
270     }
271     else {
272       context.become(initializing)
273     }
274 
275     // Now update all of the registered listeners
276     stateRegistrars foreach {
277       act => act ! ZookeeperStateEvent(currentState)
278     }
279 
280     state match {
281       case ConnectionState.RECONNECTED =>
282         // Re-register our self
283         self ! RegisterNode
284       case ConnectionState.CONNECTED =>
285         // Re-register our self
286         self ! RegisterNode
287       case _ => // Do nothing
288     }
289   }

291   private def registerForEvents(reg: RegisterZookeeperEvent) = {
292     reg.to match {
293       case ZookeeperStateEventRegistration(registrar) =>
294         stateRegistrars += registrar
295         // Send the current state of ZK to the registrar
296         registrar ! ZookeeperStateEvent(currentState)
297 
298       case ZookeeperChildEventRegistration(registrar, path, optNamespace) =>
299         // Update the child registrations
300         childRegistrars.get((path, optNamespace)).map {
301           entry =>
302             if (!entry.registrars.contains(registrar)) {
303               childRegistrars.updated((path, optNamespace), entry.copy(registrars = entry.registrars ++ Set(registrar)))
304             }
305         }.getOrElse {
306           Try({
307             // Adding a new cache
308             val cache = new PathChildrenCache(getClientContext(optNamespace), path, true)
309             cache.getListenable.addListener(this)
310             cache.start(StartMode.POST_INITIALIZED_EVENT)
311             childRegistrars += ((path, optNamespace) -> CacheEntry(cache, Set(registrar)))
312           }).recover({
313             case e: Exception => log.error(s"An error occurred trying to create a path listener for the path $path", e)
314           })
315         }
316 
317       case ZookeeperLeaderEventRegistration(registrar, path, optNamespace) =>
318         // Update the leadership registrations
319         leadershipRegistrars.get((path, optNamespace)).map {
320           entry =>
321             if (!entry.registrars.contains(registrar)) {
322               leadershipRegistrars.updated((path, optNamespace), entry.copy(registrars = entry.registrars ++ Set(registrar)))
323             }
324         }.getOrElse {
325           Try({
326             // Adding a new leader selector
327             val sel = new LeaderLatch(getClientContext(optNamespace), path)
328             sel.addListener(new LeaderListener(path, self, optNamespace))
329             sel.start
330             leadershipRegistrars += ((path, optNamespace) -> LeaderEntry(sel, Set(registrar)))
331           }).recover({
332             case e: Exception => log.error(s"An error occurred trying to create a leader selector for the path $path", e)
333           })
334         }
335     }
336 
337   }

339   private def unregisterForEvents(reg: UnregisterZookeeperEvent) = {
340     reg.to match {
341       case ZookeeperStateEventRegistration(registrar) => stateRegistrars -= registrar
342       case ZookeeperChildEventRegistration(registrar, path, optNamespace) =>
343         childRegistrars.get((path, optNamespace)).map {
344           entry =>
345             if (entry.registrars.contains(registrar)) {
346               val newEntry = ((path, optNamespace) -> entry.copy(registrars = entry.registrars.filterNot(_ == registrar)))
347               if (newEntry._2.registrars.isEmpty) {
348                 // No more registrars so we can shutdown the cache
349                 entry.cache.getListenable.removeListener(this)
350                 entry.cache.close
351                 val key = (path, optNamespace)
352                 childRegistrars -= key
353               }
354               else {
355                 childRegistrars += newEntry
356               }
357             }
358         }
359 
360       case ZookeeperLeaderEventRegistration(registrar, path, optNamespace) =>
361         leadershipRegistrars.get((path, optNamespace)).map {
362           entry =>
363             if (entry.registrars.contains(registrar)) {
364               val newEntry = ((path, optNamespace) -> entry.copy(registrars = entry.registrars.filterNot(_ == registrar)))
365               if (newEntry._2.registrars.isEmpty) {
366                 // No more registrars so we can shutdown the leader selector
367                 entry.leader.close
368                 val key = (path, optNamespace)
369                 leadershipRegistrars -= key
370               }
371               else {
372                 leadershipRegistrars += newEntry
373               }
374             }
375         }
376     }

400     }
401     else {
402       s"We are currently not able to connect to the zookeeper quorum on ${settings.quorum}"
403     }
404 
405     Future {
406       new HealthComponent("zookeeper", zookeeperState, details = msg, extra = None)
407     }

448    * Curator handler for when a child node changes for a path that we are watching
449    * @param curator
450    * @param event
451    */
452   def childEvent(curator: CuratorFramework, event: PathChildrenCacheEvent): Unit = {
453     // Ignore any initialization events
454     if (event != null && (event.getType == PathChildrenCacheEvent.Type.CHILD_ADDED ||
455       event.getType == PathChildrenCacheEvent.Type.CHILD_REMOVED ||
456       event.getType == PathChildrenCacheEvent.Type.CHILD_UPDATED)) {
457 
458       val path = event.getData.getPath
459       log.debug(s"Zookeeper child event for path ${path}")
460 
461       // Cycle through the paths to determine if any start with this change
462       for (entry <- childRegistrars) {
463         if (path.startsWith(entry._1._1)) {
464           // Notify the registered listeners
465           for (act <- entry._2.registrars) act ! ZookeeperChildEvent(event)
466         }
467       }
468     }
469   }
