123     // Set the data for the given path
124     case SetPathData(path, data, create, ephemeral, optNamespace, async) => setData(path, data, create, ephemeral, optNamespace, async)

129     // Get children for the given path
130     case GetPathChildren(path, includeData, optNamespace) => getChildren(path, includeData, optNamespace)
131     // Check if a node exists
132     case GetNodeExists(path, optNamespace) => nodeExists(path, optNamespace)
133     // Create a node
134     case CreateNode(path, ephemeral, data, optNamespace) => createNode(path, ephemeral, data, optNamespace)
135     // Delete a node
136     case DeleteNode(path, optNamespace) => deleteNode(path, optNamespace)
137     // query for service names
138     case QueryForNames(basePath) => queryForNames(basePath)
139     // query for service instances
140     case QueryForInstances(basePath, name, id) => queryForInstances(basePath, name, id)
141     // make service discoverable
142     case MakeDiscoverable(basePath, id, name, port, uriSpec) => makeDiscoverable(basePath, id, name, port, uriSpec)
143     // get a single instance from the provider
144     case GetInstance(basePath, name) => getInstance(basePath, name)
145     // get all the instances from the provider
146     case GetAllInstances(basePath, name) => getAllInstances(basePath, name)
147     // Registration messages
148     case r: RegisterZookeeperEvent => registerForEvents(r)

150     case GetLeaderRegistrars(path, optNamespace) => sender() ! leadershipRegistrars.get((path, optNamespace)).map {
151       _.registrars
152     }.getOrElse(Nil)
153     case RegisterNode =>
154       // Re-register our self

192   private def setData(path: String, data: Array[Byte], create: Boolean, ephemeral: Boolean, namespace: Option[String], async: Boolean) = {
193     def nodeCreate: String = {
194       try {
195         val mode = if (ephemeral) CreateMode.EPHEMERAL else CreateMode.PERSISTENT
196         getClientContext(namespace).create.creatingParentsIfNeeded.withMode(mode).forPath(path)
197       } catch {
198         case ne: NodeExistsException => path
199         case e: Exception =>
200           log.error(e, "An error occurred trying to create a node for the path {}", path)
201           path
202       }
203     }
204 
205     val dataPath = if (create) nodeCreate else path
206 
207     try {
208       if (async) {
209         getClientContext(namespace).setData().inBackground(callback).forPath(dataPath, data)
210       } else {
211         getClientContext(namespace).setData().forPath(dataPath, data)
212         sender() ! data.length
213       }
214     } catch {
215       case e: Exception =>
216         log.error(e, "An error occurred trying to set data for the path {}", path)
217         if (!async) sender() ! Status.Failure(e)
218     }
219   }

346   protected def handleStateChange(state: ConnectionState) = {
347     log.info("Zookeeper connection state changed to {}", state.name)
348     currentState = state
349 
350     if (currentState == ConnectionState.RECONNECTED || currentState == ConnectionState.CONNECTED) {
351       unstashAll()
352       context.become(processing)
353     }
354     else {
355       context.become(initializing)
356     }
357 
358     // Now update all of the registered listeners
359     stateRegistrars foreach {
360       act => act ! ZookeeperStateEvent(currentState)
361     }
362 
363     state match {
364       case ConnectionState.RECONNECTED =>
365         // Re-register our self
366         self ! RegisterNode
367       case ConnectionState.CONNECTED =>
368         // Re-register our self
369         self ! RegisterNode
370       case _ => // Do nothing
371     }
372   }

374   private def registerForEvents(reg: RegisterZookeeperEvent) = {
375     reg.to match {
376       case ZookeeperStateEventRegistration(registrar) =>
377         stateRegistrars += registrar
378         // Send the current state of ZK to the registrar
379         registrar ! ZookeeperStateEvent(currentState)
380 
381       case ZookeeperChildEventRegistration(registrar, path, optNamespace) =>
382         // Update the child registrations
383         childRegistrars.get((path, optNamespace)).map {
384           entry =>
385             if (!entry.registrars.contains(registrar)) {
386               childRegistrars.updated((path, optNamespace), entry.copy(registrars = entry.registrars ++ Set(registrar)))
387             }
388         }.getOrElse {
389           Try({
390             // Adding a new cache
391             val cache = new PathChildrenCache(getClientContext(optNamespace), path, true)
392             cache.getListenable.addListener(this)
393             cache.start(StartMode.POST_INITIALIZED_EVENT)
394             childRegistrars += ((path, optNamespace) -> CacheEntry(cache, Set(registrar)))
395           }).recover({
396             case e: Exception => log.error(s"An error occurred trying to create a path listener for the path $path", e)
397           })
398         }
399 
400       case ZookeeperLeaderEventRegistration(registrar, path, optNamespace) =>
401         // Update the leadership registrations
402         leadershipRegistrars.get((path, optNamespace)).map {
403           entry =>
404             if (!entry.registrars.contains(registrar)) {
405               leadershipRegistrars.updated((path, optNamespace), entry.copy(registrars = entry.registrars ++ Set(registrar)))
406             }
407         }.getOrElse {
408           Try({
409             // Adding a new leader selector
410             val sel = new LeaderLatch(getClientContext(optNamespace), path)
411             sel.addListener(new LeaderListener(path, self, optNamespace))
412             sel.start
413             leadershipRegistrars += ((path, optNamespace) -> LeaderEntry(sel, Set(registrar)))
414           }).recover({
415             case e: Exception => log.error(s"An error occurred trying to create a leader selector for the path $path", e)
416           })
417         }
418     }
419 
420   }

422   private def unregisterForEvents(reg: UnregisterZookeeperEvent) = {
423     reg.to match {
424       case ZookeeperStateEventRegistration(registrar) => stateRegistrars -= registrar
425       case ZookeeperChildEventRegistration(registrar, path, optNamespace) =>
426         childRegistrars.get((path, optNamespace)).map {
427           entry =>
428             if (entry.registrars.contains(registrar)) {
429               val newEntry = ((path, optNamespace) -> entry.copy(registrars = entry.registrars.filterNot(_ == registrar)))
430               if (newEntry._2.registrars.isEmpty) {
431                 // No more registrars so we can shutdown the cache
432                 entry.cache.getListenable.removeListener(this)
433                 entry.cache.close
434                 val key = (path, optNamespace)
435                 childRegistrars -= key
436               }
437               else {
438                 childRegistrars += newEntry
439               }
440             }
441         }
442 
443       case ZookeeperLeaderEventRegistration(registrar, path, optNamespace) =>
444         leadershipRegistrars.get((path, optNamespace)).map {
445           entry =>
446             if (entry.registrars.contains(registrar)) {
447               val newEntry = ((path, optNamespace) -> entry.copy(registrars = entry.registrars.filterNot(_ == registrar)))
448               if (newEntry._2.registrars.isEmpty) {
449                 // No more registrars so we can shutdown the leader selector
450                 entry.leader.close
451                 val key = (path, optNamespace)
452                 leadershipRegistrars -= key
453               }
454               else {
455                 leadershipRegistrars += newEntry
456               }
457             }
458         }
459     }

535   /**
536    * Curator handler for when a child node changes for a path that we are watching
537    * @param curator
538    * @param event
539    */
540   def childEvent(curator: CuratorFramework, event: PathChildrenCacheEvent): Unit = {
541     // Ignore any initialization events
542     if (event != null && (event.getType == PathChildrenCacheEvent.Type.CHILD_ADDED ||
543       event.getType == PathChildrenCacheEvent.Type.CHILD_REMOVED ||
544       event.getType == PathChildrenCacheEvent.Type.CHILD_UPDATED)) {
545 
546       val path = event.getData.getPath
547       log.debug(s"Zookeeper child event for path ${path}")
548 
549       // Cycle through the paths to determine if any start with this change
550       for (entry <- childRegistrars) {
551         if (path.startsWith(entry._1._1)) {
552           // Notify the registered listeners
553           for (act <- entry._2.registrars) act ! ZookeeperChildEvent(event)
554         }
555       }
556     }
557   }
