26    */
27   def executeDiscoverableCommand[T](basePath:String, name:String, bean:Option[CommandBean]=None)
28                                    (implicit timeout:Timeout) : Future[CommandResponse[T]]= {
29     val p = Promise[CommandResponse[T]]
30     initCommandManager onComplete {
31       case Success(_) =>
32         commandManager match {
33           case Some(cm) =>
34             getInstance(basePath, name) onComplete {
35               case Success(in) =>
36                 (cm ? ExecuteRemoteCommand(name, in.getAddress, in.getPort, bean))(timeout).mapTo[CommandResponse[T]] onComplete {
37                   case Success(s) => p success s
38                   case Failure(f) => p failure CommandException("CommandManager", f)
39                 }
40               case Failure(f) => p failure CommandException("CommandManager", f)
41             }
42           case None => p failure CommandException("CommandManager", "CommandManager not found!")
43         }
44       case Failure(f) => p failure f
45     }
46     p.future
47   }
