44   /**
45    * Restart the actor system
46    */
47   def restartActorSystem = {
48     log.get.info("Restarting the actor system")
49     system match {
50       case Some(sys) =>
51         shutdownActorSystem(false) {
52           startActorSystem()
53         }
54       case _ =>
55         externalLogger.info("There is no actor system so starting up now")
56         startActorSystem()
57     }
58   }
59 
60   /**
61    * Force a shutdown of the ActorSystem and the application's process.
62    */
63   def shutdown = {
64     log.get.info("Shutting down the harness")
65     new Thread("lifecycle") {
66       override def run() {
67         Thread.sleep(10)
68         shutdownActorSystem(false) {
69           System.exit(0)
70         }
71       }
72     }.start()
73   }

75   /**
76    * Start the actor system
77    */
78   def startActorSystem(config:Option[Config]=None) = {
79     try {
80       externalLogger.info("Creating the actor system")
81       system = config match {
82         case None => Some(HarnessActorSystem())
83         case c => Some(HarnessActorSystem(c))
84       }
85       // add the unhandled message listener so we can debug messages easily that are not being handled
86       val listener = system.get.actorOf(Props(new UnhandledEventListener))
87       system.get.eventStream.subscribe(listener, classOf[UnhandledMessage])
88 
89       log = Some(Logger(this.getClass, system.get))
90       log.get.info("Creating main harness actor")
91       implicit val sys = system.get
92       rootActor = Some(system.get.actorOf(HarnessActor.props, "system"))
93       log.get.info("Harness Started")
94     }
95     catch {
96       case t: Throwable =>
97         externalLogger.error(s"The actor system could not be started: ${t.getMessage}", t)
98         sys.exit(0)
99     }
100   }
101 
102   /**
103    * Shutdown the actor system
104    */
105   def shutdownActorSystem(block: Boolean)(f: => Unit) = {
106     log.get.info("Shutting down the main actor")
107     import scala.concurrent.ExecutionContext.Implicits.global
108 
109     // We will tell the main actor that we are shutting down. This allows it to shutdown
110     // its children and perform any needed cleanup.
111     val fut = gracefulStop(rootActor.get, 15 seconds, ShutdownSystem)
112     fut.onComplete {
113       case Success(shutdown) =>
114         log.get.info("Now shutting down the the system itself")
115         system.get.shutdown
116         // Wait for termination if it is not already complete
117         system.get.awaitTermination()
118 
119         // Set our flags
120         system = None
121         log = None
122         rootActor = None
123         externalLogger.info("The actor system has terminated")
124         // Call the passed function
125         f
126       case Failure(reason) =>
127         log.get.error("We were unable to properly shutdown the main actor", reason)
128         System.exit(0)
129     }
130 
131     if (block) {
132       Await.result(fut, 15 seconds)
133     }
134   }
135 
136   /**
137    * Add a shutdown hook so when the process is shut down that we cleanup cleanly
138    */
139   def addShutdownHook: Unit = {
140     Runtime.getRuntime.addShutdownHook(new Thread(new Runnable {
141       def run = {
142         system match {
143           case Some(sys) =>
144             sys.log.info("The shutdown hook has been called")
145             shutdownActorSystem(true) {
146               externalLogger.info("Successfully shut down")
147             }
148           case _ => externalLogger.info("Successfully shut down")
149 
150         }
151       }
152     }))
153   }
