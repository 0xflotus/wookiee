36   def getConfig(config:Option[Config]): Config = {
37     val sysConfig = {
38       if (config.isDefined) {
39         config.get
40       } else {
41         val baseConfig = ConfigFactory.load(loader, "conf/application.conf")
42         ConfigFactory.load(loader).withFallback(baseConfig).getConfig("wookie-system")
43       }
44     }
45 
46     ComponentManager.loadComponentJars(sysConfig, loader)
47     ConfigFactory.load
48 
49     externalLogger.info("Loading the service configs")
50     val configs = ServiceManager.loadConfigs(sysConfig)
51     externalLogger.info(s"${configs.size} service config(s) have been loaded: ${configs.mkString(", ")}")
52 
53     externalLogger.info("Loading the component configs")
54     val compConfigs = ComponentManager.loadComponentInfo(sysConfig)
55     externalLogger.info(s"${compConfigs.size} component config(s) have been loaded: ${compConfigs.mkString(", ")}\nIf 0 could be due to config loaded from component JARs.")
56 
57     val allConfigs = configs ++ compConfigs
58 
59     // Build the hierarchy
60     val conf = allConfigs.isEmpty match {
61       case true => sysConfig // No service configs
62       case false => allConfigs.reduce(_.withFallback(_)).withFallback(sysConfig)
63     }
64 
65     conf.resolve()
66   }
