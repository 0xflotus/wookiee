48     // First, check if the class has already been loaded

97   private def loadClassFromChildren(name: String, resolve: Boolean): Option[Class[_]] = {
98     if (childLoaders.isEmpty) {
99       None
100     }
101     else {
102       this.synchronized {
103         // Get the loaded class
104         childLoaders.filterNot(_.getLoadedClass(name) == None) match {
105           case Nil =>
106             var ret: Option[Class[_]] = None
107             for (value <- childLoaders; if ret == None) {
108               ret = value.loadClassLocally(name, resolve)
109             }
110             ret
111           case list => // Return the first if already loaded
112             list.headOption.get.getLoadedClass(name)
113         }
114       }
115     }
116   }
117 
118   private def getResourceFromChildren(name: String): URL = {
119     if (childLoaders.isEmpty) {
120       null
121     }
122     else {
123       (for {
124         value <- childLoaders
125         url = value.findResource(name)
126         if (url != null)
127       } yield url).headOption match {
128         case Some(url) => url
129         case None => null
130       }
131     }
132   }
133 
134   private def getResourcesFromChildren(name: String): java.util.Enumeration[URL] = {
135     Try(super.getResources(name)).get
136   }
137 
138   private def getResourceAsStreamFromChildren(name: String): InputStream = {
139     if (childLoaders.isEmpty) {

141     }

146         if (url != null)
147       } yield url).headOption match {
148         case Some(url) => url
149         case None => null
150       }
151     }
152   }
