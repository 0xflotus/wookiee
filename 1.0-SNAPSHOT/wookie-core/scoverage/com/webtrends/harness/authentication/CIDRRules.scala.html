33 case class CIDRRules(cidrAllow:Seq[String], cidrDeny:Seq[String]) {
34 
35   private val log = Logger.getLogger(getClass)
36 
37   private lazy val allow = { cidrAllow map (x => new SubnetUtils(x.trim).getInfo) }
38   private lazy val deny = { cidrDeny map (x => new SubnetUtils(x.trim).getInfo) }
39 
40   def checkCidrRules(ip:InetAddress) : Boolean = {
41     try {
42       deny.count(_.isInRange(ip.getHostAddress)) match {
43         case 0 if allow.length > 0 =>
44           // No denies, but we must match against allows
45           allow.count(_.isInRange(ip.getHostAddress)) match {
46             case 0 =>
47               // Not explicitly denied, but no allow matches
48               false
49             case _ =>
50               // Not denies and matches an allow
51               true
52           }
53         case 0 =>
54           // No denies and no allows to match against
55           true
56         case _ =>
57           // Denied
58           false
59       }
60     } catch {
61       case iae:IllegalArgumentException =>
62         log.info("CIDR rules do not support IPv6 yet", iae)
63         false
64     }
65   }
66 }

74   }
