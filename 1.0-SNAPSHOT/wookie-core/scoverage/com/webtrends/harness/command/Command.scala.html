27 /**
28  * A command should be a business logic that handles messages from some source (like http) then executes
29  * the logic that is defined for the end point and returns the response. It should not ever deal with the
30  * underlying messaging protocol but rather focus exclusively on the business logic. It should also not
31  * deal with any storage or caching or anything like that.
32  *
33  * @author Michael Cuthbert on 12/1/14.
34  */
35 trait Command extends HActor {
36   import context.dispatcher
37 
38   override def receive = health orElse ({
39     case ExecuteCommand(name, bean) => pipe(execute(bean)) to sender
40     case _ => // ignore all other messages to this actor
41   } : Receive)
42 
43   def path : String = s"_wt_internal/${commandName.toLowerCase}"
44 
45   /**
46    * This allows for extra functionality in your command path. Basically it will allow
47    * for multiple paths to be matched to a single command, and will map each path to a
48    * given name so that you can find out which path was mapped in the end. By default it
49    * will simply be a map of 1 element being retrieved from the default "path" function.
50    * So if you only require a single path you wouldn't worry about this function at all
51    *
52    * @return
53    */
54   def paths : Map[String, String] =  Map("default" -> path)
55 
56   /**
57    * Name of the command that will be used for the actor name
58    *
59    * @return
60    */
61   def commandName : String
62 
63   /**
64    * The primary entry point for the command, the actor for this command
65    * will ignore all other messaging and only execute through this
66    *
67    * @return
68    */
69   def execute[T](bean:Option[CommandBean]=None) : Future[CommandResponse[T]]
70 }

72 object Command {
73 
74   /**
75    * Checks the match between a test path and url path
76    *
77    * @param test The test path, like /test/$var1/ping
78    * @param uri The uri, like /test/1/ping
79    * @return Will return a command bean if matched, None if not and in the command bean the
80    *         key var1 will equal 1 as per the example above
81    */
82   def matchPath(test:String, uri:String) : Option[CommandBean] = {
83     val bean = new CommandBean()
84     val testPath = test.split("/") filter { x => x.nonEmpty }
85     val urlPath = uri.split("/") filter { x => x.nonEmpty }
86 
87     val m = urlPath.corresponds(testPath) {
88       // case for grabbing variable from the url
89       case (x, y) if y.charAt(0) == '$' =>
90         // try to normalize the segment into an INT or STRING
91         val key = y.substring(1)
92         x forall {_.isDigit} match {
93           case true => bean.addValue(key, x.toInt.asInstanceOf[Integer])
94           case false => bean.addValue(key, x)
95         }
96         true
97       // case if you want optional path values
98       case (x, y) if y.contains("|") =>
99         val matches = y.split("\\|") flatMap {
100           _ == x match {
101             case true => Some(true)
102             case false => None
103           }
104         } groupBy(_ == true)
105         // the flatMap and groupBy function will return a map with a single element pointing to all the
106         // matched path elements, all false matches will be thrown out. So if there is no matches the
107         // size of the map would be 0 otherwise it would be 1
108         matches.size == 1
109       // standard case
110       case (x, y) => x == y
111     }
112 
113     m match {
114       case true => Some(bean)
115       case false => None
116     }
117   }
118 }
