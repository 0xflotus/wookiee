39 /**
40  * Each system component needs to extend this class so that it is loaded up correctly
41  */
42 abstract class Component(name:String) extends HActor with ComponentHelper {
43 
44   import context.dispatcher
45   lazy val config = context.system.settings.config
46 
47   protected def defaultChildName:Option[String] = None
48 
49   /**
50    * Components will typically receive a couple type of messages
51    * 1. Start - Will execute after all components have loaded
52    * 2. Stop - Will execute prior to shutdown of the harness
53    *
54    * @return
55    */
56   override def receive = health orElse {
57     case StartComponent => start
58     case StopComponent => stop
59     case ConfigChange => log.info("No special config change functionality for this component.")
60     case ComponentRequest(msg, name, timeout) =>
61       val caller = sender
62       getChildActor(name) match {
63         case Some(a) => (a ? msg)(timeout) onComplete {
64           case Success(s) => caller ! ComponentResponse(s)
65           case Failure(f) => caller ! Status.Failure(f)
66         }
67         case None => sender ! Status.Failure(ComponentException(this.getClass.getName, s"$name actor not found"))
68       }
69     case ComponentMessage(msg, name) =>
70       getChildActor(name) match {
71         case Some(a) => a ! msg
72         case None => log.warn("Failed to send message to child actor [$name] for Component")
73       }
74   }
75 
76   /**
77    * Gets the child actor based on a name, if name is None then checks the default child name,
78    * if default child name is none then if will check to see if there is only a single child
79    * and return that single child
80    *
81    * @param name
82    * @return
83    */
84   protected def getChildActor(name:Option[String]) : Option[ActorRef] = {
85     name match {
86       case Some(ComponentManager.ComponentRef) => Some(self)
87       case Some(n) => context.child(n)
88       case None =>
89         defaultChildName match {
90           case Some(dn) => context.child(dn)
91           case None =>
92             if (context.children.size == 1) {
93               Some(context.children.head)
94             } else {
95               None
96             }
97         }
98     }
99   }
100 
101   /**
102    * Starts the component
103    */
104   def start = {
105     // after completion of this, we need to send the started message from the component
106     context.parent ! ComponentStarted(self.path.name)
107   }
108 
109   /**
110    * Any logic to stop the component
111    */
112   def stop = {}
113 }
114 
115 object Component {
116   def getActorPath() : String = {
117     s"${HarnessConstants.ComponentName}/"
118   }
