32 /**
33  * This is a helper class that enables developers who use to trait to interact with the ComponentManager
34  * easily on the harness
35  */
36 trait ComponentHelper {
37   this: Actor =>
38 
39   import context.dispatcher
40   var componentManagerInitialized = false
41   var componentManager:Option[ActorRef] = None
42 
43   /**
44    * This should only be called once to initialize the component manager actor. Will retry when called
45    * as many methods require this to work.
46    */
47   def initComponentHelper : Future[ActorRef] = {
48     val p = Promise[ActorRef]()
49 
50     def awaitComponentManager(timeOut: Deadline) {
51       if (timeOut.isOverdue() && !componentManagerInitialized) {
52         componentManagerInitialized = true
53         p failure ComponentException("Component Manager", "Failed to get component manager")
54       }
55       context.actorSelection(HarnessConstants.ComponentFullName).resolveOne()(1 second) onComplete {
56         case Success(s) =>
57           componentManager = Some(s)
58           componentManagerInitialized = true
59           p success s
60         case Failure(f) => awaitComponentManager(timeOut)
61       }
62     }
63 
64     componentManager match {
65       case Some(cm) => p success cm
66       case None =>
67         if (!componentManagerInitialized) {
68           val deadline = 5 seconds fromNow
69           awaitComponentManager(deadline)
70         } else {
71           p failure ComponentException("Component Manager", "Component manager did not initialize")
72         }
73     }
74     p.future
75   }
76 
77   /**
78    * Wrapper function around request that allows the developer to not have to deal with the
79    * ComponentResponse return object, and just deal with the message that they care about
80    *
81    * @param name name of the component
82    * @param msg msg to send to the component
83    * @return
84    */
85   def unwrapRequest[T, U](name:String, msg:ComponentRequest[T]) : Future[U] = {
86     val p = Promise[U]()
87     componentRequest(name, msg).mapTo[ComponentResponse[U]] onComplete {
88       case Success(s) => p success s.resp
89       case Failure(f) => p failure f
90     }
91     p.future
92   }
93 
94   def request[T](name:String, msg:Any, childName:Option[String]=None) : Future[ComponentResponse[T]] =
95     componentRequest(name, ComponentRequest(msg, childName))
96 
97   /**
98    * Simplest way to make a request directly to a component, will return a Future holding whatever the component returns
99    * @param name Name of the component
100    * @param msg Message to send it
101    */
102   def unwrapSelfRequest[T](name:String, msg:AnyRef) : Future[T] = {
103     unwrapRequest[msg.type, T](name, ComponentRequest[msg.type](msg, Some(ComponentManager.ComponentRef)))
104   }
105 
106   /**
107    * Wrapper function that allows developer to make requests to components individually without having to know about the
108    * ComponentManager as the parent that routes the messages to the various components
109    *
110    * @param name name of the component
111    * @param msg message you want to send to the component
112    * @return
113    */
114   def componentRequest[T, U](name:String, msg:ComponentRequest[T]) : Future[ComponentResponse[U]] = {
115     val p = Promise[ComponentResponse[U]]()
116     initComponentHelper onComplete {
117       case Success(cm) =>
118         (cm ? Request(name, msg))(msg.timeout).mapTo[ComponentResponse[U]] onComplete {
119           case Success(s) => p success s
120           case Failure(f) => p failure f
121         }
122       case Failure(f) => p failure f
123     }
124     p.future
125   }
126 
127   def selfMessage(name:String, msg:Any) =
128     componentMessage(name, ComponentMessage(msg, Some(ComponentManager.ComponentRef)))
129 
130   /**
131    * Wrapper function that will allow you to send any message in and it will
132    * wrap the msg within a ComponentMessage case class
133    *
134    * @param name       name of component
135    * @param msg        message to send
136    * @param childName  name of component's child, or 'self' if one wants to hit the component itself
137    */
138   def message(name:String, msg:Any, childName:Option[String]=None) =
139     componentMessage(name, ComponentMessage(msg, childName))
140 
141   /**
142    * Wrapper function that allows the developer to message components individually without having to know about the
143    * ComponentManager as the parent that routes the messages to the various components
144    *
145    * @param name name of the component
146    * @param msg message you want to send to the component
147    */
148   def componentMessage[T](name:String, msg:ComponentMessage[T]) = {
149     initComponentHelper onComplete {
150       case Success(cm) =>
151         cm ! Message(name, msg)
152       case Failure(f) => throw f
153     }
154   }
155 
156   /**
157    * Wrapper function that allows developers to get the actor reference for a particular component
158    *
159    * @param name the name of the component
160    * @param timeout implicit timeout value
161    * @return
162    */
163   def getComponent(name:String)(implicit timeout:Timeout) : Future[ActorRef] = {
164     val p = Promise[ActorRef]()
165     initComponentHelper onComplete {
166       case Success(cm) =>
167         (cm ? GetComponent(name))(timeout).mapTo[Option[ActorRef]] onComplete {
168           case Success(s) =>
169             s match {
170               case Some(ref) => p success ref
171               case None => p failure ComponentNotFoundException("Component Manager", s"component $name not found")
172             }
173           case Failure(f) => p failure f
174         }
175       case Failure(f) => p failure f
176     }
177     p.future
178   }
