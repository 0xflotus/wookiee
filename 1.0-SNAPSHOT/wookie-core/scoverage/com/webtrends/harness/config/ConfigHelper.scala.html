26 /**
27  * This helper class will keep track of a configuration local to your service/component and update
28  * the configuration whenever ConfigWatcherActor detects changes in any of our config files.
29  *
30  * @author woods
31  *         2/10/15
32  */
33 trait ConfigHelper {
34   this: Actor =>
35 
36   var renewableConfig = context.system.settings.config
37 
38   /**
39    * Should override this method and use it when ConfigChange is received, be sure
40    * to call super.renewConfiguration() though to ensure you get the new values
41    */
42   def renewConfiguration() {
43     renewableConfig = HarnessActorSystem.getConfig(None)
44   }
45 
46   def configReceive: Receive = {
47     case ConfigChange() =>
48       renewConfiguration()
49   }
50 }
