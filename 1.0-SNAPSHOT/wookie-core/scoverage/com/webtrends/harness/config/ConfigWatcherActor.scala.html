42 class ConfigWatcherActor extends HActor {
43   val configWatcher = FileSystems.getDefault.newWatchService()
44   var configDir = Paths.get(".")
45   val watchThread = new Thread(new DirectoryWatcher)
46   var brokenKeys = List()
47   var configExists = false
48 
49   override def preStart(): Unit = {
50     super.preStart()
51     ServiceManager.serviceDir(context.system.settings.config) match {
52       case Some(s) =>
53         configDir = s.toPath
54         val dirs = s.listFiles.filter(_.isDirectory)
55 
56         dirs foreach {
57           dir =>
58             val path = Paths.get(dir.getPath.concat("/conf"))
59             if (Files.exists(path)) {
60               log.info("Adding watcher to existing directory {} for any *.conf file changes", path)
61               path.register(configWatcher, Array[WatchEvent.Kind[_]](ENTRY_CREATE), SensitivityWatchEventModifier.HIGH)
62               path.register(configWatcher, Array[WatchEvent.Kind[_]](ENTRY_MODIFY), SensitivityWatchEventModifier.HIGH)
63             }
64         }
65         if (dirs.size > 0) {
66           configExists = true
67           watchThread.start()
68         }
69       case None => log.warn("Service dir does not exist, not starting watchers")
70     }
71   }
72 
73   override def postStop(): Unit = {
74     super.postStop()
75     if (watchThread.isAlive) {
76       log.info("Stopping config watcher")
77       watchThread.interrupt()
78     }
79   }
80 
81   override def checkHealth: Future[HealthComponent] = {
82     Future {
83       !configExists || (watchThread.isAlive && !watchThread.isInterrupted) match {
84         case true => HealthComponent("Config Watcher Health", ComponentState.NORMAL, "Config being watched as expected")
85         case false => HealthComponent("Config Watcher Health", ComponentState.CRITICAL, "Config changes are no longer being watched")
86       }
87     }
88   }
89 
90   private class DirectoryWatcher extends Runnable {
91     def run(): Unit = {
92       while (true) {
93 
94         // wait for key to be signaled
95         var key: Option[WatchKey] = None
96         try {
97           key = Some(configWatcher.take())
98         } catch {
99           case x: InterruptedException =>
100             return
101         }
102 
103         key.get.pollEvents().toStream.takeWhile(_.kind() != OVERFLOW) foreach {
104           event =>
105             log.info("Detected alteration on file {}", event.context().toString)
106             // The filename is the context of the event.
107             val ev = event.asInstanceOf[WatchEvent[Path]]
108             val filename = ev.context()
109 
110             try {
111               // Resolve the filename against the directory.
112               val child = configDir.resolve(filename)
113               if (filename.toString.endsWith(".conf")) {
114                 log.info("Config file change detected, {}, sending message to services/components to reload if applicable.", child.toString)
115                 context.parent ! ConfigChange()
116               } else {
117                 log.info("Ignoring change to {} as it is not a .conf file", child.toString)
118               }
119             } catch {
120               case x: IOException =>
121                 log.error("Issue reading file changed in config dir.", x)
122             }
123         }
124 
125         // Reset the key -- this step is critical if you want to receive further watch events.
126         val valid = key.get.reset()
127         if (!valid) {
128           throw new IllegalStateException("Key is no longer in a valid state, can't watch configuration anymore")
129         }
130       }
131     }
132   }
133 }
