43   def health:Receive = {
44     case CheckHealth =>
45       pipe(Try(checkHealth)
46         .recover({
47         case e: Exception =>
48           HealthComponent(getClass.getSimpleName, ComponentState.CRITICAL,
49             "Exception when trying to check the health: %s".format(e.getMessage))
50       }).get.asInstanceOf[Future[HealthComponent]]
51       ) to sender()
52   }
53 
54   /**
55    * This is the health of the current object, by default will be NORMAL
56    * In general this should be overridden to define the health of the current object
57    * For objects that simply manage other objects you shouldn't need to do anything
58    * else, as the health of the children components would be handled by their own
59    * CheckHealth function
60    *
61    * @return
62    */
63   protected def getHealth: Future[HealthComponent] = {
64     Future {
65       HealthComponent(self.path.toString, ComponentState.NORMAL, "Healthy")
66     }
67   }
68 
69   /**
70    * The actor has been asked to respond with some health information. It needs
71    * to implement this function and provide a list of components used in this service
72    * and their current state. By default the health check will simply run through all the
73    * children for the actor and get their health. Should be overridden for any custom
74    * behavior
75    * @return An instance of a health component
76    */
77   def checkHealth: Future[HealthComponent] = {
78     val p = Promise[HealthComponent]()
79 
80     val h = getHealth.onComplete {
81       case Success(s) =>
82         if (context.children.size == 0) {
83           p success s
84         } else {
85           val futureList = Future.traverse(context.children) {
86             case a: ActorRef =>
87               val healthPromise = Promise[HealthComponent]()
88               (a ? CheckHealth)(2 seconds).mapTo[HealthComponent] onComplete {
89                 case Success(suc) => healthPromise success suc
90                 case Failure(f) => healthPromise success HealthComponent(a.path.name, ComponentState.CRITICAL, s"Failure to get health of child component. ${f.getMessage}")
91               }
92               healthPromise.future
93           }
94           futureList onComplete {
95             case Failure(f) =>
96               _log.debug(f, "Failed to retrieve health of children objects")
97               p success HealthComponent(s.name, ComponentState.CRITICAL, s"Failure to get health of child components. ${f.getMessage}")
98             case Success(answers) =>
99               answers.foreach(h => s.addComponent(h))
100               p success s
101           }
102         }
103       case Failure(f) =>
104         _log.debug(f, "Failed to get health from component")
105         p success HealthComponent(self.path.toString, ComponentState.CRITICAL, f.getMessage)
106     }
107 
108     p.future
109   }
