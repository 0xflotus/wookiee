36   private val _log = Logger(this, context.system)

40   implicit val checkTimeout:Timeout =
41     ConfigUtil.getDefaultTimeout(context.system.settings.config, HarnessConstants.KeyDefaultTimeout, Timeout(10 seconds))
42 
43   def health:Receive = {
44     case CheckHealth =>
45       pipe(Try(checkHealth)
46         .recover({
47         case e: Exception =>
48           HealthComponent(getClass.getSimpleName, ComponentState.CRITICAL,
49             "Exception when trying to check the health: %s".format(e.getMessage))
50       }).get.asInstanceOf[Future[HealthComponent]]
51       ) to sender()
52   }
53 
54   /**
55    * This is the health of the current object, by default will be NORMAL
56    * In general this should be overridden to define the health of the current object
57    * For objects that simply manage other objects you shouldn't need to do anything
58    * else, as the health of the children components would be handled by their own
59    * CheckHealth function
60    *
61    * @return
62    */
63   protected def getHealth: Future[HealthComponent] = {
64     Future {
65       HealthComponent(self.path.toString, ComponentState.NORMAL, "Healthy")
66     }
67   }
68 
69   /**
70    * This is the list of child actors that should be iterated and checked for health
71    * This can be overridden in cases where one does not want to check all children for
72    * health, or some children may not support health checks, or a child is using a push
73    * based model of health reporting
74    * CheckHealth function
75    *
76    * @return
77    */
78   protected def getHealthChildren: Iterable[ActorRef] = {
79     context.children
80   }
81 
82   /**
83    * The actor has been asked to respond with some health information. It needs
84    * to implement this function and provide a list of components used in this service
85    * and their current state. By default the health check will simply run through all the
86    * children for the actor and get their health. Should be overridden for any custom
87    * behavior
88    * @return An instance of a health component
89    */
90   def checkHealth: Future[HealthComponent] = {
91     val p = Promise[HealthComponent]()
92 
93     getHealth.onComplete {
94       case Success(s) =>
95         val healthFutures = getHealthChildren map { ref =>
96           (ref ? CheckHealth).mapTo[HealthComponent] recover {
97             case ex: Exception => HealthComponent(ref.path.name, ComponentState.CRITICAL, s"Failure to get health of child component. ${ex.getMessage}")
98           }
99         }
100 
101         Future.sequence(healthFutures) onComplete {
102           case Failure(f) =>
103             _log.debug(f, "Failed to retrieve health of children objects")
104             p success HealthComponent(s.name, ComponentState.CRITICAL, s"Failure to get health of child components. ${f.getMessage}")
105           case Success(healths) =>
106             healths foreach { it => s.addComponent(it) }
107             p success s
108         }
109       case Failure(f) =>
110         _log.debug(f, "Failed to get health from component")
111         p success HealthComponent(self.path.toString, ComponentState.CRITICAL, f.getMessage)
112     }
113 
114     p.future
115   }
116 }
