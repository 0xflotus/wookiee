60   /**
61    * Rollup the overall status and critical alerts for each component
62    * @param checks
63    * @return
64    */
65   private def rollupStatuses(checks: mutable.Buffer[ComponentHealth]): ComponentHealth = {
66     // Check if all components are running normal
67     if (alerts.length == 0) {
68       ComponentHealth(ComponentState.NORMAL, "Thunderbirds are GO")
69     }
70     else {
71       val status = if (checks.forall(c => c.state == ComponentState.DEGRADED)) ComponentState.DEGRADED else ComponentState.CRITICAL
72       val details = for (c <- checks) yield c.details
73 
74       ComponentHealth(status, details.mkString("; "))
75     }
76   }
77 
78   /**
79    * Rollup alerts for all components that have a CRITICAL or DEGRADED state
80    * @param component
81    */
82   private def checkComponents(component: HealthComponent) {
83     def alertComponent(state: ComponentState.ComponentState): Boolean = {
84       if (state == ComponentState.CRITICAL || state == ComponentState.DEGRADED) true else false
85     }
86 
87     def healthDetails(component: HealthComponent): String = {
88       component.name + "[" + component.state + "] - " + component.details
89     }
90 
91     if (component.components.length == 0 && alertComponent(component.state)) {
92       alerts += ComponentHealth(component.state, healthDetails(component))
93     }
94     else {
95       if (alertComponent(component.state)) {
96         alerts += ComponentHealth(component.state, healthDetails(component))
97       }
98       component.components.foreach(checkComponents(_))
99     }
100   }
101 
102   /**
103    * Run the health checks and return the current system state
104    * @return
105    */
106   def runChecks: Future[ApplicationHealth] = {
107 
108     import context.dispatcher
109 
110     // Ask for the health of each component
111     val future = (context.actorSelection(HarnessConstants.ActorPrefix) ? CheckHealth).mapTo[Seq[HealthComponent]]
112     val p = Promise[ApplicationHealth]
113 
114     future.onComplete({
115       case Success(checks) =>
116         // Rollup alerts for any critical or degraded components
117         checks.foreach(checkComponents(_))
118         // Rollup the statuses
119         val overallHealth = rollupStatuses(alerts)
120         alerts.clear()
121         p success new ApplicationHealth(application, version, upTime, overallHealth.state, overallHealth.details, checks)
122       case Failure(e) =>
123         log.error("An error occurred while fetching the health request results", e)
124         p success new ApplicationHealth(application, version, upTime, ComponentState.CRITICAL, e.getMessage, Nil)
125     })
126 
127     p.future
128   }
