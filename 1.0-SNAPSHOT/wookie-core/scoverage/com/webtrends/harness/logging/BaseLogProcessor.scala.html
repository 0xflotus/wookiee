30   /**
31    * Make sure we have Java friendly Object instances
32    * @param params
33    * @return
34    */
35   protected def transformParams(params: Seq[Any]) = params.map(_.asInstanceOf[Object])
36 
37   /**
38    * Format the given string template. this is only used when receiving an event which includes
39    * a throwable. this is due to the fact that SLF4J does not allow template arguments when including
40    * a throwable in its calls.
41    * @param template
42    * @param arg
43    * @return
44    */
45   protected def format(template: String, arg: Seq[Any]): String = {
46     val sb = new java.lang.StringBuilder(64)
47     var p = 0
48     var rest = template
49     while (p < arg.length) {
50       val index = rest.indexOf("{}")
51       if (index == -1) {
52         sb.append(rest).append(" WARNING arguments left: ").append(arg.length - p)
53         rest = ""
54         p = arg.length
55       } else {
56         sb.append(rest.substring(0, index)).append(arg(p))
57         rest = rest.substring(index + 2)
58         p += 1
59       }
60     }
61     sb.append(rest).toString
62   }
63 
64   @inline
65   protected final def withContext(thread: Thread, timestamp: Long, altSource: Option[String])(logStatement: => Unit) {
66     if (!thread.getName.equals(Thread.currentThread.getName)) {
67       MDC.put(sourceThread, thread.getName)
68     }
69 
70     MDC.put(akkaSource, if (altSource.isDefined) altSource.get else "")
71     MDC.put(sourceTime, Helpers.currentTimeMillisToUTCString(timestamp))
72 
73     try logStatement finally {
74       MDC.remove(akkaSource)
75       MDC.remove(sourceThread)
76       MDC.remove(sourceTime)
77     }
78   }
79 }
