54   def load(context: ActorContext) {
55     Try({
56       val internalService = ConfigUtil.getDefaultValue(HarnessConstants.KeyInternalService, context.system.settings.config.getString, "")
57       if (internalService.nonEmpty) {
58         val loader = Thread.currentThread.getContextClassLoader.asInstanceOf[HarnessClassLoader]
59         val serviceName = internalService.split("\\.").reverse(0)
60         loadService(serviceName, loader.loadClass(internalService))
61       } else {
62           ServiceManager.serviceDir(context.system.settings.config) match {
63             case Some(s) =>
64               val dirs = s.listFiles.filter(_.isDirectory).sortBy(f => f.getName)
65               if (dirs.length > 0) {
66                 dirs.foreach(subPath => {
67                   Try({
68                     log.info("Loading the service(s) in {}", subPath.getPath)
69                     loadClasses(context, subPath)
70                   }).recover({
71                     case e: Throwable =>
72                       log.error(e, "Error loading the service(s) in {}", subPath.getPath)
73                   })
74                 })
75               }
76               log.info("%s Services have been loaded".format(services.size))
77 
78             case None => // ignore
79           }
80       }
81     }).recover({
82       case e: Throwable => log.error("Error loading the service(s)", e)
83     })
84   }

91   private def loadClasses(context: ActorContext, rootPath: File) = {
92     val jars = rootPath.listFiles().filter(_.getName.endsWith("jar"))
93     if (jars.length > 0) {
94 
95       val libPath = rootPath.getPath.concat("/lib")
96       log.info("Loading the dependent jars in {}", libPath)
97 
98       val allJars = Array.concat[File](jars, new File(libPath).listFiles()
99         .filter(_.getName.endsWith("jar"))
100         .filterNot(jar => harnessLibs.map(_.getName).contains(jar.getName)))
101 
102       val urls = allJars.map(_.toURI.toURL)
103       val harnessLoader = Thread.currentThread.getContextClassLoader.asInstanceOf[HarnessClassLoader]
104 
105       // Load the urls into the class loader
106       val loader = ConfigUtil.getDefaultValue(HarnessConstants.KeyServiceDistinctClassLoader, context.system.settings.config.getBoolean, true) match {
107         case true =>
108           val pcl = new ServiceClassLoader(urls, harnessLoader)
109           harnessLoader.addChildLoader(pcl)
110           pcl
111         case false =>
112           harnessLoader.addURLs(urls)
113           harnessLoader
114       }
115 
116       jars.foreach(file => processJar(context, rootPath, file, loader))
117     }
118     else {
119       log.warning("No jar was found for the service in {}", rootPath)
120     }
121   }

123   /**
124    * Process the given jar and look for any services to load
125    * @param context The service manager's context
126    * @param rootPath The services root file path
127    * @param jar The potential services jar file
128    * @param loader The class loader to use

130   private def processJar(context: ActorContext,
131                          rootPath: File,
132                          jar: File,
133                          loader: ClassLoader): Unit = {
134 
135     // Open the jar file
136     val jarFile = new JarFile(jar)
137     val loaderOpt = loader.isInstanceOf[ServiceClassLoader] match {
138       case true => Some(loader.asInstanceOf[ServiceClassLoader])
139       case false => None
140     }
141 
142     try {
143       val sClass = classOf[Service]
144 
145       val localServices = jarFile.entries().flatMap[(ServiceMetaData, (ActorSelection, Option[ServiceClassLoader]))](entry =>
146         try {
147           val entryName = entry.getName
148           if (entryName.endsWith(".class")) {
149             val clazz = loader.loadClass(entryName.replace(".class", "").replaceAll("/", "."))
150 
151             if (sClass.isAssignableFrom(clazz) && !clazz.getName.equals(sClass.getName) ) {
152 
153               val name = clazz.getSimpleName.toLowerCase
154               val manifest = jarFile.getManifest
155               val build = manifest.getMainAttributes.getValue("Implementation-Build") match {
156                 case null => "N/A"
157                 case s if s.nonEmpty => s
158                 case _ => "N/A"
159               }
160               val classPath = manifest.getMainAttributes.getValue("Class-Path") match {
161                 case null => "N/A"
162                 case s if s.nonEmpty => s
163                 case _ => "N/A"
164               }
165               val version = "%s.%s".format(manifest.getMainAttributes.getValue("Implementation-Version"), build)
166 
167               val serv = Try {
168                 val serviceActor = context.actorOf(Props(clazz.asInstanceOf[Class[_ <: Actor]]), name)
169 
170                 // Watch this actor
171                 context watch serviceActor
172                 // Get the meta data
173                 val meta = getServiceMetaDetails(context, serviceActor)
174                 log.info("{}: {}",name,loader.toString)
175 
176                 ServiceMetaData(name,
177                   version,
178                   DateTime.now(),
179                   rootPath.getAbsolutePath,
180                   serviceActor.path.toString,
181                   jar.toURI.toString,
182                   meta.supportsHttp,
183                   classPath.split(" ").toList.sorted) -> (context.actorSelection(serviceActor.path), loaderOpt)
184 
185               }.recover {
186                 case e: Throwable =>
187                   log.error(e, "Error loading the service {}", name)
188                   // Remove the actor so we can avoid a badly loaded actor
189                   context.child(name) match {
190                     case Some(actor) => context.unwatch(actor); context.stop(actor)
191                     case None => // Do nothing
192                   }
193                   None
194               }
195 
196               serv match {
197                 case Failure(t) => None
198                 case Success(None) => None
199                 case Success(meta) =>
200                   log.debug(s"Loaded service $name")
201                   Some(meta.asInstanceOf[(ServiceMetaData, (ActorSelection, Option[ServiceClassLoader]))])
202               }
203             }
204             else {
205               None
206             }
207           }
208           else {
209             None
210           }
211         } catch {
212           case e: Throwable =>
213             log.error(e, "Error loading the service: {}", entry.getName)
214             None
215         }).toMap
216 
217       services ++= localServices
218     }
219     finally {
220       jarFile.close()
221     }
222   }

224   def loadService[T](name: String, clazz: Class[T], classLoader: Option[ClassLoader] = None): Option[ActorRef] = {
225     if (!classOf[Service].isAssignableFrom(clazz)) {
226       log.error(s"Could not load service $name, not assignable from ${clazz.getName}")
227       return None
228     }
229 
230     val classLoaderFinal = classLoader match {
231       case None => Thread.currentThread.getContextClassLoader.asInstanceOf[HarnessClassLoader]
232       case Some(t) => t
233     }
234     val loaderOpt = classLoaderFinal.isInstanceOf[ServiceClassLoader] match {
235       case true => Some(classLoaderFinal.asInstanceOf[ServiceClassLoader])
236       case false => None
237     }
238     var serviceActor: Option[ActorRef] = None
239     val serv = Try {
240       serviceActor = Some(context.actorOf(Props(clazz.asInstanceOf[Class[_ <: Actor]]), name))
241 
242       // Watch this actor
243       context watch serviceActor.get
244       // Get the meta data
245       val meta = getServiceMetaDetails(context, serviceActor.get)
246       ServiceMetaData(name,
247         "1.0",
248         DateTime.now(),
249         "",
250         serviceActor.get.path.toString,
251         "",
252         meta.supportsHttp,
253         null) -> (context.actorSelection(serviceActor.get.path), loaderOpt)
254 
255     }.recover {
256       case f: InvalidActorNameException =>
257         log.info(s"Service $name already started")
258         serviceActor = getServiceByName(name)
259         None
260       case e: Throwable =>
261         log.error(e, "Error loading the service {}", name)
262         // Remove the actor so we can avoid a badly loaded actor
263         context.child(name) match {
264           case Some(actor) => context.unwatch(actor); context.stop(actor)
265           case None => // Do nothing
266         }
267         None
268     }
269 
270     serv match {
271       case Failure(t) =>
272       case Success(None) =>
273       case Success(meta) =>
274           services ++= Some(meta.asInstanceOf[(ServiceMetaData, (ActorSelection, Option[ServiceClassLoader]))])
275     }
276     serviceActor
277   }

279   def getServiceByName(name: String): Option[ActorRef] = {
280     var ref: Option[ActorRef] = None
281     services foreach { s =>
282       if (s._1.name.equals(name)) {
283         s._2._1.resolveOne(4 seconds) onComplete {
284           case Success(some) => ref = Some(some.asInstanceOf[ActorRef])
285           case Failure(f) => log.warn(s"Failed to get service $name")
286         }
287       }
288     }
289     if (!ref.isDefined) {
290       log.error(s"Was not able to find service $name")
291     }
292     ref
293   }

295   /**
296    * Send a message to the given service in order to receive it's meta data
297    * @param serviceActor The service to send the message to
298    * @return An instance of ServiceMetaData

300   private def getServiceMetaDetails(context: ActorContext, serviceActor: ActorRef): ServiceMetaDetails = {
301     implicit val timeout = Timeout(5 seconds)
302 
303     val future = (serviceActor ? GetMetaDetails).mapTo[ServiceMetaDetails]
304     // Since we are starting up the service and loading services, we shall block to make
305     // sure that everything has run its course
306     Await.result(future, timeout.duration)
307   }
