31   val externalLogger = LoggerFactory.getLogger(this.getClass)

33   /**
34    * This function helps a user create a actor based on Config from the system configuration using
35    * akka.actor.deployment {
36    *  /[ACTOR_PATH]/[ACTOR_NAME] {
37    *    router = round-robin
38    *    nr-of-instances = 5
39    *  }
40    * }
41    * This is based on the Akka FromConfig function and functionality around actor routees. This function
42    * is useful because it checks to see if configuration is available for the routees, and if not if defaults
43    * to using Round-Robin with 3 routees.
44    *
45    * @param props The props for the actor class, using defined like Props[ActorClass] or Props(clazz, construtor_arg1, ...)
46    * @param actorName The name that you wish the actor to be
47    * @param defNumRoutees DEFAULT=3 If config not found, then will default to the number of routees you put in
48    * @param context this is the implicit context that would be usually defined by the actor that you are calling this function from
49    * @return
50    */
51   def initActorFromConfig(props:Props, actorName:String, defNumRoutees:Int=3)(implicit context:ActorContext) : ActorRef = {
52     val config = context.system.settings.config
53     val deployPath = s"akka.actor.deployment.${context.self.path.toStringWithoutAddress}/$actorName"
54     if (!config.hasPath(deployPath)) {
55       // message primarily for debugging so that you can see immediately if your actor found the config
56       externalLogger.debug(s"Could not find deployment config for path [$deployPath], deploying default round robin with $defNumRoutees routees")
57       context.actorOf(RoundRobinPool(defNumRoutees).props(props), actorName)
58     } else {
59       context.actorOf(FromConfig.props(props), actorName)
60     }
61   }
62 }
