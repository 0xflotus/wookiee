27 /**
28  * A command should be a business logic that handles messages from some source (like http) then executes
29  * the logic that is defined for the end point and returns the response. It should not ever deal with the
30  * underlying messaging protocol but rather focus exclusively on the business logic. It should also not
31  * deal with any storage or caching or anything like that.
32  *
33  * @author Michael Cuthbert on 12/1/14.
34  */
35 trait Command extends HActor {
36   import context.dispatcher
37 
38   override def receive = health orElse ({
39     case ExecuteCommand(name, bean) => pipe(execute(bean)) to sender
40   } : Receive)

42   def path : String = s"_wt_internal/${commandName.toLowerCase}"

44   /**
45    * This allows for extra functionality in your command path. Basically it will allow
46    * for multiple paths to be matched to a single command, and will map each path to a
47    * given name so that you can find out which path was mapped in the end. By default it
48    * will simply be a map of 1 element being retrieved from the default "path" function.
49    * So if you only require a single path you wouldn't worry about this function at all
50    *
51    * @return
52    */
53   def paths : Map[String, String] =  Map("default" -> path)

55   /**
56    * Name of the command that will be used for the actor name
57    *
58    * @return

62   /**
63    * The primary entry point for the command, the actor for this command
64    * will ignore all other messaging and only execute through this
65    *
66    * @return
67    */
68   def execute[T](bean:Option[CommandBean]=None) : Future[CommandResponse[T]]

73   /**
74    * Checks the match between a test path and url path
75    *
76    * @param test The test path, like /test/$var1/ping
77    * @param uri The uri, like /test/1/ping
78    * @return Will return a command bean if matched, None if not and in the command bean the
79    *         key var1 will equal 1 as per the example above
80    */
81   def matchPath(test:String, uri:String) : Option[CommandBean] = {
82     val bean = new CommandBean()
83     val testPath = test.split("/") filter { x => x.nonEmpty }
84     val urlPath = uri.split("/") filter { x => x.nonEmpty }
85 
86     val m = urlPath.corresponds(testPath) {
87       // case for grabbing variable from the url
88       case (x, y) if y.charAt(0) == '$' =>
89         // try to normalize the segment into an INT or STRING
90         val key = y.substring(1)
91         x forall {_.isDigit} match {
92           case true => bean.addValue(key, x.toInt.asInstanceOf[Integer])
93           case false => bean.addValue(key, x)
94         }
95         true
96       // case if you want optional path values
97       case (x, y) if y.contains("|") =>
98         val matches = y.split("\\|") flatMap {
99           _ == x match {
100             case true => Some(true)
101             case false => None
102           }
103         } groupBy(_ == true)
104         // the flatMap and groupBy function will return a map with a single element pointing to all the
105         // matched path elements, all false matches will be thrown out. So if there is no matches the
106         // size of the map would be 0 otherwise it would be 1
107         matches.size == 1
108       // standard case
109       case (x, y) => x == y
110     }
111 
112     m match {
113       case true => Some(bean)
114       case false => None
115     }
116   }
117 }
