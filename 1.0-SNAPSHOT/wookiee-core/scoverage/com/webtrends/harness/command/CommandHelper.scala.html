31 /**
32  * A trait that you can add to any actor that will enable the actor to talk to the CommandManager easily
33  * and execute commands at will
34  *
35  * @author Michael Cuthbert on 12/10/14.
36  */
37 trait CommandHelper extends ActorLoggingAdapter {
38   this: Actor =>
39 
40   import context.dispatcher
41   var commandManagerInitialized = false
42   var commandManager:Option[ActorRef] = None
43 
44   def initCommandHelper = {
45     addCommands
46   }
47 
48   def initCommandManager : Future[Boolean] = {
49     val p = Promise[Boolean]
50     commandManagerInitialized match {
51       case true => p success commandManagerInitialized
52       case false =>
53         context.actorSelection(HarnessConstants.CommandFullName).resolveOne()(2 seconds) onComplete {
54           case Success(s) =>
55             commandManagerInitialized = true
56             commandManager = Some(s)
57             p success commandManagerInitialized
58           case Failure(f) => p failure CommandException("Component Manager", f)
59         }
60     }
61     p.future
62   }
63 
64   /**
65    * This function should be implemented by any service that wants to add
66    * any commands to make available for use
67    */
68   def addCommands = {}
69 
70   /**
71    * Wrapper that allows services to add commands to the command manager with a single command
72    *
73    * @param name name of the command you want to add
74    * @param props the props for that command actor class
75    * @return
76    */
77   def addCommandWithProps[T<:Command](name:String, props:Props) : Future[ActorRef] = {
78     implicit val timeout = Timeout(2 seconds)
79     val p = Promise[ActorRef]
80     initCommandManager onComplete {
81       case Success(_) =>
82         commandManager match {
83           case Some(cm) =>
84             (cm ? AddCommandWithProps(name, props)).mapTo[ActorRef] onComplete {
85               case Success(r) => p success r
86               case Failure(f) => p failure f
87             }
88           case None => p failure CommandException("CommandManager", "CommandManager not found!")
89         }
90       case Failure(f) => p failure f
91     }
92     p.future
93   }
94 
95   /**
96    * Wrapper that allows services add commands to the command manager with a single command
97    *
98    * @param name name of the command you want to add
99    * @param actorClass the class for the actor
100    */
101   def addCommand[T<:Command](name:String, actorClass:Class[T]) : Future[ActorRef] = {
102     implicit val timeout = Timeout(2 seconds)
103     val p = Promise[ActorRef]
104     initCommandManager onComplete {
105       case Success(_) =>
106         commandManager match {
107           case Some(cm) =>
108             (cm ? AddCommand(name, actorClass)).mapTo[ActorRef] onComplete {
109               case Success(r) => p success r
110               case Failure(f) => p failure f
111             }
112           case None => p failure CommandException("CommandManager", "CommandManager not found!")
113         }
114       case Failure(f) => p failure f
115     }
116     p.future
117   }
118 
119   /**
120    * Wrapper that allows services execute commands (remote or otherwise)
121    *
122    * @param name name of the command you want to execute
123    *             if this is a remote command the name will be the reference to the
124    *             command
125    * @param bean the bean that will be passed to the command
126    * @param server If none then we are executing a local command, if set then it is a remote command and that is the server name
127    * @param port The port of the remote server defaults to 0, as by default this function deals with local commands
128    * @return
129    */
130   def executeCommand[T](name:String, bean:Option[CommandBean]=None, server:Option[String]=None, port:Int=2552)(implicit timeout:Timeout) : Future[CommandResponse[T]] = {
131     val p = Promise[CommandResponse[T]]
132     initCommandManager onComplete {
133       case Success(_) =>
134         commandManager match {
135           case Some(cm) =>
136             val msg = server match {
137               case Some(srv) => ExecuteRemoteCommand(name, srv, port, bean)
138               case None => ExecuteCommand(name, bean)
139             }
140             (cm ? msg)(timeout).mapTo[CommandResponse[T]] onComplete {
141               case Success(s) => p success s
142               case Failure(f) => p failure CommandException("CommandManager", f)
143             }
144           case None => p failure CommandException("CommandManager", "CommandManager not found!")
145         }
146       case Failure(f) => p failure f
147     }
148     p.future
149   }
