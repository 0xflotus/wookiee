55   /**
56    * Wrapper around the addCommand function that creates a function with props, this will just get the
57    * props from the actor class
58    *
59    * @param name name of the command you want to add
60    * @param actorClass the actor class for the command
61    * @tparam T
62    * @return
63    */
64   protected def addCommand[T<:Command](name:String, actorClass:Class[T]) : Future[ActorRef] = addCommand[T](name, Props(actorClass))
65 
66   /**
67    * We add commands as children to the CommandManager, based on default routing
68    * or we use the setup defined for the command
69    *
70    * @param name
71    * @param props
72    */
73   protected def addCommand[T<:Command](name:String, props:Props) : Future[ActorRef] = {
74     // check first if the router props have been defined else
75     // use the default Round Robin approach
76     val config = context.system.settings.config
77     val aRef = if (!config.hasPath(s"akka.actor.deployment.${HarnessConstants.CommandFullName}/$name")) {
78       val nrRoutees = config.getInt(HarnessConstants.KeyCommandsNrRoutees)
79       context.actorOf(RoundRobinPool(nrRoutees).props(props), name)
80     } else {
81       context.actorOf(FromConfig.props(props), name)
82     }
83     CommandManager.addCommand(name, aRef)
84     Future { aRef }
85   }
86 
87   /**
88    * Executes a remote command and will return a commandResponse to the sender
89    *
90    * @param name The name of the command you want to execute
91    * @param server The server that has the command on
92    * @param port the port that the server is listening on
93    * @param bean
94    */
95   protected def executeRemoteCommand[T](name:String, server:String, port:Int=2552, bean:Option[CommandBean]=None) : Future[CommandResponse[T]] = {
96     val p = Promise[CommandResponse[T]]
97     context.system.settings.config.getString("akka.actor.provider") match {
98       case "akka.remote.RemoteActorRefProvider" =>
99         context.actorSelection(CommandManager.getRemoteAkkaPath(server, port)).resolveOne() onComplete {
100           case Success(ref) =>
101             (ref ? ExecuteCommand(name, bean)).mapTo[CommandResponse[T]] onComplete {
102               case Success(s) => p success s
103               case Failure(f) => p failure f
104             }
105           case Failure(f) => p failure new CommandException("CommandManager", s"Failed to find remote system [$server:$port]", Some(f))
106         }
107       case _ => p failure new CommandException("CommandManager", s"Remote provider for akka is not enabled")
108     }
109     p.future
110   }
111 
112   /**
113    * Executes a command and will return a CommandResponse to the sender
114    *
115    * @param name
116    * @param bean
117    */
118   protected def executeCommand[T](name:String, bean:Option[CommandBean]=None) : Future[CommandResponse[T]] = {
119     val p = Promise[CommandResponse[T]]
120     CommandManager.getCommand(name) match {
121       case Some(ref) =>
122         (ref ? ExecuteCommand(name, bean)).mapTo[CommandResponse[T]] onComplete {
123           case Success(s) => p success s
124           case Failure(f) => p failure f
125         }
126       case None => p failure CommandException(name, "Command not found")
127     }
128     p.future
129   }
130 }

132 object CommandManager {
133   private val externalLogger = LoggerFactory.getLogger(this.getClass)
134 
135   // map that stores the name of the command with the actor it references
136   val commandMap = mutable.Map[String, ActorRef]()
137 
138   def props = Props[CommandManager]
139 
140   def addCommand(name:String, ref:ActorRef) = {
141     externalLogger.debug(s"Command $name with path ${ref.path} inserted into Command Manager map.")
142     commandMap += (name -> ref)
143   }
144 
145   protected def removeCommand(name:String) : Boolean = {
146     commandMap.get(name) match {
147       case Some(n) =>
148         externalLogger.debug(s"Command $name with path ${n.path} removed from Command Manager map.")
149         commandMap -= name
150         true
151       case None => false
152     }
153   }
154 
155   def getCommand(name:String) : Option[ActorRef] = commandMap.get(name)
156 
157   def getRemoteAkkaPath(server:String, port:Int) : String = s"akka.tcp://server@$server:$port${HarnessConstants.CommandFullName}"
