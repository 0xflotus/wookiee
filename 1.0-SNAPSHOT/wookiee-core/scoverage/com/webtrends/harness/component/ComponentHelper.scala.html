32 /**
33  * This is a helper class that enables developers who use to trait to interact with the ComponentManager
34  * easily on the harness
35  */
36 trait ComponentHelper {
37   this: Actor =>
38 
39   import context.dispatcher
40   var componentManagerInitialized = false
41   var componentManager:Option[ActorRef] = None
42 
43   /**
44    * This should only be called once to initialize the component manager actor. Will retry when called
45    * as many methods require this to work.
46    */
47   def initComponentHelper : Future[ActorRef] = {
48     val p = Promise[ActorRef]()
49 
50     def awaitComponentManager(timeOut: Deadline) {
51       if (timeOut.isOverdue() && !componentManagerInitialized) {
52         componentManagerInitialized = true
53         p failure ComponentException("Component Manager", "Failed to get component manager")
54       } else {
55         context.actorSelection(HarnessConstants.ComponentFullName).resolveOne()(1 second) onComplete {
56           case Success(s) =>
57             componentManager = Some(s)
58             componentManagerInitialized = true
59             p success s
60           case Failure(f) => awaitComponentManager(timeOut)
61         }
62       }
63     }
64 
65     componentManager match {
66       case Some(cm) => p success cm
67       case None =>
68         if (!componentManagerInitialized) {
69           val deadline = 5 seconds fromNow
70           awaitComponentManager(deadline)
71         } else {
72           p failure ComponentException("Component Manager", "Component manager did not initialize")
73         }
74     }
75     p.future
76   }
77 
78   /**
79    * Wrapper function around request that allows the developer to not have to deal with the
80    * ComponentResponse return object, and just deal with the message that they care about
81    *
82    * @param name name of the component
83    * @param msg msg to send to the component
84    * @return
85    */
86   def unwrapRequest[T, U](name:String, msg:ComponentRequest[T]) : Future[U] = {
87     val p = Promise[U]()
88     componentRequest(name, msg).mapTo[ComponentResponse[U]] onComplete {
89       case Success(s) => p success s.resp
90       case Failure(f) => p failure f
91     }
92     p.future
93   }
94 
95   def request[T](name:String, msg:Any, childName:Option[String]=None) : Future[ComponentResponse[T]] =
96     componentRequest(name, ComponentRequest(msg, childName))
97 
98   /**
99    * Simplest way to make a request directly to a component, will return a Future holding whatever the component returns
100    * @param name Name of the component
101    * @param msg Message to send it
102    */
103   def unwrapSelfRequest[T](name:String, msg:AnyRef) : Future[T] = {
104     unwrapRequest[msg.type, T](name, ComponentRequest[msg.type](msg, Some(ComponentManager.ComponentRef)))
105   }
106 
107   /**
108    * Wrapper function that allows developer to make requests to components individually without having to know about the
109    * ComponentManager as the parent that routes the messages to the various components
110    *
111    * @param name name of the component
112    * @param msg message you want to send to the component
113    * @return
114    */
115   def componentRequest[T, U](name:String, msg:ComponentRequest[T]) : Future[ComponentResponse[U]] = {
116     val p = Promise[ComponentResponse[U]]()
117     initComponentHelper onComplete {
118       case Success(cm) =>
119         (cm ? Request(name, msg))(msg.timeout).mapTo[ComponentResponse[U]] onComplete {
120           case Success(s) => p success s
121           case Failure(f) => p failure f
122         }
123       case Failure(f) => p failure f
124     }
125     p.future
126   }
127 
128   def selfMessage(name:String, msg:Any) =
129     componentMessage(name, ComponentMessage(msg, Some(ComponentManager.ComponentRef)))
130 
131   /**
132    * Wrapper function that will allow you to send any message in and it will
133    * wrap the msg within a ComponentMessage case class
134    *
135    * @param name       name of component
136    * @param msg        message to send
137    * @param childName  name of component's child, or 'self' if one wants to hit the component itself
138    */
139   def message(name:String, msg:Any, childName:Option[String]=None) =
140     componentMessage(name, ComponentMessage(msg, childName))
141 
142   /**
143    * Wrapper function that allows the developer to message components individually without having to know about the
144    * ComponentManager as the parent that routes the messages to the various components
145    *
146    * @param name name of the component
147    * @param msg message you want to send to the component
148    */
149   def componentMessage[T](name:String, msg:ComponentMessage[T]) = {
150     initComponentHelper onComplete {
151       case Success(cm) =>
152         cm ! Message(name, msg)
153       case Failure(f) => throw f
154     }
155   }
156 
157   /**
158    * Wrapper function that allows developers to get the actor reference for a particular component
159    *
160    * @param name the name of the component
161    * @param timeout implicit timeout value
162    * @return
163    */
164   def getComponent(name:String)(implicit timeout:Timeout) : Future[ActorRef] = {
165     val p = Promise[ActorRef]()
166     initComponentHelper onComplete {
167       case Success(cm) =>
168         (cm ? GetComponent(name))(timeout).mapTo[Option[ActorRef]] onComplete {
169           case Success(s) =>
170             s match {
171               case Some(ref) => p success ref
172               case None => p failure ComponentNotFoundException("Component Manager", s"component $name not found")
173             }
174           case Failure(f) => p failure f
175         }
176       case Failure(f) => p failure f
177     }
178     p.future
179   }
