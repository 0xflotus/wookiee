92   def loadComponentJars(sysConfig:Config, loader:HarnessClassLoader) = {
93     val loadedList = mutable.MutableList[String]()
94     getComponentPath(sysConfig) match {
95       case Some(dir) =>
96         val files = dir.listFiles.filter(x => x.isDirectory | FileUtil.getExtension(x).equalsIgnoreCase("jar")) flatMap {
97           f =>
98             if (f.isDirectory) {
99               val componentName = f.getName
100               try {
101                 val co = validateComponentDir(componentName, f)
102                 // get list of all JARS and load each one
103                 Some(co._2.listFiles.filter(f => FileUtil.getExtension(f).equalsIgnoreCase("jar")) flatMap {
104                   f =>
105                     val jarName = FileUtil.getFilename(f)
106                     if (loadedList.contains(jarName)) {
107                       None
108                     } else {
109                       loadedList += jarName
110                       Some(f.getCanonicalFile.toURI.toURL)
111                     }
112                 })
113               } catch {
114                 case e: IllegalArgumentException =>
115                   externalLogger.warn(e.getMessage)
116                   None
117               }
118             } else {
119               if (loadedList.contains(FileUtil.getFilename(f))) None else Some(Array(f.getCanonicalFile.toURI.toURL))
120             }
121         } flatMap { x => x }
122         // probably should do some de-duplication
123         loader.addURLs(files)
124       case None => // ignore
125     }
126   }

167   /**
168    * Will validate a component directory and return the location of the component library dir
169    *
170    * @param componentName name of component .conf file
171    * @param folder folder in which configs can be found
172    * @return
173    * @throws IllegalArgumentException if configuration file is not found, or the lib directory is not there
174    */
175   def validateComponentDir(componentName:String, folder:File) : (File, File) = {
176     val confFile = folder.listFiles.filter(_.getName.equalsIgnoreCase("reference.conf"))
177     require(confFile.size == 1, "Conf file not found.")
178     // check the config file and if disabled then fail
179     val config = ConfigFactory.parseFile(confFile(0))
180     if (config.hasPath(s"$componentName.enabled")) {
181       require(config.getBoolean(s"$componentName.enabled"), s"$componentName not enabled")
182     }
183     val libDir = folder.listFiles.filter(f => f.isDirectory && f.getName.equalsIgnoreCase("lib"))
184     require(libDir.size == 1, "Lib directory not found.")
185     (confFile(0), libDir(0))
186   }

248    *

250    * @param msg message to send to component
251    * @param wrapResponse default is true, but if you don't want to have the response wrapped

255   def request[T, U](name:String, msg:ComponentRequest[T], wrapResponse:Boolean=true) : Future[ComponentResponse[U]] = {
256     val p = Promise[ComponentResponse[U]]
257     context.child(name) match {
258       case Some(ref) =>
259         val actorResp = (ref ? msg)(msg.timeout).mapTo[ComponentResponse[U]]
260         actorResp onComplete {
261           case Success(resp) => p success resp
262           case Failure(f) => p failure f
263         }
264       case None => p failure ComponentNotFoundException("ComponentManager", s"$name component not found")
265     }
266     p.future
267   }

291   /**
292    * Load up all the system components from the config
293    * This function needs to block quite a bit because the system requires to load components
294    * prior to anything else happening. This function will only be executed once.
295    */
296   private def initializeComponents = {
297     val cList = ComponentManager.getComponentPath(sysConfig) match {
298       case Some(dir) => dir.listFiles.filter(x => x.isDirectory || FileUtil.getExtension(x).equalsIgnoreCase("jar"))
299       case None => Array[File]()
300     }
301 
302     val libList = if (sysConfig.hasPath(HarnessConstants.KeyComponents)) {
303       sysConfig.getStringList(HarnessConstants.KeyComponents).asScala
304     } else {
305       val tempList = mutable.MutableList[String]()
306       // try find any dynamically, we may get duplicate entries, but that will be handled during the loading process
307       sysConfig.root().asScala foreach {
308         entry =>
309           try {
310             entry._2.valueType() match {
311               case ConfigValueType.OBJECT =>
312                 val c = sysConfig.getConfig(entry._1)
313                 if (c.hasPath(HarnessConstants.KeyDynamicComponent)) {
314                   if (c.getBoolean(HarnessConstants.KeyDynamicComponent)) {
315                     log.info(s"Dynamic Component detected [${entry._1}]")
316                     tempList += entry._1
317                   }
318                 }
319               case _ => //ignore
320             }
321           } catch {
322             case e:ConfigException => // if this exception occurs we know for sure that it is not a dynamic component
323           }
324       }
325       tempList.toList
326     }
327 
328     val componentsLoaded = mutable.MutableList[String]()
329     val compList = cList ++ libList
330     if (compList.length > 0) {
331       // load up configured components
332       log.info("Loading components...")
333       compList foreach {
334         compFolder =>
335           val cfName = compFolder.toString
336           try {
337             val componentName = compFolder match {
338               case f:File => getComponentName(f)
339               case s:String => s
340               case x => x.toString
341             }
342             if (!componentsLoaded.contains(componentName)) {
343               val config = ConfigUtil.prepareSubConfig(sysConfig, componentName)
344               if (config.hasPath(ComponentManager.KeyEnabled) && !config.getBoolean(ComponentManager.KeyEnabled)) {
345                 log.info(s"Component $componentName not enabled, skipping.")
346               } else {
347                 require(config.hasPath(ComponentManager.KeyManagerClass), "Manager for component not found.")
348                 val className = config.getString(ComponentManager.KeyManagerClass)
349                 loadComponentClass(componentName, className)
350               }
351               componentsLoaded += componentName
352             }
353           } catch {
354             case d:NoClassDefFoundError =>
355               componentLoadFailed(cfName, s"Could not load component [$compFolder]. No Class Def. This could be because the JAR for the component was not found in the component-path")
356             case e:ClassNotFoundException =>
357               componentLoadFailed(cfName, s"Could not load component [$cfName]. Class not found. This could be because the JAR for the component was not found in the component-path", Some(e))
358             case nf:ComponentNotFoundException =>
359               // this is the one case were we don't set the component as failed
360               log.warning(s"Could not load component [$cfName]. Component invalid. Error: ${nf.getMessage}")
361             case i:IllegalArgumentException =>
362               // this is the one case were we don't set the component as failed
363               log.warning(s"Could not load component [$cfName]. Component invalid. Error: ${i.getMessage}")
364             case c:ConfigException =>
365               componentLoadFailed(cfName, s"Could not load component [$cfName]. Configuration failure", Some(c))
366           }
367       }
368       // schedule a timeout for all components to send back success start status, and after timeout check the status
369       // and if not all started then shutdown the server
370       val startTimeout = componentTimeout.duration
371       context.system.scheduler.scheduleOnce(startTimeout,
372         new Runnable() {
373           def run = {
374             checkStartupStatus()
375           }
376         })
377     } else {
378       log.info("No components registered.")
379       sendComponentInitMessage()
380     }
381   }

383   /**
384    * This function will attempt various ways to find the component name of the jar.
385    * 1. Will check to see if the jar filename is the component name
386    * 2. Will check the mapping list from the config to see if the jar filename maps to a component name
387    * 3. Will grab the first two segments of the filename separated by "-" and use that as the component name
388    *    eg. "wookiee-socko-1.0-SNAPSHOT.jar" will evaluate the component name to "wookiee-socko"
389    * It will perform the checks in the order above
390    *
391    * @param file The file or directory that is the jar or component dir
392    * @return String option if found the name, None otherwise
393    */
394   def getComponentName(file:File) : String = {
395     val config = context.system.settings.config
396     val name = file.getName
397     if (file.isDirectory) {
398       ComponentManager.validateComponentDir(name, file)
399       name
400     } else {
401       if (config.hasPath(name)) {
402         name
403       } else if (config.hasPath(s"${HarnessConstants.KeyComponentMapping}.$name")) {
404         config.getString(s"${HarnessConstants.KeyComponentMapping}.$name")
405       } else {
406         val segments = name.split("-")
407         // example wookiee-socko-1.0-SNAPSHOT.jar
408         val fn = segments(0) + "-" + segments(1)
409         if (config.hasPath(s"$fn")) {
410           fn
411         } else {
412           throw new ComponentNotFoundException("ComponentManager", s"$name component not found")
413         }
414       }
415     }
416   }

422    * @param classLoader Class loader for harness

424   def loadComponentClass(componentName: String, className: String, classLoader: Option[HarnessClassLoader] = None): Option[ActorRef] = {
425     val hClassLoader = classLoader match {
426       case Some(loader) => loader
427       case None => Thread.currentThread.getContextClassLoader.asInstanceOf[HarnessClassLoader]
428     }
429     require(className.nonEmpty, "Manager for component not set.")
430 
431     val clazz = hClassLoader.loadClass(className)
432     var component = None: Option[ActorRef]
433     clazz match {
434       case c if classOf[Component].isAssignableFrom(c) =>
435         component = initComponentActor(componentName, clazz)
436       case _ =>
437         log.warning(s"Could not load component [$componentName]. Not an instance of Component")
438     }
439     component
440   }

450   def initComponentActor[T](componentName:String, clazz:Class[T]) : Option[ActorRef] = {
451     // check to see if the actor exists
452     // need to block in this instance we don't want the system to start prior to
453     // the system components to being fully loaded.
454     context.child(componentName) match {
455       case Some(ref) =>
456         log.info(s"Component [$componentName] already loaded.")
457         Some(ref)
458       case None =>
459         val ref = context.actorOf(Props(clazz, componentName), componentName)
460         log.info(s"Loading component [$componentName] at path [${ref.path.toString}}]")
461         // get the start timeout but default to the component timeout
462         ComponentManager.components(componentName) = ComponentState.Initializing
463         ref ! StartComponent
464         Some(ref)
465     }
466   }
