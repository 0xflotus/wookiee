27 /**
28  * This helper class will keep track of a configuration local to your service/component and update
29  * the configuration whenever ConfigWatcherActor detects changes in any of our config files.
30  *
31  * @author woods
32  *         2/10/15
33  */
34 trait ConfigHelper {
35   this: Actor =>
36 
37   var renewableConfig = context.system.settings.config
38 
39   /**
40    * Should override this method and use it when ConfigChange is received, be sure
41    * to call super.renewConfiguration() though to ensure you get the new values
42    */
43   def renewConfiguration() {
44     ConfigFactory.invalidateCaches()
45     renewableConfig = HarnessActorSystem.getConfig(None)
46   }
47 
48   def configReceive: Receive = {
49     case ConfigChange() =>
50       renewConfiguration()
51   }
52 }
