37 class ConfigWatcherActor extends HActor {
38   val configWatcher = FileSystems.getDefault.newWatchService()
39   var configDir = Paths.get(".")
40   val watchThread = new Thread(new DirectoryWatcher)
41   var brokenKeys = List()
42   var configExists = false
43 
44   override def preStart(): Unit = {
45     super.preStart()
46     val conf = context.system.settings.config
47     ServiceManager.serviceDir(conf) match {
48       case Some(s) =>
49         configDir = s.toPath
50         val dirs = s.listFiles.filter(_.isDirectory)
51 
52         dirs foreach {
53           dir =>
54             val path = Paths.get(dir.getPath.concat("/conf"))
55             if (Files.exists(path)) {
56               log.info("Adding watcher to existing directory {} for any *.conf file changes", path)
57               path.register(configWatcher, Array[WatchEvent.Kind[_]](ENTRY_CREATE, ENTRY_MODIFY), SensitivityWatchEventModifier.HIGH)
58             }
59         }
60         if (dirs.length > 0) {
61           configExists = true
62           watchThread.start()
63         }
64       case None => log.warn("Service dir does not exist, not starting watchers")
65     }
66     System.getProperty("config.file") match {
67       case s: String =>
68         val cPath = new File(s)
69         if (cPath.exists()) {
70           val path = cPath.getParentFile.toPath
71           log.info("Adding watcher to existing directory {} for any *.conf file changes", path)
72           path.register(configWatcher, Array[WatchEvent.Kind[_]](ENTRY_CREATE, ENTRY_MODIFY), SensitivityWatchEventModifier.HIGH)
73 
74           if (!configExists) {
75             configExists = true
76             watchThread.start()
77           }
78         }
79       case null => log.info("Prop config.file not set, not watching for config changes")
80     }
81   }
82 
83   override def postStop(): Unit = {
84     super.postStop()
85     if (watchThread.isAlive) {
86       log.info("Stopping config watcher")
87       watchThread.interrupt()
88     }
89   }
90 
91   override def checkHealth: Future[HealthComponent] = {
92     Future {
93       !configExists || (watchThread.isAlive && !watchThread.isInterrupted) match {
94         case true => HealthComponent("Config Watcher Health", ComponentState.NORMAL, "Config being watched as expected")
95         case false => HealthComponent("Config Watcher Health", ComponentState.CRITICAL, "Config changes are no longer being watched")
96       }
97     }
98   }
99 
100   private class DirectoryWatcher extends Runnable {
101     def run(): Unit = {
102       while (true) {
103 
104         // wait for key to be signaled
105         var key: Option[WatchKey] = None
106         try {
107           key = Some(configWatcher.take())
108         } catch {
109           case x: InterruptedException =>
110             return
111         }
112 
113         key.get.pollEvents().toStream.takeWhile(_.kind() != OVERFLOW) foreach {
114           event =>
115             log.info("Detected alteration on file {}", event.context().toString)
116             // The filename is the context of the event.
117             val ev = event.asInstanceOf[WatchEvent[Path]]
118             val filename = ev.context()
119 
120             try {
121               // Resolve the filename against the directory.
122               val child = configDir.resolve(filename)
123               if (filename.toString.endsWith(".conf")) {
124                 log.info("Config file change detected, {}, sending message to services/components to reload if applicable.", child.toString)
125                 context.parent ! ConfigChange()
126               } else {
127                 log.info("Ignoring change to {} as it is not a .conf file", child.toString)
128               }
129             } catch {
130               case x: IOException =>
131                 log.error("Issue reading file changed in config dir.", x)
132             }
133         }
134 
135         // Reset the key -- this step is critical if you want to receive further watch events.
136         val valid = key.get.reset()
137         if (!valid) {
138           throw new IllegalStateException("Key is no longer in a valid state, can't watch configuration anymore")
139         }
140       }
141     }
142   }
143 }
