31 
32   override def preStart() {
33     log.info("Health manager started: {}", context.self.path)
34   }
35 
36   override def postStop(): Unit = {
37     log.info("Health manager stopped: {}", context.self.path)
38   }
39 
40   override def receive = health orElse {
41     case HealthRequest(typ) =>
42       val caller = sender
43       log.debug("Fetching the system health")
44       import context.dispatcher
45       runChecks onComplete {
46         case Success(s) =>
47           val res = typ match {
48             case HealthResponseType.NAGIOS => "%s|%s".format(s.state.toString.toUpperCase, s.details)
49             case HealthResponseType.LB => if (s.state == ComponentState.CRITICAL) "DOWN" else "UP"
50             case _ => s
51           }
52           caller ! res
53         case Failure(f) => caller ! Status.Failure(f)
54       }
55   }
