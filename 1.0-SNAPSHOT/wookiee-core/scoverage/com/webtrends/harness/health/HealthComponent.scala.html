21 /**
22  * This is the current state of the component. A 'Normal' state refers to
23  * the system operating as it should and without any issues. A 'Degraded' state
24  * refers to the system operating with difficulty. Notification of the state should
25  * be made, however, this state does not mean that the service should be removed from
26  * service. A 'Critical' state refers to the system operating in a fully compromised
27  * state. Notification of the state should occur and the service should be taken out
28  * of service until the issue is resolved.
29  */
30 object ComponentState extends Enumeration {
31   type ComponentState = Value
32   val NORMAL, DEGRADED, CRITICAL = Value
33 }
