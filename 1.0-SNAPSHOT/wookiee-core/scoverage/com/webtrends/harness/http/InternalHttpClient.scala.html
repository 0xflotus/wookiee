35   def getResponseContent(conn: HttpURLConnection): HttpResponseData = {
36     try {
37       val connIn = conn.getInputStream
38       val headers = collection.mutable.Map.empty[String, String]
39       val statusLine = getResponseHeaders(conn, headers)
40 
41       val content = new StringBuilder
42       if (connIn != null) {
43         var in: BufferedReader = null
44         val contentEncoding = headers.getOrElse("Content-Encoding", "")
45         if (contentEncoding == "") {
46           in = new BufferedReader(new InputStreamReader(connIn))
47         } else {
48           if (contentEncoding == "gzip") {
49             in = new BufferedReader(new InputStreamReader(new GZIPInputStream(connIn)))
50           } else if (contentEncoding == "deflate") {
51             in = new BufferedReader(new InputStreamReader(new InflaterInputStream(connIn)))
52           } else {
53             throw new UnsupportedOperationException("Content encoding: " + contentEncoding)
54           }
55         }
56 
57         readAll(in, content)
58         in.close()
59       }
60 
61       HttpResponseData(statusLine, content.toString(), headers)
62     } catch {
63       case ex: Throwable =>
64         System.out.println(ex.toString)
65         getResponseErrorContent(conn)
66     }
67   }

88   /**
89    * Read in headers
90    *
91    * @param conn HTTP Connection
92    * @param headers Collection of header name-values
93    */
94   def getResponseHeaders(conn: URLConnection, headers: collection.mutable.Map[String, String]): String = {
95     headers.clear()
96     var statusLine = ""
97     var i = 0
98     while (i >= 0) {
99       val name = conn.getHeaderFieldKey(i)
100       val value = conn.getHeaderField(i)
101 
102       if (name == null && value == null) {
103         i = -1
104       } else if (name == null) {
105         // Value must be the response code like "HTTP/1.1 404 Not Found"
106         statusLine = value
107         i += 1
108       } else {
109         headers.put(name, value)
110         i += 1
111       }
112     }
113 
114     statusLine
115   }

117   private def readAll(in: BufferedReader, sb: StringBuilder) = {
118     var finished = false
119     while (!finished) {
120       val str = in.readLine()
121       if (str == null) {
122         sb.setLength(sb.length - 1) //Remove last \n
123         finished = true
124       } else {
125         sb.append(str + "\n")
126       }
127     }
128   }
