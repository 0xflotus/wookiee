46 /**
47  * This is a simple http server, which is primarily started for HealthChecks
48  * And will have basic CIDR rules applied to it by default, but will be configurable.
49  * This basic Http handler should not be extended to any functionality outside of
50  * base functionality for the core server. For that use one of the Http Components
51  * like Spray, Socko or Netty.
52  *
53  * @author Michael Cuthbert on 2/3/15.
54  */
55 class SimpleHttpServer(port:Int=8008) extends HActor with ComponentHelper {
56   import context.dispatcher
57   var httpServer:Option[HttpServer] = None
58   val healthActor = context.actorSelection(HarnessConstants.HealthFullName)
59   val servicesActor = context.actorSelection(HarnessConstants.ServicesFullName)
60   val cidrRules = if (context.system.settings.config.hasPath("cidr-rules")) {
61     Some(CIDRRules(context.system.settings.config))
62   } else {
63     None
64   }
65 
66   var portBound = false
67 
68   def respond(httpExchange: HttpExchange, response:String, responseStatus:Int=200, contentType:String="text/plain") = {
69     httpExchange.getResponseHeaders.set("Content-Type", contentType)
70     httpExchange.sendResponseHeaders(responseStatus, response.length())
71     if (response.length > 0) {
72       httpExchange.getResponseBody.write(response.getBytes)
73     }
74     httpExchange.close()
75   }
76 
77   def checkCidrRules(httpExchange: HttpExchange) : Boolean = {
78     cidrRules match {
79       case Some(s) => s.checkCidrRules(httpExchange.getRemoteAddress.getAddress)
80       case None =>
81         // no cidr rules, only allowing localhost to match
82         val ip = httpExchange.getRemoteAddress.getAddress
83         ip.isAnyLocalAddress || ip.isLoopbackAddress
84     }
85   }
86 
87   def handleHealthMessage(httpExchange: HttpExchange, hType:HealthResponseType) = {
88     (healthActor ? HealthRequest(hType)) onComplete {
89       case Success(s) =>
90         s match {
91           case ah:ApplicationHealth =>
92             respond(httpExchange, Json.build(ah).toString)
93           case resp:String => respond(httpExchange, resp)
94           case _ => respond(httpExchange, "Failed to retrieve health", 500)
95         }
96       case Failure(f) => respond(httpExchange, f.getMessage, 500)
97     }
98   }
99 
100   def handleMetricsMessage(httpExchange: HttpExchange) = {
101     componentRequest[StatusRequest, String]("wookiee-metrics", ComponentRequest(StatusRequest("string"))) onComplete {
102       case Success(s) => respond(httpExchange, s.resp)
103       case Failure(f) => respond(httpExchange, f.getMessage, 500)
104     }
105   }
106 
107   def handleServicesMessage(httpExchange: HttpExchange, path:Option[String]) = {
108     val msg = path match {
109       case None => GetMetaData(None)
110       case Some(x) => GetMetaDataByName(x)
111     }
112     (servicesActor ? msg) onComplete {
113       case Success(s) => respond(httpExchange, Json.build(s).toString, 200, "application/json")
114       case Failure(f) => respond(httpExchange, f.getMessage, 500)
115     }
116   }
117 
118   def start() = {
119     if (!isStarted) {
120       try {
121         val server = HttpServer.create(new InetSocketAddress(port), 0)
122         httpServer = Some(server)
123         val internalHandler = new HttpHandler() {
124           override def handle(httpExchange: HttpExchange): Unit = {
125             if (checkCidrRules(httpExchange)) {
126               httpExchange.getRequestURI.getPath match {
127                 case "/healthcheck" | "/healthcheck/full" =>
128                   httpExchange.getRequestURI.getQuery != null && httpExchange.getRequestURI.getQuery.contains("type=lb") match {
129                     case true => handleHealthMessage(httpExchange, HealthResponseType.LB)
130                     case false => handleHealthMessage(httpExchange, HealthResponseType.FULL)
131                   }
132                 case "/healthcheck/lb" => handleHealthMessage(httpExchange, HealthResponseType.LB)
133                 case "/healthcheck/nagios" => handleHealthMessage(httpExchange, HealthResponseType.NAGIOS)
134                 case "/ping" => respond(httpExchange, "pong: ".concat(new DateTime(System.currentTimeMillis(), DateTimeZone.UTC).toString))
135                 case "/metrics" => handleMetricsMessage(httpExchange)
136                 case x if x.startsWith("/services") =>
137                   val p = x.split("/")
138                   val path = if (p.length == 3) {
139                     Some(p(2))
140                   } else {
141                     None
142                   }
143                   handleServicesMessage(httpExchange, path)
144               }
145             } else {
146               // Could be a 403 Forbidden, but obscurity is better
147               respond(httpExchange, "", 404)
148             }
149           }
150         }
151 
152         server.createContext("/healthcheck/lb", internalHandler)
153         server.createContext("/healthcheck", internalHandler)
154         server.createContext("/healthcheck/full", internalHandler)
155         server.createContext("/healthcheck/nagios", internalHandler)
156         server.createContext("/ping", internalHandler)
157         server.createContext("/metrics", internalHandler)
158         server.createContext("/services", internalHandler)
159         server.setExecutor(Executors.newCachedThreadPool())
160         server.start()
161         log.info(s"Internal Http started on port $port")
162       } catch {
163         case ioe:IOException =>
164           log.warn(s"Internal server failed to start, HTTP port $port already bound")
165           log.trace(ioe.getMessage, ioe)
166           portBound = true
167       }
168     }
169   }
170 
171   def isStarted : Boolean = {
172     httpServer match {
173       case Some(s) => true
174       case None => false
175     }
176   }
177 
178   override def preStart() = {
179     if (!isStarted) {
180       start()
181     }
182   }
183 
184   override def postStop() = {
185     if (isStarted) {
186       log.info("Shutting down internal http server")
187       httpServer.get.stop(0)
188     }
189   }
190 
191   /**
192    * This is the health of the current object, by default will be NORMAL
193    * In general this should be overridden to define the health of the current object
194    * For objects that simply manage other objects you shouldn't need to do anything
195    * else, as the health of the children components would be handled by their own
196    * CheckHealth function
197    *
198    * @return
199    */
200   override protected def getHealth: Future[HealthComponent] = {
201     Future {
202       httpServer match {
203         case Some(s) => HealthComponent(self.path.toString, ComponentState.NORMAL, s"Internal HTTP Server started on port $port")
204         case None if portBound => HealthComponent(self.path.toString, ComponentState.NORMAL, s"Internal HTTP server not started, port [$port] already bound by http component")
205         case None => HealthComponent(self.path.toString, ComponentState.CRITICAL, "Internal HTTP Server not started")
206       }
207     }
208   }
