24 private[harness] trait AkkaLogProcessor extends BaseLogProcessor {
25 
26   val empty = ""
27   val emptyTemplate = "{}"
28 
29   /**
30    * Process akka logging events
31    * @param event the akka logging event
32    */
33   def process(event: akka.event.Logging.LogEvent) = {
34     val logger = getLogger(event)
35 
36     event match {
37       case e: akka.event.Logging.Error =>
38 
39         val msg = e.cause match {
40           case akka.event.Logging.Error.NoCause | null =>
41             withContext(event.thread, event.timestamp, Some(event.logSource)) {
42               logger.error(if (e.message != null) e.message.toString else empty, None)
43             }
44 
45           case cause =>
46             withContext(event.thread, event.timestamp, Some(event.logSource)) {
47               logger.error(if (e.message != null) e.message.toString else cause.getLocalizedMessage, e.cause)
48             }
49         }
50 
51       case e: akka.event.Logging.Warning =>
52         withContext(event.thread, event.timestamp, Some(event.logSource)) {
53           logger.warn(emptyTemplate, transformParams(Seq(e.message.asInstanceOf[AnyRef])): _*)
54         }
55 
56       case e: akka.event.Logging.Info =>
57         withContext(event.thread, event.timestamp, Some(event.logSource)) {
58           logger.info(emptyTemplate, transformParams(Seq(e.message.asInstanceOf[AnyRef])): _*)
59         }
60 
61       case e: akka.event.Logging.Debug =>
62         withContext(event.thread, event.timestamp, Some(event.logSource)) {
63           logger.debug(emptyTemplate, transformParams(Seq(e.message.asInstanceOf[AnyRef])): _*)
64         }
65     }
66   }
67 
68 
69   private def getLogger(event: akka.event.Logging.LogEvent): org.slf4j.Logger = {
70     event.logClass match {
71       case c if c == classOf[DummyClassForStringSources] => LoggerFactory getLogger event.logSource
72       case _ => LoggerFactory getLogger event.logClass
73     }
74   }
75 }
