28   private def processLoggerEvent(event: LogEvent) = event match {
29     case Error(logger, message, _, altSource, params, cause) =>
30       cause match {
31         case Some(t) => withContext(event.thread, event.timestamp, event.altSource) {
32           logger.error(format(message, params), t)
33         }
34         case None => withContext(event.thread, event.timestamp, event.altSource) {
