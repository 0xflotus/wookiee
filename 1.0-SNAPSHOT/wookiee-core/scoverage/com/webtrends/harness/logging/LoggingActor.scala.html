23 import com.webtrends.harness.health.ActorHealth

25 class LoggingActor extends Actor with ActorHealth with Slf4jLogging with ActorLoggingAdapter {
26 
27   // Are we routing non-akka logging events to this actor for processing
28   val routeLogging = context.system.settings.config.getBoolean("logging.use-actor")
29 
30   override def preStart(): Unit = {
31     if (routeLogging) {
32       Logger.registerMediator(self)
33     }
34     log.info("Logging manager started: {}", context.self.path)
35   }
36 
37   override def postStop(): Unit = {
38     if (routeLogging) {
39      Logger.unregisterMediator(self)
40     }
41     log.info("Logging manager started: {}", context.self.path)
42   }
43 
44   def receive = health orElse {
45     case InitializeLogger(_) =>
46       sender() ! LoggerInitialized
47 
48     // webtrends log events
49     case event: LogEvent => process(event)
50 
51     // akka events
52     case event: akka.event.Logging.LogEvent => process(event)
53   }
54 
55 }
