24 class ServiceClassLoader(urls: Seq[URL], parent: ClassLoader) extends URLClassLoader(urls.toArray, parent) {

26   /**
27    * This method will perform the same functionality as ClassLoader.loadClass, except that it
28    * will only locate and load the class in it's own class loader.
29    * @param name
30    * @param resolve
31    * @return
32    */
33   def loadClassLocally(name: String, resolve: Boolean) : Option[Class[_]] = {
34     // First see if the class is loaded
35     (findLoadedClass(name) match {
36       case null =>
37         // Since the class was already searched for through the parents and local loader,
38         // we will now just search for the class here
39         Try[Class[_]](findClass(name))
40       case c =>
41         Success(c)
42     }) match {
43       case Success(clazz) =>
44         val l = findLoadedClass(name)
45         if (resolve) resolveClass(clazz)
46         Some(clazz)
47       case _ =>
48         None
49     }
50   }
51 
52   def getLoadedClass(name: String): Option[Class[_]] = findLoadedClass(name) match { case null => None case clazz => Some(clazz) }
53 
54   /**
55    * Appends the specified URL to the list of URLs to search for
56    * classes and resources.
57    * 

58    * If the URL specified is null or is already in the
59    * list of URLs, or if this loader is closed, then invoking this
60    * method has no effect.
61    *
62    * @param url the URL to be added to the search path of URLs
63    */
64   def addServiceURL(url: URL) = addURL(url)
