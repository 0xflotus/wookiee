40 /**
41  * Request to return service meta data
42  */
43 class ServiceManager extends PrepareForShutdown with ServiceLoader {
44 
45   import context.dispatcher
46 
47   override val supervisorStrategy =
48     OneForOneStrategy(maxNrOfRetries = 3, withinTimeRange = 1 minute) {
49       case _: ActorInitializationException => Stop
50       case _: DeathPactException => Stop
51       case _: ActorKilledException => Restart
52       case _: Exception => Restart
53       case _: Throwable => Escalate
54     }
55 
56   override def preStart {
57 
58     load(context)
59     // Tell the harness that the services are loaded. The parent will then
60     // tell us when it is ready so that the services can be notified
61     context.parent ! ServicesReady
62     log.info("Service manager started: {}", context.self.path)
63   }
64 
65   override def postStop(): Unit = {
66     // Kill any custom class loaders
67     services.values foreach {
68       p =>
69         if (p._2.isDefined) p._2.get.close
70     }
71     services.clear
72     log.info("Service manager stopped: {}", context.self.path)
73   }
74 
75   override def receive = super.receive orElse {
76     case SystemReady =>
77       log.info("Notifying Services that we are completely ready.")
78       context.children foreach {
79         p => p ! Ready(getServiceMeta(Some(p.path)).head)
80       }
81 
82     case GetMetaData(path) =>
83       log.info("We have received a message to get service meta data")
84       val meta = getServiceMeta(path)
85       path match {
86         case None => sender() ! meta // Asked for all services
87         case _ => sender() ! meta.head // Asked for only a specific service
88       }
89 
90     case LoadService(name, clazz) =>
91       log.info(s"We have received a message to load service $name")
92       sender ! loadService(name, clazz)
93 
94     case GetMetaDataByName(service) =>
95       log.info("We have received a message to get service meta data")
96       services.filter(_._1.name.equalsIgnoreCase(service)).keys.headOption match {
97         case Some(meta) => sender() ! meta
98         case None => sender() ! Status.Failure(new NoSuchElementException(s"Could not locate the meta information for service $service"))
99       }
100 
101     case RestartService(service) =>
102       log.info(s"We have received a message to restart the service ${service}")
103       val meta = services.filter(_._1.name.equalsIgnoreCase(service)).keys.headOption match {
104         case Some(meta) => services.get(meta).get._1 ! Kill
105         case None =>
106       }
107 
108     case c: ConfigChange =>
109       log.info("Sending config change message to all services...")
110       context.children foreach {
111         p => p ! c
112       }
113 
114     case Terminated(service) =>
115       log.info("Service {} terminated", service.path.name)
116       // Find and nuke the classloader
117       services.filter(_._1.akkaPath.equals(service.path.toString)) foreach {
118         p =>
119           if (p._2._2.isDefined) {
120             p._2._2.get.close
121           }
122           services.remove(p._1)
123       }
124   }
125 
126   private def getServiceMeta(servicePath: Option[ActorPath]): Seq[ServiceMetaData] = {
127     try {
128       log.debug("Service meta requested")
129       servicePath match {
130         case None =>
131           services.keys.toSeq
132         case _ =>
133           Seq(services.filter(p => ActorPath.fromString(p._1.akkaPath).equals(servicePath.get)).keys.headOption.get)
134       }
135     } catch {
136       case e: Throwable =>
137         log.error("Error fetching service meta information", e)
138         Nil
139     }
140   }
141 
142 
143   /**
144    * This is the health of the current object, by default will be NORMAL
145    * In general this should be overridden to define the health of the current object
146    * For objects that simply manage other objects you shouldn't need to do anything
147    * else, as the health of the children components would be handled by their own
148    * CheckHealth function
149    *
150    * @return
151    */
152   override protected def getHealth: Future[HealthComponent] = {
153     log.debug("Service health requested")
154     Future {
155       if (services.size == 0) {
156           HealthComponent(ServiceManager.ServiceManagerName, ComponentState.CRITICAL, "There are no services currently installed")
157       } else if (context.children.size != services.size) {
158           HealthComponent(ServiceManager.ServiceManagerName, ComponentState.CRITICAL, s"There are ${services.size} installed, but only ${context.children.size} that were successfully loaded")
159       } else {
160         HealthComponent(ServiceManager.ServiceManagerName, ComponentState.NORMAL, s"Currently managing ${services.size} service(s)")
161       }
162     }
163   }
164 }

178   /**
179    * Load the configuration files for the deployed services
180    * @param sysConfig
181    * @return
182    */
183   def loadConfigs(sysConfig: Config): Seq[Config] = {
184     serviceDir(sysConfig) match {
185       case Some(s) =>
186         val dirs = s.listFiles.filter(_.isDirectory)
187 
188         val configs = dirs flatMap {
189           dir =>
190             val path = dir.getPath.concat("/conf")
191             log.info("Checking the directory {} for any *.conf files to load", path)
192             for {
193               file <- getConfigFiles(path)
194               conf = allCatch either (ConfigFactory.parseFile(file)) match {
195                 case Left(fail) => log.error(s"Could not load the config file ${file.getAbsolutePath}", fail); None
196                 case Right(value) => Some(value)
197               }
198               if (conf.isDefined)
199             } yield conf.get
200         }
201         configs
202       case None => Seq()
203     }
204   }
205 
206   /**
207    * Get the services directory
208    * @param config The systems main config
209    * @return The service root path, this is option, so if none then not found
210    */
211   def serviceDir(config: Config): Option[File] = {
212     val file = FileSystems.getDefault.getPath(config.getString(HarnessConstants.KeyServicePath)).toFile
213     if (file.exists()) {
214       Some(file)
215     } else {
216       None
217     }
218   }
219 
220   private def getConfigFiles(path: String): Seq[File] = {
221     val root = new File(path)
222     root.exists match {
223       case true =>
224         root.listFiles(new FilenameFilter {
225           def accept(dir: File, name: String): Boolean = name.endsWith(".conf")
226         })
227       case false =>
228         Seq.empty
229     }
230   }
