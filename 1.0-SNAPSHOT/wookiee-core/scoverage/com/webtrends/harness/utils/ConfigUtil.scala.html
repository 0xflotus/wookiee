21 import java.util.concurrent.TimeUnit
22 
23 import akka.util.Timeout
24 import com.typesafe.config.{ConfigException, Config, ConfigFactory}
25 
26 object ConfigUtil {
27 
28   lazy val referenceConfig = ConfigFactory.defaultReference
29 
30   /**
31    * Gets a sub config based on the current config
32    *
33    * @param config root config
34    * @param path path where the sub config resides
35    * @return
36    */
37   def prepareSubConfig(config: Config, path: String): Config = {
38     val c = config.withFallback(referenceConfig)
39     c.checkValid(referenceConfig, path)
40     c.getConfig(path)
41   }
42 
43   /**
44    * Gets a default value from the config
45    *
46    * @param path path to retrieve the value from
47    * @param f function to execute to retrieve the value
48    * @param default the default value you want set if is not available
49    * @tparam T The type for the expected return object
50    * @return Option value with expected type
51    */
52   def getDefaultValue[T](path:String, f: (String) => T, default:T) : T = {
53     try {
54       f(path)
55     } catch {
56       case c:ConfigException => default
57     }
58   }
59 
60   /**
61    * Gets the default timeout from a config
62    *
63    * @param config config containing the timeout
64    * @param path path to the timeout
65    * @param unit TimeUnit
66    * @param default default if not found in config
67    * @return Option value with Timeout
68    */
69   def getDefaultTimeout(config:Config, path:String, default:Timeout, unit:TimeUnit=TimeUnit.SECONDS) : Timeout = {
70     if (config.hasPath(path)) {
71       val duration = config.getDuration(path, unit)
72       Timeout(duration, unit)
73     } else {
74       default
75     }
76   }
77 }
