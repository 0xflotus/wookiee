30 object FileUtil {
31 
32   /**
33    * Gets the filename only for the given file, minus the extension so filename.ext will return filename
34    *
35    * @param f
36    * @return
37    */
38   def getFilename(f:File) : String = {
39     val name = f.getName
40     name.substring(0, name.lastIndexOf('.'))
41   }
42 
43   /**
44    * Gets the extension only for the given file, so filename.ext will return ext
45    *
46    * @param f
47    * @return
48    */
49   def getExtension(f:File) : String = {
50     val name = f.getName
51     name.substring(name.lastIndexOf('.') + 1)
52   }
53 
54   /**
55    * Gets the contents of a file as a string. Obviously don't use this with files that have the potential of
56    * being large.
57    *
58    * @param f
59    * @return
60    */
61   def getFileAsString(f:File) : String = {
62     Source.fromFile(f).mkString
63   }
64 
65   /**
66    * Checks to see if the supplied file is a symbolic link and if it is will return
67    * the real location of the file
68    *
69    * @param f
70    * @return
71    */
72   def getSymLink(f:File) : File = {
73     if (f == null)
74       throw new NullPointerException("File must not be null")
75     val path = FileSystems.getDefault.getPath(f.getPath)
76     if (Files.isSymbolicLink(path)) {
77       f.getCanonicalFile
78     } else {
79       f.getAbsoluteFile
80     }
81   }
82 }
