72 private class JsonParser extends JavaTokenParsers {
73   def obj: Parser[Map[String, Any]] = "{" ~> repsep(member, ",") <~ "}" ^^ (Map.empty ++ _)
74 
75   def arr: Parser[List[Any]] = "[" ~> repsep(value, ",") <~ "]"
76 
77   def member: Parser[(String, Any)] = string ~ ":" ~ value ^^ {
78     case name ~ ":" ~ value => (name, value)
79   }
80 
81   def number: Parser[Any] = floatingPointNumber ^^ {
82     case num if num.matches(".*[.eE].*") => BigDecimal(num)
83     case num => {
84       val rv = num.toLong
85       if (rv >= Int.MinValue && rv <= Int.MaxValue) rv.toInt else rv
86     }
87   }
88 
89   lazy val stringParser = (new EscapedStringParser)
90 
91   def string: Parser[String] = "\"(\\\\\\\\|\\\\\"|[^\"])*+\"".r ^^ { escapedStr =>
92     stringParser.parse(escapedStr)
93   }
94 
95   def value: Parser[Any] = obj | arr | string | number |
96     "null" ^^ (x => null) | "true" ^^ (x => true) | "false" ^^ (x => false)
97 
98   def parse(s: String) = {
99     parseAll(value, s) match {
100       case Success(result, _) => result
101       case x @ Failure(msg, z) => throw new JsonException(x.toString)
102       case x @ Error(msg, _) => throw new JsonException(x.toString)
103     }
104   }

108 /**
109  * An explanation of Scala types and their JSON representations.
110  *
111  * Natively supported scalar types are: Boolean, Int, Long, String.
112  * Collections are Sequence[T], Map[String, T] where T includes the scalars defined above, or
113  * recursive Sequence or Map. You are in flavor country.
114  */
115 object Json {
116   private[Json] def quotedChar(codePoint: Int) = {
117     codePoint match {
118       case c if c > 0xffff =>
119         val chars = Character.toChars(c)
120         "\\u%04x\\u%04x".format(chars(0).toInt, chars(1).toInt)
121       case c if c > 0x7e => "\\u%04x".format(c.toInt)
122       case c => c.toChar
123     }
124   }
125 
126   /**
127    * Quote a string according to "JSON rules".
128    */
129   def quote(s: String) = {
130     val charCount = s.codePointCount(0, s.length)
131     "\"" + 0.to(charCount - 1).map { idx =>
132       s.codePointAt(s.offsetByCodePoints(0, idx)) match {
133         case 0x0d => "\\r"
134         case 0x0a => "\\n"
135         case 0x0c => "\\f"
136         case 0x09 => "\\t"
137         case 0x22 => "\\\""
138         case 0x5c => "\\\\"
139         case 0x2f => "\\/"     // to avoid sending "140         case c => quotedChar(c)
141       }
142     }.mkString("") + "\""
143   }
144 
145   /**
146    * Returns a JSON representation of the given object, as a JsonQuoted object.
147    */
148   def build(obj: Any): JsonQuoted = {
149     val rv = obj match {
150       case JsonQuoted(body) => body
151       case null => "null"
152       case x: Boolean => x.toString
153       case x: Number => x.toString
154       case array: Array[_] => array.map(build(_).body).mkString("[", ",", "]")
155       case list: Seq[_] =>
156         list.map(build(_).body).mkString("[", ",", "]")
157       case map: collection.mutable.LinkedHashMap[_, _] =>
158         map.map { case (k, v) => quote(k.toString) + ":" + build(v).body }.mkString("{", ",", "}")
159       case map: scala.collection.Map[_, _] =>
160         Sorting.stableSort[(Any, Any), String](map.iterator.toList, { case (k, v) => k.toString }).map { case (k, v) =>
161           quote(k.toString) + ":" + build(v).body
162         }.mkString("{", ",", "}")
163       case x: JsonSerializable => x.toJson()
164       case x =>
165         quote(x.toString)
166     }
167     JsonQuoted(rv)
168   }
169 
170   /**
171    * Parses a JSON String representation into its native Scala reprsentation.
172    */
173   def parse(s: String): Any = (new JsonParser).parse(s)
174 }
