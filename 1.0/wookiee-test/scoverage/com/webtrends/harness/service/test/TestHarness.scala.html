68 class TestHarness(conf:Config) {
69   var services = Map[String, ActorRef]()
70   var components = Map[String, ActorRef]()
71   var serviceManager: Option[ActorRef] = None
72   var componentManager: Option[ActorRef] = None
73   var commandManager: Option[ActorRef] = None
74   var config = conf.withFallback(defaultConfig)
75   config = config.withFallback(config.getConfig("wookiee-system")).resolve()
76 
77   implicit val timeout = Timeout(4000, TimeUnit.MILLISECONDS)
78 
79   def start(serviceMap:Option[Map[String, Class[_ <: Service]]]=None,
80             componentMap:Option[Map[String, Class[_ <: Component]]]=None, logLevel:Level=Level.ERROR) : TestHarness = {
81     Harness.externalLogger.info("Starting Harness...")
82     Harness.externalLogger.info(s"Test Harness Config: ${config.toString}")
83     Harness.addShutdownHook
84     Harness.startActorSystem(Some(config))
85     // after we have started the TestHarness we need to set the serviceManager, ComponentManager and CommandManager from the Harness
86     harnessReadyCheck(10.seconds.fromNow)
87     Await.result(TestHarness.rootActor.get ? GetManagers, 5.seconds) match {
88       case m =>
89         val map = m.asInstanceOf[Map[String, ActorRef]]
90         serviceManager = map.get(ServicesName)
91         commandManager = map.get(CommandName)
92         componentManager = map.get(ComponentName)
93         TestHarness.log.get.info("Managers all accounted for")
94     }
95     setLogLevel(logLevel)
96     if (componentMap.isDefined) {
97       loadComponents(componentMap.get)
98     }
99     if (serviceMap.isDefined) {
100       loadServices(serviceMap.get)
101     }
102     componentManager.foreach {_ ! SystemReady}
103     serviceManager.foreach {_ ! SystemReady}
104     this
105   }
106 
107   def stop = {
108     Harness.shutdownActorSystem(false) {
109       // wait a second to make sure it shutdown correctly
110       Thread.sleep(1000)
111     }
112   }
113 
114   def setLogLevel(level:Level) = {
115     TestHarness.log.get.setLogLevel(level)
116   }
117 
118   def harnessReadyCheck(timeOut: Deadline) {
119     while(!timeOut.isOverdue() && !Await.result(TestHarness.rootActor.get ? ReadyCheck, 10.seconds).asInstanceOf[Boolean]) {
120     }
121 
122     if (timeOut.isOverdue()) {
123       throw new IllegalStateException("HarnessActor did not start up")
124     }
125   }
126 
127   def getService(service: String): Option[ActorRef] = {
128     services.get(service)
129   }
130 
131   def getComponent(component: String): Option[ActorRef] = {
132     components.get(component)
133   }
134 
135   def loadComponents(componentMap: Map[String, Class[_ <: Component]]) = {
136     componentMap foreach { p =>
137       componentReady(5.seconds.fromNow, p._1, p._2.getCanonicalName)
138     }
139   }
140   
141   def loadServices(serviceMap: Map[String, Class[_ <: Service]]) = {
142     serviceMap foreach { p =>
143       serviceReady(5.seconds.fromNow, p._1, p._2)
144     }
145   }
146 
147   private def componentReady(timeOut: Deadline, componentName: String, componentClass: String) {
148     if (timeOut.isOverdue()) {
149       throw new IllegalStateException(s"Component $componentName did not start up")
150     }
151     Await.result(componentManager.get ? LoadComponent(componentName, componentClass), 5.seconds) match {
152       case Some(m) =>
153         val component = m.asInstanceOf[ActorRef]
154         TestHarness.log.get.info(s"Loaded component $componentName, ${component.path.toString}")
155         components += (componentName -> component)
156       case None =>
157         throw new Exception("Component not returned")
158     }
159   }
160 
161   private def serviceReady(timeOut: Deadline, serviceName: String, serviceClass: Class[_ <: Service]) {
162     if (timeOut.isOverdue()) {
163       throw new IllegalStateException(s"Service $serviceName did not start up")
164     }
165     Await.result(serviceManager.get ? LoadService(serviceName, serviceClass), 3.seconds) match {
166       case Some(m) =>
167         val service = m.asInstanceOf[ActorRef]
168         TestHarness.log.get.info(s"Loaded service $serviceName, ${service.path.toString}")
169         services += (serviceName -> service)
170       case None =>
171         throw new Exception("Service not returned")
172     }
173   }
