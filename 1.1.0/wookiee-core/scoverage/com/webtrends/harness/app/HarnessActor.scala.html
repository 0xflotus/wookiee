80   private val config = context.system.settings.config
81 
82   implicit val checkTimeout = getDefaultTimeout(config, HarnessConstants.KeyDefaultTimeout, Timeout(5 seconds))
83   val startupTimeout = getDefaultTimeout(config, HarnessConstants.KeyStartupTimeout, Timeout(20 seconds))
84   val prepareShutdownTimeout = getDefaultTimeout(config, HarnessConstants.PrepareToShutdownTimeout, Timeout(5 seconds))

86   var running: Option[Boolean] = Some(false)
87 
88   override val supervisorStrategy =
89     OneForOneStrategy(maxNrOfRetries = 3, withinTimeRange = 1 minute, loggingEnabled = true) {
90       case _: ActorInitializationException => Stop
91       case _: DeathPactException => Stop
92       case _: ActorKilledException => Restart
93       case _: Exception => Restart
94       case _: Throwable => Escalate

97   // The service manager will be created after we have started and all of the system's actors are able to receive messages
98   var serviceActor: Option[ActorRef] = None
99   var componentActor: Option[ActorRef] = None
100   var commandManager: Option[ActorRef] = None
101 
102   // The actor that watches for changes in the harness configuration file and sends out messages when config changes are detected
103   var configWatcherActor: Option[ActorRef] = None
104 
105   override def preStart(): Unit = initialize
106 
107   override def receive = initializing
108 
109   def initializing: Receive = {
110     case CheckHealth => pipe(getHealth(true)) to sender
111     case ComponentInitializationComplete => initializationComplete
112     case ShutdownSystem => shutdownCoreServices
113     case ReadyCheck => sender ! running.get
114   }
115 
116   def processing: Receive = {
117     case CheckHealth => pipe(getHealth(false)) to sender
118 
119     case ServicesReady =>
120       // This message is sent from the service manager that tells us the services are loaded.
121       // Notify the services, components and commands that we are all ready to go
122       serviceActor match {
123         case Some(sa) => sa ! SystemReady
124         case None => // ignore
125       }
126       componentActor match {
127         case Some(ca) => ca ! SystemReady
128         case None => // ignore
129       }
130       commandManager match {
131         case Some(cm) => cm ! SystemReady
132         case None => // ignore
133       }
134     case ReadyCheck => sender ! running.get
135     case GetManagers =>
136       sender ! Map[String, Option[ActorRef]](
137         HarnessConstants.CommandName -> commandManager,
138         HarnessConstants.ServicesName -> serviceActor,
139         HarnessConstants.ComponentName -> componentActor
140       ).collect { case (key, Some(value)) => key -> value }
141     case RestartSystem => Harness.restartActorSystem
142     case c: ConfigChange =>
143       log.info("Received message to reload services/components due to config change")
144       serviceActor.get ! c
145       componentActor.get ! c
146     case ShutdownSystem => shutdownCoreServices
147   }
148 
149   /**
150    * Start the core services
151    */
152   private def initialize = {
153     startHealth
154     startConfigWatcher
155     if (!config.hasPath(HarnessConstants.KeyCommandsEnabled) || config.getBoolean(HarnessConstants.KeyCommandsEnabled)) {
156       // initialize the command manager right at the beginning
157       commandManager = Some(context.actorOf(CommandManager.props, HarnessConstants.CommandName))
158       log.info("Command Manager started: {}", commandManager)
159     }
160     componentActor = Some(context.actorOf(ComponentManager.props, HarnessConstants.ComponentName))
161     log.info("Component manager started: {}", componentActor)
162     componentActor.get ! InitializeComponents
163   }
164 
165   private def initializationComplete: Unit = {
166     // Wait for the child actors above to be loaded before calling on the services
167     Future.traverse(context.children)(child => (child ? Identify("xyz123"))(startupTimeout)) onComplete {
168       case Success(_) =>
169         context.become(processing)
170         // Load any services
171         serviceActor = Some(context.actorOf(ServiceManager.props, HarnessConstants.ServicesName))
172         log.info("Harness manager started: {}", context.self.path)
173         // in general the internal http should always start, but in the cases where you want to turn it off
174         // you can just disable it in the config using internal-http.enable = false
175         // it will also fail silently with a warning if another http component is using the same port as it.
176         if (ConfigUtil.getDefaultValue(HarnessConstants.KeyInternalHttpEnabled, config.getBoolean, true)) {
177           startInternalHTTP(ConfigUtil.getDefaultValue(HarnessConstants.KeyInternalHttpPort, config.getInt, 8080))
178         }
179         running = Some(true)
180       case Failure(t) =>
181         log.error("Error loading the main harness actors", t)
182     }
183   }
184 
185   /**
186    * Complete the shutdown process. This will be called after clustering has been shutdown.
187    */
188   private def shutdownCoreServices: Unit = {
189     log.info("Starting the shutdown process")
190     if (running.isDefined && running.get) {
191       val tmpService = serviceActor
192       val tmpComp = componentActor
193       val tmpCmd = commandManager;
194       prepareForShutdown(tmpService, tmpCmd, tmpComp) andThen {
195         case _ => gracefulShutdown
196       }
197     }
198   }
199 
200   private def prepareForShutdown(actorRefs: Option[ActorRef]*): Future[Unit] = {
201     Future {
202       log.info(s"Prepare For Shutdown")
203       for {
204         optRef <- actorRefs
205         ref <- optRef
206       } yield {
207         ref ! PrepareForShutdown
208       }
209       //Give the message time to propagate through the system.
210       Thread sleep prepareShutdownTimeout.duration.toMillis
211     }
212   }
213 
214 
215   private def gracefulShutdown:Unit = {
216     log.info(s"Starting graceful shutdown")
217     gracefulStop(serviceActor.get, 5 seconds).onComplete {
218       case _ => // in the case of a failure just continue shutting down
219         // Shutdown the components second
220         gracefulStop(componentActor.get, 5 seconds) onComplete {
221           case _ => // We don't care if we could not shutdown properly. Any errors were logged so
222             // just continue
223             // Shutdown the children
224             if (context.children.size > 0) {
225               Future.sequence(context.children map (a => gracefulStop(a, 10 seconds)))
226                 .onComplete {
227                 case Success(_) =>
228                   log.info("Harness subsystems have been shutdown")
229                   context.stop(self)
230                   running = Some(false)
231                 case Failure(fail) =>
232                   log.info("Harness subsystems have not been shutdown properly")
233                   context.stop(self)
234                   running = Some(false)
235               }
236             }
237         }
238     }
239   }

242   /**
243    * Fetch the health of this actor and all of its children.
244    * @return A Future that contains a sequence of the children's HealthComponent
245    */
246   private def getHealth(initializing: Boolean): Future[Seq[HealthComponent]] = {
247     log.debug("We have received a message to check our health")
248     if (initializing) {
249       Future {
250         Seq(HealthComponent("system", ComponentState.DEGRADED, s"The system is still initializing"))
251       }
252     } else {
253       // Call the sections and get their health
254       val future = Future.traverse(context.children)(sys => sys match {
255         case a: ActorRef =>
256           (a ? CheckHealth).mapTo[HealthComponent]
257       })
258 
259       val p = Promise[Seq[HealthComponent]]()
260       future.onComplete({
261         case Failure(f) =>
262           log.error("Error fetching health", f)
263           p failure (f)
264         case Success(answers) =>
265           p success answers.toSeq
266       })
267 
268       p.future
269     }
270   }
